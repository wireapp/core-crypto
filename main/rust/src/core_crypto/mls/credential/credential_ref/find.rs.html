12/// Filters to narrow down the set of credentials returned from various credential-finding methods.
13///
14/// Filters which are unset allow any value.
15///
16/// ## Example
17///
18/// ```ignore
19/// # use core_crypto::{CredentialRef, CredentialFindFilters, ClientIdRef};
20/// # use openmls::prelude::CredentialType;
21/// # let database = todo!();
22/// # let client_id = ClientIdRef::new(&[]);
23/// # smol::block_on(async {
24/// // get all basic credentials for a client
25/// let credentials = CredentialRef::find(
26///     &database,
27///     CredentialFindFilters::builder()
28///         .client_id(&client_id)
29///         .credential_type(CredentialType::Basic)
30///         .build()
31/// ).await.unwrap();
32/// # println!("{credentials:?}");
33/// # });
34/// ```
35#[derive(Debug, Default, typed_builder::TypedBuilder)]
36pub struct FindFilters<'a> {
37    /// Client ID to search for
38    #[builder(default, setter(strip_option))]
39    pub client_id: Option<&'a ClientIdRef>,
40    /// Public key to search for
41    #[builder(default, setter(strip_option))]
42    pub public_key: Option<&'a [u8]>,
43    /// Ciphersuite to search for
44    #[builder(default, setter(strip_option))]
45    pub ciphersuite: Option<Ciphersuite>,
46    /// Credential type to search for
47    #[builder(default, setter(strip_option))]
48    pub credential_type: Option<CredentialType>,
49    /// unix timestamp (seconds) of point of earliest validity to search for
50    #[builder(default, setter(strip_option))]
51    pub earliest_validity: Option<u64>,
52}
53
54impl CredentialRef {
55    /// Find all credentials in the database matching the provided filters.
56    //
57    // Our database does not currently support indices or even in-db searching, so this moves all data
58    // from the DB to the runtime, decodes everything, and then filters. This is obviously suboptimal,
59    // but that's only going to be improved with WPB-20839.
60    pub(crate) async fn find(database: &Database, find_filters: FindFilters<'_>) -> Result<Vec<Self>> {
61        let FindFilters {
62            client_id,
63            ciphersuite,
64            credential_type,
65            public_key,
66            earliest_validity,
67        } = find_filters;
68
69        let partial_credentials = database
70            .find_all::<StoredCredential>(EntityFindParams::default())
71            .await
72            .map_err(KeystoreError::wrap("finding all credentials"))?
73            .into_iter()
74            .filter(|stored| {
75                client_id.is_none_or(|client_id| client_id.as_ref() == stored.session_id)
76                    && earliest_validity.is_none_or(|earliest_validity| earliest_validity == stored.created_at)
77                    && ciphersuite.is_none_or(|ciphersuite| u16::from(ciphersuite) == stored.ciphersuite)
78                    && public_key.is_none_or(|public_key| public_key == stored.public_key)
79            })
80            .map(|stored| -> Result<_> {
81                let mls_credential = MlsCredential::tls_deserialize_exact(&stored.credential)
82                    .map_err(Error::tls_deserialize("Credential"))?;
83                Ok((mls_credential, stored))
84            });
85
86        let mut out = Vec::new();
87        for partial in partial_credentials {
88            let (ref mls_credential, ref stored_credential) = partial?;
89
90            if credential_type.is_some_and(|credential_type| credential_type != mls_credential.credential_type()) {
91                // credential type did not match
92                continue;
93            }
94
95            if let Ok(r#type) = mls_credential.credential_type().try_into()
96                && let Ok(ciphersuite) = stored_credential.ciphersuite.try_into()
97            {
98                out.push(Self {
99                    client_id: ClientId(stored_credential.session_id.clone()),
100                    r#type,
101                    ciphersuite,
102                    earliest_validity: stored_credential.created_at,
103                    public_key: stored_credential.public_key.to_owned(),
104                })
105            }
106        }
107
108        Ok(out)
109    }
110
111    /// Load all credentials from the database
112    pub async fn get_all(database: &Database) -> Result<Vec<Self>> {
113        Self::find(database, FindFilters::default()).await
114    }
115}
