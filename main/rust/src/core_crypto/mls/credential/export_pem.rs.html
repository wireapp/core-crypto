1//! It can be desirable to export a credential as a PEM file.
2//!
3//! For basic credentials, this is a simple PEM-encoding of the credential's public key.
4//!
5//! For x509 credentials, this exports the full certificate chain in a composite file.
6
7use std::debug_assert_matches;
8
9use openmls::prelude::MlsCredentialType;
10use pem::{self, Pem};
11
12use crate::{Credential, CredentialType};
13
14impl Credential {
15    /// A basic credential is nothing but a signing key, so wrap that up in PEM format and call it a day.
16    fn export_basic_pem(&self) -> String {
17        debug_assert_matches!(self.credential_type, CredentialType::Basic);
18        debug_assert_matches!(
19            self.mls_credential.mls_credential(),
20            MlsCredentialType::Basic(_),
21            "coding error: attempted to export basic pem given x509 credential"
22        );
23
24        let public_key = self.signature_key_pair.public();
25        let pem = Pem::new("PUBLIC KEY", public_key);
26        pem::encode(&pem)
27    }
28
29    /// Encode a composite PEM document containing the certificate chain for this credential.
30    fn export_x509_pem(&self) -> String {
31        debug_assert_matches!(self.credential_type, CredentialType::X509);
32        let MlsCredentialType::X509(credential) = self.mls_credential.mls_credential() else {
33            panic!("coding error: attempted to export x509 pem given basic credential");
34        };
35
36        let mut pem_document = String::new();
37
38        // It's not documented, but OpenMLS stores the certificates in its `Certificate` type
39        // leaf-first. One presumes that each subsequent certificate in the chain signs the previous.
40        // https://github.com/wireapp/openmls/blob/c9cde17076508968c9cbead5728454f0a1f60c4f/openmls/src/credentials/mod.rs#L189-L190
41        // This seems to be the conventional ordering for a composite PEM document as well.
42        //
43        // We encode manually instead of using `pem::encode_many` in order to reduce the amount of
44        // unnecessary allocation; we still have to allocate for all the certificates' bytes and also
45        // their output string, but at least we're not also allocating a new vector.
46        for certificate in &credential.certificates {
47            let pem = Pem::new("CERTIFICATE", certificate.as_slice());
48            pem_document.push_str(&pem::encode(&pem));
49        }
50
51        pem_document
52    }
53
54    /// Export a PEM string containing the public portion of this credential.
55    ///
56    /// - Basic credentials export their public key.
57    /// - x509 credentials export the full certificate chain. This enables external tools such as `openssl` to validate
58    ///   the certificate chain.
59    pub fn export_pem(&self) -> String {
60        match self.credential_type {
61            CredentialType::Basic => self.export_basic_pem(),
62            CredentialType::X509 => self.export_x509_pem(),
63        }
64    }
65}
