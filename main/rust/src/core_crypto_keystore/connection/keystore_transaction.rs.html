1//! The methods in this module handle keystore transactions.
2//!
3//! Keystore transactions are "fake", in-memory persistence of database operations over time.
4//! They're required because actual [`rusqlite::Transaction`] is `!Send + !Sync`, and we need
5//! `Send` at a minimum in order to keep the transaction around and manipulate it concurrently
6//! from various tasks.
7
8use crate::{
9    CryptoKeystoreError, CryptoKeystoreResult, Database,
10    traits::{BorrowPrimaryKey, Entity, EntityDatabaseMutation, EntityDeleteBorrowed},
11    transaction::KeystoreTransaction,
12};
13
14/// These impls control the keystore transaction lifecycle.
15impl Database {
16    /// Waits for the current transaction to be committed or rolled back, then starts a new one.
17    pub async fn new_transaction(&self) -> CryptoKeystoreResult<()> {
18        let semaphore = self.transaction_semaphore.acquire_arc().await;
19        let mut transaction_guard = self.transaction.lock().await;
20        debug_assert!(
21            transaction_guard.is_none(),
22            "transaction already existed despite acquiring semaphore"
23        );
24
25        // we'll need to adjust the `KeystoreTransaction` constructor not to require an Arc version of the semaphore
26        // guard
27        let transaction = KeystoreTransaction::new(semaphore).await?;
28        *transaction_guard = Some(transaction);
29        Ok(())
30    }
31
32    /// Start a new transaction if no other transaction is currently in progress.
33    ///
34    /// If a transaction is currently in progress, this will produce a `TransactionInProgress` error.
35    pub async fn try_new_immediate_transaction(&self) -> CryptoKeystoreResult<()> {
36        let semaphore = self
37            .transaction_semaphore
38            .try_acquire_arc()
39            .ok_or(CryptoKeystoreError::TransactionInProgress)?;
40        let mut transaction_guard = self.transaction.lock().await;
41        debug_assert!(
42            transaction_guard.is_none(),
43            "transaction already existed despite acquiring semaphore"
44        );
45
46        // we'll need to adjust the `KeystoreTransaction` constructor not to require an Arc version of the semaphore
47        // guard
48        let transaction = KeystoreTransaction::new(semaphore).await?;
49        *transaction_guard = Some(transaction);
50        Ok(())
51    }
52
53    pub async fn commit_transaction(&self) -> CryptoKeystoreResult<()> {
54        let mut transaction_guard = self.transaction.lock().await;
55        let Some(transaction) = transaction_guard.take() else {
56            return Err(CryptoKeystoreError::MutatingOperationWithoutTransaction);
57        };
58        debug_assert!(
59            transaction_guard.is_none(),
60            "taking the transaction leaves `None` behind"
61        );
62
63        transaction.commit(self).await?;
64        Ok(())
65    }
66
67    pub async fn rollback_transaction(&self) -> CryptoKeystoreResult<()> {
68        let mut transaction_guard = self.transaction.lock().await;
69        if transaction_guard.is_none() {
70            return Err(CryptoKeystoreError::MutatingOperationWithoutTransaction);
71        };
72        *transaction_guard = None;
73        Ok(())
74    }
75
76    /// Do an operation on a keystore transaction on this database.
77    ///
78    /// This is a convenience method abstracting over [`Self::new_transaction`],
79    /// [`Self::commit_transaction`], and [`Self::rollback_transaction`].
80    ///
81    /// If the operation succeeds, the transaction is committed.
82    /// Otherwise, it is rolled back.
83    pub async fn transactionally<R>(
84        &self,
85        operation: impl AsyncFnOnce(&KeystoreTransaction) -> CryptoKeystoreResult<R>,
86    ) -> CryptoKeystoreResult<R> {
87        // we don't actually delegate to the internal lifecycle constructs because
88        // they include mutations and checks we don't need, given that we
89        // know that this function's lifetime exceeds that of the internal transaction.
90        let semaphore = self.transaction_semaphore.acquire_arc().await;
91        let transaction = KeystoreTransaction::new(semaphore).await?;
92
93        let result = operation(&transaction).await;
94        if result.is_ok() {
95            transaction.commit(self).await?;
96        }
97        // otherwise implicit abort on tx drop
98
99        result
100    }
101
102    /// Do an operation on an existing keystore transaction.
103    ///
104    /// This does not create, commit, or abort an existing transaction; it just provides a standardized
105    /// helper to acquire it while creating appropriate errors.
106    pub(crate) async fn with_transaction<R>(
107        &self,
108        operation: impl AsyncFnOnce(&KeystoreTransaction) -> CryptoKeystoreResult<R>,
109    ) -> CryptoKeystoreResult<R> {
110        let guard = self.transaction.lock().await;
111        let Some(transaction) = guard.as_ref() else {
112            return Err(CryptoKeystoreError::MutatingOperationWithoutTransaction);
113        };
114        operation(transaction).await
115    }
116}

118/// These impls are convenience methods to modify the keystore transaction.
119///
120/// Be aware that we are likely to remove these in the future: WPB-23951
121impl Database {
122    pub async fn save<E>(&self, entity: E) -> CryptoKeystoreResult<E::AutoGeneratedFields>
123    where
124        E: Entity + EntityDatabaseMutation + Send + Sync,
125    {
126        self.with_transaction(async |transaction| transaction.save(entity).await)
127            .await
128    }
129
130    pub async fn remove<E>(&self, id: &E::PrimaryKey) -> CryptoKeystoreResult<()>
131    where
132        E: Entity + EntityDatabaseMutation,
133    {
134        self.with_transaction(async |transaction| transaction.remove::<E>(id).await)
135            .await
136    }
137
138    pub async fn remove_borrowed<E>(&self, id: &E::BorrowedPrimaryKey) -> CryptoKeystoreResult<()>
139    where
140        E: Entity + EntityDatabaseMutation + BorrowPrimaryKey + EntityDeleteBorrowed,
141    {
142        self.with_transaction(async |transaction| transaction.remove_borrowed::<E>(id).await)
143            .await
144    }
145
146    /// Restore an entity that was deleted in this transaction by removing it from the deleted list. This is
147    /// idempotent: if the entity doesn't exist in the deleted list, do nothing.
148    ///
149    /// NOTE: This will only work if the entity has been added in an earlier transaction, because otherwise,
150    /// removing its id from the deleted list wouldn't suffice: we'd need to replay its insertion.
151    pub async fn restore<E>(&self, id: &E::BorrowedPrimaryKey) -> CryptoKeystoreResult<()>
152    where
153        E: Entity + EntityDatabaseMutation + BorrowPrimaryKey + EntityDeleteBorrowed,
154    {
155        self.with_transaction(async |transaction| transaction.restore::<E>(id).await)
156            .await
157    }
