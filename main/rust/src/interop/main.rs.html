72#[cfg(not(target_family = "wasm"))]
73fn run_test() -> Result<()> {
74    use std::time::{Duration, Instant};
75
76    use tokio::net::{TcpListener, TcpStream};
77
78    env_logger::init();
79
80    // Check if we have a correct pwd
81    let current_dir = std::env::current_dir()?;
82    if current_dir.ends_with("interop") {
83        let new_cwd = current_dir.parent().unwrap();
84        log::info!("cwd was {current_dir:?}; setting it to {new_cwd:?}");
85        std::env::set_current_dir(new_cwd)?;
86    }
87    let tempdir = tempfile::tempdir()?;
88
89    // because cannot use `#[tokio::main]` on wasm target because tokio does not compile on this target
90    let runtime = tokio::runtime::Builder::new_multi_thread()
91        .enable_all()
92        .build()
93        .unwrap();
94
95    runtime.block_on(async {
96        util::cp_wasm_files(tempdir.path().to_path_buf()).await?;
97
98        let spinner = util::RunningProcess::new("Starting HTTP server...", false);
99        let (server, server_task) = util::bind_http_server(tempdir.path().to_path_buf()).await;
100        let http_server_hwnd = tokio::task::spawn(server_task);
101        spinner.success(format!("HTTP server started {server} [OK]"));
102
103        let mut spinner = util::RunningProcess::new("Starting WebDriver [ChromeDriver & GeckoDriver]...", false);
104        let chrome_driver_addr = TcpListener::bind("127.0.0.1:0").await?.local_addr()?;
105        let mut chrome_webdriver = util::start_webdriver_chrome(&chrome_driver_addr).await?;
106        spinner.update("Sleeping to wait for Webdrivers to get ready...");
107        let timeout = Duration::from_secs(5);
108        let start = Instant::now();
109        while start.elapsed() < timeout {
110            let chrome_ready = TcpStream::connect(&chrome_driver_addr).await.is_ok();
111            if chrome_ready {
112                break;
113            }
114            tokio::time::sleep(Duration::from_millis(100)).await;
115        }
116        spinner.success("WebDriver [OK]");
117
118        run_mls_test(&chrome_driver_addr, &server).await?;
119
120        #[cfg(feature = "proteus")]
121        run_proteus_test(&chrome_driver_addr, &server).await?;
122
123        chrome_webdriver.kill().await?;
124        http_server_hwnd.abort();
125        Ok(())
126    })
127}

134#[cfg(not(target_family = "wasm"))]
135async fn run_mls_test(chrome_driver_addr: &std::net::SocketAddr, web_server: &std::net::SocketAddr) -> Result<()> {
136    use core_crypto::*;
137    use rand::distributions::DistString;
138
139    log::info!("Using ciphersuite {CIPHERSUITE_IN_USE}");
140
141    let mut spinner = util::RunningProcess::new("[MLS] Step 0: Initializing clients & env...", true);
142    let db = Database::open(ConnectionType::InMemory, &DatabaseKey::generate())
143        .await
144        .unwrap();
145
146    let mut clients = create_mls_clients(chrome_driver_addr, web_server).await;
147
148    let conversation_id: ConversationId = MLS_CONVERSATION_ID.into();
149    let config = MlsConversationConfiguration {
150        ciphersuite: CIPHERSUITE_IN_USE.into(),
151        ..Default::default()
152    };
153    let cc = CoreCrypto::new(db);
154    spinner.update("initialized cc...");
155
156    let master_client_id = ClientId::from(b"interop master client".as_slice());
157    let credential = Credential::basic(
158        CIPHERSUITE_IN_USE.into(),
159        master_client_id.clone(),
160        mls_crypto_provider::RustCrypto::default(),
161    )?;
162    spinner.update("created credential...");
163
164    let transaction = cc.new_transaction().await?;
165    let success_provider = Arc::new(MlsTransportSuccessProvider::default());
166    transaction
167        .mls_init(
168            master_client_id.clone().into(),
169            &[CIPHERSUITE_IN_USE.into()],
170            success_provider.clone(),
171        )
172        .await?;
173    let credential_ref = transaction.add_credential(credential).await?;
174    transaction
175        .new_conversation(&conversation_id, &credential_ref, config)
176        .await?;
177
178    spinner.success("[MLS] Step 0: Initializing clients [OK]");
179
180    let spinner = util::RunningProcess::new("[MLS] Step 1: Fetching KeyPackages from clients...", true);
181
182    use tls_codec::Deserialize as _;
183    let mut key_packages = vec![];
184    for c in clients.iter() {
185        let kp = c.get_keypackage().await?;
186        let kp = KeyPackageIn::tls_deserialize(&mut kp.as_slice())?;
187        key_packages.push(kp);
188    }
189
190    spinner.success("[MLS] Step 1: KeyPackages [OK]");
191
192    let spinner = util::RunningProcess::new("[MLS] Step 2: Adding clients to conversation...", true);
193
194    transaction
195        .conversation(&conversation_id)
196        .await?
197        .add_members(key_packages)
198        .await?;
199
200    let conversation_add_msg = success_provider.latest_welcome_message().await;
201    let welcome_raw = conversation_add_msg.tls_serialize_detached()?;
202
203    for c in clients.iter_mut() {
204        let conversation_id_from_welcome = c.process_welcome(&welcome_raw).await?;
205        assert_eq!(conversation_id_from_welcome, conversation_id.as_ref());
206    }
207
208    spinner.success("[MLS] Step 2: Added clients [OK]");
209
210    let mut spinner = util::RunningProcess::new(
211        format!("[MLS] Step 3: Roundtripping messages [0/{ROUNDTRIP_MSG_AMOUNT}]"),
212        true,
213    );
214
215    let mut prng = rand::thread_rng();
216    let mut message;
217    for i in 1..=ROUNDTRIP_MSG_AMOUNT {
218        message = rand::distributions::Alphanumeric.sample_string(&mut prng, 16);
219
220        log::info!(
221            "Master client [{}] >>> {}",
222            hex::encode(master_client_id.as_slice()),
223            message
224        );
225
226        let mut message_to_decrypt = transaction
227            .conversation(&conversation_id)
228            .await?
229            .encrypt_message(&message)
230            .await?;
231
232        for c in clients.iter_mut() {
233            let decrypted_message_raw = c
234                .decrypt_message(conversation_id.as_ref(), &message_to_decrypt)
235                .await?
236                .ok_or_else(|| {
237                    anyhow!(
238                        "[MLS] No message, something went very wrong [Client = {}]",
239                        c.client_type()
240                    )
241                })?;
242
243            let decrypted_message = String::from_utf8(decrypted_message_raw)?;
244
245            log::info!(
246                "{} [{}] <<< {}",
247                c.client_name(),
248                hex::encode(c.client_id()),
249                decrypted_message
250            );
251
252            assert_eq!(
253                decrypted_message,
254                message,
255                "[MLS] Messages differ [Client = {}]",
256                c.client_type()
257            );
258
259            message_to_decrypt = c
260                .encrypt_message(conversation_id.as_ref(), decrypted_message.as_bytes())
261                .await?;
262        }
263
264        let decrypted_master_raw = transaction
265            .conversation(&conversation_id)
266            .await
267            .unwrap()
268            .decrypt_message(message_to_decrypt)
269            .await?
270            .app_msg
271            .ok_or_else(|| anyhow!("[MLS] No message received on master client"))?;
272
273        let decrypted_master = String::from_utf8(decrypted_master_raw)?;
274
275        assert_eq!(decrypted_master, message);
276
277        spinner.update(format!(
278            "[MLS] Step 3: Roundtripping messages... [{i}/{ROUNDTRIP_MSG_AMOUNT}]"
279        ));
280    }
281
282    spinner.success(format!(
283        "[MLS] Step 3: Roundtripping {ROUNDTRIP_MSG_AMOUNT} messages... [OK]"
284    ));
285
286    let spinner = util::RunningProcess::new("[MLS] Step 4: Deleting clients...", true);
287    for client in &mut clients {
288        client.wipe().await?;
289    }
290    spinner.success("[MLS] Step 4: Deleting clients [OK]");
291
292    Ok(())
293}

295#[cfg(all(not(target_family = "wasm"), feature = "proteus"))]
296async fn run_proteus_test(chrome_driver_addr: &std::net::SocketAddr, web_server: &std::net::SocketAddr) -> Result<()> {
297    use core_crypto::*;
298
299    let spinner = util::RunningProcess::new("[Proteus] Step 0: Initializing clients & env...", true);
300
301    let mut clients = create_proteus_clients(chrome_driver_addr, web_server).await;
302
303    for client in &mut clients {
304        client.init().await?;
305    }
306
307    let db = Database::open(ConnectionType::InMemory, &DatabaseKey::generate())
308        .await
309        .unwrap();
310
311    let master_client = CoreCrypto::new(db);
312    let transaction = master_client.new_transaction().await?;
313    transaction.proteus_init().await?;
314
315    let master_fingerprint = master_client.proteus_fingerprint().await?;
316
317    spinner.success("[Proteus] Step 0: Initializing clients [OK]");
318
319    let mut spinner = util::RunningProcess::new("[Proteus] Step 1: Fetching PreKeys from clients...", true);
320
321    let mut client_type_mapping = std::collections::HashMap::new();
322    let mut prekeys = std::collections::HashMap::new();
323    for c in clients.iter_mut() {
324        let client_name = c.client_name();
325        spinner.update(format!("[Proteus] Step 1: PreKeys - {client_name}"));
326        let fingerprint = c.fingerprint().await?;
327        spinner.update(format!("[Proteus] Step 1: PreKeys - {fingerprint}@{client_name}"));
328        client_type_mapping.insert(fingerprint.clone(), client_name.to_string());
329        let prekey = c.get_prekey().await?;
330        prekeys.insert(fingerprint, prekey);
331    }
332
333    spinner.success("[Proteus] Step 1: PreKeys [OK]");
334
335    let mut spinner = util::RunningProcess::new("[Proteus] Step 2: Creating sessions...", true);
336
337    let mut master_sessions = vec![];
338    let mut messages = std::collections::HashMap::new();
339    const PROTEUS_INITIAL_MESSAGE: &[u8] = b"Hello world!";
340    for (fingerprint, prekey) in prekeys {
341        spinner.update(format!(
342            "[Proteus] Step 2: Session master -> {fingerprint}@{}",
343            client_type_mapping[&fingerprint]
344        ));
345        let session_arc = transaction.proteus_session_from_prekey(&fingerprint, &prekey).await?;
346        let mut session = session_arc.write().await;
347        messages.insert(fingerprint, session.encrypt(PROTEUS_INITIAL_MESSAGE)?);
348        master_sessions.push(session.identifier().to_string());
349    }
350
351    let session_id_with_master = format!("session-{master_fingerprint}");
352    for c in clients.iter_mut() {
353        let fingerprint = c.fingerprint().await?;
354        spinner.update(format!(
355            "[Proteus] Step 2: Session {fingerprint}@{} -> master",
356            c.client_name()
357        ));
358        let message = messages.remove(&fingerprint).unwrap();
359        let message_recv = c.session_from_message(&session_id_with_master, &message).await?;
360        assert_eq!(PROTEUS_INITIAL_MESSAGE, message_recv);
361    }
362
363    assert!(messages.is_empty());
364
365    spinner.success("[Proteus] Step 2: Creating sessions [OK]");
366
367    let mut spinner = util::RunningProcess::new(
368        format!("[Proteus] Step 3: Roundtripping messages [0/{ROUNDTRIP_MSG_AMOUNT}]"),
369        true,
370    );
371
372    let mut prng = rand::thread_rng();
373    let mut message = [0u8; 128];
374    let mut master_messages_to_decrypt = std::collections::HashMap::new();
375    for i in 0..ROUNDTRIP_MSG_AMOUNT {
376        use rand::RngCore as _;
377
378        prng.fill_bytes(&mut message);
379
380        let mut messages_to_decrypt = transaction.proteus_encrypt_batched(&master_sessions, &message).await?;
381
382        for c in clients.iter_mut() {
383            let fingerprint = c.fingerprint().await?;
384            let decrypted_message = c
385                .decrypt(
386                    &session_id_with_master,
387                    &messages_to_decrypt.remove(&fingerprint).unwrap(),
388                )
389                .await?;
390
391            assert_eq!(
392                decrypted_message,
393                message,
394                "[Proteus] Messages differ [Client = {}::{}]",
395                c.client_name(),
396                c.client_type(),
397            );
398
399            let ciphertext = c.encrypt(&session_id_with_master, &decrypted_message).await?;
400            master_messages_to_decrypt.insert(fingerprint, ciphertext);
401        }
402
403        for (fingerprint, encrypted) in master_messages_to_decrypt.drain() {
404            let decrypted = transaction.proteus_decrypt(&fingerprint, &encrypted).await?;
405            assert_eq!(decrypted, message);
406        }
407
408        spinner.update(format!(
409            "[Proteus] Step 3: Roundtripping messages... [{i}/{ROUNDTRIP_MSG_AMOUNT}]"
410        ));
411    }
412
413    clients.clear();
414
415    spinner.success(format!(
416        "[Proteus] Step 3: Roundtripping {ROUNDTRIP_MSG_AMOUNT} messages... [OK]"
417    ));
418
419    let spinner = util::RunningProcess::new("[Proteus] Step 4: Deleting clients...", true);
420    for client in &mut clients {
421        client.wipe().await?;
422    }
423    spinner.success("[Proteus] Step 4: Deleting clients [OK]");
424
425    Ok(())
