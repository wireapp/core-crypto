9/// Extend an [`Entity`] with db-mutating operations which can be performed when provided with a transaction.
10pub trait EntityDatabaseMutation: Entity + Into<dynamic_dispatch::Entity> {
11    /// The `pre_save` method might generate or update some fields of the item. The canonical example is an `updated_at`
12    /// field.
13    ///
14    /// This type must contain a copy of each modification to the item, so that the caller of a `.save(entity)`
15    /// function can know what has changed and what the new values are.
16    ///
17    /// For types which are not edited on save, `()` is perfectly acceptable for this type.
18    type AutoGeneratedFields: Default;
19
20    /// Adjust self before saving.
21    ///
22    /// This will be called by the transaction's implementation of `save_mut` and should not be called
23    /// as part of the `save` implementation.
24    ///
25    /// Returns, in a format specific to the entity, all the data which has been generated or updated
26    /// within this method.
27    fn pre_save(&mut self) -> CryptoKeystoreResult<Self::AutoGeneratedFields> {
28        Ok(Default::default())
29    }
30
31    /// Use the transaction's interface to save this entity to the database
32    fn save(&self, tx: &Transaction) -> CryptoKeystoreResult<()>;
33
34    /// Use the transaction's interface to count the number of entities of this type in the database.
35    fn count(tx: &Transaction) -> CryptoKeystoreResult<u32>;
36
37    /// Use the transaction's inteface to delete this entity from the database.
38    ///
39    /// Returns `true` if at least one entity was deleted, or `false` if the id was not found in the database.
40    ///
41    /// For entites whose primary key has a distinct borrowed type, it is best to implement this as a direct
42    /// passthrough:
43    ///
44    /// ```rust,ignore
45    /// fn delete(tx: &Self::Transaction, key: &Self::PrimaryKey) -> CoreCryptoKeystoreResult<bool> {
46    ///     <Self as EntityTransactionDeleteBorrowed<'a>>::delete_borrowed(tx, id).await
47    /// }
48    /// ```
49    fn delete(tx: &Transaction, id: &Self::PrimaryKey) -> CryptoKeystoreResult<bool>;
50}

52/// Extend an [`Entity`] with db-mutating operations which can be performed when provided with a transaction.
53pub trait EntityDeleteBorrowed: EntityDatabaseMutation + BorrowPrimaryKey {
54    /// Delete an entity by a borrowed form of its primary key.
55    fn delete_borrowed(tx: &Transaction, id: &Self::BorrowedPrimaryKey) -> CryptoKeystoreResult<bool>
56    where
57        for<'pk> &'pk Self::BorrowedPrimaryKey: KeyType;
58}
