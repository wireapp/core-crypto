72    /// Builds an instance holding private key material. This instance has to be used in the whole
73    /// enrollment process then dropped to clear secret key material.
74    ///
75    /// # Parameters
76    /// * `client_id` - client identifier e.g. `b7ac11a4-8f01-4527-af88-1c30885a7931:6add501bacd1d90e@example.com`
77    /// * `display_name` - human readable name displayed in the application e.g. `Smith, Alice M (QA)`
78    /// * `handle` - user handle e.g. `alice.smith.qa@example.com`
79    /// * `expiry_sec` - generated x509 certificate expiry in seconds
80    #[allow(clippy::too_many_arguments)]
81    pub fn try_new(
82        client_id: ClientId,
83        display_name: String,
84        handle: String,
85        team: Option<String>,
86        expiry_sec: u32,
87        backend: &MlsCryptoProvider,
88        ciphersuite: MlsCiphersuite,
89        sign_keypair: Option<E2eiSignatureKeypair>,
90        #[cfg(not(target_family = "wasm"))] refresh_token: Option<refresh_token::RefreshToken>,
91    ) -> Result<Self> {
92        let alg = ciphersuite.try_into()?;
93        let sign_sk = match sign_keypair {
94            Some(kp) => kp,
95            None => Self::new_sign_key(ciphersuite, backend)?,
96        };
97
98        let client_id = QualifiedE2eiClientId::try_from(client_id.as_slice())?;
99        let client_id = String::try_from(client_id)?;
100        let expiry = core::time::Duration::from_secs(u64::from(expiry_sec));
101        let delegate = RustyE2eIdentity::try_new(alg, sign_sk.clone()).map_err(Error::from)?;
102        Ok(Self {
103            delegate,
104            sign_sk,
105            client_id,
106            display_name,
107            handle,
108            team,
109            expiry,
110            directory: None,
111            account: None,
112            user_authz: None,
113            device_authz: None,
114            valid_order: None,
115            finalize: None,
116            ciphersuite,
117            #[cfg(not(target_family = "wasm"))]
118            refresh_token,
119        })
120    }

244    /// Generates a new client Dpop JWT token. It demonstrates proof of possession of the nonces
245    /// (from wire-server & acme server) and will be verified by the acme server when verifying the
246    /// challenge (in order to deliver a certificate).
247    ///
248    /// Then send it to
249    /// [`POST /clients/{id}/access-token`](https://staging-nginz-https.zinfra.io/api/swagger-ui/#/default/post_clients__cid__access_token)
250    /// on wire-server.
251    ///
252    /// # Parameters
253    /// * `expiry_secs` - of the client Dpop JWT. This should be equal to the grace period set in Team Management
254    /// * `backend_nonce` - you get by calling `GET /clients/token/nonce` on wire-server.
255    ///   See endpoint [definition](https://staging-nginz-https.zinfra.io/api/swagger-ui/#/default/get_clients__client__nonce)
256    /// * `expiry` - token expiry
257    #[allow(clippy::too_many_arguments)]
258    pub fn create_dpop_token(&self, expiry_secs: u32, backend_nonce: String) -> Result<String> {
259        let expiry = core::time::Duration::from_secs(expiry_secs as u64);
260        let authz = self
261            .device_authz
262            .as_ref()
263            .ok_or(Error::OutOfOrderEnrollment("You must first call 'newAuthzResponse()'"))?;
264        let challenge = match authz {
265            E2eiAcmeAuthorization::Device { challenge, .. } => challenge,
266            E2eiAcmeAuthorization::User { .. } => return Err(Error::ImplementationError),
267        };
268        Ok(self.new_dpop_token(
269            &self.client_id,
270            self.display_name.as_str(),
271            challenge,
272            backend_nonce,
273            self.handle.as_str(),
274            self.team.clone(),
275            expiry,
276        )?)
277    }

316    /// Creates a new challenge request.
317    ///
318    /// See [RFC 8555 Section 7.5.1](https://www.rfc-editor.org/rfc/rfc8555.html#section-7.5.1).
319    ///
320    /// # Parameters
321    /// * `id_token` - you get back from Identity Provider
322    /// * `refresh_token` - you get back from Identity Provider to renew the access token
323    /// * `oidc_challenge` - you found after [Self::new_authz_response]
324    /// * `account` - you got from [Self::new_account_response]
325    /// * `previous_nonce` - `replay-nonce` response header from `POST /acme/{provisioner-name}/authz/{authz-id}`
326    pub fn new_oidc_challenge_request(
327        &mut self,
328        id_token: String,
329        #[cfg(not(target_family = "wasm"))] refresh_token: String,
330        previous_nonce: String,
331    ) -> Result<Json> {
332        #[cfg(not(target_family = "wasm"))]
333        {
334            if refresh_token.is_empty() {
335                return Err(Error::InvalidRefreshToken);
336            }
337        }
338        let authz = self
339            .user_authz
340            .as_ref()
341            .ok_or(Error::OutOfOrderEnrollment("You must first call 'newAuthzResponse()'"))?;
342        let challenge = match authz {
343            E2eiAcmeAuthorization::User { challenge, .. } => challenge,
344            E2eiAcmeAuthorization::Device { .. } => return Err(Error::ImplementationError),
345        };
346        let account = self.account.as_ref().ok_or(Error::OutOfOrderEnrollment(
347            "You must first call 'newAccountResponse()'",
348        ))?;
349        let challenge = self.acme_oidc_challenge_request(id_token, challenge, account, previous_nonce)?;
350        let challenge = serde_json::to_vec(&challenge)?;
351        #[cfg(not(target_family = "wasm"))]
352        {
353            self.refresh_token.replace(refresh_token.into());
354        }
355        Ok(challenge)
356    }

358    /// Parses the response from `POST /acme/{provisioner-name}/challenge/{challenge-id}` for the OIDC challenge
359    ///
360    /// See [RFC 8555 Section 7.5.1](https://www.rfc-editor.org/rfc/rfc8555.html#section-7.5.1).
361    ///
362    /// # Parameters
363    /// * `challenge` - http response body
364    pub async fn new_oidc_challenge_response(
365        &mut self,
366        #[cfg(not(target_family = "wasm"))] backend: &MlsCryptoProvider,
367        challenge: Json,
368    ) -> Result<()> {
369        let challenge = serde_json::from_slice(&challenge[..])?;
370        self.acme_new_challenge_response(challenge)?;
371
372        #[cfg(not(target_family = "wasm"))]
373        {
374            // Now that the OIDC challenge is valid, we can store the refresh token for future uses. Note
375            // that we could have persisted it at the end of the enrollment but what if the next enrollment
376            // steps fail ? Is it a reason good enough not to persist the token and ask the user to
377            // authenticate again: probably not.
378            let refresh_token = self.refresh_token.take().ok_or(Error::OutOfOrderEnrollment(
379                "You must first call 'new_oidc_challenge_request()'",
380            ))?;
381            refresh_token.replace(backend).await?;
382        }
383        Ok(())
384    }

458    /// Creates a request for finally fetching the x509 certificate.
459    ///

462    /// # Parameters

472            .take()
473            .ok_or(Error::OutOfOrderEnrollment("You must first call 'finalizeResponse()'"))?;
474        let certificate = self.acme_x509_certificate_request(finalize, account, previous_nonce)?;

479    pub(crate) async fn certificate_response(
480        &mut self,
481        certificate_chain: String,
482        env: &wire_e2e_identity::prelude::x509::revocation::PkiEnvironment,
483    ) -> Result<Vec<Vec<u8>>> {
484        let order = self.valid_order.take().ok_or(Error::OutOfOrderEnrollment(
485            "You must first call 'checkOrderResponse()'",
486        ))?;
487        let certificates = self.acme_x509_certificate_response(certificate_chain, order, Some(env))?;
488
489        // zeroize the private material
490        self.sign_sk.zeroize();
491        self.delegate.sign_kp.zeroize();
492        self.delegate.acme_kp.zeroize();
493
494        #[cfg(not(target_family = "wasm"))]
495        self.refresh_token.zeroize();
496
497        Ok(certificates)
498    }

500    pub(crate) async fn stash(self, backend: &MlsCryptoProvider) -> Result<EnrollmentHandle> {
501        // should be enough to prevent collisions
502        const HANDLE_SIZE: usize = 32;
503
504        let content = serde_json::to_vec(&self)?;
505        let handle = backend
506            .crypto()
507            .random_vec(HANDLE_SIZE)
508            .map_err(MlsError::wrap("generating random vector of bytes"))?;
509        backend
510            .key_store()
511            .save_e2ei_enrollment(&handle, &content)
512            .await
513            .map_err(KeystoreError::wrap("saving e2ei enrollment"))?;
514        Ok(handle)
515    }

623    pub(crate) async fn failsafe_ctx(
624        ctxs: &mut [&mut SessionContext],
625        sc: SignatureScheme,
626    ) -> std::sync::Arc<Option<X509TestChain>> {
627        let mut found_test_chain = None;
628        for ctx in ctxs.iter() {
629            if ctx.x509_test_chain.is_some() {
630                found_test_chain.replace(ctx.x509_test_chain.clone());
631                break;
632            }
633        }
634
635        let found_test_chain = found_test_chain.unwrap_or_else(|| Some(X509TestChain::init_empty(sc)).into());
636
637        // Propagate the chain
638        for ctx in ctxs.iter_mut() {
639            if ctx.x509_test_chain.is_none() {
640                ctx.replace_x509_chain(found_test_chain.clone());
641            }
642        }
643
644        let x509_test_chain = found_test_chain.as_ref().as_ref().unwrap();
645
646        for ctx in ctxs {
647            let _ = x509_test_chain.register_with_central(&ctx.context).await;
648        }
649        found_test_chain
650    }
