21#[cfg_attr(not(target_family = "wasm"), uniffi::export)]

23impl CoreCryptoFfi {
24    /// See [core_crypto::mls::conversation::Conversation::get_device_identities]
25    #[cfg_attr(target_family = "wasm", wasm_bindgen(unchecked_return_type = "WireIdentity[]"))]
26    pub async fn get_device_identities(
27        &self,
28        conversation_id: &ConversationId,
29        device_ids: Vec<ClientIdMaybeArc>,
30    ) -> CoreCryptoResult<DeviceIdentities> {
31        let conversation = self
32            .inner
33            .get_raw_conversation(conversation_id.as_ref())
34            .await
35            .map_err(RecursiveError::mls_client("getting raw conversation"))?;
36        let device_ids = device_ids.into_iter().map(|id| id.as_cc()).collect::<Vec<_>>();
37        let wire_identities = conversation
38            .get_device_identities(&device_ids)
39            .await?
40            .into_iter()
41            .map(WireIdentity::from)
42            .collect::<Vec<_>>();
43        #[cfg(target_family = "wasm")]
44        let wire_identities =
45            serde_wasm_bindgen::to_value(&wire_identities).expect("device identities can always be serialized");
46        Ok(wire_identities)
47    }
48
49    /// See [core_crypto::mls::conversation::Conversation::get_user_identities]
50    #[cfg_attr(
51        target_family = "wasm",
52        wasm_bindgen(unchecked_return_type = "Map<string, WireIdentity[]>")
53    )]
54    pub async fn get_user_identities(
55        &self,
56        conversation_id: &ConversationId,
57        user_ids: Vec<String>,
58    ) -> CoreCryptoResult<UserIdentities> {
59        let conversation = self
60            .inner
61            .get_raw_conversation(conversation_id.as_ref())
62            .await
63            .map_err(RecursiveError::mls_client("getting raw conversation"))?;
64        let identities = conversation.get_user_identities(user_ids.as_slice()).await?;
65        let identities = identities
66            .into_iter()
67            .map(|(k, v)| (k, v.into_iter().map(WireIdentity::from).collect()))
68            .collect::<HashMap<_, Vec<_>>>();
69        #[cfg(target_family = "wasm")]
70        let identities = serde_wasm_bindgen::to_value(&identities).expect("user identities can always be serialized");
71        Ok(identities)
72    }
73}
