19/// A Conversation Guard wraps a `GroupStoreValue<MlsConversation>`.
20///
21/// By doing so, it permits mutable accesses to the conversation. This in turn
22/// means that we don't have to duplicate the entire `MlsConversation` API
23/// on `TransactionContext`.
24#[derive(Debug)]
25pub struct ConversationGuard {
26    inner: GroupStoreValue<MlsConversation>,
27    central_context: TransactionContext,
28}
29
30#[cfg_attr(target_os = "unknown", async_trait::async_trait(?Send))]
31#[cfg_attr(not(target_os = "unknown"), async_trait::async_trait)]
32impl<'inner> ConversationWithMls<'inner> for ConversationGuard {
33    type Context = TransactionContext;
34    type Conversation = RwLockReadGuard<'inner, MlsConversation>;
35
36    async fn context(&self) -> Result<TransactionContext> {
37        Ok(self.central_context.clone())
38    }
39
40    async fn conversation(&'inner self) -> RwLockReadGuard<'inner, MlsConversation> {
41        self.inner.read().await
42    }
43}
44
45impl ConversationGuard {
46    pub(crate) fn new(inner: GroupStoreValue<MlsConversation>, central_context: TransactionContext) -> Self {
47        Self { inner, central_context }
48    }
49
50    /// Perform an operation on a mutable reference to this conversation.
51    ///
52    /// Errors will be propagated.
53    /// When the operation does not error, [`MlsConversation::persist_group_when_changed`] will be called automatically.
54    /// This ensures that persistence cannot be forgotten.
55    ///
56    /// We choose to implement this as a closure instead of a lightweight holding a reference to the coversation
57    /// which calls that method on `Drop` because this way we can ensure we do _not_ automatically call it when there is
58    /// an error.
59    pub(crate) async fn conversation_mut<T>(
60        &mut self,
61        operation: impl AsyncFnOnce(&mut MlsConversation, &Database) -> Result<T>,
62    ) -> Result<T> {
63        // we can't get the database if the transaction context has been invalidated,
64        // and we want to have that error first before evaluating anything in the operation.
65        let database = self
66            .central_context
67            .database()
68            .await
69            .map_err(RecursiveError::transaction("getting database from context"))?;
70        let mut guard = self.inner.write().await;
71        let ok_result = operation(&mut guard, &database).await?;
72        guard.persist_group_when_changed(&database, false).await?;
73        Ok(ok_result)
74    }
75
76    async fn transport(&self) -> Result<Arc<dyn MlsTransport>> {
77        let transport = self.session().await?.transport.clone();
78        Ok(transport)
79    }
80
81    async fn database(&self) -> Result<Database> {
82        self.central_context
83            .database()
84            .await
85            .map_err(RecursiveError::transaction("getting database from context"))
86            .map_err(Into::into)
87    }
88
89    /// Destroys a group locally
90    ///
91    /// # Errors
92    /// KeyStore errors, such as IO
93    pub async fn wipe(&mut self) -> Result<()> {
94        let provider = self.crypto_provider().await?;
95        let mut group_store = self
96            .central_context
97            .mls_groups()
98            .await
99            .map_err(RecursiveError::transaction("getting mls groups"))?;
100
101        let id = self
102            .conversation_mut(async |conversation, _| {
103                conversation.wipe_associated_entities(&provider).await?;
104                Ok(conversation.id().to_owned())
105            })
106            .await?;
107        provider
108            .key_store()
109            .mls_group_delete(&id)
110            .await
111            .map_err(KeystoreError::wrap("deleting mls group"))?;
112        let _ = group_store.remove(&id);
113
114        Ok(())
115    }
116
117    pub(crate) async fn credential(&self) -> Result<Arc<Credential>> {
118        let client = self.session().await?;
119        let inner = self.conversation().await;
120        inner
121            .find_current_credential(&client)
122            .await
123            .map_err(|_| Error::IdentityInitializationError)
124    }
125
126    fn group_info(group_info: Option<GroupInfo>) -> Result<MlsGroupInfoBundle> {
127        let group_info = group_info.ok_or(LeafError::MissingGroupInfo)?;
128        MlsGroupInfoBundle::try_new_full_plaintext(group_info)
129    }
130}
