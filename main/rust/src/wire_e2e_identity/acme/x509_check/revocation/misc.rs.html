9pub(crate) fn check_crl_valid_at_toi(toi: u64, crl: &CertificateList) -> bool {
10    if toi == 0 {
11        return false;
12    }
13
14    if crl.tbs_cert_list.this_update.to_unix_duration().as_secs() > toi {
15        return false;
16    }
17
18    if let Some(nu) = crl.tbs_cert_list.next_update
19        && nu.to_unix_duration().as_secs() < toi
20    {
21        return false;
22    }
23
24    true
25}

27pub(crate) fn get_dp_from_crl(crl: &CertificateList) -> Option<Vec<u8>> {
28    if let Some(exts) = &crl.tbs_cert_list.crl_extensions {
29        for ext in exts {
30            if ext.extn_id == ID_CE_ISSUING_DISTRIBUTION_POINT
31                && let Some(enc_dp) = IssuingDistributionPoint::from_der(ext.extn_value.as_bytes())
32                    .ok()
33                    .and_then(|idp| idp.distribution_point)
34                    .and_then(|dp| dp.to_der().ok())
35            {
36                return Some(enc_dp);
37            }
38        }
39    }
40    None
41}

43pub(crate) fn get_dps_from_cert(cert: &PDVCertificate) -> Option<Vec<Vec<u8>>> {
44    match cert.get_extension(&ID_CE_CRL_DISTRIBUTION_POINTS) {
45        Ok(Some(PDVExtension::CrlDistributionPoints(crl_dps))) => Some(
46            crl_dps
47                .0
48                .iter()
49                .filter_map(|crl_dp| crl_dp.distribution_point.as_ref().and_then(|dp| dp.to_der().ok()))
50                .collect::<Vec<_>>(),
51        ),
52        _ => None,
53    }
