1//! Ciphersuites in bindings
2//!
3//! Both wasm-bindgen and uniffi support emitting enums, as long as they directly implement the enum;
4//! it doesn't work on newtypes around external enums. We therefore redefine the ciphersuites enum
5//! here with appropriate annotations such that it gets exported to all relevant bindings.
6
7use core_crypto::{CiphersuiteName, MlsCiphersuite};
8#[cfg(target_family = "wasm")]
9use wasm_bindgen::prelude::*;
10
11use crate::{CoreCryptoError, CoreCryptoResult};
12
13/// MLS ciphersuites.
14#[allow(non_camel_case_types)]
15#[allow(clippy::upper_case_acronyms)]
16#[derive(Default, Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, derive_more::TryFrom)]
17#[try_from(repr)]
18#[repr(u16)]
19#[cfg_attr(target_family = "wasm", wasm_bindgen, derive(serde::Serialize, serde::Deserialize))]
20#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Enum))]
21pub enum Ciphersuite {
22    /// DH KEM x25519 | AES-GCM 128 | SHA2-256 | Ed25519
23    #[default]
24    MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519 = 0x0001,
25
26    /// DH KEM P256 | AES-GCM 128 | SHA2-256 | EcDSA P256
27    MLS_128_DHKEMP256_AES128GCM_SHA256_P256 = 0x0002,
28
29    /// DH KEM x25519 | Chacha20Poly1305 | SHA2-256 | Ed25519
30    MLS_128_DHKEMX25519_CHACHA20POLY1305_SHA256_Ed25519 = 0x0003,
31
32    /// DH KEM x448 | AES-GCM 256 | SHA2-512 | Ed448
33    MLS_256_DHKEMX448_AES256GCM_SHA512_Ed448 = 0x0004,
34
35    /// DH KEM P521 | AES-GCM 256 | SHA2-512 | EcDSA P521
36    MLS_256_DHKEMP521_AES256GCM_SHA512_P521 = 0x0005,
37
38    /// DH KEM x448 | Chacha20Poly1305 | SHA2-512 | Ed448
39    MLS_256_DHKEMX448_CHACHA20POLY1305_SHA512_Ed448 = 0x0006,
40
41    /// DH KEM P384 | AES-GCM 256 | SHA2-384 | EcDSA P384
42    MLS_256_DHKEMP384_AES256GCM_SHA384_P384 = 0x0007,
43}

45impl From<Ciphersuite> for CiphersuiteName {
46    #[inline]
47    fn from(value: Ciphersuite) -> Self {
48        (value as u16)
49            .try_into()
50            .expect("ffi Ciphersuite is a subset of mls Ciphersuite")
51    }
52}

54impl From<CiphersuiteName> for Ciphersuite {
55    #[inline]

59            .expect("mls Ciphersuite is a subset of ffi Ciphersuite")
60    }

63impl From<Ciphersuite> for MlsCiphersuite {
64    #[inline]
65    fn from(value: Ciphersuite) -> Self {
66        CiphersuiteName::from(value).into()
67    }

70impl From<MlsCiphersuite> for Ciphersuite {
71    #[inline]
72    fn from(value: MlsCiphersuite) -> Self {
73        CiphersuiteName::from(value).into()
74    }
75}

77/// Construct a ciphersuite enum instance from its discriminant.
78#[cfg_attr(target_family = "wasm", wasm_bindgen(js_name = ciphersuiteFromU16))]
79#[cfg_attr(not(target_family = "wasm"), uniffi::export)]
80pub fn ciphersuite_from_u16(discriminant: u16) -> CoreCryptoResult<Ciphersuite> {
81    Ciphersuite::try_from(discriminant).map_err(CoreCryptoError::generic())
82}

86#[cfg_attr(not(target_family = "wasm"), uniffi::export)]
