1use std::borrow::Cow;
2
3/// A Key Type is a type which can act as a key for a database.
4///
5/// This might be a primary key, in which case the key uniquely identifies either 0 or 1 entries in the database.
6/// Or it might be a search key, in which case the key could match any number of entries.
7pub trait KeyType: Send + Sync + Sized {
8    /// Get a unique binary representation of this key.
9    ///
10    /// For simple keys it can just be the borrowed form of the key itself,
11    /// but for complex keys it could be the run length encoding of each field of the key.
12    fn bytes(&self) -> Cow<'_, [u8]>;
13}
14
15/// An owned key type can be converted to from arbitrary bytes.
16pub trait OwnedKeyType: 'static + KeyType {
17    /// Parse some bytes into an instance of this type.
18    ///
19    /// We're just going with `Option` instead of `CryptoKeystoreResult` for now because
20    /// the hopeful assumption is that this is going to be a rare occurrence that doesn't
21    /// need much explanation.
22    fn from_bytes(bytes: &[u8]) -> Option<Self>;
23}
24
25macro_rules! impl_keytype {
26    ($t:ty, |$self:ident| $bytes:expr) => {
27        impl KeyType for $t {
28            fn bytes(&$self) -> Cow<'_, [u8]> {
29                $bytes.into()
30            }
31        }
32    };
33    ($t:ty, |$self:ident| $bytes:expr, |$bytes_id:ident| $from_bytes:expr) => {
34        impl_keytype!($t, |$self| $bytes);
35
36        impl OwnedKeyType for $t {
37            fn from_bytes($bytes_id: &[u8]) -> Option<Self> {
38                $from_bytes
39            }
40        }
41    };
42
43
44}
45
46// useful for unique entities; non-allocating
47impl_keytype!((), |self| Vec::new(), |bytes| bytes.is_empty().then_some(()));
48impl_keytype!(&[u8], |self| *self);
49impl_keytype!(Vec<u8>, |self| self.as_slice(), |bytes| Some(bytes.into()));
50impl_keytype!(&str, |self| self.as_bytes());
51impl_keytype!(String, |self| self.as_bytes(), |bytes| str::from_utf8(bytes)
52    .ok()
53    .map(ToOwned::to_owned));
54
55macro_rules! impl_keytype_for_integer {
56    ($t:ty) => {
57        impl_keytype!($t, |self| Vec::from(self.to_le_bytes()), |bytes| {
58            let array = bytes.try_into().ok()?;
59            Some(<$t>::from_le_bytes(array))
60        });
61    };
62}
