147    /// See [core_crypto::transaction_context::TransactionContext::get_or_create_client_keypackages]
148    pub async fn client_keypackages(
149        &self,
150        ciphersuite: Ciphersuite,
151        credential_type: CredentialType,
152        amount_requested: u32,
153    ) -> CoreCryptoResult<Vec<KeyPackageMaybeArc>> {
154        let kps = self
155            .inner
156            .get_or_create_client_keypackages(ciphersuite.into(), credential_type.into(), amount_requested as usize)
157            .await
158            .map_err(RecursiveError::transaction("getting or creating client keypackages"))?;
159
160        kps.into_iter()
161            .map(|kp| {
162                kp.tls_serialize_detached()
163                    .map(key_package_coerce_maybe_arc)
164                    .map_err(core_crypto::mls::conversation::Error::tls_serialize("keypackage"))
165                    .map_err(RecursiveError::mls_conversation("serializing keypackage"))
166                    .map_err(Into::into)
167            })
168            .collect::<CoreCryptoResult<_>>()
169    }

188        &self,
189        conversation_id: &ConversationId,
190        creator_credential_type: CredentialType,
191        config: ConversationConfiguration,
192    ) -> CoreCryptoResult<()> {
193        let mut lower_cfg = MlsConversationConfiguration {

207            .await?;
208
209        self.inner
210            .new_conversation(conversation_id.as_ref(), creator_credential_type.into(), lower_cfg)
211            .await?;
212        Ok(())

229    /// See [core_crypto::mls::conversation::ConversationGuard::add_members]
230    pub async fn add_clients_to_conversation(
231        &self,
232        conversation_id: &ConversationId,
233        key_packages: Vec<KeyPackageMaybeArc>,
234    ) -> CoreCryptoResult<NewCrlDistributionPoints> {
235        let key_packages = key_packages
236            .into_iter()
237            .map(|kp| {
238                KeyPackageIn::tls_deserialize(&mut kp.as_slice())
239                    .map_err(core_crypto::mls::conversation::Error::tls_deserialize("keypackage"))
240                    .map_err(RecursiveError::mls_conversation("adding members to conversation"))
241                    .map_err(Into::into)
242            })
243            .collect::<CoreCryptoResult<Vec<_>>>()?;
244
245        let mut conversation = self.inner.conversation(conversation_id.as_ref()).await?;
246        let distribution_points = conversation.add_members(key_packages).await?.into();
247        Ok(distribution_points)
248    }

292    /// See [core_crypto::mls::conversation::ConversationGuard::decrypt_message]
293    pub async fn decrypt_message(
294        &self,
295        conversation_id: &ConversationId,
296        payload: Vec<u8>,
297    ) -> CoreCryptoResult<DecryptedMessage> {
298        let conversation_result = self.inner.conversation(conversation_id.as_ref()).await;
299        let decrypted_message = match conversation_result {
300            Err(TransactionError::PendingConversation(mut pending)) => {
301                pending.try_process_own_join_commit(&payload).await?
302            }
303            Ok(mut conversation) => conversation.decrypt_message(&payload).await?,
304            Err(e) => Err(e)?,
305        };
306
307        decrypted_message.try_into()
308    }

351    /// Add a [`Credential`][crate::Credential] to this client.
352    ///
353    /// Note that while an arbitrary number of credentials can be generated,
354    /// those which are added to a CC instance must be distinct in credential type,
355    /// signature scheme, and the timestamp of creation. This timestamp has only
356    /// 1 second of resolution, limiting the number of credentials which
357    /// can be added. This is a known limitation and will be relaxed in the future.
358    pub async fn add_credential(&self, credential: CredentialMaybeArc) -> CoreCryptoResult<CredentialRef> {
359        #[cfg(not(target_family = "wasm"))]
360        let credential = std::sync::Arc::unwrap_or_clone(credential);
361        let credential_ref = self.inner.add_credential(credential.0).await?;
362        Ok(credential_ref.into())
363    }

388    /// Get all credentials from this client which match the provided parameters.
389    ///
390    /// Parameters which are unset or `None` match anything. Those with a particular value find only credentials matching that value.
391    pub async fn find_credentials(
392        &self,
393        client_id: Option<ClientIdMaybeArc>,
394        public_key: Option<Vec<u8>>,
395        ciphersuite: Option<Ciphersuite>,
396        credential_type: Option<CredentialType>,
397        earliest_validity: Option<u64>,
398    ) -> CoreCryptoResult<Vec<CredentialRefMaybeArc>> {
399        let client_id = client_id.as_ref().map(|client_id| client_id.as_cc());
400        let client_id = client_id.as_ref().map(|client_id| client_id.as_ref());
401
402        let signature_scheme = ciphersuite
403            .map(CryptoCiphersuite::from)
404            .map(|ciphersuite| ciphersuite.signature_algorithm());
405
406        let credential_type = credential_type.map(core_crypto::CredentialType::from);
407
408        let find_filters = CredentialFindFilters {
409            client_id,
410            public_key: public_key.as_deref(),
411            signature_scheme,
412            credential_type,
413            earliest_validity,
414        };
415
416        self.inner
417            .find_credentials(find_filters)
418            .await
419            .map(|credentials| {
420                credentials
421                    .into_iter()
422                    .map(CredentialRef::from)
423                    .map(CredentialRef::into_maybe_arc)
424                    .collect()
425            })
426            .map_err(Into::into)
427    }
