7/// A `HistorySecret` encodes sufficient client state that it can be used to instantiate an
8/// ephemeral client.
9#[derive(core_crypto_macros::Debug, Clone, PartialEq, Eq)]
10#[cfg_attr(target_family = "wasm", wasm_bindgen(getter_with_clone))]
11#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Record))]
12pub struct HistorySecret {
13    /// Client id of the associated history client
14    #[cfg_attr(target_family = "wasm", wasm_bindgen(readonly, js_name = clientId))]
15    pub client_id: ClientIdMaybeArc,
16    /// Opaque secret data sufficient to reconstruct a history client.
17    #[cfg_attr(target_family = "wasm", wasm_bindgen(readonly))]
18    pub data: Vec<u8>,
19}
20
21#[cfg(target_family = "wasm")]
22#[wasm_bindgen]
23impl HistorySecret {
24    /// Construct a history secret from client id and raw data
25    #[wasm_bindgen(constructor)]
26    pub fn new(client_id: ClientIdMaybeArc, data: &[u8]) -> Result<HistorySecret, wasm_bindgen::JsError> {
27        Ok(HistorySecret {
28            client_id,
29            data: data.into(),
30        })
31    }
32}

34impl TryFrom<&CoreCryptoHistorySecret> for HistorySecret {
35    type Error = CoreCryptoError;
36
37    fn try_from(value: &CoreCryptoHistorySecret) -> Result<Self, Self::Error> {
38        let client_id = value.client_id.clone();
39        rmp_serde::to_vec(&value)
40            .map_err(CoreCryptoError::generic())
41            .map(|secret| HistorySecret {
42                client_id: ClientId::from_cc(client_id),
43                data: secret,
44            })
45    }
46}

48async fn history_client_inner(history_secret: HistorySecret) -> CoreCryptoResult<CoreCryptoFfi> {
49    let secret =
50        rmp_serde::from_slice::<CoreCryptoHistorySecret>(&history_secret.data).map_err(CoreCryptoError::generic())?;
51    CoreCrypto::history_client(secret)
52        .await
53        .map(|inner| CoreCryptoFfi { inner })
54        .map_err(Into::into)
55}

57#[cfg(target_family = "wasm")]
58#[wasm_bindgen]
59impl CoreCryptoFfi {
60    /// Instantiate a history client.
61    ///
62    /// This client exposes the full interface of `CoreCrypto`, but it should only be used to decrypt messages.
63    /// Other use is a logic error.
64    pub async fn history_client(history_secret: HistorySecret) -> CoreCryptoResult<Self> {
65        history_client_inner(history_secret).await
66    }
67}
