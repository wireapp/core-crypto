3//! Unfortunately, there is no good way to reuse code between uniffi and wasm for an interface

16/// Application data packaged to be encrypted and transmitted in an MLS application message.
17//
18// TODO: We derive Constructor here only because we need to construct an instance in interop.
19// Remove it once we drop the FFI client from interop.
20#[derive(Debug, derive_more::From, derive_more::Into, derive_more::Deref, derive_more::Constructor)]
21pub struct MlsTransportData(core_crypto::TransportData);

23uniffi::custom_type!(MlsTransportData, Vec<u8>, {
24    lower: |key| key.0.to_vec(),
25    try_lift: |vec| {
26        Ok(MlsTransportData(core_crypto::TransportData::from(vec)))
27    }
28});

30/// Used by CoreCrypto to send commits or application messages to the delivery service.
31///
32/// This trait must be implemented before calling any functions that produce commits.
33#[uniffi::export(with_foreign)]
34#[cfg_attr(target_os = "unknown", async_trait::async_trait(?Send))]
35#[cfg_attr(not(target_os = "unknown"), async_trait::async_trait)]
36pub trait MlsTransport: Send + Sync {
37    /// Send a commit bundle to the corresponding endpoint.
38    async fn send_commit_bundle(&self, commit_bundle: CommitBundle) -> MlsTransportResult;
39    /// Prepare a history secret before transmission.
40    async fn prepare_for_transport(&self, history_secret: HistorySecretFfi) -> MlsTransportData;
41}
42
43#[derive(derive_more::Constructor)]
44struct MlsTransportShim {
45    callbacks: Arc<dyn MlsTransport>,
46    #[cfg(feature = "cancellable-transactions")]
47    cancellation_slot: Arc<CancellationSlot>,
48}
49
50impl std::fmt::Debug for MlsTransportShim {
51    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> std::fmt::Result {
52        f.debug_tuple("MlsTransportShim")
53            .field(&fmt::from_fn(|f| write!(f, "{:p}", Arc::as_ptr(&self.callbacks))))
54            .finish()
55    }
56}
57
58#[cfg_attr(target_os = "unknown", async_trait::async_trait(?Send))]
59#[cfg_attr(not(target_os = "unknown"), async_trait::async_trait)]
60impl core_crypto::MlsTransport for MlsTransportShim {
61    async fn send_commit_bundle(&self, commit_bundle: CryptoCommitBundle) -> core_crypto::Result<()> {
62        let commit_bundle = CommitBundle::try_from(commit_bundle)
63            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
64
65        #[cfg(feature = "cancellable-transactions")]
66        {
67            return race_callback(
68                &self.cancellation_slot,
69                self.callbacks.send_commit_bundle(commit_bundle).map_err(Into::into),
70            )
71            .await;
72        }
73
74        #[cfg(not(feature = "cancellable-transactions"))]
75        self.callbacks
76            .send_commit_bundle(commit_bundle)
77            .await
78            .map_err(Into::into)
79    }
80
81    async fn prepare_for_transport(&self, secret: &HistorySecret) -> core_crypto::Result<core_crypto::TransportData> {
82        // This callback is expected to perform only short-running local preparation,
83        // so it is intentionally not raced against transaction cancellation.
84        let client_id = ClientId::from(secret.client_id.clone()).into();
85        let history_secret = rmp_serde::to_vec(&secret)
86            .map(|secret| HistorySecretFfi {
87                client_id,
88                data: secret,
89            })
90            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
91        Ok(self.callbacks.prepare_for_transport(history_secret).await.into())
92    }
93}

95/// In uniffi, `MlsTransport` is a trait which we need to wrap, and we need to provide the transport shim with the
96/// cancellation slot of core crypto.
97#[cfg(feature = "cancellable-transactions")]
98pub(crate) fn callback_shim(
99    callbacks: Arc<dyn MlsTransport>,
100    cancellation_slot: Arc<CancellationSlot>,
101) -> Arc<dyn core_crypto::MlsTransport> {
102    Arc::new(MlsTransportShim::new(callbacks, cancellation_slot))
103}
104
105#[cfg(not(feature = "cancellable-transactions"))]
106pub(crate) fn callback_shim(callbacks: Arc<dyn MlsTransport>) -> Arc<dyn core_crypto::MlsTransport> {
107    Arc::new(MlsTransportShim::new(callbacks))
108}

110#[cfg(feature = "cancellable-transactions")]
111async fn race_callback<T>(
112    slot: &CancellationSlot,
113    callback: impl Future<Output = core_crypto::Result<T>>,
114) -> core_crypto::Result<T> {
115    let Some(token) = slot
116        .current()
117        .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?
118    else {
119        return callback.await;
120    };
121
122    // Futures used with select_biased!{} need to be `Fuse` instances. select_biased!{} prefers the first declared
123    // future in case both finish at the same time.
124    let result = futures_util::select_biased! {
125        _ = token.cancelled().fuse() => Err(core_crypto::Error::ErrorDuringMlsTransport("cancelled via cancellation token".into())),
126        result = callback.fuse() => result,
127    };
128
129    result
130}
