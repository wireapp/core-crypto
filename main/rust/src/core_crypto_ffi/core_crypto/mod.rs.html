43/// CoreCrypto wraps around MLS and Proteus implementations and provides a transactional interface for each.
44#[derive(Debug)]
45#[cfg_attr(target_family = "wasm", wasm_bindgen)]
46#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Object))]
47pub struct CoreCryptoFfi {
48    pub(crate) inner: core_crypto::CoreCrypto,
49}
50
51/// Free function to construct a new `CoreCryptoFfi` instance.
52///
53/// This is necessary because in uniffi async constructors are not supported.
54///
55/// MLS or proteus can be initialized  with [core_crypto::transaction_context::TransactionContext::mls_init] or
56/// [core_crypto::transaction_context::TransactionContext::proteus_init], respectively.
57#[cfg(not(target_family = "wasm"))]
58#[uniffi::export]
59pub async fn core_crypto_new(database: DatabaseMaybeArc) -> CoreCryptoResult<CoreCryptoFfi> {
60    CoreCryptoFfi::new(database).await
61}
62
63impl CoreCryptoFfi {
64    /// Instantiate CC
65    pub async fn new(database: DatabaseMaybeArc) -> CoreCryptoResult<Self> {
66        let configuration = SessionConfig::builder().database(database.to_cc()).build().validate()?;
67        CoreCryptoFfi::from_config(configuration).await
68    }
69
70    async fn from_config(configuration: ValidatedSessionConfig) -> CoreCryptoResult<Self> {
71        #[cfg(target_family = "wasm")]
72        console_error_panic_hook::set_once();
73
74        let client = Session::try_new(configuration).await?;
75        let inner = core_crypto::CoreCrypto::from(client);
76
77        Ok(Self { inner })
78    }
79}
80
81#[cfg(target_family = "wasm")]
82#[wasm_bindgen]
83impl CoreCryptoFfi {
84    /// Asynchronously instantiate CC.
85    ///
86    /// MLS or proteus can be initialized  with [core_crypto::transaction_context::TransactionContext::mls_init] or
87    /// [core_crypto::transaction_context::TransactionContext::proteus_init], respectively.
88    pub async fn async_new(database: DatabaseMaybeArc) -> CoreCryptoResult<CoreCryptoFfi> {
89        CoreCryptoFfi::new(database).await
90    }
91
92    /// See [Session::close]
93    // Note that this is implemented only for Wasm; Uniffi already generates a `close` method which suffices.
94    pub async fn close(self) -> CoreCryptoResult<()> {
95        self.inner.take().close().await.map_err(Into::into)
96    }
97}
