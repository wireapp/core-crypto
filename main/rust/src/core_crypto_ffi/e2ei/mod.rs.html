10/// The end-to-end identity verification state of a conversation.
11///
12/// Note: this does not check pending state (pending commit, pending proposals), so it does not
13/// consider members about to be added or removed.
14#[derive(Debug, Copy, Clone, uniffi::Enum)]
15#[repr(u8)]
16pub enum E2eiConversationState {
17    /// All clients have a valid E2EI certificate.
18    Verified = 1,
19    /// Some clients are either still using Basic credentials or their certificate has expired.
20    NotVerified,
21    /// All clients are still using Basic credentials.
22    ///
23    /// Note: if all clients have expired certificates, `NotVerified` is returned instead.
24    NotEnabled,
25}
26
27impl From<core_crypto::E2eiConversationState> for E2eiConversationState {
28    fn from(value: core_crypto::E2eiConversationState) -> Self {
29        match value {
30            core_crypto::E2eiConversationState::Verified => Self::Verified,
31            core_crypto::E2eiConversationState::NotVerified => Self::NotVerified,
32            core_crypto::E2eiConversationState::NotEnabled => Self::NotEnabled,
33        }
34    }
35}

37impl TryFrom<FfiCiphersuite> for JwsAlgorithm {
38    type Error = CoreCryptoError;
39
40    fn try_from(value: FfiCiphersuite) -> Result<Self, Self::Error> {
41        match value {
42            FfiCiphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519 => Ok(Self::Ed25519),
43            FfiCiphersuite::MLS_128_DHKEMP256_AES128GCM_SHA256_P256 => Ok(Self::P256),
44            FfiCiphersuite::MLS_256_DHKEMP384_AES256GCM_SHA384_P384 => Ok(Self::P384),
45            FfiCiphersuite::MLS_256_DHKEMP521_AES256GCM_SHA512_P521 => Ok(Self::P521),
46            _ => Err(CoreCryptoError::ad_hoc(
47                "ciphersuite is not supported for certificate acquisition",
48            )),
49        }
50    }
51}

53/// Configuration for an X509 credential acquisition flow.
54#[derive(Debug, Clone, uniffi::Record)]
55pub struct X509CredentialAcquisitionConfiguration {
56    /// ACME server hostname.
57    pub acme_url: String,
58    /// OIDC identity provider URL.
59    pub idp_url: String,
60    /// Ciphersuite of the acquired credential.
61    pub ciphersuite: FfiCiphersuite,
62    /// User-visible display name.
63    pub display_name: String,
64    /// Wire client id for the device acquiring the credential.
65    pub client_id: Arc<ClientId>,
66    /// Wire handle without the domain suffix.
67    pub handle: String,
68    /// Wire domain.
69    pub domain: String,
70    /// Optional Wire team id.
71    pub team: Option<String>,
72    /// Certificate validity period in seconds.
73    pub validity_period_secs: u64,
74}
75
76impl X509CredentialAcquisitionConfiguration {
77    fn try_into_core(self) -> CoreCryptoResult<wire_e2e_identity::acquisition::X509CredentialConfiguration> {
78        let client_id = std::str::from_utf8(self.client_id.as_ref().0.as_ref()).map_err(CoreCryptoError::generic())?;
79        let client_id =
80            wire_e2e_identity::E2eiClientId::try_from_qualified(client_id).map_err(CoreCryptoError::generic())?;
81        let sign_alg: JwsAlgorithm = self.ciphersuite.try_into()?;
82
83        Ok(wire_e2e_identity::acquisition::X509CredentialConfiguration {
84            acme_url: self.acme_url,
85            idp_url: self.idp_url,
86            sign_alg,
87            hash_alg: HashAlgorithm::SHA256,
88            display_name: self.display_name,
89            client_id,
90            handle: self.handle,
91            domain: self.domain,
92            team: self.team,
93            validity_period: std::time::Duration::from_secs(self.validity_period_secs),
94        })
95    }
96}

98/// X509 credential acquisition flow.
99///
100/// This allows acquiring a X509 credential for a CoreCrypto client.
101#[derive(uniffi::Object)]
102pub struct X509CredentialAcquisition {
103    state: Mutex<AcquisitionState>,
104    ciphersuite: FfiCiphersuite,
105}
106
107enum AcquisitionState {
108    Initialized(Box<wire_e2e_identity::X509CredentialAcquisition>),
109    InProgress,
110    Finalized,
111}
112
113fn signing_key_bytes(sign_alg: wire_e2e_identity::JwsAlgorithm, signing_key_pem: &str) -> CoreCryptoResult<Vec<u8>> {
114    match sign_alg {
115        JwsAlgorithm::Ed25519 => Ed25519KeyPair::from_pem(signing_key_pem).map(|key| key.to_bytes()),
116        JwsAlgorithm::P256 => ES256KeyPair::from_pem(signing_key_pem).map(|key| key.to_bytes()),
117        JwsAlgorithm::P384 => ES384KeyPair::from_pem(signing_key_pem).map(|key| key.to_bytes()),
118        JwsAlgorithm::P521 => ES512KeyPair::from_pem(signing_key_pem).map(|key| key.to_bytes()),
119    }
120    .map_err(CoreCryptoError::generic())
121}
122
123fn credential_from_acquisition_result(
124    ciphersuite: FfiCiphersuite,
125    signing_key_pem: &str,
126    certificate_chain: Vec<x509_cert::Certificate>,
127) -> CoreCryptoResult<Credential> {
128    let sign_alg = ciphersuite.try_into()?;
129    let signing_key = signing_key_bytes(sign_alg, signing_key_pem)?;
130    let certificate_chain = certificate_chain
131        .into_iter()
132        .map(|cert| {
133            cert.to_der().map_err(|err| CoreCryptoError::E2ei {
134                e2ei_error: err.to_string(),
135            })
136        })
137        .collect::<CoreCryptoResult<Vec<_>>>()?;
138
139    let certificate_bundle = core_crypto::CertificateBundle::from_raw(
140        certificate_chain,
141        signing_key,
142        core_crypto::CipherSuite::from(ciphersuite).signature_algorithm(),
143    );
144
145    core_crypto::Credential::x509(ciphersuite.into(), certificate_bundle)
146        .map(Credential)
147        .map_err(Into::into)
148}

150#[uniffi::export]
151impl X509CredentialAcquisition {
152    /// Create a new credential acquisition
153    #[uniffi::constructor]
154    pub fn new(
155        pki_environment: Arc<PkiEnvironment>,
156        config: X509CredentialAcquisitionConfiguration,
157    ) -> CoreCryptoResult<Self> {
158        let ciphersuite = config.ciphersuite;
159        let inner = wire_e2e_identity::X509CredentialAcquisition::try_new(
160            Arc::new(pki_environment.as_ref().clone().into()),
161            config.try_into_core()?,
162        )?;
163
164        Ok(Self {
165            state: Mutex::new(AcquisitionState::Initialized(inner.into())),
166            ciphersuite,
167        })
168    }
169
170    /// Complete the DPoP and OIDC challenges and return the acquired X509 credential.
171    pub async fn finalize(&self) -> CoreCryptoResult<Credential> {
172        let state = {
173            let mut state = self.state.lock().await;
174            std::mem::replace(&mut *state, AcquisitionState::InProgress)
175        };
176
177        let result = match state {
178            AcquisitionState::Initialized(inner) => inner
179                .complete_dpop_challenge()
180                .await
181                .map_err(|err| CoreCryptoError::E2ei {
182                    e2ei_error: err.to_string(),
183                })?
184                .complete_oidc_challenge()
185                .await
186                .map_err(|err| CoreCryptoError::E2ei {
187                    e2ei_error: err.to_string(),
188                }),
189            AcquisitionState::InProgress => {
190                return Err(CoreCryptoError::ad_hoc(
191                    "x509 credential acquisition is already in progress",
192                ));
193            }
194            AcquisitionState::Finalized => {
195                return Err(CoreCryptoError::ad_hoc(
196                    "x509 credential acquisition has already been finalized",
197                ));
198            }
199        };
200
201        *self.state.lock().await = AcquisitionState::Finalized;
202        let (signing_key_pem, certificate_chain) = result?;
203        credential_from_acquisition_result(self.ciphersuite, signing_key_pem.as_str(), certificate_chain)
204    }
205}
