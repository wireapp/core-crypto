1use std::collections::HashMap;
2
3use core_crypto_keystore::{entities::PersistedMlsGroup, traits::FetchFromDatabase};
4use openmls::group::MlsGroup;
5
6use super::Result;
7use crate::{ConversationConfiguration, ConversationId, KeystoreError, Session, mls::conversation::Conversation};
8
9impl Conversation {
10    /// restore the conversation from a persistence-saved serialized Group State.
11    fn from_serialized_state(session: Session, buf: Vec<u8>) -> Result<Self> {
12        let group: MlsGroup =
13            core_crypto_keystore::deser(&buf).map_err(KeystoreError::wrap("deserializing group state"))?;
14        let id = ConversationId::from(group.group_id().as_slice());
15        let configuration = ConversationConfiguration {
16            cipher_suite: group.ciphersuite().into(),
17            ..Default::default()
18        };
19        let group = group.into();
20
21        Ok(Self {
22            id,
23            group,
24            configuration,
25            session,
26        })
27    }
28
29    /// Load a conversation from the database
30    pub(crate) async fn load(session: Session, id: impl AsRef<[u8]>) -> Result<Option<Self>> {
31        let group = session
32            .database()
33            .get_borrowed::<PersistedMlsGroup>(id.as_ref())
34            .await
35            .map_err(KeystoreError::wrap("finding a persisted mls group"))?;
36        let Some(mut group) = group else { return Ok(None) };
37        let conversation = Self::from_serialized_state(session, std::mem::take(&mut group.state))?;
38        Ok(Some(conversation))
39    }
40
41    /// Effectively [`Database::mls_groups_restore`] but with better types
42    pub(crate) async fn load_all(session: Session) -> Result<HashMap<ConversationId, Self>> {
43        let groups = session
44            .database()
45            .load_all::<PersistedMlsGroup>()
46            .await
47            .map_err(KeystoreError::wrap("finding all persisted mls groups"))?;
48        groups
49            .into_iter()
50            .map(|group| {
51                // we can't just destructure the fields straight out of the group, because we derive `Zeroize`, which
52                // zeroizes on drop, which means we are forced to clone all the group's fields, because
53                // otherwise the drop impl couldn't run.
54                let conversation = Self::from_serialized_state(session.clone(), group.state.clone())?;
55                Ok((group.id.clone().into(), conversation))
56            })
57            .collect()
58    }
59}
