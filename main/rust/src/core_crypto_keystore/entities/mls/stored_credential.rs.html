5/// This type exists so that we can efficiently search for credentials by a variety of metrics at the database level.
6///
7/// This includes some but not all of the fields in `core_crypto::CredentialFindFilters`: those that are actually stored
8/// in the database, and do not require deserializing the `credential` field.
9#[derive(Debug, Default, Clone, Copy, serde::Serialize)]
10pub struct CredentialFindFilters<'a> {
11    /// Hash of public key to search for.
12    pub hash: Option<Sha256Hash>,
13    /// Public key to search for
14    pub public_key: Option<&'a [u8]>,
15    /// Session / Client id to search for
16    pub session_id: Option<&'a [u8]>,
17    /// Ciphersuite to search for
18    pub ciphersuite: Option<u16>,
19    /// unix timestamp (seconds) of point of earliest validity to search for
20    pub earliest_validity: Option<u64>,
21}
22
23impl<'a> KeyType for CredentialFindFilters<'a> {
24    fn bytes(&self) -> std::borrow::Cow<'_, [u8]> {
25        postcard::to_stdvec(self)
26            .expect("serializing these filters cannot fail")
27            .into()
28    }
29}

31/// Entity representing a persisted `Credential`
32#[derive(core_crypto_macros::Debug, Clone, PartialEq, Eq, Zeroize, serde::Serialize, serde::Deserialize)]
33#[zeroize(drop)]
34pub struct StoredCredential {
35    /// Note: this is not a unique identifier, but the session id this credential belongs to.
36    #[sensitive]
37    pub session_id: Vec<u8>,
38    #[sensitive]
39    pub credential: Vec<u8>,
40    pub created_at: u64,
41    pub ciphersuite: u16,
42    #[sensitive]
43    pub public_key: Vec<u8>,
44    #[sensitive]
45    pub private_key: Vec<u8>,
46}

48impl crate::traits::UnifiedEntity for StoredCredential {
49    const COLLECTION_NAME: &'static str = "mls_credentials";
50
51    fn get(conn: &rusqlite::Connection, key: &crate::Sha256Hash) -> crate::CryptoKeystoreResult<Option<Self>> {
52        use rusqlite::OptionalExtension as _;
53        let mut stmt = conn.prepare_cached(
54            "SELECT session_id, credential, unixepoch(created_at) AS created_at, ciphersuite, public_key, private_key \
55             FROM mls_credentials WHERE public_key_sha256 = ?",
56        )?;
57        stmt.query_row([key], |row| {
58            Ok(Self {
59                session_id: row.get("session_id")?,
60                credential: row.get("credential")?,
61                created_at: row.get("created_at")?,
62                ciphersuite: row.get("ciphersuite")?,
63                public_key: row.get("public_key")?,
64                private_key: row.get("private_key")?,
65            })
66        })
67        .optional()
68        .map_err(Into::into)
69    }
70
71    fn count(conn: &rusqlite::Connection) -> crate::CryptoKeystoreResult<u32> {
72        crate::entities::helpers::count_helper::<Self>(conn)
73    }
74
75    fn load_all(conn: &rusqlite::Connection) -> crate::CryptoKeystoreResult<Vec<Self>> {
76        let mut stmt = conn.prepare_cached(
77            "SELECT session_id, credential, unixepoch(created_at) AS created_at, ciphersuite, public_key, private_key \
78             FROM mls_credentials",
79        )?;
80        stmt.query_map([], |row| {
81            Ok(Self {
82                session_id: row.get("session_id")?,
83                credential: row.get("credential")?,
84                created_at: row.get("created_at")?,
85                ciphersuite: row.get("ciphersuite")?,
86                public_key: row.get("public_key")?,
87                private_key: row.get("private_key")?,
88            })
89        })?
90        .collect::<Result<_, _>>()
91        .map_err(Into::into)
92    }
93}

95impl crate::traits::UnifiedEntityDatabaseMutation for StoredCredential {
96    type AutoGeneratedFields = u64;
97
98    fn pre_save(&mut self) -> crate::CryptoKeystoreResult<Self::AutoGeneratedFields> {
99        let created_at = std::time::SystemTime::now()
100            .duration_since(std::time::SystemTime::UNIX_EPOCH)
101            .map_err(|_| crate::CryptoKeystoreError::TimestampError)?
102            .as_secs();
103        self.created_at = created_at;
104        Ok(created_at)
105    }
106
107    fn save(&self, tx: &rusqlite::Transaction) -> crate::CryptoKeystoreResult<()> {
108        use crate::traits::PrimaryKey as _;
109        let mut stmt = tx.prepare_cached(
110            "INSERT INTO mls_credentials \
111             (public_key_sha256, public_key, session_id, credential, created_at, ciphersuite, private_key) \
112             VALUES \
113             (:public_key_sha256, :public_key, :session_id, :credential, datetime(:created_at, 'unixepoch'), :ciphersuite, :private_key)",
114        )?;
115        stmt.execute(rusqlite::named_params![
116            ":public_key_sha256": self.primary_key(),
117            ":public_key": self.public_key,
118            ":session_id": self.session_id,
119            ":credential": self.credential,
120            ":created_at": self.created_at,
121            ":ciphersuite": self.ciphersuite,
122            ":private_key": self.private_key,
123        ])?;
124        Ok(())
125    }
126
127    fn count(tx: &rusqlite::Transaction) -> crate::CryptoKeystoreResult<u32> {
128        crate::entities::helpers::count_helper_tx::<Self>(tx)
129    }
130
131    fn delete(tx: &rusqlite::Transaction, id: &crate::Sha256Hash) -> crate::CryptoKeystoreResult<bool> {
132        crate::entities::helpers::delete_helper::<Self>(tx, "public_key_sha256", *id)
133    }
134}

136impl<'a> crate::traits::UnifiedSearchableEntity<CredentialFindFilters<'a>> for StoredCredential {
137    fn find_all_matching(
138        conn: &rusqlite::Connection,
139        filters: &CredentialFindFilters<'a>,
140    ) -> crate::CryptoKeystoreResult<Vec<Self>> {
141        if let Some(hash) = filters
142            .hash
143            .or_else(|| filters.public_key.map(crate::Sha256Hash::hash_from))
144        {
145            return <Self as crate::traits::UnifiedEntity>::get(conn, &hash).map(|opt| {
146                opt.into_iter()
147                    .filter(|c| <Self as crate::traits::UnifiedSearchableEntity<_>>::matches(c, filters))
148                    .collect()
149            });
150        }
151
152        let mut query =
153            "SELECT session_id, credential, unixepoch(created_at) AS created_at, ciphersuite, public_key, private_key \
154             FROM mls_credentials \
155             WHERE (true OR :ciphersuite OR :created_at OR :session_id) "
156                .to_owned();
157
158        if filters.ciphersuite.is_some() {
159            query.push_str("AND ciphersuite = :ciphersuite ");
160        }
161        if filters.earliest_validity.is_some() {
162            query.push_str("AND unixepoch(created_at) = :created_at ");
163        }
164        if filters.session_id.is_some() {
165            query.push_str("AND session_id = :session_id ");
166        }
167
168        let mut stmt = conn.prepare(&query)?;
169        stmt.query_map(
170            rusqlite::named_params![
171                ":ciphersuite": filters.ciphersuite,
172                ":created_at": filters.earliest_validity,
173                ":session_id": filters.session_id,
174            ],
175            |row| {
176                Ok(Self {
177                    session_id: row.get("session_id")?,
178                    credential: row.get("credential")?,
179                    created_at: row.get("created_at")?,
180                    ciphersuite: row.get("ciphersuite")?,
181                    public_key: row.get("public_key")?,
182                    private_key: row.get("private_key")?,
183                })
184            },
185        )?
186        .collect::<Result<_, _>>()
187        .map_err(Into::into)
188    }
189
190    fn matches(&self, filters: &CredentialFindFilters<'a>) -> bool {
191        use crate::traits::PrimaryKey as _;
192        filters.hash.is_none_or(|hash| hash == self.primary_key())
193            && filters.public_key.is_none_or(|pk| pk == self.public_key)
194            && filters.session_id.is_none_or(|sid| sid == self.session_id)
195            && filters.ciphersuite.is_none_or(|cs| cs == self.ciphersuite)
196            && filters.earliest_validity.is_none_or(|ev| ev == self.created_at)
197    }
