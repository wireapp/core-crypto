8impl ProteusCentral {
9    /// Decrypt a proteus message for an already existing session
10    /// Note: This cannot be used for handshake messages, see [ProteusCentral::session_from_message]
11    pub(crate) async fn decrypt(
12        &mut self,
13        keystore: &Database,
14        session_id: &str,
15        ciphertext: &[u8],
16    ) -> Result<Vec<u8>> {
17        let session = self
18            .proteus_sessions
19            .get_or_fetch(session_id, keystore)
20            .await?
21            .ok_or(LeafError::ConversationNotFound(session_id.as_bytes().into()))
22            .map_err(ProteusError::wrap("getting session"))?;
23
24        let plaintext = session.decrypt(keystore, ciphertext).await?;
25        Self::session_save_by_ref(keystore, session).await?;
26
27        Ok(plaintext)
28    }
29
30    /// Encrypt a message for a session
31    pub(crate) async fn encrypt(&mut self, keystore: &Database, session_id: &str, plaintext: &[u8]) -> Result<Vec<u8>> {
32        let session = self
33            .session(session_id, keystore)
34            .await?
35            .ok_or(LeafError::ConversationNotFound(session_id.as_bytes().into()))
36            .map_err(ProteusError::wrap("getting session"))?;
37
38        let ciphertext = session.encrypt(plaintext)?;
39        Self::session_save_by_ref(keystore, session).await?;
40
41        Ok(ciphertext)
42    }
43
44    /// Encrypts a message for a list of sessions
45    /// This is mainly used for conversations with multiple clients, this allows to minimize FFI roundtrips
46    pub(crate) async fn encrypt_batched(
47        &mut self,
48        keystore: &Database,
49        sessions: &[impl AsRef<str>],
50        plaintext: &[u8],
51    ) -> Result<HashMap<String, Vec<u8>>> {
52        // unfortunately we can't write this as an iterator chain because
53        // the operations are async
54        let mut acc = HashMap::new();
55        for session_id in sessions {
56            if let Some(session) = self.session(session_id.as_ref(), keystore).await? {
57                let identifier = session.identifier.clone();
58                let ciphertext = session.encrypt(plaintext)?;
59                Self::session_save_by_ref(keystore, session).await?;
60                acc.insert(identifier, ciphertext);
61            }
62        }
63        Ok(acc)
64    }
65}

75    async fn can_talk_with_proteus() {
76        #[cfg(not(target_os = "unknown"))]
77        let (path, db_file) = tmp_db_file();
78        #[cfg(target_os = "unknown")]
79        let (path, _) = tmp_db_file();
80
81        let session_id = uuid::Uuid::new_v4().hyphenated().to_string();
82
83        let key = DatabaseKey::generate();
84        let keystore = core_crypto_keystore::Database::open(ConnectionType::Persistent(&path), &key)
85            .await
86            .unwrap();
87        keystore.new_transaction().await.unwrap();
88
89        let mut alice = ProteusCentral::try_new(&keystore).await.unwrap();
90
91        let mut bob = CryptoboxLike::init();
92        let bob_pk_bundle = bob.new_prekey();
93
94        alice
95            .session_from_prekey(&session_id, &bob_pk_bundle.serialise().unwrap())
96            .await
97            .unwrap();
98
99        let message = b"Hello world";
100
101        let encrypted = alice.encrypt(&keystore, &session_id, message).await.unwrap();
102        let decrypted = bob.decrypt(&session_id, &encrypted).await;
103        assert_eq!(decrypted, message);
104
105        let encrypted = bob.encrypt(&session_id, message);
106        let decrypted = alice.decrypt(&keystore, &session_id, &encrypted).await.unwrap();
107        assert_eq!(decrypted, message);
108
109        keystore.commit_transaction().await.unwrap();
110        keystore.wipe().await.unwrap();
111        #[cfg(not(target_os = "unknown"))]
112        drop(db_file);
113    }
