24impl TransactionContext {
25    /// Creates an enrollment instance with private key material you can use in order to fetch
26    /// a new x509 certificate from the acme server.
27    ///
28    /// # Parameters
29    /// * `client_id` - client identifier e.g. `b7ac11a4-8f01-4527-af88-1c30885a7931:6add501bacd1d90e@example.com`
30    /// * `display_name` - human readable name displayed in the application e.g. `Smith, Alice M (QA)`
31    /// * `handle` - user handle e.g. `alice.smith.qa@example.com`
32    /// * `expiry_sec` - generated x509 certificate expiry in seconds
33    pub async fn e2ei_new_enrollment(
34        &self,
35        client_id: ClientId,
36        display_name: String,
37        handle: String,
38        team: Option<String>,
39        expiry_sec: u32,
40        ciphersuite: MlsCiphersuite,
41    ) -> Result<E2eiEnrollment> {
42        let signature_keypair = None; // fresh install without a Basic client. Supplying None will generate a new keypair
43        E2eiEnrollment::try_new(
44            client_id,
45            display_name,
46            handle,
47            team,
48            expiry_sec,
49            &self
50                .mls_provider()
51                .await
52                .map_err(RecursiveError::transaction("getting mls provider"))?,
53            ciphersuite,
54            signature_keypair,
55            false, // fresh install so no refresh token registered yet
56        )
57        .map_err(RecursiveError::e2e_identity("creating new enrollment"))
58        .map_err(Into::into)
59    }

61    /// Parses the ACME server response from the endpoint fetching x509 certificates and uses it
62    /// to initialize the MLS client with a certificate
63    pub async fn e2ei_mls_init_only(
64        &self,
65        enrollment: &mut E2eiEnrollment,
66        certificate_chain: String,
67        nb_init_key_packages: Option<usize>,
68    ) -> Result<NewCrlDistributionPoints> {
69        let sk = enrollment
70            .get_sign_key_for_mls()
71            .map_err(RecursiveError::e2e_identity("creating new enrollment"))?;
72        let cs = *enrollment.ciphersuite();
73        let certificate_chain = enrollment
74            .certificate_response(
75                certificate_chain,
76                self.mls_provider()
77                    .await
78                    .map_err(RecursiveError::transaction("getting mls provider"))?
79                    .authentication_service()
80                    .borrow()
81                    .await
82                    .as_ref()
83                    .ok_or(Error::PkiEnvironmentUnset)?,
84            )
85            .await
86            .map_err(RecursiveError::e2e_identity("getting certificate response"))?;
87
88        let crl_new_distribution_points = self.extract_dp_on_init(&certificate_chain[..]).await?;
89
90        let private_key = CertificatePrivateKey {
91            value: sk,
92            signature_scheme: cs.signature_algorithm(),
93        };
94
95        let cert_bundle = CertificateBundle {
96            certificate_chain,
97            private_key,
98        };
99        let identifier = ClientIdentifier::X509(HashMap::from([(cs.signature_algorithm(), cert_bundle)]));
100        self.mls_init(identifier, vec![cs], nb_init_key_packages)
101            .await
102            .map_err(RecursiveError::transaction("initializing mls"))?;
103        Ok(crl_new_distribution_points)
104    }

106    /// When x509 new credentials are registered this extracts the new CRL Distribution Point from the end entity certificate
107    /// and all the intermediates
108    async fn extract_dp_on_init(&self, certificate_chain: &[Vec<u8>]) -> Result<NewCrlDistributionPoints> {
109        use x509_cert::der::Decode as _;
110
111        // Own intermediates are not provided by smallstep in the /federation endpoint so we got to intercept them here, at issuance
112        let size = certificate_chain.len();
113        let mut crl_new_distribution_points = HashSet::new();
114        if size > 1 {
115            for int in certificate_chain.iter().skip(1).rev() {
116                let mut crl_dp = self
117                    .e2ei_register_intermediate_ca_der(int)
118                    .await
119                    .map_err(RecursiveError::transaction("registering intermediate ca der"))?;
120                if let Some(crl_dp) = crl_dp.take() {
121                    crl_new_distribution_points.extend(crl_dp);
122                }
123            }
124        }
125
126        let ee = certificate_chain.first().ok_or(Error::InvalidCertificateChain)?;
127        let ee = x509_cert::Certificate::from_der(ee)
128            .map_err(crate::mls::credential::Error::DecodeX509)
129            .map_err(RecursiveError::mls_credential("decoding x509 credential"))?;
130        let mut ee_crl_dp = extract_crl_uris(&ee).map_err(RecursiveError::e2e_identity("extracting crl urls"))?;
131        if let Some(crl_dp) = ee_crl_dp.take() {
132            crl_new_distribution_points.extend(crl_dp);
133        }
134
135        get_new_crl_distribution_points(
136            &self
137                .mls_provider()
138                .await
139                .map_err(RecursiveError::transaction("getting mls provider"))?,
140            crl_new_distribution_points,
141        )
142        .await
143        .map_err(RecursiveError::mls_credential("getting new crl distribution points"))
144        .map_err(Into::into)
145    }

148#[cfg(test)]
149mod tests {
150    use crate::e2e_identity::enrollment::test_utils as e2ei_utils;
151    use crate::mls::conversation::Conversation as _;
152    use crate::test_utils::x509::X509TestChain;
153    use crate::{prelude::*, test_utils::*};
154
155    #[apply(all_cred_cipher)]
156    async fn e2e_identity_should_work(case: TestContext) {
157        use e2ei_utils::E2EI_CLIENT_ID_URI;
158
159        let session = SessionContext::new_uninitialized(&case).await;
160        Box::pin(async move {
161            let x509_test_chain = X509TestChain::init_empty(case.signature_scheme());
162
163            let is_renewal = false;
164
165            let (mut enrollment, cert) = e2ei_utils::e2ei_enrollment(
166                &session,
167                &case,
168                &x509_test_chain,
169                Some(E2EI_CLIENT_ID_URI),
170                is_renewal,
171                e2ei_utils::init_enrollment,
172                e2ei_utils::noop_restore,
173            )
174            .await
175            .unwrap();
176
177            session
178                .transaction
179                .e2ei_mls_init_only(&mut enrollment, cert, Some(INITIAL_KEYING_MATERIAL_COUNT))
180                .await
181                .unwrap();
182
183            // verify the created client can create a conversation
184            let conversation = case
185                .create_conversation_with_credential_type(MlsCredentialType::X509, [&session])
186                .await;
187            conversation
188                .guard()
189                .await
190                .encrypt_message("Hello e2e identity !")
191                .await
192                .unwrap();
193            assert_eq!(
194                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
195                E2eiConversationState::Verified
196            );
197            assert!(
198                session
199                    .transaction
200                    .e2ei_is_enabled(case.signature_scheme())
201                    .await
202                    .unwrap()
203            );
204        })
205        .await
206    }
