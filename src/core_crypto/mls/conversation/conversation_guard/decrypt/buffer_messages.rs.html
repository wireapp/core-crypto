1//! Sometimes, while still in epoch `n`, we receive messages for epoch `n + 1`.
2//! This is because our DS implementation doesn't guarantee that messages arrive in order.
3//! This module deals with buffering these messages until we receive the commit that advances the
4//! group epoch.
5

18#[derive(Clone, Copy, PartialEq, Eq)]
19pub(crate) enum MessageRestorePolicy {
20    /// Retrieve and decrypt pending messages, then clear them from the keystore.
21    DecryptAndClear,
22    /// Clear pending messages from the keystore without decrypting them.
23    ClearOnly,
24}
25
26impl ConversationGuard {
27    pub(super) async fn buffer_future_message(&self, message: impl AsRef<[u8]>) -> Result<()> {
28        let backend = self.crypto_provider().await?;
29        let keystore = backend.keystore();
30        let conversation = self.conversation().await;
31        let pending_msg = MlsPendingMessage {
32            foreign_id: conversation.id().clone(),
33            message: message.as_ref().to_vec(),
34        };
35        keystore
36            .save::<MlsPendingMessage>(pending_msg)
37            .await
38            .map_err(KeystoreError::wrap("saving pending mls message"))?;
39        Ok(())
40    }
41
42    pub(super) async fn restore_and_clear_pending_messages(
43        &mut self,
44    ) -> Result<Option<Vec<MlsBufferedConversationDecryptMessage>>> {
45        let pending_messages = self
46            .restore_pending_messages(MessageRestorePolicy::DecryptAndClear)
47            .await?;
48
49        if pending_messages.is_some() {
50            let conversation = self.conversation().await;
51            let backend = self.crypto_provider().await?;
52            info!(group_id = Obfuscated::from(conversation.id()); "Clearing all buffered messages for conversation");
53            backend
54                .key_store()
55                .remove::<MlsPendingMessage, _>(conversation.id())
56                .await
57                .map_err(KeystoreError::wrap("removing MlsPendingMessage from keystore"))?;
58        }
59
60        Ok(pending_messages)
61    }

63    #[cfg_attr(target_family = "wasm", async_recursion::async_recursion(?Send))]
64    #[cfg_attr(not(target_family = "wasm"), async_recursion::async_recursion)]
65    pub(crate) async fn restore_pending_messages(
66        &mut self,
67        policy: MessageRestorePolicy,
68    ) -> Result<Option<Vec<MlsBufferedConversationDecryptMessage>>> {
69        let result = async move {
70            let conversation = self.conversation().await;
71            let conversation_id = conversation.id();
72            let backend = self.crypto_provider().await?;
73            let keystore = backend.keystore();
74            if policy == MessageRestorePolicy::ClearOnly {
75                if keystore
76                    .find::<MlsPendingMessage>(conversation_id)
77                    .await
78                    .map_err(KeystoreError::wrap("finding mls pending message by group id"))?
79                    .is_some()
80                {
81                    keystore
82                        .remove::<MlsPendingMessage, _>(conversation_id)
83                        .await
84                        .map_err(KeystoreError::wrap("removing mls pending message"))?;
85                }
86                return Ok(None);
87            }
88
89            let mut pending_messages = keystore
90                .find_all::<MlsPendingMessage>(EntityFindParams::default())
91                .await
92                .map_err(KeystoreError::wrap("finding all mls pending messages"))?
93                .into_iter()
94                .filter(|pm| pm.foreign_id == *conversation_id)
95                .map(|m| -> Result<_> {
96                    let msg = MlsMessageIn::tls_deserialize(&mut m.message.as_slice())
97                        .map_err(Error::tls_deserialize("mls message in"))?;
98                    let ct = match msg.body_as_ref() {
99                        MlsMessageInBody::PublicMessage(m) => m.content_type(),
100                        MlsMessageInBody::PrivateMessage(m) => m.content_type(),
101                        _ => return Err(Error::InappropriateMessageBodyType),
102                    };
103                    Ok((ct as u8, msg))
104                })
105                .collect::<Result<Vec<_>>>()?;
106
107            // We want to restore application messages first, then Proposals & finally Commits
108            // luckily for us that's the exact same order as the [ContentType] enum
109            pending_messages.sort_by(|(a, _), (b, _)| a.cmp(b));
110
111            info!(group_id = Obfuscated::from(conversation_id); "Attempting to restore {} buffered messages", pending_messages.len());
112
113            // Need to drop conversation to allow borrowing `self` again.
114            drop(conversation);
115
116            let mut decrypted_messages = Vec::with_capacity(pending_messages.len());
117            for (_, m) in pending_messages {
118                let decrypted = self
119                    .decrypt_message_inner(m, RecursionPolicy::None)
120                    .await?;
121                decrypted_messages.push(decrypted.into());
122            }
123
124            let decrypted_messages = (!decrypted_messages.is_empty()).then_some(decrypted_messages);
125
126            Ok(decrypted_messages)
127        }
128            .await;
129        if let Err(e) = &result {
130            error!(error:% = e; "Error restoring pending messages");
131        }
132        result
133    }
