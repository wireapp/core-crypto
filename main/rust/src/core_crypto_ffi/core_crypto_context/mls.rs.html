59    /// See [core_crypto::transaction_context::TransactionContext::mls_init]
60    pub async fn mls_init(
61        &self,
62        client_id: ClientIdMaybeArc,
63        ciphersuites: Ciphersuites,
64        nb_key_package: Option<u32>,
65    ) -> CoreCryptoResult<()> {
66        let nb_key_package = nb_key_package
67            .map(usize::try_from)
68            .transpose()
69            .map_err(CoreCryptoError::generic())?;
70        self.inner
71            .mls_init(
72                ClientIdentifier::Basic(client_id.as_cc()),
73                ciphersuites.into_iter().map(Into::into).collect(),
74                nb_key_package,
75            )
76            .await?;
77        Ok(())
78    }

153    /// See [core_crypto::transaction_context::TransactionContext::get_or_create_client_keypackages]
154    pub async fn client_keypackages(
155        &self,
156        ciphersuite: Ciphersuite,
157        credential_type: CredentialType,
158        amount_requested: u32,
159    ) -> CoreCryptoResult<Vec<KeyPackageMaybeArc>> {
160        let kps = self
161            .inner
162            .get_or_create_client_keypackages(ciphersuite.into(), credential_type.into(), amount_requested as usize)
163            .await
164            .map_err(RecursiveError::transaction("getting or creating client keypackages"))?;
165
166        kps.into_iter()
167            .map(|kp| {
168                kp.tls_serialize_detached()
169                    .map(key_package_coerce_maybe_arc)
170                    .map_err(core_crypto::mls::conversation::Error::tls_serialize("keypackage"))
171                    .map_err(RecursiveError::mls_conversation("serializing keypackage"))
172                    .map_err(Into::into)
173            })
174            .collect::<CoreCryptoResult<_>>()
175    }

192    /// See [core_crypto::transaction_context::TransactionContext::new_conversation]
193    pub async fn create_conversation(
194        &self,
195        conversation_id: &ConversationId,
196        creator_credential_type: CredentialType,
197        config: ConversationConfiguration,
198    ) -> CoreCryptoResult<()> {
199        let mut lower_cfg = MlsConversationConfiguration {
200            custom: config.custom.into(),
201            ciphersuite: config.ciphersuite.map(Into::into).unwrap_or_default(),
202            ..Default::default()
203        };
204
205        self.inner
206            .set_raw_external_senders(
207                &mut lower_cfg,
208                config
209                    .external_senders
210                    .into_iter()
211                    .map(|external_sender| external_sender.copy_bytes()),
212            )
213            .await?;
214
215        self.inner
216            .new_conversation(conversation_id.as_ref(), creator_credential_type.into(), lower_cfg)
217            .await?;
218        Ok(())
219    }

235    /// See [core_crypto::mls::conversation::ConversationGuard::add_members]
236    pub async fn add_clients_to_conversation(
237        &self,
238        conversation_id: &ConversationId,
239        key_packages: Vec<KeyPackageMaybeArc>,
240    ) -> CoreCryptoResult<NewCrlDistributionPoints> {
241        let key_packages = key_packages
242            .into_iter()
243            .map(|kp| {
244                KeyPackageIn::tls_deserialize(&mut kp.as_slice())
245                    .map_err(core_crypto::mls::conversation::Error::tls_deserialize("keypackage"))
246                    .map_err(RecursiveError::mls_conversation("adding members to conversation"))
247                    .map_err(Into::into)
248            })
249            .collect::<CoreCryptoResult<Vec<_>>>()?;
250
251        let mut conversation = self.inner.conversation(conversation_id.as_ref()).await?;
252        let distribution_points = conversation.add_members(key_packages).await?.into();
253        Ok(distribution_points)
254    }

298    /// See [core_crypto::mls::conversation::ConversationGuard::decrypt_message]
299    pub async fn decrypt_message(
300        &self,
301        conversation_id: &ConversationId,
302        payload: Vec<u8>,
303    ) -> CoreCryptoResult<DecryptedMessage> {
304        let conversation_result = self.inner.conversation(conversation_id.as_ref()).await;
305        let decrypted_message = match conversation_result {
306            Err(TransactionError::PendingConversation(mut pending)) => {
307                pending.try_process_own_join_commit(&payload).await?
308            }
309            Ok(mut conversation) => conversation.decrypt_message(&payload).await?,
310            Err(e) => Err(e)?,
311        };
312
313        decrypted_message.try_into()
314    }
