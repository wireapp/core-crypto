45impl Entity {
46    /// Downcast this entity to an instance of the requested type.
47    ///
48    /// This increments the smart pointer counter instead of cloning the potentially large item instance.
49    pub(crate) fn downcast<E>(&self) -> Option<Arc<E>>
50    where
51        E: EntityBase + Send + Sync,
52    {
53        match self {
54            Entity::ConsumerData(consumer_data) => consumer_data.clone().downcast_arc(),
55            Entity::HpkePrivateKey(stored_hpke_private_key) => stored_hpke_private_key.clone().downcast_arc(),
56            Entity::StoredKeypackage(stored_keypackage) => stored_keypackage.clone().downcast_arc(),
57            Entity::PskBundle(stored_psk_bundle) => stored_psk_bundle.clone().downcast_arc(),
58            Entity::EncryptionKeyPair(stored_encryption_key_pair) => stored_encryption_key_pair.clone().downcast_arc(),
59            Entity::StoredEpochEncryptionKeypair(stored_epoch_encryption_keypair) => {
60                stored_epoch_encryption_keypair.clone().downcast_arc()
61            }
62            Entity::StoredCredential(stored_credential) => stored_credential.clone().downcast_arc(),
63            Entity::StoredBufferedCommit(stored_buffered_commit) => stored_buffered_commit.clone().downcast_arc(),
64            Entity::PersistedMlsGroup(persisted_mls_group) => persisted_mls_group.clone().downcast_arc(),
65            Entity::PersistedMlsPendingGroup(persisted_mls_pending_group) => {
66                persisted_mls_pending_group.clone().downcast_arc()
67            }
68            Entity::MlsPendingMessage(mls_pending_message) => mls_pending_message.clone().downcast_arc(),
69            Entity::StoredE2eiEnrollment(stored_e2ei_enrollment) => stored_e2ei_enrollment.clone().downcast_arc(),
70            Entity::E2eiAcmeCA(e2ei_acme_ca) => e2ei_acme_ca.clone().downcast_arc(),
71            Entity::E2eiIntermediateCert(e2ei_intermediate_cert) => e2ei_intermediate_cert.clone().downcast_arc(),
72            Entity::E2eiCrl(e2ei_crl) => e2ei_crl.clone().downcast_arc(),
73            #[cfg(target_family = "wasm")]
74            Entity::E2eiRefreshToken(e2ei_refresh_token) => e2ei_refresh_token.clone().downcast_arc(),
75            #[cfg(feature = "proteus-keystore")]
76            Entity::ProteusIdentity(proteus_identity) => proteus_identity.clone().downcast_arc(),
77            #[cfg(feature = "proteus-keystore")]
78            Entity::ProteusPrekey(proteus_prekey) => proteus_prekey.clone().downcast_arc(),
79            #[cfg(feature = "proteus-keystore")]
80            Entity::ProteusSession(proteus_session) => proteus_session.clone().downcast_arc(),
81        }
82    }

84    pub(crate) async fn execute_save(&self, tx: &TransactionWrapper<'_>) -> CryptoKeystoreResult<()> {
85        match self {
86            Entity::ConsumerData(consumer_data) => consumer_data.set_and_replace(tx).await.map(|_| ()),
87            Entity::HpkePrivateKey(mls_hpke_private_key) => mls_hpke_private_key.save(tx).await,
88            Entity::StoredKeypackage(mls_key_package) => mls_key_package.save(tx).await,
89            Entity::PskBundle(mls_psk_bundle) => mls_psk_bundle.save(tx).await,
90            Entity::EncryptionKeyPair(mls_encryption_key_pair) => mls_encryption_key_pair.save(tx).await,
91            Entity::StoredEpochEncryptionKeypair(mls_epoch_encryption_key_pair) => {
92                mls_epoch_encryption_key_pair.save(tx).await
93            }
94            Entity::StoredCredential(mls_credential) => mls_credential.save(tx).await,
95            Entity::StoredBufferedCommit(mls_pending_commit) => mls_pending_commit.save(tx).await,
96            Entity::PersistedMlsGroup(persisted_mls_group) => persisted_mls_group.save(tx).await,
97            Entity::PersistedMlsPendingGroup(persisted_mls_pending_group) => persisted_mls_pending_group.save(tx).await,
98            Entity::MlsPendingMessage(mls_pending_message) => mls_pending_message.save(tx).await,
99            Entity::StoredE2eiEnrollment(e2ei_enrollment) => e2ei_enrollment.save(tx).await,
100            Entity::E2eiAcmeCA(e2ei_acme_ca) => e2ei_acme_ca.set_and_replace(tx).await.map(|_| ()),
101            Entity::E2eiIntermediateCert(e2ei_intermediate_cert) => e2ei_intermediate_cert.save(tx).await,
102            #[cfg(target_family = "wasm")]
103            Entity::E2eiRefreshToken(e2ei_refresh_token) => e2ei_refresh_token.set_and_replace(tx).await.map(|_| ()),
104            Entity::E2eiCrl(e2ei_crl) => e2ei_crl.save(tx).await,
105            #[cfg(feature = "proteus-keystore")]
106            Entity::ProteusSession(record) => record.save(tx).await,
107            #[cfg(feature = "proteus-keystore")]
108            Entity::ProteusIdentity(record) => record.save(tx).await,
109            #[cfg(feature = "proteus-keystore")]
110            Entity::ProteusPrekey(record) => record.save(tx).await,
111        }
112    }
