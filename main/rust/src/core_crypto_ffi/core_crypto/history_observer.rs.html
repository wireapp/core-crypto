7#[derive(Debug, thiserror::Error, uniffi::Error)]
8#[uniffi(flat_error)]
9pub enum NewHistoryClientReportingError {
10    #[error("panic or otherwise unexpected error from foreign code")]
11    Ffi(#[from] uniffi::UnexpectedUniFFICallbackError),
12}
13
14/// An `HistoryObserver` is notified whenever a new history client is created.
15#[uniffi::export(with_foreign)]
16#[cfg_attr(target_family = "wasm", async_trait::async_trait(?Send))]
17#[cfg_attr(not(target_family = "wasm"), async_trait::async_trait)]
18pub trait HistoryObserver: Send + Sync {
19    /// This function will be called every time a new history client is created.
20    ///
21    /// The `secret` parameter is the secret associated with the new history client
22    ///
23    /// <div class="warning">
24    /// This function must not block! Foreign implementors of this interface can
25    /// spawn a task indirecting the notification, or (unblocking) send the notification
26    /// on some kind of channel, or anything else, as long as the operation completes
27    /// quickly.
28    /// </div>
29    ///
30    /// Though the signature includes an error type, that error is only present because
31    /// it is required by `uniffi` in order to handle panics. This function should suppress
32    /// and ignore internal errors instead of propagating them, to the maximum extent possible.
33    async fn history_client_created(
34        &self,
35        conversation_id: Arc<ConversationId>,
36        secret: HistorySecret,
37    ) -> Result<(), NewHistoryClientReportingError>;
38}
39
40/// This shim bridges the public `HistoryObserver` interface with the internal one defined by `core-crypto`.
41///
42/// This is slightly unfortunate, as it introduces an extra layer of indirection before a change notice can
43/// actually reach its foreign target. However, the orphan rule prevents us from just tying the two traits
44/// together directly, so this is the straightforward way to accomplish that.
45struct ObserverShim(Arc<dyn HistoryObserver>);
46
47#[cfg_attr(target_family = "wasm", async_trait(?Send))]
48#[cfg_attr(not(target_family = "wasm"), async_trait)]
49impl core_crypto::mls::HistoryObserver for ObserverShim {
50    async fn history_client_created(
51        &self,
52        conversation_id: core_crypto::ConversationId,
53        secret: &core_crypto::HistorySecret,
54    ) {
55        let Ok(secret) = HistorySecret::try_from(secret) else {
56            // weird that we couldn't convert this but ¯\_(ツ)_/¯
57            log::warn!(
58                conversation_id = conversation_id;
59                "failed to convert to ffi history secret during creation notification");
60            return;
61        };
62        if let Err(err) = self
63            .0
64            .history_client_created(Arc::new(ConversationId(conversation_id.clone().into())), secret)
65            .await
66        {
67            // we don't _care_ if an error is thrown by the notification function, per se,
68            // but this would probably be useful information for downstream debugging efforts
69            log::warn!(
70                conversation_id = conversation_id,
71                err = log::kv::Value::from_dyn_error(&err);
72                "caught an error when attempting to notify the history observer of a new history client"
73            );
74        }
75    }
76}

78#[uniffi::export]
79impl CoreCryptoFfi {
80    /// Add a history observer to this client.
81    ///
82    /// This function should be called 0 or 1 times in a session's lifetime. If called
83    /// when an history observer already exists, this will return an error.
84    pub async fn register_history_observer(&self, history_observer: Arc<dyn HistoryObserver>) -> CoreCryptoResult<()> {
85        let shim = Arc::new(ObserverShim(history_observer));
86        self.inner
87            .mls_session()
88            .await?
89            .register_history_observer(shim)
90            .await
91            .map_err(CoreCryptoError::generic())
92    }
