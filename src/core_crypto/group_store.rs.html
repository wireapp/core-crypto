33impl GroupStoreEntity for MlsConversation {
34    type RawStoreValue = core_crypto_keystore::entities::PersistedMlsGroup;
35    type IdentityType = ();
36
37    #[cfg(test)]
38    fn id(&self) -> &[u8] {
39        self.id().as_slice()
40    }
41
42    async fn fetch_from_id(
43        id: &[u8],
44        _: Option<Self::IdentityType>,
45        keystore: &impl FetchFromDatabase,
46    ) -> crate::Result<Option<Self>> {
47        let result = keystore
48            .find::<Self::RawStoreValue>(id)
49            .await
50            .map_err(KeystoreError::wrap("finding mls conversation from keystore by id"))?;
51        let Some(store_value) = result else {
52            return Ok(None);
53        };
54
55        let conversation = Self::from_serialized_state(store_value.state.clone(), store_value.parent_id.clone())
56            .map_err(RecursiveError::mls_conversation("deserializing mls conversation"))?;
57        // If the conversation is not active, pretend it doesn't exist
58        Ok(conversation.group.is_active().then_some(conversation))
59    }
60
61    #[cfg(test)]
62    async fn fetch_all(keystore: &impl FetchFromDatabase) -> Result<Vec<Self>> {
63        let all_conversations = keystore
64            .find_all::<Self::RawStoreValue>(EntityFindParams::default())
65            .await
66            .map_err(KeystoreError::wrap("finding all mls conversations"))?;
67        Ok(all_conversations
68            .iter()
69            .filter_map(|c| {
70                let conversation = Self::from_serialized_state(c.state.clone(), c.parent_id.clone()).unwrap();
71                conversation.group.is_active().then_some(conversation)
72            })
73            .collect::<Vec<_>>())
74    }
75}

81/// LRU-cache based group/session store
82/// Uses a hybrid memory limiter based on both amount of elements and total memory usage
83/// As with all LRU caches, eviction is based on oldest elements
84pub(crate) struct GroupStore<V: GroupStoreEntity>(LruMap<V>);
85
86impl<V: GroupStoreEntity> std::fmt::Debug for GroupStore<V> {
87    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
88        f.debug_struct("GroupStore")
89            .field("length", &self.0.len())
90            .field("memory_usage", &self.0.memory_usage())
91            .field(
92                "entries",
93                &self
94                    .0
95                    .iter()
96                    .map(|(k, v)| format!("{k:?}={v:?}"))
97                    .collect::<Vec<String>>()
98                    .join("\n"),
99            )
100            .finish()
101    }
102}

126impl<V: GroupStoreEntity> GroupStore<V> {
127    #[allow(dead_code)]
128    pub(crate) fn new_with_limit(len: u32) -> Self {
129        let limiter = HybridMemoryLimiter::new(Some(len), None);
130        let store = schnellru::LruMap::new(limiter);
131        Self(store)
132    }
133
134    #[allow(dead_code)]
135    pub(crate) fn new(count: Option<u32>, memory: Option<usize>) -> Self {
136        let limiter = HybridMemoryLimiter::new(count, memory);
137        let store = schnellru::LruMap::new(limiter);
138        Self(store)
139    }
140
141    #[allow(dead_code)]
142    pub(crate) fn contains_key(&self, k: &[u8]) -> bool {
143        self.0.peek(k).is_some()
144    }
145
146    pub(crate) async fn get_fetch(
147        &mut self,
148        k: &[u8],
149        keystore: &impl FetchFromDatabase,
150        identity: Option<V::IdentityType>,
151    ) -> crate::Result<Option<GroupStoreValue<V>>> {
152        // Optimistic cache lookup
153        if let Some(value) = self.0.get(k) {
154            return Ok(Some(value.clone()));
155        }
156
157        // Not in store, fetch the thing in the keystore
158        let inserted_value = V::fetch_from_id(k, identity, keystore).await?.map(|value| {
159            let value_to_insert = Arc::new(async_lock::RwLock::new(value));
160            self.insert_prepped(k.to_vec(), value_to_insert.clone());
161            value_to_insert
162        });
163        Ok(inserted_value)
164    }
165
166    /// Returns the value from the keystore.
167    /// WARNING: the returned value is not attached to the keystore and mutations on it will be
168    /// lost when the object is dropped
169    pub(crate) async fn fetch_from_keystore(
170        k: &[u8],
171        keystore: &impl FetchFromDatabase,
172        identity: Option<V::IdentityType>,
173    ) -> crate::Result<Option<V>> {
174        V::fetch_from_id(k, identity, keystore).await
175    }
176
177    #[cfg(test)]
178    pub(crate) async fn get_fetch_all(&mut self, keystore: &impl FetchFromDatabase) -> Result<Vec<GroupStoreValue<V>>> {
179        let all = V::fetch_all(keystore)
180            .await?
181            .into_iter()
182            .map(|g| {
183                let id = g.id().to_vec();
184                let to_insert = Arc::new(async_lock::RwLock::new(g));
185                self.insert_prepped(id, to_insert.clone());
186                to_insert
187            })
188            .collect::<Vec<_>>();
189        Ok(all)
190    }
191
192    fn insert_prepped(&mut self, k: Vec<u8>, prepped_entity: GroupStoreValue<V>) {
193        self.0.insert(k, prepped_entity);
194    }
195
196    pub(crate) fn insert(&mut self, k: Vec<u8>, entity: V) {
197        let value_to_insert = Arc::new(async_lock::RwLock::new(entity));
198        self.insert_prepped(k, value_to_insert)
199    }
200
201    pub(crate) fn try_insert(&mut self, k: Vec<u8>, entity: V) -> Result<(), V> {
202        let value_to_insert = Arc::new(async_lock::RwLock::new(entity));
203
204        if self.0.insert(k, value_to_insert.clone()) {
205            Ok(())
206        } else {
207            // This is safe because we just built the value
208            Err(Arc::into_inner(value_to_insert).unwrap().into_inner())
209        }
210    }
211
212    pub(crate) fn remove(&mut self, k: &[u8]) -> Option<GroupStoreValue<V>> {
213        self.0.remove(k)
214    }
215
216    pub(crate) fn get(&mut self, k: &[u8]) -> Option<&mut GroupStoreValue<V>> {
217        self.0.get(k)
218    }
219}

221pub(crate) struct HybridMemoryLimiter {
222    mem: schnellru::ByMemoryUsage,
223    len: schnellru::ByLength,
224}
225
226pub(crate) const MEMORY_LIMIT: usize = 100_000_000;
227pub(crate) const ITEM_LIMIT: u32 = 100;
228
229impl HybridMemoryLimiter {
230    // in the wasm case, we ignore the suggested memory limit
231    #[cfg_attr(target_family = "wasm", expect(unused_variables))]
232    pub(crate) fn new(count: Option<u32>, memory: Option<usize>) -> Self {
233        #[cfg(target_family = "wasm")]
234        let memory_limit = MEMORY_LIMIT;
235
236        #[cfg(not(target_family = "wasm"))]
237        let memory_limit = memory
238            .or_else(|| {
239                let system = sysinfo::System::new_with_specifics(
240                    sysinfo::RefreshKind::nothing().with_memory(sysinfo::MemoryRefreshKind::nothing().with_ram()),
241                );
242
243                let available_sys_memory = system.available_memory();
244                (available_sys_memory > 0).then_some(available_sys_memory as usize)
245            })
246            .unwrap_or(MEMORY_LIMIT);
247
248        let mem = schnellru::ByMemoryUsage::new(memory_limit);
249        let len = schnellru::ByLength::new(count.unwrap_or(ITEM_LIMIT));
250
251        Self { mem, len }
252    }
253}
254
255impl Default for HybridMemoryLimiter {
256    fn default() -> Self {
257        Self::new(None, None)
258    }
259}

261impl<K, V> schnellru::Limiter<K, V> for HybridMemoryLimiter {
262    type KeyToInsert<'a> = K;
263    type LinkType = u32;
264
265    fn is_over_the_limit(&self, length: usize) -> bool {
266        <schnellru::ByLength as schnellru::Limiter<K, V>>::is_over_the_limit(&self.len, length)
267    }
268
269    fn on_insert(&mut self, length: usize, key: Self::KeyToInsert<'_>, value: V) -> Option<(K, V)> {
270        <schnellru::ByLength as schnellru::Limiter<K, V>>::on_insert(&mut self.len, length, key, value)
271    }
272
273    // Both underlying limiters have dummy implementations here
274    fn on_replace(
275        &mut self,
276        _length: usize,
277        _old_key: &mut K,
278        _new_key: Self::KeyToInsert<'_>,
279        _old_value: &mut V,
280        _new_value: &mut V,
281    ) -> bool {
282        true
283    }
284    fn on_removed(&mut self, _key: &mut K, _value: &mut V) {}
285    fn on_cleared(&mut self) {}
286
287    fn on_grow(&mut self, new_memory_usage: usize) -> bool {
288        <schnellru::ByMemoryUsage as schnellru::Limiter<K, V>>::on_grow(&mut self.mem, new_memory_usage)
289    }
290}

391        use schnellru::{LruMap, UnlimitedCompact};
392        let mut lru: LruMap<Vec<u8>, DummyValue, UnlimitedCompact> =
393            LruMap::<Vec<u8>, DummyValue, UnlimitedCompact>::new(UnlimitedCompact);

397        let memory_usage_step_1 = lru.memory_usage();
398        lru.insert(2usize.to_le_bytes().to_vec(), "20".into());
399        lru.insert(3usize.to_le_bytes().to_vec(), "30".into());
400        lru.insert(4usize.to_le_bytes().to_vec(), "40".into());
401        let memory_usage_step_2 = lru.memory_usage();
402        assert_ne!(memory_usage_step_1, memory_usage_step_2);
403
404        let mut store = TestGroupStore::new(None, Some(memory_usage_step_2));
405        assert_eq!(store.guaranteed_capacity(), 0);
406        assert_eq!(store.memory_usage(), 0);
407        store.try_insert(1usize.to_le_bytes().to_vec(), "10".into()).unwrap();
408        assert_eq!(store.guaranteed_capacity(), 3);
409        assert!(store.memory_usage() <= memory_usage_step_1);
410        store.try_insert(2usize.to_le_bytes().to_vec(), "20".into()).unwrap();
411        store.try_insert(3usize.to_le_bytes().to_vec(), "30".into()).unwrap();
412        for i in 1..=3usize {

415                DummyValue::from(format!("{}", i * 10).as_str())

419        assert!(store.memory_usage() <= memory_usage_step_1);
420        assert!(store.try_insert(4usize.to_le_bytes().to_vec(), "40".into()).is_ok());
421        for i in (1usize..=4).rev() {
422            assert_eq!(
