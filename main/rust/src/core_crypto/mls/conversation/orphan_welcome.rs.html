1//! This deals with DS inconsistencies. When a Welcome message is received, the client might have
2//! already deleted its associated KeyPackage (and encryption key).
3//! Feel free to remove this when this is no longer a problem !!!
4
5#[cfg(test)]
6mod tests {
7
8    use openmls::prelude::KeyPackage;
9    use openmls_traits::OpenMlsCryptoProvider;
10
11    use super::super::error::Error;
12    use crate::test_utils::*;
13
14    #[apply(all_cred_cipher)]
15    pub async fn orphan_welcome_should_generate_external_commit(case: TestContext) {
16        let [alice, bob] = case.sessions().await;
17        Box::pin(async move {
18            let conversation = case.create_conversation([&alice]).await;
19
20                let bob_kp = bob.rand_key_package(&case).await;
21                let bob_kp_ref = KeyPackage::from(bob_kp.clone())
22                    .hash_ref(bob.transaction.mls_provider().await.unwrap().crypto())
23                    .unwrap();
24
25                // Alice invites Bob with a KeyPackage...
26                conversation.guard().await
27                    .add_members(vec![bob_kp])
28                    .await
29                    .unwrap();
30
31                // ...Bob deletes locally (with the associated private key) before processing the Welcome
32                bob.transaction.delete_keypackages([bob_kp_ref]).await.unwrap();
33
34                let welcome = alice.mls_transport().await.latest_welcome_message().await;
35
36                // in that case a dedicated error is thrown for clients to identify this case
37                // and rejoin with an external commit
38                let process_welcome = bob
39                    .transaction
40                    .process_welcome_message(welcome.into(), case.custom_cfg())
41                    .await;
42                assert!(matches!(
43                    process_welcome.unwrap_err(),
44                    crate::transaction_context::Error::Recursive(crate::RecursiveError::MlsConversation { source, .. }) if matches!(*source, Error::OrphanWelcome)
45                ));
46            })
47        .await;
48    }
49}
