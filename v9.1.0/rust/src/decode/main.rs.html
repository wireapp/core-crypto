113/// Utility for decoding various wire formats for mls and proteus
114#[derive(Parser, Debug)]
115#[clap(name = "decode", version)]
116pub struct App {
117    #[clap(subcommand)]
118    command: Command,
119}
120
121#[derive(Debug, Subcommand)]
122enum Command {
123    /// Decode and display a proteus prekey bundle
124    PrekeyBundle { bundle: FileOrStdin<String> },
125    /// Decode and display a proteus message
126    ProteusMessage {
127        /// File containing a base64 encoded proteus message, or `-` to read from stdin.
128        message: FileOrStdin<String>,
129    },
130    /// Decode and display an MLS message
131    MlsMessage {
132        /// File containing a base64 encoded mls message, or `-` to read from stdin.
133        message: FileOrStdin<String>,
134        /// Display raw mls tls decoded structure.
135        #[arg(short, long)]
136        raw_message: bool,
137        /// Display all members if group info is provided.
138        #[arg(short, long)]
139        members: bool,
140        /// Display decoded basic identities if group info is provided.
141        #[arg(short, long)]
142        identities: bool,
143    },
144}

146fn main() -> Result<(), Box<dyn std::error::Error>> {
147    let app = App::parse();
148    match app.command {
149        Command::PrekeyBundle { bundle } => {
150            let input: String = bundle.contents()?;
151            let bytes = base64::prelude::BASE64_STANDARD.decode(input)?;
152            let bundle = PreKeyBundle::deserialise(&bytes)?;
153            println!("{:#?}", ProteusPreKeyBundle::from(bundle));
154            Ok(())
155        }
156        Command::ProteusMessage { message } => {
157            let input: String = message.contents()?;
158            let bytes = base64::prelude::BASE64_STANDARD.decode(input)?;
159            let message = Envelope::deserialise(&bytes)?;
160            println!("{:#?}", ProteusEnvelope::from(message));
161            Ok(())
162        }
163        Command::MlsMessage {
164            message,
165            raw_message,
166            members,
167            identities,
168        } => {
169            let input: String = message.contents()?;
170            let bytes = base64::prelude::BASE64_STANDARD.decode(input)?;
171            let mut group_info = None;
172
173            if let Ok(message) = MlsMessage::mls_decode(&mut bytes.as_slice()) {
174                if raw_message {
175                    println!("{message:#?}");
176                }
177                group_info = message.into_group_info();
178            }
179
180            if let Ok(gi) = GroupInfo::mls_decode(&mut bytes.as_slice()) {
181                if raw_message {
182                    println!("{gi:#?}");
183                }
184                group_info = Some(gi);
185            }
186
187            if let Some(group_info) = group_info {
188                if members {
189                    print_members(&group_info);
190                }
191
192                if identities {
193                    print_basic_identities(&group_info);
194                }
195            }
196
197            Ok(())
198        }
199    }
200}
201
202fn print_members(group_info: &GroupInfo) {
203    group_info.extensions().iter().for_each(|ext| {
204        if ext.extension_type == ExtensionType::RATCHET_TREE
205            && let Some(tree) = RatchetTreeExt::from_bytes(ext.extension_data.as_slice()).ok()
206        {
207            tree.tree_data.roster().members().iter().for_each(|member| {
208                println!("{:#?}", member);
209            });
210        }
211    })
212}
213
214fn print_basic_identities(group_info: &GroupInfo) {
215    group_info.extensions().iter().for_each(|ext| {
216        if ext.extension_type == ExtensionType::RATCHET_TREE
217            && let Some(tree) = RatchetTreeExt::from_bytes(ext.extension_data.as_slice()).ok()
218        {
219            tree.tree_data.roster().members().iter().for_each(|member| {
220                if let Some(bytes) = member.signing_identity.credential.as_basic().map(|f| &f.identifier) {
221                    println!(
222                        "leaf index: {:#?} identity: {:#?}",
223                        member.index,
224                        str::from_utf8(bytes).unwrap_or("invalid utf-8")
225                    );
226                }
227            });
228        }
229    });
