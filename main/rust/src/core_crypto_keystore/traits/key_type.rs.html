1use std::borrow::Cow;
2
3/// A Key Type is a type which can act as a key for a database.
4///
5/// This might be a primary key, in which case the key uniquely identifies either 0 or 1 entries in the database.
6/// Or it might be a search key, in which case the key could match any number of entries.
7pub trait KeyType: Send + Sync {
8    /// Get a unique binary representation of this key.
9    ///
10    /// For simple keys it can just be the borrowed form of the key itself,
11    /// but for complex keys it could be the run length encoding of each field of the key.
12    fn bytes(&self) -> Cow<'_, [u8]>;
13}
14
15// useful for unique entities; non-allocating
16impl KeyType for () {
17    fn bytes(&self) -> Cow<'_, [u8]> {
18        Vec::new().into()
19    }
20}
21
22macro_rules! impl_keytype {
23    ($t:ty, |$self:ident| $impl:expr) => {
24        impl KeyType for $t {
25            fn bytes(&$self) -> Cow<'_, [u8]> {
26                $impl.into()
27            }
28        }
29    };
30}
31
32impl_keytype!(&[u8], |self| *self);
33impl_keytype!(Vec<u8>, |self| self.as_slice());
34impl_keytype!(&str, |self| self.as_bytes());
35impl_keytype!(String, |self| self.as_bytes());
36
37macro_rules! impl_keytype_for_integer {
38    ($t:ty) => {
39        impl KeyType for $t {
40            fn bytes(&self) -> Cow<'_, [u8]> {
41                Vec::from(self.to_le_bytes()).into()
42            }
43        }
44    };
45}
46

48impl_keytype_for_integer!(u16);
49impl_keytype_for_integer!(u32);
50impl_keytype_for_integer!(u64);
51impl_keytype_for_integer!(u128);
52impl_keytype_for_integer!(i8);
53impl_keytype_for_integer!(i16);
54impl_keytype_for_integer!(i32);
