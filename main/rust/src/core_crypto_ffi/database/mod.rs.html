11/// The database acting as a core crypto keystore.
12#[derive(Debug, derive_more::From, derive_more::Into, Clone, derive_more::Deref, uniffi::Object)]
13pub struct Database(core_crypto_keystore::Database);
14
15impl Database {
16    /// Open or create a [Database].
17    pub async fn open(location: &str, key: Arc<DatabaseKey>) -> CoreCryptoResult<Self> {
18        core_crypto_keystore::Database::open(core_crypto_keystore::ConnectionType::Persistent(location), key.as_ref())
19            .await
20            .map(Database)
21            .map_err(CoreCryptoError::generic())
22    }
23
24    /// Create an in-memory [Database] whose data will be lost when the instance is dropped.
25    pub async fn in_memory(key: Arc<DatabaseKey>) -> CoreCryptoResult<Self> {
26        core_crypto_keystore::Database::open(core_crypto_keystore::ConnectionType::InMemory, key.as_ref())
27            .await
28            .map(Database)
29            .map_err(CoreCryptoError::generic())
30    }
31}
32
33#[uniffi::export]
34impl Database {
35    /// Get the location of the database
36    /// Returns null if in-memory
37    pub async fn get_location(&self) -> CoreCryptoResult<Option<String>> {
38        self.location().await.map_err(CoreCryptoError::generic())
39    }
40}

42/// Open or create a [Database].
43#[uniffi::export]
44pub async fn open_database(location: &str, key: Arc<DatabaseKey>) -> CoreCryptoResult<Database> {
45    Database::open(location, key).await
46}

48/// Create an in-memory [Database] whose data will be lost when the instance is dropped.
49#[uniffi::export]
50pub async fn in_memory_database(key: Arc<DatabaseKey>) -> CoreCryptoResult<Database> {
51    Database::in_memory(key).await
52}

54/// Export a copy of the database to the specified path.
55///
56/// This creates a fully vacuumed and optimized copy of the database using SQLite's VACUUM INTO command.
57/// The copy will be encrypted with the same key as the source database.
58///
59/// # Platform Support
60/// This method is only useful on platforms using SQLCipher (iOS, Android, JVM, native).
61/// On WASM platforms, this function will return an error.
62///
63/// # Arguments
64/// * `database` - The database instance to export
65/// * `destination_path` - The file path where the database copy should be created
66//
67// These feature flags are ugly, but here's how they work:
68//
69// - If the `wasm` feature is enabled, exlude this; we don't want it.
70// - If we're compiling for wasm without the `wasm` feature, we have to have a function body, so just return an error
71// - If we're compiling for non-wasm, we have the real impl
72//
73// This works out because we only ever build real wasm builds with the `wasm` feature flag enabled.
74// The error could therefore be an `unimplemented!` but it's harmless to return a real error in this case.
75#[cfg(not(feature = "wasm"))]
76#[cfg_attr(all(not(feature = "wasm"), target_family = "wasm"), expect(unused_variables))]
77#[uniffi::export]
78pub async fn export_database_copy(database: &Database, destination_path: &str) -> CoreCryptoResult<()> {
79    // we need a noop here for the case where we're compiling for wasm but without feature wasm
80    #[cfg(target_family = "wasm")]
81    {
82        Err(CoreCryptoError::ad_hoc(
83            "export_database_copy is not implemented for wasm",
84        ))
85    }
86    #[cfg(not(target_family = "wasm"))]
87    {
88        database
89            .0
90            .export_copy(destination_path)
91            .await
92            .map_err(CoreCryptoError::generic())
93    }
