9/// Helper for working with the new transasction interface.
10///
11/// This helper serves two purposes: to present a `FnOnce` interface for transactions,
12/// and to allow the extraction of data from within transactions.
13///
14/// ## Extracting Data
15///
16/// The `CoreCryptoCommand` interface requires some kind of interior mutability to extract
17/// any data: it takes an immutable reference to the implementing item, and returns the unit struct
18/// in the success case.
19///
20/// That pattern is relatively arcane and verbose, particularly when we just want to smuggle out
21/// some data from within the transaction. This helper is intended to ease and automate
22/// that process.
23///
24/// Use it like this (pseudocode):
25///
26/// ```ignore
27/// // an extractor is always `Arc`-wrapped
28/// let extractor: Arc<_> = TransactionHelper::new(async move |context| {
29///     // return whatever you need from the transaction here
30/// });
31/// core_crypto.transaction(extractor.clone()).await?;
32/// let return_value = extractor.into_return_value();
33/// ```
34///
35/// ## Panics
36///
37/// `TransactionHelper` is a one-shot item. Attempting to use the
38/// same extractor in two different transactions will cause a panic.
39pub struct TransactionHelper<T, F> {
40    func: Mutex<Option<F>>,
41    return_value: OnceCell<T>,
42}
43
44impl<T, F, Fut> TransactionHelper<T, F>
45where
46    F: FnOnce(Arc<CoreCryptoContext>) -> Fut + Send + Sync,
47    Fut: Future<Output = CoreCryptoResult<T>> + Send,
48    T: Send + Sync,
49{
50    pub fn new(func: F) -> Arc<Self> {
51        Arc::new(Self {
52            func: Mutex::new(Some(func)),
53            return_value: OnceCell::new(),
54        })
55    }
56
57    /// Get the return value from the internal function.
58    ///
59    /// ## Panics
60    ///
61    /// - If there exists more than one strong reference to this extractor
62    /// - If the inner function was never called
63    /// - If the inner function returned an `Err` variant
64    ///
65    /// In general if you call this after a call like
66    ///
67    /// ```ignore
68    /// core_crypto.transaction(extractor.clone())?;
69    /// ```
70    ///
71    /// then this will be fine.
72    pub fn into_return_value(self: Arc<Self>) -> T {
73        Arc::into_inner(self)
74            .expect("there should exist exactly one strong ref right now")
75            .return_value
76            .into_inner()
77            .expect("return value should be initialized")
78    }
79
80    /// Safely get the return value from the internal function.
81    ///
82    /// If there exists more than one strong reference to this item, or
83    /// the inner function was never called or returned an `Err` variant,
84    /// this will return `None`.
85    pub fn try_into_return_value(self: Arc<Self>) -> Option<T> {
86        Arc::into_inner(self)?.return_value.into_inner()
87    }
88}
89
90#[async_trait::async_trait]
91impl<T, F, Fut> CoreCryptoCommand for TransactionHelper<T, F>
92where
93    F: FnOnce(Arc<CoreCryptoContext>) -> Fut + Send + Sync,
94    Fut: Future<Output = CoreCryptoResult<T>> + Send,
95    T: Send + Sync,
96{
97    async fn execute(&self, context: Arc<CoreCryptoContext>) -> CoreCryptoResult<()> {
98        let func = self
99            .func
100            .lock()
101            .await
102            .take()
103            .expect("inner function must only be called once");
104        let return_value = func(context).await?;
105        let set_result = self.return_value.set(return_value).await;
106        if set_result.is_err() {
107            // can't just `.expect()` here because `T` is not `Debug`
108            // though TBH this would be a really weird case; we should already have
109            // paniced getting `func` above
110            panic!("return value was previously set");
111        }
112        Ok(())
113    }
114}
