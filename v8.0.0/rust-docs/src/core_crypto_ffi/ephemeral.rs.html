7/// A `HistorySecret` encodes sufficient client state that it can be used to instantiate an
8/// ephemeral client.
9#[derive(Debug, Clone, PartialEq, Eq)]
10#[cfg_attr(target_family = "wasm", wasm_bindgen(getter_with_clone))]
11#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Record))]
12pub struct HistorySecret {
13    #[cfg_attr(target_family = "wasm", wasm_bindgen(readonly, js_name = clientId))]
14    pub client_id: ClientIdMaybeArc,
15    #[cfg_attr(target_family = "wasm", wasm_bindgen(readonly))]
16    pub data: Vec<u8>,
17}
18
19#[cfg(target_family = "wasm")]
20#[wasm_bindgen]
21impl HistorySecret {
22    #[wasm_bindgen(constructor)]
23    pub fn new(client_id: ClientIdMaybeArc, data: &[u8]) -> Result<HistorySecret, wasm_bindgen::JsError> {
24        Ok(HistorySecret {
25            client_id,
26            data: data.into(),
27        })
28    }
29}

31impl TryFrom<&CoreCryptoHistorySecret> for HistorySecret {
32    type Error = CoreCryptoError;
33
34    fn try_from(value: &CoreCryptoHistorySecret) -> Result<Self, Self::Error> {
35        let client_id = value.client_id.clone();
36        rmp_serde::to_vec(&value)
37            .map_err(CoreCryptoError::generic())
38            .map(|secret| HistorySecret {
39                client_id: ClientId::from_cc(client_id),
40                data: secret,
41            })
42    }
43}

45async fn history_client_inner(history_secret: HistorySecret) -> CoreCryptoResult<CoreCrypto> {
46    let secret =
47        rmp_serde::from_slice::<CoreCryptoHistorySecret>(&history_secret.data).map_err(CoreCryptoError::generic())?;
48    CoreCryptoFfi::history_client(secret)
49        .await
50        .map(|inner| CoreCrypto { inner })
51        .map_err(Into::into)
52}
53
54#[cfg(target_family = "wasm")]
55#[wasm_bindgen]
56impl CoreCrypto {
57    /// Instantiate a history client.
58    ///
59    /// This client exposes the full interface of `CoreCrypto`, but it should only be used to decrypt messages.
60    /// Other use is a logic error.
61    pub async fn history_client(history_secret: HistorySecret) -> CoreCryptoResult<Self> {
62        history_client_inner(history_secret).await
63    }
64}
