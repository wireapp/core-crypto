7pub(crate) async fn start_webdriver_chrome(addr: &std::net::SocketAddr) -> Result<tokio::process::Child> {
8    let chromedriver_path = std::env::var_os("CHROMEDRIVER_PATH")
9        .map(PathBuf::from)
10        .or_else(|| which("chromedriver").ok())
11        .expect("chromedriver must be in PATH or CHROMEDRIVER_PATH must be set");
12    Ok(tokio::process::Command::new(chromedriver_path)
13        .arg(format!("--port={}", addr.port()))
14        .stdout(std::process::Stdio::null())
15        .stderr(std::process::Stdio::null())
16        .spawn()?)
17}

19pub(crate) async fn setup_browser(
20    client: &std::net::SocketAddr,
21    server: &std::net::SocketAddr,
22    folder: &str,
23) -> Result<fantoccini::Client> {
24    let spinner = RunningProcess::new("Starting Fantoccini remote browser...", false);
25    let mut caps_json = serde_json::json!({
26        "goog:chromeOptions": {
27            "args": [
28                "headless=shell",
29                "disable-dev-shm-usage",
30                "no-sandbox"
31            ]
32        }
33    });
34
35    if let Ok(chrome_path) = std::env::var("CHROME_PATH") {
36        caps_json["goog:chromeOptions"]["binary"] = chrome_path.into();
37    }
38
39    let serde_json::Value::Object(caps) = caps_json else {
40        unreachable!("`serde_json::json!()` did not produce an object when provided an object. Something is broken.")
41    };
42
43    let browser = fantoccini::ClientBuilder::native()
44        .capabilities(caps)
45        .connect(&format!("http://{client}"))
46        .await?;
47    browser.goto(&format!("http://{server}/{folder}/index.html")).await?;
48
49    spinner.success("Browser [OK]");
50    Ok(browser)
51}
