33/// Represents a x509 certificate chain supplied by the client
34/// It can fetch it after an end-to-end identity process where it can get back a certificate
35/// from the Authentication Service
36#[derive(Clone)]
37pub struct CertificateBundle {
38    /// x509 certificate chain
39    /// First entry is the leaf certificate and each subsequent is its issuer
40    pub certificate_chain: Vec<Vec<u8>>,
41    /// Leaf certificate private key
42    pub private_key: CertificatePrivateKey,
43}
44
45impl fmt::Debug for CertificateBundle {
46    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
47        use base64::prelude::*;
48
49        #[derive(derive_more::Debug)]
50        #[debug("{}", BASE64_STANDARD.encode(_0))]
51        // this only exists for the debug impl, which is ignored by the dead code check
52        #[expect(dead_code)]
53        struct CertificateDebugHelper<'a>(&'a Vec<u8>);
54
55        let certificates = self
56            .certificate_chain
57            .iter()
58            .map(CertificateDebugHelper)
59            .collect::<Vec<_>>();
60        f.debug_struct("CertificateBundle")
61            .field("certificate_chain", &certificates)
62            .field("private_key", &self.private_key)
63            .finish()
64    }
65}

67impl CertificateBundle {
68    /// Reads the client_id from the leaf certificate
69    pub fn get_client_id(&self) -> Result<ClientId> {
70        let leaf = self.certificate_chain.first().ok_or(Error::InvalidIdentity)?;
71
72        let hash_alg = match self.private_key.signature_scheme {
73            SignatureScheme::ECDSA_SECP256R1_SHA256 | SignatureScheme::ED25519 => HashAlgorithm::SHA256,
74            SignatureScheme::ECDSA_SECP384R1_SHA384 => HashAlgorithm::SHA384,
75            SignatureScheme::ED448 | SignatureScheme::ECDSA_SECP521R1_SHA512 => HashAlgorithm::SHA512,
76        };
77
78        let identity = leaf
79            .extract_identity(None, hash_alg)
80            .map_err(|_| Error::InvalidIdentity)?;
81        let client_id = identity
82            .client_id
83            .parse::<WireQualifiedClientId>()
84            .map_err(RecursiveError::e2e_identity("parsing wire qualified client id"))?;
85        Ok(client_id.into())
86    }
87
88    /// Reads the 'Not Before' claim from the leaf certificate
89    pub fn get_created_at(&self) -> Result<u64> {
90        let leaf = self.certificate_chain.first().ok_or(Error::InvalidIdentity)?;
91        leaf.extract_created_at().map_err(|_| Error::InvalidIdentity)
92    }
93}

95impl Credential {
96    /// Create a new x509 credential from a certificate bundle.
97    pub fn x509(cert: CertificateBundle) -> Result<Self> {
98        let earliest_validity = cert.get_created_at().map_err(RecursiveError::mls_credential(
99            "getting credential 'not before' claim from leaf cert in Credential::x509",
100        ))?;
101        let (sk, ..) = cert.private_key.into_parts();
102        let chain = cert.certificate_chain;
103
104        let kp = CertificateKeyPair::new(sk, chain.clone()).map_err(MlsError::wrap("creating certificate key pair"))?;
105
106        let credential = MlsCredential::new_x509(chain).map_err(MlsError::wrap("creating x509 credential"))?;
107
108        let cb = Credential {
109            credential_type: CredentialType::X509,
110            mls_credential: credential,
111            signature_key_pair: kp.0,
112            earliest_validity,
113        };
114        Ok(cb)
115    }
116}

119fn new_rand_client(domain: Option<String>) -> (String, String) {
120    let rand_str = |n: usize| {
121        use rand::distributions::{Alphanumeric, DistString as _};
122        Alphanumeric.sample_string(&mut rand::thread_rng(), n)
123    };
124    let user_id = uuid::Uuid::new_v4().to_string();
125    let domain = domain.unwrap_or_else(|| format!("{}.com", rand_str(6)));
126    let client_id = wire_e2e_identity::prelude::E2eiClientId::try_new(user_id, rand::random::<u64>(), &domain)
127        .unwrap()
128        .to_qualified();
129    (client_id, domain)
130}

160    pub fn new_with_expiration(
161        handle: &str,
162        display_name: &str,
163        client_id: Option<&crate::e2e_identity::id::QualifiedE2eiClientId>,
164        cert_keypair: Option<PkiKeypair>,
165        signer: &crate::test_utils::x509::X509Certificate,
166        expiration: Option<std::time::Duration>,
167    ) -> Self {
168        // here in our tests client_id is generally just "alice" or "bob"
169        // so we will use it to augment handle & display_name
170        // and not a real client_id, instead we'll generate a random one
171        let domain = "world.com";
172        let (client_id, domain) = client_id
173            .map(|cid| {
174                let cid = String::from_utf8(cid.to_vec()).unwrap();
175                (cid, domain.to_string())
176            })
177            .unwrap_or_else(|| new_rand_client(Some(domain.to_string())));
178
179        let mut cert_params = crate::test_utils::x509::CertificateParams {
180            domain: domain.into(),
181            common_name: Some(display_name.to_string()),
182            handle: Some(handle.to_string()),
183            client_id: Some(client_id.to_string()),
184            cert_keypair,
185            ..Default::default()
186        };
187
188        if let Some(expiration) = expiration {
189            cert_params.expiration = expiration;
190        }
191
192        let cert = signer.create_and_sign_end_identity(cert_params);
193        Self::from_certificate_and_issuer(&cert, signer)
194    }
