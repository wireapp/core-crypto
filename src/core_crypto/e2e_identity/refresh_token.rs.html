11/// An OIDC refresh token managed by CoreCrypto to benefit from encryption-at-rest
12#[derive(Debug, serde::Serialize, serde::Deserialize, Zeroize, derive_more::From, derive_more::Deref)]
13#[zeroize(drop)]
14pub struct RefreshToken(String);
15
16impl RefreshToken {
17    pub(crate) async fn find(key_store: &impl FetchFromDatabase) -> Result<RefreshToken> {
18        key_store
19            .find_unique::<E2eiRefreshToken>()
20            .await
21            .map_err(KeystoreError::wrap("finding refresh token"))?
22            .try_into()
23    }
24
25    pub(crate) async fn replace(self, backend: &MlsCryptoProvider) -> Result<()> {
26        let keystore = backend.keystore();
27        let rt = E2eiRefreshToken::from(self);
28        keystore
29            .save(rt)
30            .await
31            .map_err(KeystoreError::wrap("replacing refresh token"))?;
32        Ok(())
33    }
34}
35
36impl E2eiEnrollment {
37    /// Lets clients retrieve the OIDC refresh token to try to renew the user's authorization.
38    /// If it's expired, the user needs to reauthenticate and they will update the refresh token
39    /// in [E2eiEnrollment::new_oidc_challenge_request]
40    pub fn get_refresh_token(&self) -> Result<&str> {
41        self.refresh_token
42            .as_ref()
43            .map(|rt| rt.as_str())
44            .ok_or(Error::OutOfOrderEnrollment(
45                "No OIDC refresh token registered yet or it has been persisted",
46            ))
47    }
48}
49
50impl TryFrom<E2eiRefreshToken> for RefreshToken {
51    type Error = Error;
52
53    fn try_from(mut entity: E2eiRefreshToken) -> Result<Self> {
54        let content = std::mem::take(&mut entity.content);
55        let content = String::from_utf8(content).map_err(|_| Error::InvalidRefreshToken)?;
56        Ok(Self(content))
57    }
58}
59
60impl From<RefreshToken> for E2eiRefreshToken {
61    fn from(mut rt: RefreshToken) -> Self {
62        let content = std::mem::take(&mut rt.0);
63        Self {
64            content: content.into_bytes(),
65        }
66    }
