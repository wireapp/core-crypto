7impl MlsPendingMessage {
8    pub async fn find_all_by_conversation_id(
9        conn: &mut <Self as EntityBase>::ConnectionType,
10        conversation_id: &[u8],
11        params: EntityFindParams,
12    ) -> crate::CryptoKeystoreResult<Vec<Self>> {
13        let mut conn = conn.conn().await;
14        let transaction = conn.transaction()?;
15        let query: String = format!(
16            "SELECT rowid FROM mls_pending_messages WHERE id = ? {}",
17            params.to_sql()
18        );
19
20        let mut stmt = transaction.prepare_cached(&query)?;
21        let rows = stmt.query_map([conversation_id], |r| r.get(0))?;
22        rows.map(|rowid_result| {
23            let rowid = rowid_result?;
24            use std::io::Read as _;
25
26            let mut blob = transaction.blob_open(rusqlite::MAIN_DB, "mls_pending_messages", "id", rowid, true)?;
27            let mut conversation_id = vec![];
28            blob.read_to_end(&mut conversation_id)?;
29            blob.close()?;
30
31            let mut blob = transaction.blob_open(rusqlite::MAIN_DB, "mls_pending_messages", "message", rowid, true)?;
32            let mut message = vec![];
33            blob.read_to_end(&mut message)?;
34            blob.close()?;
35
36            Ok(Self {
37                foreign_id: conversation_id,
38                message,
39            })
40        })
41        .collect()
42    }
43}

55    }

57    async fn find_one(_: &mut Self::ConnectionType, _: &StringEntityId) -> crate::CryptoKeystoreResult<Option<Self>> {
58        panic!("Must not be called. The intended usage is to call MlsPendingMessage::find_all_by_conversation_id().")
59    }
60
61    async fn find_all(
62        conn: &mut Self::ConnectionType,
63        params: EntityFindParams,
64    ) -> crate::CryptoKeystoreResult<Vec<Self>> {
65        let mut conn = conn.conn().await;
66        let transaction = conn.transaction()?;
67        let query: String = format!("SELECT rowid FROM mls_pending_messages {}", params.to_sql());
68
69        let mut stmt = transaction.prepare_cached(&query)?;
70        let mut rows = stmt.query_map([], |r| r.get(0))?;
71        let entities = rows.try_fold(Vec::new(), |mut acc, rowid_result| {
72            use std::io::Read as _;
73            let rowid = rowid_result?;
74
75            let mut blob = transaction.blob_open(rusqlite::MAIN_DB, "mls_pending_messages", "id", rowid, true)?;
76            let mut id = vec![];
77            blob.read_to_end(&mut id)?;
78            blob.close()?;
79
80            let mut blob = transaction.blob_open(rusqlite::MAIN_DB, "mls_pending_messages", "message", rowid, true)?;
81            let mut message = vec![];
82            blob.read_to_end(&mut message)?;
83            blob.close()?;
84
85            acc.push(Self {
86                foreign_id: id,
87                message,
88            });
89            crate::CryptoKeystoreResult::Ok(acc)
90        })?;
91
92        Ok(entities)
93    }

124#[async_trait::async_trait]
125impl EntityTransactionExt for MlsPendingMessage {
126    async fn save(&self, transaction: &TransactionWrapper<'_>) -> CryptoKeystoreResult<()> {
127        Self::ConnectionType::check_buffer_size(self.foreign_id.len())?;
128        Self::ConnectionType::check_buffer_size(self.message.len())?;
129
130        let zid = rusqlite::blob::ZeroBlob(self.foreign_id.len() as i32);
131        let zmsg = rusqlite::blob::ZeroBlob(self.message.len() as i32);
132
133        let id_bytes = &self.foreign_id;
134
135        use rusqlite::ToSql as _;
136        transaction.execute(
137            "INSERT INTO mls_pending_messages (id, message) VALUES(?, ?)",
138            [&zid.to_sql()?, &zmsg.to_sql()?],
139        )?;
140        let rowid = transaction.last_insert_rowid();
141
142        let mut blob = transaction.blob_open(rusqlite::MAIN_DB, "mls_pending_messages", "id", rowid, false)?;
143        use std::io::Write as _;
144        blob.write_all(id_bytes)?;
145        blob.close()?;
146
147        let mut blob = transaction.blob_open(rusqlite::MAIN_DB, "mls_pending_messages", "message", rowid, false)?;
148        blob.write_all(&self.message)?;
149        blob.close()?;
150
151        Ok(())
152    }
153
154    async fn delete_fail_on_missing_id(
155        transaction: &TransactionWrapper<'_>,
156        id: StringEntityId<'_>,
157    ) -> CryptoKeystoreResult<()> {
158        let updated = transaction.execute("DELETE FROM mls_pending_messages WHERE id = ?", [id.as_slice()])?;
159
160        if updated > 0 {
161            Ok(())
162        } else {
163            Err(Self::to_missing_key_err_kind().into())
164        }
165    }
