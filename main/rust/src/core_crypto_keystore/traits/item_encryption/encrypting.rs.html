3use crate::CryptoKeystoreResult;

5/// This trait produces an encrypted form of this struct, where all sensitive fields have been
6/// encrypted using the [`EncryptData`][super::EncryptData] helper.
7///
8/// This is quite likely to be handled automatically by a macro, depending on how annoying it is to implement
9/// everywhere.
10///
11/// ## Example
12///
13/// ```rust,ignore
14/// // Foo is an Entity
15/// struct Foo {
16///     id: Vec<u8>,
17///     sensitive_data: Vec<u8>, // sensitive!
18/// }
19///
20/// #[derive(serde::Serialize)]
21/// struct EncryptedFoo<'a> {
22///     id: &'a Vec<u8>,
23///     sensitive_data: Vec<u8>,
24/// }
25///
26/// impl<'a> Encrypting<'a> for Foo {
27///     type EncryptedForm: EncryptedFoo<'a>;
28///
29///     fn encrypt(&'a self, cipher: &aes_gcm::Aes256Gcm) -> CryptoKeystoreResult<EncryptedFoo<'a>> {
30///         Ok(EncryptedFoo {
31///             id: &self.id,
32///             sensitive_data: self.encrypt_data(cipher, &self.sensitive_data)?,
33///         })
34///     }
35/// }
36/// ```
37///
38/// This can then be used like:
39///
40/// ```rust,ignore
41/// let json = serde_json::to_string(&foo.encrypt(cipher)?)?;
42/// ```
43pub trait Encrypting<'a> {
44    /// This type must be serializable, but can depend on the lifetime of `self` to reduce copying.
45    type EncryptedForm: 'a + Serialize;
46
47    /// Make an instance of the encrypted form of this struct, for which all sensitive fields have been encrypted.
48    fn encrypt(&'a self, cipher: &aes_gcm::Aes256Gcm) -> CryptoKeystoreResult<Self::EncryptedForm>;
49}
