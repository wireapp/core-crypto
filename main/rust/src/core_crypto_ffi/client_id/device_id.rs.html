16#[uniffi::export(Eq, Hash)]
17/// A Device ID.
18pub struct DeviceId(u64);
19
20#[uniffi::export]
21impl DeviceId {
22    /// New device id from an unsigned 64-bit integer.
23    #[uniffi::constructor]
24    pub fn new(id: u64) -> Self {
25        Self(id)
26    }
27
28    /// Construct a `DeviceId` from 8 bytes encoded as a hex string.
29    #[uniffi::constructor]
30    pub fn from_hex_string(hex_string: &str) -> CoreCryptoResult<Self> {
31        u64::from_str_radix(hex_string, 16)
32            .map(Self)
33            .map_err(|e| CoreCryptoError::Other {
34                msg: format!("invalid device id: {e}"),
35            })
36    }
37
38    /// Encode the `DeviceId` as a 16-character lowercase hex string, with leading 0s.
39    pub fn to_hex_string(&self) -> String {
40        format!("{:016x}", self.0)
41    }
42
43    /// Get the number corresponding to this `DeviceId`.
44    pub fn to_u64(&self) -> u64 {
45        self.0
46    }
47}
