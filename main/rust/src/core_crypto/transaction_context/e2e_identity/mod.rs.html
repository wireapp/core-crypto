23impl TransactionContext {
24    /// Creates an enrollment instance with private key material you can use in order to fetch
25    /// a new x509 certificate from the acme server.
26    ///
27    /// # Parameters
28    /// * `client_id` - client identifier e.g. `b7ac11a4-8f01-4527-af88-1c30885a7931:6add501bacd1d90e@example.com`
29    /// * `display_name` - human readable name displayed in the application e.g. `Smith, Alice M (QA)`
30    /// * `handle` - user handle e.g. `alice.smith.qa@example.com`
31    /// * `expiry_sec` - generated x509 certificate expiry in seconds
32    pub async fn e2ei_new_enrollment(
33        &self,
34        client_id: ClientId,
35        display_name: String,
36        handle: String,
37        team: Option<String>,
38        expiry_sec: u32,
39        ciphersuite: Ciphersuite,
40    ) -> Result<E2eiEnrollment> {
41        let signature_keypair = None; // fresh install without a Basic client. Supplying None will generate a new keypair
42        E2eiEnrollment::try_new(
43            client_id,
44            display_name,
45            handle,
46            team,
47            expiry_sec,
48            &self
49                .mls_provider()
50                .await
51                .map_err(RecursiveError::transaction("getting mls provider"))?,
52            ciphersuite,
53            signature_keypair,
54            false, // fresh install so no refresh token registered yet
55        )
56        .map_err(RecursiveError::e2e_identity("creating new enrollment"))
57        .map_err(Into::into)
58    }

60    /// Parses the ACME server response from the endpoint fetching x509 certificates and uses it
61    /// to initialize the MLS client with a certificate
62    pub async fn e2ei_mls_init_only(
63        &self,
64        enrollment: &mut E2eiEnrollment,
65        certificate_chain: String,
66    ) -> Result<NewCrlDistributionPoints> {
67        let mls_provider = self
68            .mls_provider()
69            .await
70            .map_err(RecursiveError::transaction("getting mls provider"))?;
71
72        let sk = enrollment
73            .get_sign_key_for_mls()
74            .map_err(RecursiveError::e2e_identity("creating new enrollment"))?;
75        let cs = *enrollment.ciphersuite();
76        let certificate_chain = enrollment
77            .certificate_response(
78                certificate_chain,
79                mls_provider
80                    .authentication_service()
81                    .borrow()
82                    .await
83                    .as_ref()
84                    .ok_or(Error::PkiEnvironmentUnset)?,
85            )
86            .await
87            .map_err(RecursiveError::e2e_identity("getting certificate response"))?;
88
89        let crl_new_distribution_points = self.extract_dp_on_init(&certificate_chain[..]).await?;
90
91        let private_key = CertificatePrivateKey {
92            value: sk.into(),
93            signature_scheme: cs.signature_algorithm(),
94        };
95
96        let cert_bundle = CertificateBundle {
97            certificate_chain,
98            private_key,
99        };
100
101        let mut credential = Credential::x509(cert_bundle.clone()).map_err(RecursiveError::mls_credential(
102            "creating credential from certificate bundle in e2ei_mls_init_only",
103        ))?;
104        // we don't need to keep the credential ref; this credential will be loaded in mls_init later on
105        credential
106            .save(&mls_provider.keystore())
107            .await
108            .map_err(RecursiveError::mls_credential(
109                "saving credential in e2ei_mls_init_only",
110            ))?;
111
112        let identifier = ClientIdentifier::X509(HashMap::from([(cs.signature_algorithm(), cert_bundle)]));
113        self.mls_init(identifier, &[cs])
114            .await
115            .map_err(RecursiveError::transaction("initializing mls"))?;
116        Ok(crl_new_distribution_points)
117    }

119    /// When x509 new credentials are registered this extracts the new CRL Distribution Point from the end entity certificate
120    /// and all the intermediates
121    async fn extract_dp_on_init(&self, certificate_chain: &[Vec<u8>]) -> Result<NewCrlDistributionPoints> {
122        use x509_cert::der::Decode as _;
123
124        // Own intermediates are not provided by smallstep in the /federation endpoint so we got to intercept them here, at issuance
125        let size = certificate_chain.len();
126        let mut crl_new_distribution_points = HashSet::new();
127        if size > 1 {
128            for int in certificate_chain.iter().skip(1).rev() {
129                let mut crl_dp = self
130                    .e2ei_register_intermediate_ca_der(int)
131                    .await
132                    .map_err(RecursiveError::transaction("registering intermediate ca der"))?;
133                if let Some(crl_dp) = crl_dp.take() {
134                    crl_new_distribution_points.extend(crl_dp);
135                }
136            }
137        }
138
139        let ee = certificate_chain.first().ok_or(Error::InvalidCertificateChain)?;
140        let ee = x509_cert::Certificate::from_der(ee)
141            .map_err(crate::mls::credential::Error::DecodeX509)
142            .map_err(RecursiveError::mls_credential("decoding x509 credential"))?;
143        let mut ee_crl_dp = extract_crl_uris(&ee).map_err(RecursiveError::e2e_identity("extracting crl urls"))?;
144        if let Some(crl_dp) = ee_crl_dp.take() {
145            crl_new_distribution_points.extend(crl_dp);
146        }
147
148        get_new_crl_distribution_points(
149            &self
150                .mls_provider()
151                .await
152                .map_err(RecursiveError::transaction("getting mls provider"))?,
153            crl_new_distribution_points,
154        )
155        .await
156        .map_err(RecursiveError::mls_credential("getting new crl distribution points"))
157        .map_err(Into::into)
158    }

161#[cfg(test)]
162mod tests {
163    use crate::{
164        e2e_identity::enrollment::test_utils as e2ei_utils,
165        mls::conversation::Conversation as _,
166        test_utils::{x509::X509TestChain, *},
167        *,

169
170    #[apply(all_cred_cipher)]
171    async fn e2e_identity_should_work(case: TestContext) {
172        use e2ei_utils::E2EI_CLIENT_ID_URI;
173
174        let session = SessionContext::new_uninitialized(&case).await;
175        Box::pin(async move {
176            let owned_x509_test_chain;
177            let x509_test_chain = match session.x509_chain() {
178                Some(chain) => chain,
179                None => {
180                    owned_x509_test_chain = X509TestChain::init_empty(case.signature_scheme());
181                    &owned_x509_test_chain
182                }
183            };
184
185            let is_renewal = false;
186
187            let (mut enrollment, cert) = e2ei_utils::e2ei_enrollment(
188                &session,
189                &case,
190                x509_test_chain,
191                Some(E2EI_CLIENT_ID_URI),
192                is_renewal,
193                e2ei_utils::init_enrollment,
194                e2ei_utils::noop_restore,
195            )
196            .await
197            .unwrap();
198
199            session
200                .transaction
201                .e2ei_mls_init_only(&mut enrollment, cert)
202                .await
203                .unwrap();
204
205            // verify the created client can create a conversation
206            let conversation = case
207                .create_conversation_with_credential_type(CredentialType::X509, [&session])
208                .await;
209            conversation
210                .guard()
211                .await
212                .encrypt_message("Hello e2e identity !")
213                .await
214                .unwrap();
215            assert_eq!(
216                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
217                E2eiConversationState::Verified
218            );
219            assert!(
220                session
221                    .transaction
222                    .e2ei_is_enabled(case.signature_scheme())
223                    .await
224                    .unwrap()
225            );
226        })
227        .await
228    }
