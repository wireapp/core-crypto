19fn generate_nonce() -> String {
20    let nonce = uuid::Uuid::new_v4();
21    base64::prelude::BASE64_URL_SAFE_NO_PAD.encode(nonce)
22}

26async fn wire_api(nonces: &Mutex<Nonces>, req: Request<Incoming>) -> http::Result<Response<Full<Bytes>>> {
27    let (parts, body) = req.into_parts();
28    let path = parts.uri.path();
29    let paths = path.split('/').filter(|p| !p.is_empty()).collect::<Vec<&str>>();
30    let header = |k: &str| {
31        parts
32            .headers
33            .get(k)
34            .and_then(|d| d.to_str().ok())
35            .and_then(|v| base64::prelude::BASE64_URL_SAFE_NO_PAD.decode(v).ok())
36            .and_then(|v| String::from_utf8(v).ok())
37            .unwrap_or_else(|| panic!("No header '{k}'"))
38    };
39    Ok(match (parts.method, paths.as_slice()) {
40        (Method::GET, ["clients", device_id, "nonce"]) => {
41            let nonce = generate_nonce();
42            nonces
43                .lock()
44                .unwrap()
45                .insert(device_id.to_string(), nonce.clone().into());
46            Response::builder()
47                .status(StatusCode::OK)
48                .body(Full::new(nonce.into()))?
49        }
50        (Method::POST, ["clients", device_id, "access-token"]) => {
51            let bytes = body.collect().await.unwrap().to_bytes();
52            let context: HashMap<String, String> = serde_json::from_slice(&bytes).unwrap();
53            let client_id = &context["client-id"];
54            let client_id = ClientId::try_from_uri(client_id).unwrap();
55
56            let dpop = header("dpop");
57            let backend_nonce = nonces.lock().unwrap()[*device_id].clone();
58
59            // verify this tests has a valid API handler expecting the right deviceId encoding
60            let received_device_id = u64::from_str_radix(device_id, 16).unwrap();
61            assert_eq!(received_device_id, client_id.device_id);
62
63            match generate_access_token(&context, &dpop, client_id, backend_nonce) {
64                Ok(body) => {
65                    let body = serde_json::to_vec(&body).unwrap().into();
66                    Response::builder().status(StatusCode::OK).body(body).unwrap()
67                }
68                Err(_) => Response::builder()
69                    .status(StatusCode::UNAUTHORIZED)
70                    .body("".into())
71                    .unwrap(),
72            }
73        }
74        (Method::GET, ["callback"]) => handle_callback(parts.uri).await?,
75        _ => not_found()?,
76    })
77}

79async fn handle_callback(uri: http::Uri) -> http::Result<Response<Full<Bytes>>> {
80    let req_uri: url::Url = format!("http://localhost{uri}").parse().unwrap();
81    let authorization_code = req_uri
82        .query_pairs()
83        .find_map(|(k, v)| match k.as_ref() {
84            "code" => Some(v.to_string()),
85            _ => None,
86        })
87        .unwrap();
88    let resp = Response::builder()
89        .status(StatusCode::OK)
90        .body(authorization_code.into())
91        .unwrap();
92    Ok(resp)
93}

95fn generate_access_token(
96    ctx: &HashMap<String, String>,
97    dpop: &str,
98    client_id: ClientId,
99    nonce: BackendNonce,
100) -> RustyJwtResult<serde_json::Value> {
101    let backend_kp: Pem = ctx["backend-kp"].as_str().into();
102    let hash_alg: HashAlgorithm = ctx["hash-alg"].parse().unwrap();
103    let htu: Htu = ctx["wire-server-uri"].as_str().try_into().unwrap();
104    let handle = QualifiedHandle::from_str(ctx["handle"].as_str()).unwrap();
105    let display_name = ctx["display_name"].as_str();
106    let team: Team = ctx["team"].as_str().into();
107
108    let leeway = 2;
109    let max_expiry = 2082008461;
110    let access_token = RustyJwtTools::generate_access_token(
111        dpop,
112        &client_id,
113        handle,
114        display_name,
115        team,
116        nonce,
117        htu,
118        Htm::Post,
119        leeway,
120        max_expiry,
121        backend_kp,
122        hash_alg,
123        5,
124        core::time::Duration::from_secs(360),
125    )?;
126
127    Ok(serde_json::json!({
128        "expires_in": 2082008461,
129        "token": access_token,
130        "type": "DPoP"
131    }))
132}

140#[tokio::main(flavor = "current_thread")]
141async fn main() {
142    let addr = SocketAddrV4::new(Ipv4Addr::LOCALHOST, 0);
143    let listener = TcpListener::bind(&addr).await.unwrap();
144    let addr = listener.local_addr().unwrap();
145    println!("{addr}");
146
147    let nonces: Arc<Mutex<Nonces>> = Mutex::new(HashMap::new()).into();
148
149    while let Ok((stream, _)) = listener.accept().await {
150        let io = TokioIo::new(stream);
151        let cloned = Arc::clone(&nonces);
152        tokio::spawn(async move {
153            let service = hyper::service::service_fn(|req| wire_api(&cloned, req));
154            if let Err(err) = http1::Builder::new().serve_connection(io, service).await {
155                eprintln!("server error: {err}");
156            }
157        });
158    }
159}
