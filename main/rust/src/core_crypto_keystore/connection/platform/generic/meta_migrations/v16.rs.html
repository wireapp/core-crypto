10pub(crate) fn meta_migration(conn: &mut rusqlite::Connection) -> CryptoKeystoreResult<()> {
11    let tx = conn.transaction()?;
12    let mut stmt = tx.prepare(&format!(
13        "SELECT
14            {credential_table}.rowid AS cred_rowid,
15            id,
16            credential,
17            unixepoch(created_at) AS created_at,
18            signature_scheme,
19            pk,
20            keypair,
21            credential_id
22         FROM {credential_table}, {keypair_table}
23         WHERE {keypair_table}.credential_id = {credential_table}.id",
24        credential_table = StoredCredential::COLLECTION_NAME,
25        keypair_table = "mls_signature_keypairs"
26    ))?;
27
28    let mut rows = stmt.query([])?;
29    while let Some(row) = rows.next()? {
30        let v5 = V5Credential {
31            id: row.get("id")?,
32            credential: row.get("credential")?,
33            created_at: row.get("created_at")?,
34        };
35
36        let kp = StoredSignatureKeypair {
37            signature_scheme: row.get("signature_scheme")?,
38            pk: row.get("pk")?,
39            keypair: row.get("keypair")?,
40            credential_id: row.get("credential_id")?,
41        };
42
43        // Insert the new credential into temporary mls_credentials_new table, that will be renamed in the next
44        // migration
45        if let Some(c) = migrate_to_new_credential(&v5, &kp)? {
46            tx.execute(
47                "INSERT INTO mls_credentials_new (
48                        session_id,
49                        credential,
50                        created_at,
51                        signature_scheme,
52                        public_key,
53                        private_key
54                    )
55                    VALUES (?1, ?2, datetime(?3, 'unixepoch'), ?4, ?5, ?6)",
56                (
57                    c.session_id.clone(),
58                    c.credential.clone(),
59                    c.created_at,
60                    c.signature_scheme,
61                    c.public_key.clone(),
62                    c.private_key.clone(),
63                ),
64            )?;
65
66            // Delete this credential from the old table, so that migration only happens once
67            let rowid = row.get::<_, i32>("cred_rowid")?;
68            tx.execute(
69                "DELETE FROM mls_credentials
70                        WHERE rowid = ?1",
71                (rowid,),
72            )?;
73        }
74    }
75
76    drop(rows);
77    drop(stmt);
78
79    tx.commit()?;
80
81    Ok(())
82}
