26impl TransactionContext {
27    /// Creates an enrollment instance with private key material you can use in order to fetch
28    /// a new x509 certificate from the acme server.
29    ///
30    /// # Parameters
31    /// * `client_id` - client identifier e.g. `b7ac11a4-8f01-4527-af88-1c30885a7931:6add501bacd1d90e@example.com`
32    /// * `display_name` - human readable name displayed in the application e.g. `Smith, Alice M (QA)`
33    /// * `handle` - user handle e.g. `alice.smith.qa@example.com`
34    /// * `expiry_sec` - generated x509 certificate expiry in seconds
35    pub async fn e2ei_new_enrollment(
36        &self,
37        client_id: ClientId,
38        display_name: String,
39        handle: String,
40        team: Option<String>,
41        expiry_sec: u32,
42        ciphersuite: Ciphersuite,
43    ) -> Result<E2eiEnrollment> {
44        let signature_keypair = None; // fresh install without a Basic client. Supplying None will generate a new keypair
45        E2eiEnrollment::try_new(
46            client_id,
47            display_name,
48            handle,
49            team,
50            expiry_sec,
51            &self
52                .mls_provider()
53                .await
54                .map_err(RecursiveError::transaction("getting mls provider"))?,
55            ciphersuite,
56            signature_keypair,
57            false, // fresh install so no refresh token registered yet
58        )
59        .map_err(RecursiveError::e2e_identity("creating new enrollment"))
60        .map_err(Into::into)
61    }

63    /// Parses the ACME server response from the endpoint fetching x509 certificates and uses it
64    /// to initialize the MLS client with a certificate
65    pub async fn e2ei_mls_init_only(
66        &self,
67        enrollment: &mut E2eiEnrollment,
68        certificate_chain: String,
69        transport: Arc<dyn MlsTransport>,
70    ) -> Result<(CredentialRef, NewCrlDistributionPoints)> {
71        let pki_environment = self
72            .pki_environment()
73            .await
74            .map_err(RecursiveError::transaction("getting pki environment"))?;
75
76        let sk = enrollment
77            .get_sign_key_for_mls()
78            .map_err(RecursiveError::e2e_identity("creating new enrollment"))?;
79        let ciphersuite = *enrollment.ciphersuite();
80        let certificate_chain = enrollment
81            .certificate_response(
82                certificate_chain,
83                pki_environment
84                    .mls_pki_env_provider()
85                    .borrow()
86                    .await
87                    .as_ref()
88                    .ok_or(Error::PkiEnvironmentUnset)?,
89            )
90            .await
91            .map_err(RecursiveError::e2e_identity("getting certificate response"))?;
92
93        let crl_new_distribution_points = self.extract_dp_on_init(&certificate_chain[..]).await?;
94
95        let private_key = CertificatePrivateKey::new(sk);
96
97        let cert_bundle = CertificateBundle {
98            certificate_chain,
99            private_key,
100            signature_scheme: ciphersuite.signature_algorithm(),
101        };
102
103        let mut credential = Credential::x509(ciphersuite, cert_bundle.clone()).map_err(
104            RecursiveError::mls_credential("creating credential from certificate bundle in e2ei_mls_init_only"),
105        )?;
106        let database = &self
107            .database()
108            .await
109            .map_err(RecursiveError::transaction("Getting database from transaction context"))?;
110        let credential_ref = credential.save(database).await.map_err(RecursiveError::mls_credential(
111            "saving credential in e2ei_mls_init_only",
112        ))?;
113
114        let identifier = ClientIdentifier::X509(HashMap::from([(ciphersuite.signature_algorithm(), cert_bundle)]));
115        self.mls_init(identifier, transport)
116            .await
117            .map_err(RecursiveError::transaction("initializing mls"))?;
118        Ok((credential_ref, crl_new_distribution_points))
119    }

121    /// When x509 new credentials are registered this extracts the new CRL Distribution Point from the end entity
122    /// certificate and all the intermediates
123    async fn extract_dp_on_init(&self, certificate_chain: &[Vec<u8>]) -> Result<NewCrlDistributionPoints> {
124        use x509_cert::der::Decode as _;
125
126        // Own intermediates are not provided by smallstep in the /federation endpoint so we got to intercept them here,
127        // at issuance
128        let size = certificate_chain.len();
129        let mut crl_new_distribution_points = HashSet::new();
130        if size > 1 {
131            for int in certificate_chain.iter().skip(1).rev() {
132                let mut crl_dp = self
133                    .e2ei_register_intermediate_ca_der(int)
134                    .await
135                    .map_err(RecursiveError::transaction("registering intermediate ca der"))?;
136                if let Some(crl_dp) = crl_dp.take() {
137                    crl_new_distribution_points.extend(crl_dp);
138                }
139            }
140        }
141
142        let ee = certificate_chain.first().ok_or(Error::InvalidCertificateChain)?;
143        let ee = x509_cert::Certificate::from_der(ee)
144            .map_err(crate::mls::credential::Error::DecodeX509)
145            .map_err(RecursiveError::mls_credential("decoding x509 credential"))?;
146        let mut ee_crl_dp = extract_crl_uris(&ee).map_err(RecursiveError::e2e_identity("extracting crl urls"))?;
147        if let Some(crl_dp) = ee_crl_dp.take() {
148            crl_new_distribution_points.extend(crl_dp);
149        }
150
151        let database = self
152            .database()
153            .await
154            .map_err(RecursiveError::transaction("getting database"))?;
155        get_new_crl_distribution_points(&database, crl_new_distribution_points)
156            .await
157            .map_err(RecursiveError::mls_credential("getting new crl distribution points"))
158            .map_err(Into::into)
159    }

175    async fn e2e_identity_should_work(mut case: TestContext) {
176        use e2ei_utils::E2EI_CLIENT_ID_URI;
177
178        let db = case.create_in_memory_database().await;
179        let cc = CoreCrypto::new(db);
180        let tx = cc.new_transaction().await.unwrap();
181        Box::pin(async move {
182            let chain = X509TestChain::init_empty(case.signature_scheme());
183
184            let is_renewal = false;
185
186            let (mut enrollment, cert) = e2ei_utils::e2ei_enrollment(
187                &tx,
188                &case,
189                &chain,
190                E2EI_CLIENT_ID_URI,
191                is_renewal,
192                e2ei_utils::init_enrollment,
193                e2ei_utils::noop_restore,
194            )
195            .await
196            .unwrap();
197            let transport = Arc::new(CoreCryptoTransportSuccessProvider::default());
198
199            let (credential_ref, _) = tx.e2ei_mls_init_only(&mut enrollment, cert, transport).await.unwrap();
200
201            let session = SessionContext::new_from_cc(&case, cc, Some(&chain)).await;
202
203            // verify the created client can create a conversation
204            let conversation = case
205                .create_conversation_with_credentials([(&session, &credential_ref)])
206                .await;
207            conversation
208                .guard()
209                .await
210                .encrypt_message("Hello e2e identity !")
211                .await
212                .unwrap();
213            assert_eq!(
214                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
215                E2eiConversationState::Verified
216            );
217            assert!(session.transaction.e2ei_is_enabled(case.ciphersuite()).await.unwrap());
218        })
219        .await
220    }
