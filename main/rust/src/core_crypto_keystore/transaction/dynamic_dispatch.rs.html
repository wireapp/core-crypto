161pub async fn execute_save(tx: &TransactionWrapper<'_>, entity: &Entity) -> CryptoKeystoreResult<()> {
162    match entity {
163        Entity::ConsumerData(consumer_data) => consumer_data.replace(tx).await,
164        Entity::SignatureKeyPair(mls_signature_key_pair) => mls_signature_key_pair.save(tx).await,
165        Entity::HpkePrivateKey(mls_hpke_private_key) => mls_hpke_private_key.save(tx).await,
166        Entity::MlsKeyPackage(mls_key_package) => mls_key_package.save(tx).await,
167        Entity::PskBundle(mls_psk_bundle) => mls_psk_bundle.save(tx).await,
168        Entity::EncryptionKeyPair(mls_encryption_key_pair) => mls_encryption_key_pair.save(tx).await,
169        Entity::MlsEpochEncryptionKeyPair(mls_epoch_encryption_key_pair) => {
170            mls_epoch_encryption_key_pair.save(tx).await
171        }
172        Entity::MlsCredential(mls_credential) => mls_credential.save(tx).await,
173        Entity::MlsBufferedCommit(mls_pending_commit) => mls_pending_commit.save(tx).await,
174        Entity::PersistedMlsGroup(persisted_mls_group) => persisted_mls_group.save(tx).await,
175        Entity::PersistedMlsPendingGroup(persisted_mls_pending_group) => persisted_mls_pending_group.save(tx).await,
176        Entity::MlsPendingMessage(mls_pending_message) => mls_pending_message.save(tx).await,
177        Entity::E2eiEnrollment(e2ei_enrollment) => e2ei_enrollment.save(tx).await,
178        #[cfg(target_family = "wasm")]
179        Entity::E2eiRefreshToken(e2ei_refresh_token) => e2ei_refresh_token.replace(tx).await,
180        Entity::E2eiAcmeCA(e2ei_acme_ca) => e2ei_acme_ca.replace(tx).await,
181        Entity::E2eiIntermediateCert(e2ei_intermediate_cert) => e2ei_intermediate_cert.save(tx).await,
182        Entity::E2eiCrl(e2ei_crl) => e2ei_crl.save(tx).await,
183        #[cfg(feature = "proteus-keystore")]
184        Entity::ProteusSession(record) => record.save(tx).await,
185        #[cfg(feature = "proteus-keystore")]
186        Entity::ProteusIdentity(record) => record.save(tx).await,
187        #[cfg(feature = "proteus-keystore")]
188        Entity::ProteusPrekey(record) => record.save(tx).await,
189    }
190}
