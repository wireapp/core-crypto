1//! Implement the MLS Transport interface.
2//!
3//! Unfortunately, there is no good way to reuse code between uniffi and wasm for an interface
4//! like this; the fundamental techniques in use are very different. So we just have to feature-gate
5//! everything.
6use std::{fmt, sync::Arc};
7
8use core_crypto::{HistorySecret, MlsCommitBundle};
9
10use crate::{ClientId, CommitBundle, HistorySecret as HistorySecretFfi, error::mls_transport::MlsTransportResult};

12/// Application data packaged to be encrypted and transmitted in an MLS application message.
13//
14// TODO: We derive Constructor here only because we need to construct an instance in interop.
15// Remove it once we drop the FFI client from interop.
16#[derive(Debug, derive_more::From, derive_more::Into, derive_more::Deref, derive_more::Constructor)]
17pub struct MlsTransportData(core_crypto::MlsTransportData);
18
19uniffi::custom_type!(MlsTransportData, Vec<u8>, {
20    lower: |key| key.0.to_vec(),
21    try_lift: |vec| {
22        Ok(MlsTransportData(core_crypto::MlsTransportData::from(vec)))
23    }
24});

26/// Used by CoreCrypto to send commits or application messages to the delivery service.
27///
28/// This trait must be implemented before calling any functions that produce commits.
29#[uniffi::export(with_foreign)]
30#[cfg_attr(target_os = "unknown", async_trait::async_trait(?Send))]
31#[cfg_attr(not(target_os = "unknown"), async_trait::async_trait)]
32pub trait MlsTransport: Send + Sync {
33    /// Send a commit bundle to the corresponding endpoint.
34    async fn send_commit_bundle(&self, commit_bundle: CommitBundle) -> MlsTransportResult;
35    /// Prepare a history secret before transmission.
36    async fn prepare_for_transport(&self, history_secret: HistorySecretFfi) -> MlsTransportData;
37}
38
39#[derive(derive_more::Constructor)]
40struct MlsTransportShim(Arc<dyn MlsTransport>);
41
42impl std::fmt::Debug for MlsTransportShim {
43    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> std::fmt::Result {
44        f.debug_tuple("MlsTransportShim")
45            .field(&fmt::from_fn(|f| write!(f, "{:p}", Arc::as_ptr(&self.0))))
46            .finish()
47    }
48}
49
50#[cfg_attr(target_os = "unknown", async_trait::async_trait(?Send))]
51#[cfg_attr(not(target_os = "unknown"), async_trait::async_trait)]
52impl core_crypto::MlsTransport for MlsTransportShim {
53    async fn send_commit_bundle(&self, commit_bundle: MlsCommitBundle) -> core_crypto::Result<()> {
54        let commit_bundle = CommitBundle::try_from(commit_bundle)
55            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
56        self.0.send_commit_bundle(commit_bundle).await.map_err(Into::into)
57    }
58
59    async fn prepare_for_transport(
60        &self,
61        secret: &HistorySecret,
62    ) -> core_crypto::Result<core_crypto::MlsTransportData> {
63        let client_id = ClientId::from(secret.client_id.clone()).into();
64        let history_secret = rmp_serde::to_vec(&secret)
65            .map(|secret| HistorySecretFfi {
66                client_id,
67                data: secret,
68            })
69            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
70        Ok(self.0.prepare_for_transport(history_secret).await.into())
71    }
72}
73
74/// In uniffi, `MlsTransport` is a trait which we need to wrap
75pub(crate) fn callback_shim(callbacks: Arc<dyn MlsTransport>) -> Arc<dyn core_crypto::MlsTransport> {
76    Arc::new(MlsTransportShim::new(callbacks))
