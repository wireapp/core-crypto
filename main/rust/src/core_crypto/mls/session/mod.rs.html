33/// A MLS Session enables a user device to communicate via the MLS protocol.
34///
35/// This closely maps to the `Client` term in [RFC 9720], but we avoid that term to avoid ambiguity;
36/// `Client` is very overloaded with distinct meanings.
37///
38/// There is one `Session` per user per device. A session can contain many MLS groups/conversations.
39///
40/// It is cheap to clone a `Session` because everything heavy is wrapped inside an [Arc].
41///
42/// [RFC 9720]: https://www.rfc-editor.org/rfc/rfc9420.html
43#[derive(Clone, derive_more::Debug)]
44pub struct Session {
45    pub(crate) inner: Arc<RwLock<Option<SessionInner>>>,
46    pub(crate) crypto_provider: MlsCryptoProvider,
47    pub(crate) transport: Arc<RwLock<Option<Arc<dyn MlsTransport + 'static>>>>,
48    #[debug("EpochObserver")]
49    pub(crate) epoch_observer: Arc<RwLock<Option<Arc<dyn EpochObserver + 'static>>>>,
50    #[debug("HistoryObserver")]
51    pub(crate) history_observer: Arc<RwLock<Option<Arc<dyn HistoryObserver + 'static>>>>,
52}

73    /// Creates a new [Session]. Does not initialize MLS or Proteus.
74    ///
75    /// ## Errors
76    ///
77    /// Failures in the initialization of the KeyStore can cause errors, such as IO, the same kind
78    /// of errors can happen when the groups are being restored from the KeyStore or even during
79    /// the client initialization (to fetch the identity signature).
80    pub async fn try_new(database: &Database) -> crate::mls::Result<Self> {
81        // cloning a database is relatively cheap; it's all arcs inside
82        let database = database.to_owned();
83        // Init backend (crypto + rand + keystore)
84        let mls_backend = MlsCryptoProvider::new(database);
85
86        // We create the core crypto instance first to enable creating a transaction from it and
87        // doing all subsequent actions inside a single transaction, though it forces us to clone
88        // a few Arcs and locks.
89        let session = Self {
90            crypto_provider: mls_backend,
91            inner: Default::default(),
92            transport: Arc::new(None.into()),
93            epoch_observer: Arc::new(None.into()),
94            history_observer: Arc::new(None.into()),
95        };
96
97        let cc = CoreCrypto::from(session);
98        let context = cc
99            .new_transaction()
100            .await
101            .map_err(RecursiveError::transaction("starting new transaction"))?;
102
103        context
104            .init_pki_env()
105            .await
106            .map_err(RecursiveError::transaction("initializing pki environment"))?;
107        context
108            .finish()
109            .await
110            .map_err(RecursiveError::transaction("finishing transaction"))?;
111
112        Ok(cc.mls)
113    }

121    /// Initializes the client.
122    ///
123    /// Loads any cryptographic material already present in the keystore, but does not create any.
124    /// If no credentials are present in the keystore, then one _must_ be created and added to the
125    /// session before it can be used.
126    pub async fn init(&self, identifier: ClientIdentifier, signature_schemes: &[SignatureScheme]) -> Result<()> {
127        self.ensure_unready().await?;
128        let client_id = identifier.get_id()?.into_owned();
129
130        // we want to find all credentials matching this identifier, having a valid signature scheme.
131        // the `CredentialRef::find` API doesn't allow us to easily find those credentials having
132        // one of a set of signature schemes, meaning we have two paths here:
133        // we could either search unbound by signature schemes and then filter for valid ones here,
134        // or we could iterate over the list of signature schemes and build up a set of credential refs.
135        // as there are only a few signature schemes possible and the cost of a find operation is non-trivial,
136        // we choose the first option.
137        // we might revisit this choice after WPB-20844 and WPB-21819.
138        let mut credential_refs = CredentialRef::find(
139            &self.crypto_provider.keystore(),
140            CredentialFindFilters::builder().client_id(&client_id).build(),
141        )
142        .await
143        .map_err(RecursiveError::mls_credential_ref(
144            "loading matching credential refs while initializing a client",
145        ))?;
146        credential_refs.retain(|credential_ref| signature_schemes.contains(&credential_ref.signature_scheme()));
147
148        let mut identities = Identities::new(credential_refs.len());
149        let credentials_cache = CredentialRef::load_stored_credentials(&self.crypto_provider.keystore())
150            .await
151            .map_err(RecursiveError::mls_credential_ref(
152                "loading credential ref cache while initializing session",
153            ))?;
154
155        for credential_ref in credential_refs {
156            for credential_result in
157                credential_ref
158                    .load_from_cache(&credentials_cache)
159                    .map_err(RecursiveError::mls_credential_ref(
160                        "loading credential list in session init",
161                    ))?
162            {
163                let credential = credential_result
164                    .map_err(RecursiveError::mls_credential_ref("loading credential in session init"))?;
165
166                match identities.push_credential(credential).await {
167                    Err(Error::CredentialConflict) => {
168                        // this is what we get for not having real primary keys in our DB
169                        // no harm done though; no need to propagate this error
170                    }
171                    Ok(_) => {}
172                    Err(err) => {
173                        return Err(RecursiveError::MlsClient {
174                            context: "adding credential to identities in init",
175                            source: Box::new(err),
176                        }
177                        .into());
178                    }
179                }
180            }
181        }
182
183        self.replace_inner(SessionInner {
184            id: client_id,
185            identities,
186        })
187        .await;
188
189        Ok(())
190    }

192    /// Resets the client to an uninitialized state.

199    pub(crate) async fn is_ready(&self) -> bool {
200        let inner_lock = self.inner.read().await;
201        inner_lock.is_some()
202    }

204    async fn ensure_unready(&self) -> Result<()> {
205        if self.is_ready().await {
206            Err(Error::UnexpectedlyReady)
207        } else {
208            Ok(())
209        }
210    }

212    async fn replace_inner(&self, new_inner: SessionInner) {
213        let mut inner_lock = self.inner.write().await;
214        *inner_lock = Some(new_inner);

217    /// Get an immutable view of an `MlsConversation`.
218    ///
219    /// Because it operates on the raw conversation type, this may be faster than
220    /// [crate::transaction_context::TransactionContext::conversation]. for transient and immutable
221    /// purposes. For long-lived or mutable purposes, prefer the other method.
222    pub async fn get_raw_conversation(&self, id: &ConversationIdRef) -> Result<ImmutableConversation> {
223        let raw_conversation = GroupStore::fetch_from_keystore(id, &self.crypto_provider.keystore(), None)
224            .await
225            .map_err(RecursiveError::root("getting conversation by id"))?
226            .ok_or_else(|| LeafError::ConversationNotFound(id.to_owned()))?;
227        Ok(ImmutableConversation::new(raw_conversation, self.clone()))
228    }

230    /// Returns the client's most recent public signature key as a buffer.
231    /// Used to upload a public key to the server in order to verify client's messages signature.
232    ///
233    /// # Arguments
234    /// * `ciphersuite` - a callback to be called to perform authorization
235    /// * `credential_type` - of the credential to look for
236    pub async fn public_key(
237        &self,
238        ciphersuite: Ciphersuite,
239        credential_type: CredentialType,
240    ) -> crate::mls::Result<Vec<u8>> {
241        let cb = self
242            .find_most_recent_credential(ciphersuite.signature_algorithm(), credential_type)
243            .await
244            .map_err(RecursiveError::mls_client("finding most recent credential"))?;
245        Ok(cb.signature_key_pair.to_public_vec())
246    }

248    /// Checks if a given conversation id exists locally
249    pub async fn conversation_exists(&self, id: &ConversationIdRef) -> Result<bool> {
250        match self.get_raw_conversation(id).await {
251            Ok(_) => Ok(true),
252            Err(Error::Leaf(LeafError::ConversationNotFound(_))) => Ok(false),
253            Err(e) => Err(e),
254        }

288    /// Restore from an external [`HistorySecret`].
289    pub(crate) async fn restore_from_history_secret(&self, history_secret: HistorySecret) -> Result<()> {
290        self.ensure_unready().await?;
291
292        // store the client id (with some other stuff)
293        self.replace_inner(SessionInner {
294            id: history_secret.client_id.clone(),
295            identities: Identities::new(0),
296        })
297        .await;
298
299        // store the key package
300        history_secret
301            .key_package
302            .store(&self.crypto_provider)
303            .await
304            .map_err(MlsError::wrap("storing key package encapsulation"))?;
305
306        Ok(())
307    }

309    /// Retrieves the client's client id. This is free-form and not inspected.
310    pub async fn id(&self) -> Result<ClientId> {
311        match &*self.inner.read().await {
312            None => Err(Error::MlsNotInitialized),
313            Some(SessionInner { id, .. }) => Ok(id.clone()),
314        }
315    }

317    /// Returns whether this client is E2EI capable
318    pub async fn is_e2ei_capable(&self) -> bool {
319        match &*self.inner.read().await {
320            None => false,
321            Some(SessionInner { identities, .. }) => identities
322                .iter()
323                .any(|cred| cred.credential_type() == CredentialType::X509),
324        }
325    }

388            let keystore = self.crypto_provider.keystore();
389            let credential = keystore.count::<StoredCredential>().await.unwrap();
390            let encryption_keypair = keystore.count::<StoredEncryptionKeyPair>().await.unwrap();
391            let epoch_encryption_keypair = keystore.count::<StoredEpochEncryptionKeypair>().await.unwrap();
392            let enrollment = keystore.count::<StoredE2eiEnrollment>().await.unwrap();
393            let group = keystore.count::<PersistedMlsGroup>().await.unwrap();
394            let hpke_private_key = keystore.count::<StoredHpkePrivateKey>().await.unwrap();
395            let key_package = keystore.count::<StoredKeypackage>().await.unwrap();
396            let pending_group = keystore.count::<PersistedMlsPendingGroup>().await.unwrap();
397            let pending_messages = keystore.count::<MlsPendingMessage>().await.unwrap();
398            let psk_bundle = keystore.count::<StoredPskBundle>().await.unwrap();
399            EntitiesCount {
400                credential,
401                encryption_keypair,
402                epoch_encryption_keypair,
403                enrollment,
404                group,
405                hpke_private_key,
406                key_package,
407                pending_group,
408                pending_messages,
409                psk_bundle,
