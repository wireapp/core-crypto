17    /// A secret key derived from the group secret.
18    ///
19    /// This is intended to be used for AVS.
20    #[derive(Debug, Clone, PartialEq, Eq, Hash)]
21    #[uniffi::export(Eq, Hash, Display)]
22    SecretKey infallibly wraps core_crypto::mls::conversation::SecretKey; copy_bytes

24impl fmt::Display for SecretKey {
25    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
26        let hash = obfuscate::compute_hash(&self.0);
27        let mut buffer = [0; 20];
28        hex::encode_to_slice(hash, &mut buffer).expect("hash has constant length 10 therefore encodes to 20 bytes");
29        // SAFETY: hex crate always produces valid utf-8
30        let data = unsafe { str::from_utf8_unchecked(&buffer) };
31        f.write_str(data)
32    }

45/// MLS Group Information
46///
47/// This is used when joining by external commit.
48/// It can be found within the `GroupInfoBundle` within a `CommitBundle`.
49//
50// We can't use the `bytes_wrapper` macro here because it requires certain
51// trait implementations for its implementation, and `VerifiableGroupInfo`
52// does not and cannot implement these because despite the import path it is
53// an OpenMLS type, not a CoreCrypto type.
54//
55// So we just reimplement it manually.

57pub struct GroupInfo(pub(crate) core_crypto::VerifiableGroupInfo);

279    /// Decrypts an MLS message received in the given conversation.
280    /// **Note**: this will discard any local pending operations.
281    pub async fn decrypt_message(
282        &self,
283        conversation_id: &ConversationId,
284        payload: Vec<u8>,
285    ) -> CoreCryptoResult<DecryptedMessage> {
286        let conversation_result = self.inner.conversation(conversation_id.as_ref()).await;
287        let decrypted_message = match conversation_result {
288            Err(TransactionError::PendingConversation(mut pending)) => {
289                pending.try_process_own_join_commit(&payload).await?
290            }
291            Ok(mut conversation) => conversation.decrypt_message(&payload).await?,
292            Err(e) => Err(e)?,
293        };
294
295        Ok(decrypted_message.into())
296    }

457impl CoreCryptoContext {
458    /// Get all credentials from this client which match the provided parameters.
459    ///
460    /// Parameters which are unset or `None` match anything. Those with a particular value find only credentials
461    /// matching that value.
462    async fn find_credentials_inner(
463        &self,
464        client_id: Option<Arc<ClientId>>,
465        public_key: Option<Vec<u8>>,
466        ciphersuite: Option<Ciphersuite>,
467        credential_type: Option<CredentialType>,
468        earliest_validity: Option<u64>,
469    ) -> CoreCryptoResult<Vec<Arc<CredentialRef>>> {
470        let client_id = client_id.as_ref().map(|c| c.as_ref().as_ref());
471
472        let ciphersuite = ciphersuite.map(CryptoCiphersuite::from);
473
474        let credential_type = credential_type.map(core_crypto::CredentialType::from);
475
476        let find_filters = CredentialFindFilters {
477            client_id,
478            public_key_hash: public_key.map(Sha256Hash::hash_from),
479            ciphersuite,
480            credential_type,
481            earliest_validity,
482        };
483
484        self.inner
485            .find_credentials(find_filters)
486            .await
487            .map(|credentials| credentials.into_iter().map(CredentialRef::from).map(Arc::new).collect())
488            .map_err(Into::into)
489    }
