11/// MLS produces these kinds of error
12#[derive(Debug, thiserror::Error)]
13#[cfg_attr(target_family = "wasm", derive(strum::AsRefStr))]
14#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Error))]
15#[allow(missing_docs)] // error variants are self-describing
16pub enum MlsError {
17    /// The byte vector included in this error variant is the raw conversation id.
18    ///
19    /// We cannot provide a proper `ConversationId` instance because of a uniffi bug:
20    /// <https://github.com/mozilla/uniffi-rs/issues/2409>.
21    #[error("Conversation already exists")]
22    ConversationAlreadyExists { conversation_id: Vec<u8> },
23    #[error("We already decrypted this message once")]
24    DuplicateMessage,
25    #[error("Incoming message is for a future epoch. We will buffer it until the commit for that epoch arrives")]
26    BufferedFutureMessage,
27    #[error("Incoming message is from an epoch too far in the future to buffer.")]
28    WrongEpoch,
29    #[error(
30        "Incoming message is a commit for which we have not yet received all the proposals. Buffering until all proposals have arrived."
31    )]
32    BufferedCommit,
33    #[error("The epoch in which message was encrypted is older than allowed")]
34    MessageEpochTooOld,
35    #[error("Tried to decrypt a commit created by self which is likely to have been replayed by the DS")]
36    SelfCommitIgnored,
37    #[error(
38        "You tried to join with an external commit but did not merge it yet. We will reapply this message for you when you merge your external commit"
39    )]
40    UnmergedPendingGroup,
41    #[error("The received proposal is deemed stale and is from an older epoch.")]
42    StaleProposal,
43    #[error("The received commit is deemed stale and is from an older epoch.")]
44    StaleCommit,
45    /// This happens when the DS cannot flag KeyPackages as claimed or not. In this scenario, a client
46    /// requests their old KeyPackages to be deleted but one has already been claimed by another client to create a Welcome.
47    /// In that case the only solution is that the client receiving such a Welcome tries to join the group
48    /// with an External Commit instead
49    #[error(
50        "Although this Welcome seems valid, the local KeyPackage it references has already been deleted locally. Join this group with an external commit"
51    )]
52    OrphanWelcome,
53    #[error("Message rejected by the delivery service. Reason: {reason}")]
54    MessageRejected { reason: String },
55    #[error("{msg}")]
56    Other { msg: String },
57}
58
59impl From<core_crypto::MlsError> for MlsError {
60    #[inline]
61    fn from(e: core_crypto::MlsError) -> Self {
62        Self::Other {
63            msg: e.innermost_error_message(),
64        }
65    }
66}

68#[cfg(target_family = "wasm")]
69impl JsErrorContext for MlsError {
70    fn get_context(&self) -> JsValue {
71        let context = Object::new();
72        let inner_context = Object::new();
73        match &self {
74            e @ MlsError::ConversationAlreadyExists { conversation_id } => {
75                context.set_field("type", e.as_ref());
76                inner_context.set_field(
77                    "conversationId",
78                    serde_wasm_bindgen::to_value(conversation_id).expect("constructing json array"),
79                );
80            }
81            e @ (MlsError::DuplicateMessage
82            | MlsError::BufferedFutureMessage
83            | MlsError::WrongEpoch
84            | MlsError::BufferedCommit
85            | MlsError::MessageEpochTooOld
86            | MlsError::SelfCommitIgnored
87            | MlsError::UnmergedPendingGroup
88            | MlsError::StaleProposal
89            | MlsError::StaleCommit
90            | MlsError::OrphanWelcome) => context.set_field("type", e.as_ref()),
91            e @ MlsError::MessageRejected { reason } => {
92                context.set_field("type", e.as_ref());
93                inner_context.set_field("reason", reason);
94            }
95            e @ MlsError::Other { msg } => {
96                context.set_field("type", e.as_ref());
97                inner_context.set_field("msg", msg);
98            }
99        }
100        context.set_field("context", inner_context);
101        context.into()
102    }
103}
