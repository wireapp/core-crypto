71/// The base layer for [Conversation].
72/// The trait is only exposed internally.
73#[cfg_attr(target_family = "wasm", async_trait::async_trait(?Send))]
74#[cfg_attr(not(target_family = "wasm"), async_trait::async_trait)]
75pub(crate) trait ConversationWithMls<'a> {
76    /// [Client] or [CentralContext] both implement [HasClientAndProvider].
77    type Context: HasClientAndProvider;
78
79    type Conversation: Deref<Target = MlsConversation> + Send;
80
81    async fn context(&self) -> Result<Self::Context>;
82
83    async fn conversation(&'a self) -> Self::Conversation;
84
85    async fn mls_provider(&self) -> Result<MlsCryptoProvider> {
86        self.context()
87            .await?
88            .mls_provider()
89            .await
90            .map_err(RecursiveError::mls("getting mls provider"))
91            .map_err(Into::into)
92    }
93
94    async fn mls_client(&self) -> Result<Client> {
95        self.context()
96            .await?
97            .client()
98            .await
99            .map_err(RecursiveError::mls("getting mls client"))
100            .map_err(Into::into)
101    }
102}

104/// The `Conversation` trait provides a set of operations that can be done on
105/// an **immutable** conversation.
106// We keep the super trait internal intentionally, as it is not meant to be used by the public API,
107// hence #[expect(private_bounds)].
108#[expect(private_bounds)]
109#[cfg_attr(target_family = "wasm", async_trait::async_trait(?Send))]
110#[cfg_attr(not(target_family = "wasm"), async_trait::async_trait)]
111pub trait Conversation<'a>: ConversationWithMls<'a> {
112    /// Returns the epoch of a given conversation
113    async fn epoch(&'a self) -> u64 {
114        self.conversation().await.group().epoch().as_u64()
115    }
116
117    /// Returns the ciphersuite of a given conversation
118    async fn ciphersuite(&'a self) -> MlsCiphersuite {
119        self.conversation().await.ciphersuite()
120    }
121
122    /// Derives a new key from the one in the group, to be used elsewhere.
123    ///
124    /// # Arguments
125    /// * `key_length` - the length of the key to be derived. If the value is higher than the
126    ///     bounds of `u16` or the context hash * 255, an error will be returned
127    ///
128    /// # Errors
129    /// OpenMls secret generation error
130    async fn export_secret_key(&'a self, key_length: usize) -> Result<Vec<u8>> {
131        const EXPORTER_LABEL: &str = "exporter";
132        const EXPORTER_CONTEXT: &[u8] = &[];
133        let backend = self.mls_provider().await?;
134        let inner = self.conversation().await;
135        inner
136            .group()
137            .export_secret(&backend, EXPORTER_LABEL, EXPORTER_CONTEXT, key_length)
138            .map_err(MlsError::wrap("exporting secret key"))
139            .map_err(Into::into)
140    }
141
142    /// Exports the clients from a conversation
143    ///
144    /// # Arguments
145    /// * `conversation_id` - the group/conversation id
146    async fn get_client_ids(&'a self) -> Vec<ClientId> {
147        let inner = self.conversation().await;
148        inner
149            .group()
150            .members()
151            .map(|kp| ClientId::from(kp.credential.identity()))
152            .collect()
153    }
154
155    /// Returns the raw public key of the single external sender present in this group.
156    /// This should be used to initialize a subconversation
157    async fn get_external_sender(&'a self) -> Result<Vec<u8>> {
158        let inner = self.conversation().await;
159        let ext_senders = inner
160            .group()
161            .group_context_extensions()
162            .external_senders()
163            .ok_or(Error::MissingExternalSenderExtension)?;
164        let ext_sender = ext_senders.first().ok_or(Error::MissingExternalSenderExtension)?;
165        let ext_sender_public_key = ext_sender.signature_key().as_slice().to_vec();
166        Ok(ext_sender_public_key)
167    }
168
169    /// Indicates when to mark a conversation as not verified i.e. when not all its members have a X509
170    /// Credential generated by Wire's end-to-end identity enrollment
171    async fn e2ei_conversation_state(&'a self) -> Result<E2eiConversationState> {
172        let backend = self.mls_provider().await?;
173        let authentication_service = backend.authentication_service();
174        authentication_service.refresh_time_of_interest().await;
175        let inner = self.conversation().await;
176        let state = Client::compute_conversation_state(
177            inner.ciphersuite(),
178            inner.group.members_credentials(),
179            MlsCredentialType::X509,
180            authentication_service.borrow().await.as_ref(),
181        )
182        .await;
183        Ok(state)
184    }
185
186    /// From a given conversation, get the identity of the members supplied. Identity is only present for
187    /// members with a Certificate Credential (after turning on end-to-end identity).
188    /// If no member has a x509 certificate, it will return an empty Vec
189    async fn get_device_identities(&'a self, device_ids: &[ClientId]) -> Result<Vec<WireIdentity>> {
190        if device_ids.is_empty() {
191            return Err(Error::CallerError(
192                "This function accepts a list of IDs as a parameter, but that list was empty.",
193            ));
194        }
195        let mls_provider = self.mls_provider().await?;
196        let auth_service = mls_provider.authentication_service();
197        auth_service.refresh_time_of_interest().await;
198        let auth_service = auth_service.borrow().await;
199        let env = auth_service.as_ref();
200        let conversation = self.conversation().await;
201        conversation
202            .members_with_key()
203            .into_iter()
204            .filter(|(id, _)| device_ids.contains(&ClientId::from(id.as_slice())))
205            .map(|(_, c)| {
206                c.extract_identity(conversation.ciphersuite(), env)
207                    .map_err(RecursiveError::mls_credential("extracting identity"))
208            })
209            .collect::<Result<Vec<_>, _>>()
210            .map_err(Into::into)
211    }
212
213    /// From a given conversation, get the identity of the users (device holders) supplied.
214    /// Identity is only present for devices with a Certificate Credential (after turning on end-to-end identity).
215    /// If no member has a x509 certificate, it will return an empty Vec.
216    ///
217    /// Returns a Map with all the identities for a given users. Consumers are then recommended to
218    /// reduce those identities to determine the actual status of a user.
219    async fn get_user_identities(&'a self, user_ids: &[String]) -> Result<HashMap<String, Vec<WireIdentity>>> {
220        if user_ids.is_empty() {
221            return Err(Error::CallerError(
222                "This function accepts a list of IDs as a parameter, but that list was empty.",
223            ));
224        }
225        let mls_provider = self.mls_provider().await?;
226        let auth_service = mls_provider.authentication_service();
227        auth_service.refresh_time_of_interest().await;
228        let auth_service = auth_service.borrow().await;
229        let env = auth_service.as_ref();
230        let conversation = self.conversation().await;
231        let user_ids = user_ids.iter().map(|uid| uid.as_bytes()).collect::<Vec<_>>();
232
233        conversation
234            .members_with_key()
235            .iter()
236            .filter_map(|(id, c)| UserId::try_from(id.as_slice()).ok().zip(Some(c)))
237            .filter(|(uid, _)| user_ids.contains(uid))
238            .map(|(uid, c)| {
239                let uid = String::try_from(uid).map_err(RecursiveError::mls_client("getting user identities"))?;
240                let identity = c
241                    .extract_identity(conversation.ciphersuite(), env)
242                    .map_err(RecursiveError::mls_credential("extracting identity"))?;
243                Ok((uid, identity))
244            })
245            .process_results(|iter| iter.into_group_map())
246    }
247}

254/// This is a wrapper on top of the OpenMls's [MlsGroup], that provides Core Crypto specific functionality
255///
256/// This type will store the state of a group. With the [MlsGroup] it holds, it provides all
257/// operations that can be done in a group, such as creating proposals and commits.
258/// More information [here](https://messaginglayersecurity.rocks/mls-architecture/draft-ietf-mls-architecture.html#name-general-setting)
259#[derive(Debug)]
260#[allow(dead_code)]
261pub struct MlsConversation {
262    pub(crate) id: ConversationId,
263    pub(crate) parent_id: Option<ConversationId>,
264    pub(crate) group: MlsGroup,
265    configuration: MlsConversationConfiguration,
266}

268impl MlsConversation {
269    /// Creates a new group/conversation
270    ///
271    /// # Arguments
272    /// * `id` - group/conversation identifier
273    /// * `author_client` - the client responsible for creating the group
274    /// * `creator_credential_type` - kind of credential the creator wants to join the group with
275    /// * `config` - group configuration
276    /// * `backend` - MLS Provider that will be used to persist the group
277    ///
278    /// # Errors
279    /// Errors can happen from OpenMls or from the KeyStore
280    pub async fn create(
281        id: ConversationId,
282        author_client: &Client,
283        creator_credential_type: MlsCredentialType,
284        configuration: MlsConversationConfiguration,
285        backend: &MlsCryptoProvider,
286    ) -> Result<Self> {
287        let (cs, ct) = (configuration.ciphersuite, creator_credential_type);
288        let cb = author_client
289            .get_most_recent_or_create_credential_bundle(backend, cs.signature_algorithm(), ct)
290            .await
291            .map_err(RecursiveError::mls_client("getting or creating credential bundle"))?;
292
293        let group = MlsGroup::new_with_group_id(
294            backend,
295            &cb.signature_key,
296            &configuration.as_openmls_default_configuration()?,
297            openmls::prelude::GroupId::from_slice(id.as_slice()),
298            cb.to_mls_credential_with_key(),
299        )
300        .await
301        .map_err(MlsError::wrap("creating group with id"))?;
302
303        let mut conversation = Self {
304            id,
305            group,
306            parent_id: None,
307            configuration,
308        };
309
310        conversation
311            .persist_group_when_changed(&backend.keystore(), true)
312            .await?;
313
314        Ok(conversation)
315    }

339    /// Internal API: restore the conversation from a persistence-saved serialized Group State.
340    pub(crate) fn from_serialized_state(buf: Vec<u8>, parent_id: Option<ConversationId>) -> Result<Self> {
341        let group: MlsGroup =
342            core_crypto_keystore::deser(&buf).map_err(KeystoreError::wrap("deserializing group state"))?;
343        let id = ConversationId::from(group.group_id().as_slice());
344        let configuration = MlsConversationConfiguration {
345            ciphersuite: group.ciphersuite().into(),
346            ..Default::default()
347        };
348
349        Ok(Self {
350            id,
351            group,
352            parent_id,
353            configuration,
354        })
355    }

366    /// Returns all members credentials from the group/conversation
367    pub fn members(&self) -> HashMap<Vec<u8>, Credential> {
368        self.group.members().fold(HashMap::new(), |mut acc, kp| {
369            let credential = kp.credential;
370            let id = credential.identity().to_vec();
371            acc.entry(id).or_insert(credential);
372            acc
373        })
374    }
375
376    /// Get actual group members and subtract pending remove proposals
377    pub fn members_in_next_epoch(&self) -> Vec<ClientId> {
378        let pending_removals = self.pending_removals();
379        let existing_clients = self
380            .group
381            .members()
382            .filter_map(|kp| {
383                if !pending_removals.contains(&kp.index) {
384                    Some(kp.credential.identity().into())
385                } else {
386                    trace!(client_index:% = kp.index; "Client is pending removal");
387                    None
388                }
389            })
390            .collect::<HashSet<_>>();
391        existing_clients.into_iter().collect()
392    }
393
394    /// Gather pending remove proposals
395    fn pending_removals(&self) -> Vec<LeafNodeIndex> {
396        self.group
397            .pending_proposals()
398            .filter_map(|proposal| match proposal.proposal() {
399                Proposal::Remove(remove) => Some(remove.removed()),
400                _ => None,
401            })
402            .collect::<Vec<_>>()
403    }
404
405    /// Returns all members credentials with their signature public key from the group/conversation
406    pub fn members_with_key(&self) -> HashMap<Vec<u8>, CredentialWithKey> {
407        self.group.members().fold(HashMap::new(), |mut acc, kp| {
408            let credential = kp.credential;
409            let id = credential.identity().to_vec();
410            let signature_key = SignaturePublicKey::from(kp.signature_key);
411            let credential = CredentialWithKey {
412                credential,
413                signature_key,
414            };
415            acc.entry(id).or_insert(credential);
416            acc
417        })
418    }

420    pub(crate) async fn persist_group_when_changed(&mut self, keystore: &CryptoKeystore, force: bool) -> Result<()> {
421        if force || self.group.state_changed() == openmls::group::InnerState::Changed {
422            keystore
423                .mls_group_persist(
424                    &self.id,
425                    &core_crypto_keystore::ser(&self.group).map_err(KeystoreError::wrap("serializing group state"))?,
426                    self.parent_id.as_deref(),
427                )
428                .await
429                .map_err(KeystoreError::wrap("persisting mls group"))?;
430
431            self.group.set_state(openmls::group::InnerState::Persisted);
432        }
433
434        Ok(())
435    }

455    pub(crate) async fn find_current_credential_bundle(&self, client: &Client) -> Result<Arc<CredentialBundle>> {
456        let own_leaf = self.group.own_leaf().ok_or(LeafError::InternalMlsError)?;
457        let sc = self.ciphersuite().signature_algorithm();
458        let ct = self
459            .own_credential_type()
460            .map_err(RecursiveError::mls_conversation("getting own credential type"))?;
461
462        client
463            .find_credential_bundle_by_public_key(sc, ct, own_leaf.signature_key())
464            .await
465            .map_err(RecursiveError::mls_client("finding current credential bundle"))
466            .map_err(Into::into)
467    }

470        let sc = self.ciphersuite().signature_algorithm();
471        let ct = self
472            .own_credential_type()
473            .map_err(RecursiveError::mls_conversation("getting own credential type"))?;

475        client
476            .find_most_recent_credential_bundle(sc, ct)
477            .await
478            .map_err(RecursiveError::mls_client("finding most recent credential bundle"))
479            .map_err(Into::into)
480    }

483impl CentralContext {
484    /// Acquire a conversation guard.
485    ///
486    /// This helper struct permits mutations on a conversation.
487    pub async fn conversation(&self, id: &ConversationId) -> Result<ConversationGuard> {
488        let keystore = self
489            .mls_provider()
490            .await
491            .map_err(RecursiveError::root("getting mls provider"))?
492            .keystore();
493        let inner = self
494            .mls_groups()
495            .await
496            .map_err(RecursiveError::root("getting mls groups"))?
497            .get_fetch(id, &keystore, None)
498            .await
499            .map_err(RecursiveError::root("fetching conversation from mls groups by id"))?;
500
501        if let Some(inner) = inner {
502            return Ok(ConversationGuard::new(inner, self.clone()));
503        }
504        // Check if there is a pending conversation with
505        // the same id
506        let pending = self.pending_conversation(id).await.map(Error::PendingConversation)?;
507        Err(pending)
508    }
509
510    pub(crate) async fn pending_conversation(&self, id: &ConversationId) -> Result<PendingConversation> {
511        let keystore = self
512            .keystore()
513            .await
514            .map_err(RecursiveError::root("getting keystore"))?;
515        let Some(pending_group) = keystore
516            .find::<PersistedMlsPendingGroup>(id)
517            .await
518            .map_err(KeystoreError::wrap("finding persisted mls pending group"))?
519        else {
520            return Err(LeafError::ConversationNotFound(id.clone()).into());
521        };
522        Ok(PendingConversation::new(pending_group, self.clone()))
523    }
524}
