1//! A MLS group is a distributed object scattered across many parties. We use a Delivery Service
2//! to orchestrate those parties. So when we create a commit, a mutable operation, it has to be
3//! validated by the Delivery Service. But it might occur that another group member did the
4//! exact same thing at the same time. So if we arrive second in this race, we must "rollback" the commit
5//! we created and accept ("merge") the other one.
6//! A client would
7//! * Create a commit
8//! * Send the commit to the Delivery Service
9//! * When Delivery Service responds
10//!     * 200 OK --> use [MlsConversation::commit_accepted][super::MlsConversation::commit_accepted] to merge the commit
11//!     * 409 CONFLICT --> do nothing. [ConversationGuard::decrypt_message] will restore the proposals not committed
12//!     * 5xx --> retry
13
14use super::{ConversationGuard, Result};
15use crate::mls::conversation::Error;
16
17impl ConversationGuard {
18    /// Allows to remove a pending commit. Use this when backend rejects the commit
19    /// you just sent e.g. if permissions have changed meanwhile.
20    ///
21    /// **CAUTION**: only use this when you had an explicit response from the Delivery Service
22    /// e.g. 403. Do not use otherwise e.g. 5xx responses, timeout etc..
23    /// **DO NOT** use when Delivery Service responds 409, pending state will be renewed
24    /// in [ConversationGuard::decrypt_message]
25    ///
26    ///
27    /// # Errors
28    /// When there is no pending commit
29    pub(crate) async fn clear_pending_commit(&mut self) -> Result<()> {
30        let database = self.database().await?;
31        self.conversation_mut(async |conversation| {
32            if conversation.group.pending_commit().is_some() {
33                conversation.group.clear_pending_commit();
34                conversation.persist_group_when_changed(&database, true).await?;
35                log::info!(group_id = conversation.id(); "Cleared pending commit.");
36                Ok(())
37            } else {
38                Err(Error::PendingCommitNotFound)
39            }
40        })
41        .await
42    }
43
44    /// Clear a pending commit if it exists. Unlike [Self::clear_pending_commit],
45    /// don't throw an error if there is none.
46    pub(crate) async fn ensure_no_pending_commit(&mut self) -> Result<()> {
47        match self.clear_pending_commit().await {
48            Err(Error::PendingCommitNotFound) => Ok(()),
49            result => result,
50        }
51    }
52}

54#[cfg(test)]
55mod tests {
56    use super::*;
57    use crate::test_utils::*;
58
59    mod clear_pending_commit {
60        use super::*;
61
62        #[apply(all_cred_cipher)]
63        pub async fn should_remove_commit(case: TestContext) {
64            let [alice] = case.sessions().await;
65            Box::pin(async move {
66                let conversation = case.create_conversation([&alice]).await;
67                assert!(!conversation.has_pending_commit().await);
68
69                let conversation = conversation.update_unmerged().await.finish();
70                assert!(conversation.has_pending_commit().await);
71                conversation.guard().await.clear_pending_commit().await.unwrap();
72                assert!(!conversation.has_pending_commit().await);
73            })
74            .await
75        }
76
77        #[apply(all_cred_cipher)]
78        pub async fn should_fail_when_pending_commit_absent(case: TestContext) {
79            let [alice] = case.sessions().await;
80            Box::pin(async move {
81                let conversation = case.create_conversation([&alice]).await;
82                assert!(!conversation.has_pending_commit().await);
83                let clear = conversation.guard().await.clear_pending_commit().await;
84                assert!(matches!(clear.unwrap_err(), Error::PendingCommitNotFound))
85            })
86            .await
87        }
88
89        #[apply(all_cred_cipher)]
90        pub async fn should_clean_associated_key_material(case: TestContext) {
91            let [session] = case.sessions().await;
92            Box::pin(async move {
93                let conversation = case.create_conversation([&session]).await;
94                assert!(!conversation.has_pending_commit().await);
95
96                let init = session.transaction.count_entities().await;
97
98                let conversation = conversation.update_unmerged().await.finish();
99                assert!(conversation.has_pending_commit().await);
100
101                conversation.guard().await.clear_pending_commit().await.unwrap();
102                assert!(!conversation.has_pending_commit().await);
103
104                // This whole flow should be idempotent.
105                // Here we verify that we are indeed deleting the `EncryptionKeyPair` created
106                // for the Update commit
107                let after_clear_commit = session.transaction.count_entities().await;
108                assert_eq!(init, after_clear_commit);
109            })
110            .await
111        }
112    }
