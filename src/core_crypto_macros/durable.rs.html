6
7    let ast = match syn::parse2::<syn::ItemFn>(item.clone().into()) {
8        Ok(ast) => ast,
9        Err(e) => return compile_error(item, e),
10    };
11    if ast.sig.asyncness.is_none() {

13    }
14

20        #(#attrs)*
21        #vis async fn #name(#inputs) #ret {
22            let _result = #body;

28    };
