83fn downcast<E, T>(input: &Arc<E>) -> Option<Arc<T>>
84where
85    E: 'static + Send + Sync,
86    T: 'static + Send + Sync,
87{
88    let dynamic = input.clone() as Arc<dyn Any + Send + Sync>;
89    dynamic.downcast().ok()
90}

129    pub(crate) fn execute_save(&self, tx: &Transaction<'_>) -> CryptoKeystoreResult<()> {
130        match self {
131            Entity::ConsumerData(consumer_data) => consumer_data.set_and_replace(tx).map(|_| ()),
132            Entity::HpkePrivateKey(mls_hpke_private_key) => mls_hpke_private_key.save(tx),
133            Entity::StoredKeypackage(mls_key_package) => mls_key_package.save(tx),
134            Entity::PskBundle(mls_psk_bundle) => mls_psk_bundle.save(tx),
135            Entity::EncryptionKeyPair(mls_encryption_key_pair) => mls_encryption_key_pair.save(tx),
136            Entity::StoredEpochEncryptionKeypair(mls_epoch_encryption_key_pair) => {
137                mls_epoch_encryption_key_pair.save(tx)
138            }
139            Entity::StoredCredential(mls_credential) => mls_credential.save(tx),
140            Entity::StoredBufferedCommit(mls_pending_commit) => mls_pending_commit.save(tx),
141            Entity::PersistedMlsGroup(persisted_mls_group) => persisted_mls_group.save(tx),
142            Entity::PersistedMlsPendingGroup(persisted_mls_pending_group) => persisted_mls_pending_group.save(tx),
143            Entity::MlsPendingMessage(mls_pending_message) => mls_pending_message.save(tx),
144            Entity::StoredE2eiEnrollment(e2ei_enrollment) => e2ei_enrollment.save(tx),
145            Entity::E2eiAcmeCA(e2ei_acme_ca) => e2ei_acme_ca.set_and_replace(tx).map(|_| ()),
146            Entity::E2eiIntermediateCert(e2ei_intermediate_cert) => e2ei_intermediate_cert.save(tx),
147            #[cfg(target_os = "unknown")]
148            Entity::E2eiRefreshToken(e2ei_refresh_token) => e2ei_refresh_token.set_and_replace(tx).map(|_| ()),
149            Entity::E2eiCrl(e2ei_crl) => e2ei_crl.save(tx),
150            #[cfg(feature = "proteus-keystore")]
151            Entity::ProteusSession(record) => record.save(tx),
152            #[cfg(feature = "proteus-keystore")]
153            Entity::ProteusIdentity(record) => record.save(tx),
154            #[cfg(feature = "proteus-keystore")]
155            Entity::ProteusPrekey(record) => record.save(tx),
156        }
157    }
