10/// Indicates the state of a Conversation regarding end-to-end identity.
11///
12/// Note: this does not check pending state (pending commit, pending proposals) so it does not
13/// consider members about to be added/removed
14#[derive(Debug, Copy, Clone)]
15#[cfg_attr(target_family = "wasm", wasm_bindgen)]
16#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Enum))]
17#[repr(u8)]
18pub enum E2eiConversationState {
19    /// All clients have a valid E2EI certificate
20    Verified = 1,
21    /// Some clients are either still Basic or their certificate is expired
22    NotVerified,
23    /// All clients are still Basic. If all client have expired certificates, [E2eiConversationState::NotVerified] is returned.
24    NotEnabled,
25}
26
27impl From<core_crypto::E2eiConversationState> for E2eiConversationState {
28    fn from(value: core_crypto::E2eiConversationState) -> Self {
29        match value {
30            core_crypto::E2eiConversationState::Verified => Self::Verified,
31            core_crypto::E2eiConversationState::NotVerified => Self::NotVerified,
32            core_crypto::E2eiConversationState::NotEnabled => Self::NotEnabled,
33        }
34    }
35}
