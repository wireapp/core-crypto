1use std::{collections::HashMap, sync::Arc};

3use core_crypto::{mls::conversation::Conversation as _, transaction_context::Error as TransactionError};

10#[uniffi::export]
11impl CoreCryptoContext {
12    /// Returns the end-to-end identity verification state of the given conversation.
13    pub async fn e2ei_conversation_state(
14        &self,
15        conversation_id: &ConversationId,
16    ) -> CoreCryptoResult<E2eiConversationState> {
17        let conversation = self.inner.conversation(conversation_id.as_ref()).await?;
18        conversation
19            .e2ei_conversation_state()
20            .await
21            .map(Into::into)
22            .map_err(Into::into)
23    }
24
25    /// Returns true if end-to-end identity is enabled for the given ciphersuite.
26    pub async fn e2ei_is_enabled(&self, ciphersuite: Ciphersuite) -> CoreCryptoResult<bool> {
27        self.inner
28            .e2ei_is_enabled(ciphersuite.into())
29            .await
30            .map_err(Into::<TransactionError>::into)
31            .map_err(Into::into)
32    }
33
34    /// Returns the E2EI identity claims for the specified devices in the given conversation.
35    pub async fn get_device_identities(
36        &self,
37        conversation_id: &ConversationId,
38        device_ids: Vec<Arc<ClientId>>,
39    ) -> CoreCryptoResult<Vec<WireIdentity>> {
40        let device_ids = device_ids.iter().map(|c| c.as_ref().as_ref()).collect::<Vec<_>>();
41
42        let conversation = self.inner.conversation(conversation_id.as_ref()).await?;
43        let wire_ids = conversation.get_device_identities(device_ids.as_slice()).await?;
44        Ok(wire_ids.into_iter().map(Into::into).collect())
45    }
46
47    /// Returns the E2EI identity claims for the specified users in the given conversation, grouped by user ID.
48    pub async fn get_user_identities(
49        &self,
50        conversation_id: &ConversationId,
51        user_ids: Vec<String>,
52    ) -> CoreCryptoResult<UserIdentities> {
53        let conversation = self.inner.conversation(conversation_id.as_ref()).await?;
54        let user_ids = conversation.get_user_identities(user_ids.as_slice()).await?;
55        let user_ids = user_ids
56            .into_iter()
57            .map(|(k, v)| -> CoreCryptoResult<_> {
58                let identities = v.into_iter().map(WireIdentity::from).collect::<Vec<_>>();
59                Ok((k, identities))
60            })
61            .collect::<CoreCryptoResult<HashMap<_, _>>>()?;
62        Ok(user_ids)
63    }

67        self.inner.e2ei_is_pki_env_setup().await
68    }
