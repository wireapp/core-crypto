21    Ffi(#[from] uniffi::UnexpectedUniFFICallbackError),

24/// An `HistoryObserver` is notified whenever a new history client is created.
25#[cfg(not(target_family = "wasm"))]
26#[uniffi::export(with_foreign)]
27#[async_trait]
28pub trait HistoryObserver: Send + Sync {
29    /// This function will be called every time a new history client is created.
30    ///
31    /// The `secret` parameter is the secret associated with the new history client
32    ///
33    /// <div class="warning">
34    /// This function must not block! Foreign implementors of this interface can
35    /// spawn a task indirecting the notification, or (unblocking) send the notification
36    /// on some kind of channel, or anything else, as long as the operation completes
37    /// quickly.
38    /// </div>
39    ///
40    /// Though the signature includes an error type, that error is only present because
41    /// it is required by `uniffi` in order to handle panics. This function should suppress
42    /// and ignore internal errors instead of propagating them, to the maximum extent possible.
43    async fn history_client_created(
44        &self,
45        conversation_id: ConversationId,
46        secret: HistorySecret,
47    ) -> Result<(), NewHistoryClientReportingError>;
48}

50/// This shim bridges the public `HistoryObserver` interface with the internal one defined by `core-crypto`.
51///
52/// This is slightly unfortunate, as it introduces an extra layer of indirection before a change notice can
53/// actually reach its foreign target. However, the orphan rule prevents us from just tying the two traits
54/// together directly, so this is the straightforward way to accomplish that.
55#[cfg(not(target_family = "wasm"))]
56struct ObserverShim(Arc<dyn HistoryObserver>);
57
58#[cfg(not(target_family = "wasm"))]
59#[async_trait]
60impl core_crypto::mls::HistoryObserver for ObserverShim {
61    async fn history_client_created(
62        &self,
63        conversation_id: ConversationId,
64        secret: &core_crypto::prelude::HistorySecret,
65    ) {
66        let secret = HistorySecret::try_from(secret)
67            .expect("conversion from the HistorySecret rust type to the FFI type should work");
68
69        if let Err(err) = self.0.history_client_created(conversation_id.clone(), secret).await {
70            // we don't _care_ if an error is thrown by the notification function, per se,
71            // but this would probably be useful information for downstream debugging efforts
72            log::warn!(
73                conversation_id = Obfuscated::new(&conversation_id),
74                err = log::kv::Value::from_dyn_error(&err);
75                "caught an error when attempting to notify the history observer of a new history client"
76            );
77        }
78    }
79}

84    /// Add a history observer to this client.
85    ///
86    /// This function should be called 0 or 1 times in a session's lifetime. If called

97/// An `HistoryObserver` is notified whenever a new history client is created.
98#[cfg(target_family = "wasm")]
99#[wasm_bindgen]
100#[derive(derive_more::Debug)]
101#[debug("HistoryObserver")]
102pub struct HistoryObserver {
103    this_context: JsValue,
104    history_client_created: js_sys::Function,
105}
106
107#[cfg(target_family = "wasm")]
108// SAFETY: we promise that we're only ever using this in a single-threaded context
109unsafe impl Send for HistoryObserver {}
110#[cfg(target_family = "wasm")]
111// SAFETY: we promise that we're only ever using this in a single-threaded context
112unsafe impl Sync for HistoryObserver {}
113
114#[cfg(target_family = "wasm")]
115#[wasm_bindgen]
116impl HistoryObserver {
117    /// Create a new history observer.
118    ///
119    /// This function should be hidden on the JS side of things! The JS bindings should have an `interface HistoryObserver`
120    /// which has the method defined, and the bindings themselves should destructure an instance implementing that
121    /// interface appropriately to construct this.
122    ///
123    /// - `this_context` is the instance itself, which will be bound to `this` within the function bodies
124    /// - `history_client_created`: A function of the form `(conversation_id: Uint8Array, secret: HistorySecret) -> Promise<void>`.
125    ///
126    ///   Called every time a history client is created.
127    #[wasm_bindgen(constructor)]
128    pub fn new(this_context: JsValue, history_client_created: js_sys::Function) -> CoreCryptoResult<Self> {
129        // we can't do much type-checking here unfortunately, but we can at least validate that the incoming functions have the right length
130        if history_client_created.length() != 2 {
131            return Err(CoreCryptoError::ad_hoc(format!(
132                "`history_client_created` must accept 2 arguments but accepts {}",
133                history_client_created.length()
134            )));
135        }
136        Ok(Self {
137            this_context,
138            history_client_created,
139        })
140    }
141}

143#[cfg(target_family = "wasm")]
144impl HistoryObserver {
145    /// Call the JS `history_client_created` function
146    ///
147    /// This blocks if the JS side of things blocks.
148    ///
149    /// This is extracted as its own function instead of being implemented inline within the
150    /// `impl HistoryObserver for HistoryObserver` block mostly to consolidate error-handling.
151    async fn history_client_created(&self, conversation_id: ConversationId, secret: JsValue) -> Result<(), JsValue> {
152        let conversation_id = Uint8Array::from(conversation_id.as_slice());
153
154        let promise = self
155            .history_client_created
156            .call2(&self.this_context, &conversation_id.into(), &secret)?
157            .dyn_into::<Promise>()?;
158        // we don't actually care what the result of executing the notification promise is; we'll ignore it if it exists
159        JsFuture::from(promise).await?;
160        Ok(())
161    }
162}

208#[cfg(target_family = "wasm")]
209impl kv::ToValue for LoggableJsValue {
210    fn to_value(&self) -> kv::Value<'_> {
211        // can't get a borrowed str from `JsValue`, so can't directly
212        // convert into a string; oh well; fallback should catch it
213        if let Some(f) = self.0.as_f64() {
214            return f.into();
215        }
216        if let Some(b) = self.0.as_bool() {
217            return b.into();
218        }
219        if self.0.is_null() || self.0.is_undefined() {
220            return kv::Value::null();
221        }
222        kv::Value::from_debug(&self.0)
223    }
