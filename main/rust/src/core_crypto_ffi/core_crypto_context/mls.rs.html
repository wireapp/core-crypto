141    /// See [core_crypto::transaction_context::TransactionContext::get_or_create_client_keypackages]
142    pub async fn client_keypackages(
143        &self,
144        ciphersuite: Ciphersuite,
145        credential_type: CredentialType,
146        amount_requested: u32,
147    ) -> CoreCryptoResult<Vec<KeyPackageMaybeArc>> {
148        let kps = self
149            .inner
150            .get_or_create_client_keypackages(ciphersuite.into(), credential_type.into(), amount_requested as usize)
151            .await
152            .map_err(RecursiveError::transaction("getting or creating client keypackages"))?;

159                    .map_err(RecursiveError::mls_conversation("serializing keypackage"))
160                    .map_err(Into::into)
161            })
162            .collect::<CoreCryptoResult<_>>()
163    }

201            .await?;
202

204            .new_conversation(conversation_id.as_ref(), creator_credential_type.into(), lower_cfg)
205            .await?;
206        Ok(())

223    /// See [core_crypto::mls::conversation::ConversationGuard::add_members]
224    pub async fn add_clients_to_conversation(
225        &self,
226        conversation_id: &ConversationId,
227        key_packages: Vec<KeyPackageMaybeArc>,
228    ) -> CoreCryptoResult<NewCrlDistributionPoints> {
229        let key_packages = key_packages
230            .into_iter()
231            .map(|kp| {
232                KeyPackageIn::tls_deserialize(&mut kp.as_slice())
233                    .map_err(core_crypto::mls::conversation::Error::tls_deserialize("keypackage"))
234                    .map_err(RecursiveError::mls_conversation("adding members to conversation"))
235                    .map_err(Into::into)
236            })
237            .collect::<CoreCryptoResult<Vec<_>>>()?;
238
239        let mut conversation = self.inner.conversation(conversation_id.as_ref()).await?;
240        let distribution_points = conversation.add_members(key_packages).await?.into();
241        Ok(distribution_points)
242    }

286    /// See [core_crypto::mls::conversation::ConversationGuard::decrypt_message]
287    pub async fn decrypt_message(
288        &self,
289        conversation_id: &ConversationId,
290        payload: Vec<u8>,
291    ) -> CoreCryptoResult<DecryptedMessage> {
292        let conversation_result = self.inner.conversation(conversation_id.as_ref()).await;
293        let decrypted_message = match conversation_result {
294            Err(TransactionError::PendingConversation(mut pending)) => {
295                pending.try_process_own_join_commit(&payload).await?
296            }
297            Ok(mut conversation) => conversation.decrypt_message(&payload).await?,
298            Err(e) => Err(e)?,
299        };
300
301        decrypted_message.try_into()
302    }

380    /// Get all credentials from this client which match the provided parameters.
381    ///
382    /// Parameters which are unset or `None` match anything. Those with a particular value find only credentials
383    /// matching that value.
384    pub async fn find_credentials(
385        &self,
386        client_id: Option<ClientIdMaybeArc>,
387        public_key: Option<Vec<u8>>,
388        ciphersuite: Option<Ciphersuite>,
389        credential_type: Option<CredentialType>,
390        earliest_validity: Option<u64>,
391    ) -> CoreCryptoResult<Vec<CredentialRefMaybeArc>> {
392        let client_id = client_id.as_ref().map(|client_id| client_id.as_cc());
393        let client_id = client_id.as_ref().map(|client_id| client_id.as_ref());
394
395        let signature_scheme = ciphersuite
396            .map(CryptoCiphersuite::from)
397            .map(|ciphersuite| ciphersuite.signature_algorithm());
398
399        let credential_type = credential_type.map(core_crypto::CredentialType::from);
400
401        let find_filters = CredentialFindFilters {
402            client_id,
403            public_key: public_key.as_deref(),
404            signature_scheme,
405            credential_type,
406            earliest_validity,
407        };
408
409        self.inner
410            .find_credentials(find_filters)
411            .await
412            .map(|credentials| {
413                credentials
414                    .into_iter()
415                    .map(CredentialRef::from)
416                    .map(CredentialRef::into_maybe_arc)
417                    .collect()
418            })
419            .map_err(Into::into)
420    }
