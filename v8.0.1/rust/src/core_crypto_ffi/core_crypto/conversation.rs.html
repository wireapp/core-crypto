9bytes_wrapper!(
10    /// A unique identifier for a single conversation.
11    ///
12    /// The backend provides an opaque string identifying a new conversation.
13    /// Construct an instance of this newtype to pass that identifier to Rust.
14    #[derive(Debug, PartialEq, Eq, Hash, Clone)]
15    #[cfg_attr(target_family = "wasm", derive(serde::Serialize, serde::Deserialize))]
16    #[cfg_attr(not(target_family = "wasm"), uniffi::export(Debug, Eq, Hash))]
17    ConversationId
18);

20#[cfg_attr(not(target_family = "wasm"), uniffi::export)]
21#[cfg_attr(target_family = "wasm", wasm_bindgen)]
22impl CoreCrypto {
23    /// See [core_crypto::mls::conversation::Conversation::epoch]
24    pub async fn conversation_epoch(&self, conversation_id: &ConversationId) -> CoreCryptoResult<u64> {
25        let conversation = self
26            .inner
27            .get_raw_conversation(conversation_id)
28            .await
29            .map_err(RecursiveError::mls_client("getting raw conversation by id"))?;
30        Ok(conversation.epoch().await)
31    }
32
33    /// See [core_crypto::mls::conversation::Conversation::ciphersuite]
34    pub async fn conversation_ciphersuite(&self, conversation_id: &ConversationId) -> CoreCryptoResult<Ciphersuite> {
35        let cs = self
36            .inner
37            .get_raw_conversation(conversation_id)
38            .await
39            .map_err(RecursiveError::mls_client("getting raw conversation by id"))?
40            .ciphersuite()
41            .await;
42        Ok(Ciphersuite::from(core_crypto::prelude::CiphersuiteName::from(cs)))
43    }
44
45    /// See [core_crypto::prelude::Session::conversation_exists]
46    pub async fn conversation_exists(&self, conversation_id: &ConversationId) -> CoreCryptoResult<bool> {
47        self.inner
48            .conversation_exists(conversation_id)
49            .await
50            .map_err(RecursiveError::mls_client("getting conversation existence by id"))
51            .map_err(Into::into)
52    }
53
54    /// See [core_crypto::mls::conversation::Conversation::get_client_ids]
55    pub async fn get_client_ids(&self, conversation_id: &ConversationId) -> CoreCryptoResult<Vec<ClientIdMaybeArc>> {
56        let conversation = self
57            .inner
58            .get_raw_conversation(conversation_id)
59            .await
60            .map_err(RecursiveError::mls_client("getting raw conversation"))?;
61        Ok(conversation
62            .get_client_ids()

64            .into_iter()

69    /// See [core_crypto::mls::conversation::Conversation::get_external_sender]
70    pub async fn get_external_sender(&self, conversation_id: &ConversationId) -> CoreCryptoResult<Vec<u8>> {
71        let conversation = self
72            .inner
73            .get_raw_conversation(conversation_id)
74            .await
75            .map_err(RecursiveError::mls_client("getting raw conversation"))?;
