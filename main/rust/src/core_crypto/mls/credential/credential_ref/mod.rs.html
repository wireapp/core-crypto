14/// A reference to a credential which has been stored in a session.
15///
16/// Credentials can be quite large; we'd really like to avoid passing them
17/// back and forth across the FFI boundary more than is strictly required.
18/// Therefore, we use this type which is substantially more compact.
19///
20/// Created with [`TransactionContext::add_credential`][crate::transaction_context::TransactionContext::add_credential].
21///
22/// This reference is _not_ a literal reference in memory.
23/// It is instead a key with which a credential can be retrieved.
24/// This means that it is stable over time and across the FFI boundary.
25#[derive(
26    core_crypto_macros::Debug, Clone, derive_more::From, derive_more::Into, serde::Serialize, serde::Deserialize,
27)]
28pub struct CredentialRef {
29    client_id: ClientId,
30    public_key_hash: Sha256Hash,
31    r#type: CredentialType,
32    ciphersuite: Ciphersuite,
33    // first unix timestamp at which the credential is valid
34    earliest_validity: u64,
35}
36
37impl CredentialRef {
38    /// Construct an instance from a credential.
39    // not an actual `From` impl in order to keep it crate-private
40    pub(crate) fn from_credential(credential: &super::Credential) -> Self {
41        Self {
42            client_id: credential.client_id().to_owned(),
43            public_key_hash: Sha256Hash::hash_from(credential.signature_key_pair.public()),
44            r#type: credential.credential_type(),
45            ciphersuite: credential.ciphersuite(),
46            earliest_validity: credential.earliest_validity,
47        }
48    }
49
50    /// Get the client ID associated with this credential
51    pub fn client_id(&self) -> &ClientIdRef {
52        self.client_id.as_ref()
53    }
54
55    /// Get the SHA256 hash of the public key associated with this credential
56    pub fn public_key_hash(&self) -> Sha256Hash {
57        self.public_key_hash
58    }
59
60    /// Get the credential type associated with this credential
61    pub fn r#type(&self) -> CredentialType {
62        self.r#type
63    }
64
65    /// Get the signature scheme associated with this credential
66    pub fn signature_scheme(&self) -> SignatureScheme {
67        self.ciphersuite.signature_algorithm()
68    }
69
70    /// Get the ciphersuite associated with this credential
71    pub fn ciphersuite(&self) -> Ciphersuite {
72        self.ciphersuite
73    }
74
75    /// Get the unix timestamp of the earliest validity of this credential.
76    pub fn earliest_validity(&self) -> u64 {
77        self.earliest_validity
78    }
79}
