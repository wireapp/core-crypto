23impl TransactionContext {
24    /// Creates an enrollment instance with private key material you can use in order to fetch
25    /// a new x509 certificate from the acme server.
26    ///
27    /// # Parameters
28    /// * `client_id` - client identifier e.g. `b7ac11a4-8f01-4527-af88-1c30885a7931:6add501bacd1d90e@example.com`
29    /// * `display_name` - human readable name displayed in the application e.g. `Smith, Alice M (QA)`
30    /// * `handle` - user handle e.g. `alice.smith.qa@example.com`
31    /// * `expiry_sec` - generated x509 certificate expiry in seconds
32    pub async fn e2ei_new_enrollment(
33        &self,
34        client_id: ClientId,
35        display_name: String,
36        handle: String,
37        team: Option<String>,
38        expiry_sec: u32,
39        ciphersuite: MlsCiphersuite,
40    ) -> Result<E2eiEnrollment> {
41        let signature_keypair = None; // fresh install without a Basic client. Supplying None will generate a new keypair
42        E2eiEnrollment::try_new(
43            client_id,
44            display_name,
45            handle,
46            team,
47            expiry_sec,
48            &self
49                .mls_provider()
50                .await
51                .map_err(RecursiveError::transaction("getting mls provider"))?,
52            ciphersuite,
53            signature_keypair,
54            false, // fresh install so no refresh token registered yet
55        )
56        .map_err(RecursiveError::e2e_identity("creating new enrollment"))
57        .map_err(Into::into)
58    }

60    /// Parses the ACME server response from the endpoint fetching x509 certificates and uses it
61    /// to initialize the MLS client with a certificate
62    pub async fn e2ei_mls_init_only(
63        &self,
64        enrollment: &mut E2eiEnrollment,
65        certificate_chain: String,
66    ) -> Result<NewCrlDistributionPoints> {
67        let sk = enrollment
68            .get_sign_key_for_mls()
69            .map_err(RecursiveError::e2e_identity("creating new enrollment"))?;
70        let cs = *enrollment.ciphersuite();
71        let certificate_chain = enrollment
72            .certificate_response(
73                certificate_chain,
74                self.mls_provider()
75                    .await
76                    .map_err(RecursiveError::transaction("getting mls provider"))?
77                    .authentication_service()
78                    .borrow()
79                    .await
80                    .as_ref()
81                    .ok_or(Error::PkiEnvironmentUnset)?,
82            )
83            .await
84            .map_err(RecursiveError::e2e_identity("getting certificate response"))?;
85
86        let crl_new_distribution_points = self.extract_dp_on_init(&certificate_chain[..]).await?;
87
88        let private_key = CertificatePrivateKey {
89            value: sk,
90            signature_scheme: cs.signature_algorithm(),
91        };
92
93        let cert_bundle = CertificateBundle {
94            certificate_chain,
95            private_key,
96        };
97        let identifier = ClientIdentifier::X509(HashMap::from([(cs.signature_algorithm(), cert_bundle)]));
98        self.mls_init(identifier, &[cs])
99            .await
100            .map_err(RecursiveError::transaction("initializing mls"))?;
101        Ok(crl_new_distribution_points)
102    }

104    /// When x509 new credentials are registered this extracts the new CRL Distribution Point from the end entity certificate
105    /// and all the intermediates
106    async fn extract_dp_on_init(&self, certificate_chain: &[Vec<u8>]) -> Result<NewCrlDistributionPoints> {
107        use x509_cert::der::Decode as _;
108
109        // Own intermediates are not provided by smallstep in the /federation endpoint so we got to intercept them here, at issuance
110        let size = certificate_chain.len();
111        let mut crl_new_distribution_points = HashSet::new();
112        if size > 1 {
113            for int in certificate_chain.iter().skip(1).rev() {
114                let mut crl_dp = self
115                    .e2ei_register_intermediate_ca_der(int)
116                    .await
117                    .map_err(RecursiveError::transaction("registering intermediate ca der"))?;
118                if let Some(crl_dp) = crl_dp.take() {
119                    crl_new_distribution_points.extend(crl_dp);
120                }
121            }
122        }
123
124        let ee = certificate_chain.first().ok_or(Error::InvalidCertificateChain)?;
125        let ee = x509_cert::Certificate::from_der(ee)
126            .map_err(crate::mls::credential::Error::DecodeX509)
127            .map_err(RecursiveError::mls_credential("decoding x509 credential"))?;
128        let mut ee_crl_dp = extract_crl_uris(&ee).map_err(RecursiveError::e2e_identity("extracting crl urls"))?;
129        if let Some(crl_dp) = ee_crl_dp.take() {
130            crl_new_distribution_points.extend(crl_dp);
131        }
132
133        get_new_crl_distribution_points(
134            &self
135                .mls_provider()
136                .await
137                .map_err(RecursiveError::transaction("getting mls provider"))?,
138            crl_new_distribution_points,
139        )
140        .await
141        .map_err(RecursiveError::mls_credential("getting new crl distribution points"))
142        .map_err(Into::into)
143    }

146#[cfg(test)]
147mod tests {
148    use crate::{
149        e2e_identity::enrollment::test_utils as e2ei_utils,
150        mls::conversation::Conversation as _,
151        test_utils::{x509::X509TestChain, *},

155    #[apply(all_cred_cipher)]
156    async fn e2e_identity_should_work(case: TestContext) {
157        use e2ei_utils::E2EI_CLIENT_ID_URI;
158
159        let session = SessionContext::new_uninitialized(&case).await;
160        Box::pin(async move {
161            let x509_test_chain = X509TestChain::init_empty(case.signature_scheme());
162
163            let is_renewal = false;
164
165            let (mut enrollment, cert) = e2ei_utils::e2ei_enrollment(
166                &session,
167                &case,
168                &x509_test_chain,
169                Some(E2EI_CLIENT_ID_URI),
170                is_renewal,
171                e2ei_utils::init_enrollment,
172                e2ei_utils::noop_restore,
173            )
174            .await
175            .unwrap();
176
177            session
178                .transaction
179                .e2ei_mls_init_only(&mut enrollment, cert)
180                .await
181                .unwrap();
182
183            // verify the created client can create a conversation
184            let conversation = case
185                .create_conversation_with_credential_type(MlsCredentialType::X509, [&session])
186                .await;
187            conversation
188                .guard()
189                .await
190                .encrypt_message("Hello e2e identity !")
191                .await
192                .unwrap();
193            assert_eq!(
194                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
195                E2eiConversationState::Verified
196            );
197            assert!(
198                session
199                    .transaction
200                    .e2ei_is_enabled(case.signature_scheme())
201                    .await
202                    .unwrap()
203            );
204        })
205        .await
206    }
