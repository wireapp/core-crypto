13impl TransactionContext {
14    /// Create a conversation from an existing MLS group.
15    ///
16    /// For example, by external commit.
17    ///
18    /// This effectively goes through the following steps:
19    ///
20    /// 1. Produce a [`Conversation`] from the group.
21    /// 2. Persist that conversation in the database.
22    /// 3. Persist that conversation in the conversation cache.
23    /// 4. Return the cached entry for that conversation.
24    ///
25    /// Note that this does not check whether or not the conversation already exists.
26    pub(crate) async fn persist_conversation_from_mls_group(
27        &self,
28        mut group: MlsGroup,
29        configuration: ConversationConfiguration,
30    ) -> Result<ConversationMut> {
31        let id = ConversationId::from(group.group_id().as_slice());
32        let session = self.session().await.map_err(RecursiveError::transaction(
33            "getting session from tx context to persist",
34        ))?;
35
36        // we're actually out of order from the docs, beause this leads to a better data flow
37        let database = self.database().await?;
38        let group_state = core_crypto_keystore::ser(&group).map_err(KeystoreError::wrap("serializing group state"))?;
39        database
40            .mls_group_persist(&id, &group_state, None)
41            .await
42            .map_err(KeystoreError::wrap("persisting mls group"))?;
43        group.set_state(InnerState::Persisted);
44
45        // now that we're persisted, construct a conversation
46        let conversation = Conversation::new(id, group.into(), configuration, session);
47        let mut group_store = self.mls_groups().await?;
48
49        let inner = group_store.insert(conversation);
50        Ok(ConversationMut::new(inner, self.clone()))
51    }
52
53    /// Create a MLS conversation from an MLS Welcome message
54    ///
55    /// Unlike [`Self::persist_conversation_from_mls_group`], this _does_ check whether the conversation
56    /// already exists or is pending. If it does, returns [`LeafError::ConversationAlreadyExists`].
57    pub(crate) async fn persist_conversation_from_welcome_message(
58        &self,
59        welcome: Welcome,
60        configuration: ConversationConfiguration,
61    ) -> Result<ConversationMut> {
62        let mls_group_config =
63            configuration
64                .as_openmls_default_configuration()
65                .map_err(RecursiveError::mls_conversation(
66                    "converting configuration to openmls default",
67                ))?;
68
69        let crypto_provider = self.crypto_provider().await?;
70
71        let group = MlsGroup::new_from_welcome(&crypto_provider, &mls_group_config, welcome, None)
72            .await
73            .map_err(|err| {
74                use openmls::prelude::WelcomeError;
75                if matches!(
76                    err,
77                    WelcomeError::NoMatchingKeyPackage | WelcomeError::NoMatchingEncryptionKey
78                ) {
79                    ConversationError::OrphanWelcome
80                } else {
81                    OpenMlsError::wrap("group could not be created from welcome")(err).into()
82                }
83            })
84            .map_err(RecursiveError::mls_conversation(
85                "creating mls group from welcome message",
86            ))?;
87
88        let id = ConversationId::from(group.group_id().as_slice());
89
90        if self.conversation_exists(&id).await? || self.pending_conversation_exists(&id).await? {
91            return Err(LeafError::ConversationAlreadyExists(id).into());
92        }
93
94        self.persist_conversation_from_mls_group(group, configuration).await
95    }
