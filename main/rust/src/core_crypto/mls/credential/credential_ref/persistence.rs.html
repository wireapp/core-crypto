1//! Persistence for [`CredentialRef`], i.e. loading actual credentials from the keystore given a ref.
2//!
3//! It is not logically required that these methods are crate-private, but they aren't likely to be
4//! useful to end users. Clients building on the CC API can't do anything useful with a full [`Credential`],
5//! and it's wasteful to transfer one across the FFI boundary.
6
7use core_crypto_keystore::{Sha256Hash, entities::StoredCredential, traits::FetchFromDatabase as _};
8use mls_crypto_provider::Database;
9
10use super::{Error, Result};
11use crate::{Credential, CredentialRef, KeystoreError, RecursiveError};
12
13impl CredentialRef {
14    /// Helper to prefetch relevant keypairs when loading multiple credentials at a time.
15    ///
16    /// Only useful when preparing to call [`Self::load_from_cache`] multiple times.
17    /// For loading a single credential, prefer [`Self::load`].
18    pub(crate) async fn load_stored_credentials(database: &Database) -> Result<Vec<StoredCredential>> {
19        let credentials = database
20            .load_all::<StoredCredential>()
21            .await
22            .map_err(KeystoreError::wrap("finding all mls credentials"))?;
23        Ok(credentials)
24    }
25
26    /// Load the credential which matches this ref from the database.
27    ///
28    /// Note that this does not attach the credential to any Session; it just does the data manipulation.
29    pub(crate) async fn load(&self, database: &Database) -> Result<Credential> {
30        database
31            .get::<StoredCredential>(&Sha256Hash::hash_from(self.public_key()))
32            .await
33            .map_err(KeystoreError::wrap("finding credential"))?
34            .ok_or(Error::CredentialNotFound)
35            .and_then(|stored_credential| {
36                Credential::try_from(&stored_credential)
37                    .map_err(RecursiveError::mls_credential(
38                        "creating credential from stored credential",
39                    ))
40                    .map_err(Into::into)
41            })
42    }
43
44    /// Filter and map the provided stored credentials (_cache_). This results in [Credential]s which match
45    /// this [CredentialRef]'s conditions.
46    ///
47    /// If you are only loading a single credential ref, it may be simpler to call [`Self::load`].
48    pub(crate) fn load_from_cache(&self, cache: &[StoredCredential]) -> Result<Option<Credential>> {
49        cache
50            .iter()
51            .filter(|stored_credential| stored_credential.public_key == self.public_key)
52            .map(|stored_credential| {
53                Credential::try_from(stored_credential)
54                    .map_err(RecursiveError::mls_credential(
55                        "creating credential from stored credential",
56                    ))
57                    .map_err(Into::into)
58            })
59            .map(|credential_result| {
60                credential_result.and_then(|credential| {
61                    Credential::validate_mls_credential(
62                        &credential.mls_credential,
63                        self.client_id(),
64                        &credential.signature_key_pair,
65                    )?;
66                    Ok(credential)
67                })
68            })
69            .next()
70            .transpose()
71    }
72}
