27pub struct TransactionWrapper<'conn> {
28    transaction: Transaction<'conn>,
29}
30
31impl TransactionWrapper<'_> {
32    // this is async just to conform with the wasm impl
33    pub(crate) async fn commit_tx(self) -> CryptoKeystoreResult<()> {
34        // It's really not ideal to do potentially-heavy IO such as committing a transaction
35        // within the async context, because Rust async depends on inserting cooperative yields
36        // in appropriate places, and blocking functions simply do not have those. This is going
37        // to bind up the whole async executor every time we try to commit the transaction, for
38        // the entire duration of the execution of the transaction.
39        //
40        // We can't even do `unblock(|| transaction.commit())` here becase `Transaction: !Send`.
41        //
42        // Hopefully either WPB-14326, WPB-14327, or WPB-15766 will open a path to a unified
43        // async database which can give us better performance characteristics than this.
44        self.transaction.commit().map_err(Into::into)
45    }
46}
47
48impl<'conn> From<Transaction<'conn>> for TransactionWrapper<'conn> {
49    fn from(transaction: Transaction<'conn>) -> Self {
50        TransactionWrapper { transaction }
51    }
52}

54impl<'conn> Deref for TransactionWrapper<'conn> {
55    type Target = Transaction<'conn>;
56
57    fn deref(&self) -> &Self::Target {
58        &self.transaction
59    }
60}

73pub struct SqlCipherConnection {

78// SAFETY: An `Arc` is unnecessary as `SqlCipherConnection: !Clone`, and there is a `Mutex`
79// internally which ensures unique access.
80unsafe impl Send for SqlCipherConnection {}

85impl SqlCipherConnection {
86    fn init_with_key(path: &str, key: &str) -> CryptoKeystoreResult<Self> {
87        #[allow(unused_mut)]
88        let mut conn = rusqlite::Connection::open(path)?;
89        cfg_if::cfg_if! {
90            if #[cfg(feature = "log-queries")] {
91                fn log_query(q: &str) {
92                    log::info!("{}", q);
93                }
94
95                conn.trace(Some(log_query));
96            }
97        }
98
99        conn.pragma_update(None, "key", key)?;
100
101        // ? iOS WAL journaling fix; see details here: https://github.com/sqlcipher/sqlcipher/issues/255
102        #[cfg(feature = "ios-wal-compat")]
103        Self::handle_ios_wal_compat(&conn, path)?;
104
105        // Enable WAL journaling mode
106        conn.pragma_update(None, "journal_mode", "wal")?;
107
108        // Disable FOREIGN KEYs - The 2 step blob writing process invalidates foreign key checks unfortunately
109        conn.pragma_update(None, "foreign_keys", "OFF")?;
110
111        let conn = Mutex::new(conn);
112
113        let mut conn = Self {
114            path: path.into(),
115            conn,
116        };
117        conn.run_migrations()?;
118
119        Ok(conn)
120    }
121
122    fn init_with_key_in_memory(_path: &str, key: &str) -> CryptoKeystoreResult<Self> {
123        #[allow(unused_mut)]
124        let mut conn = rusqlite::Connection::open("")?;
125        cfg_if::cfg_if! {
126            if #[cfg(feature = "log-queries")] {
127                fn log_query(q: &str) {
128                    log::info!("{}", q);
129                }
130
131                conn.trace(Some(log_query));
132            }
133        }
134        conn.pragma_update(None, "key", key)?;
135
136        // Disable FOREIGN KEYs - The 2 step blob writing process invalidates foreign key checks unfortunately
137        conn.pragma_update(None, "foreign_keys", "OFF")?;
138
139        let conn = Mutex::new(conn);
140
141        let mut conn = Self { path: "".into(), conn };
142
143        // Need to run migrations also in memory to make sure expected tables exist.
144        conn.run_migrations()?;
145
146        Ok(conn)
147    }

149    pub async fn conn(&self) -> MutexGuard<rusqlite::Connection> {
150        self.conn.lock().await
151    }
152
153    pub fn conn_blocking(&self) -> MutexGuard<rusqlite::Connection> {
154        self.conn.lock_blocking()
155    }

157    pub async fn wipe(self) -> CryptoKeystoreResult<()> {
158        if self.path.is_empty() {
159            return Ok(());
160        }
161
162        let path = self.path.clone();
163
164        unblock(|| self.close()).await?;
165        async_fs::remove_file(&path).await?;
166        Ok(())
167    }
168
169    fn close(self) -> CryptoKeystoreResult<()> {
170        let conn = self.conn.into_inner();
171        conn.close().map_err(|(_, e)| e.into())
172    }

174    /// To prevent iOS from killing backgrounded apps using a WAL-journaled file,
175    /// we need to leave the first 32 bytes as plaintext, this way, iOS can see the
176    /// `SQLite Format 3\0` magic bytes and identify the file as a SQLite database
177    /// and when it does so, it treats this file "specially" and avoids killing the app
178    /// when doing background work
179    /// See more: https://github.com/sqlcipher/sqlcipher/issues/255
180    #[cfg(feature = "ios-wal-compat")]
181    fn handle_ios_wal_compat(conn: &rusqlite::Connection, path: &str) -> CryptoKeystoreResult<()> {
182        const ERR_SEC_ITEM_NOT_FOUND: i32 = -25300;
183        const LEGACY_ACCT_NAME: &str = "keystore_salt";
184        const WIRE_SERVICE_NAME: &str = "wire.com";
185        use security_framework::passwords as ios_keychain;
186        use sha2::Digest as _;
187
188        #[allow(non_upper_case_globals)]
189        // This is to make sure that macOS/iOS keychain items that we create (see above for the *why*)
190        // are accessible in the background through a `kSecAttrAccessibleAfterFirstUnlock` attribute
191        // More on the topic: https://developer.apple.com/documentation/security/keychain_services/keychain_items/restricting_keychain_item_accessibility
192        // More here on the specific attribute: https://developer.apple.com/documentation/security/ksecattraccessibleafterfirstunlock?language=swift
193        fn mark_password_as_accessible(key: &str) -> security_framework::base::Result<()> {
194            use core_foundation::{
195                base::TCFType,
196                dictionary::CFDictionary,
197                string::{CFString, CFStringRef},
198            };
199            use security_framework::base::Error;
200            use security_framework_sys::{
201                base::errSecSuccess,
202                item::{kSecAttrAccount, kSecAttrService, kSecClass, kSecClassGenericPassword},
203            };
204
205            // Import raw symbols from CoreFoundation
206            //
207            // SAFETY: we promise that these symbols will appear when we link this
208            unsafe extern "C" {
209                pub static kSecAttrAccessibleAfterFirstUnlock: CFStringRef;
210                pub static kSecAttrAccessible: CFStringRef;
211            }
212
213            macro_rules! wrap_under_get_rule {
214                ($external_ref:expr) => {
215                    // SAFETY: The method `CFString::wrap_under_get_rule` is required by rustc to be marked as unsafe
216                    // because accessing any static item via FFI is intrinsically unsafe. For this reason, we can't
217                    // just create a safe wrapper function here; rustc immediately flags it with an [E0133] because it
218                    // needs that FFI static item to be its parameter.
219                    //
220                    // There isn't safety documentation anywhere in the CFString or TCFType documentation, and
221                    // the implementation does only [basic checks], so we can assume that the main property we
222                    // need to uphold is "the reference is not null".
223                    //
224                    // [E0133]: https://doc.rust-lang.org/stable/error_codes/E0133.html
225                    // [basic checks]: https://github.com/servo/core-foundation-rs/blob/core-foundation-v0.10.0/core-foundation/src/lib.rs#L91-L95
226                    unsafe { CFString::wrap_under_get_rule($external_ref) }
227                };
228            }
229
230            // Create a query that matches a:
231            let query_params = CFDictionary::from_CFType_pairs(&[
232                // Class GenericPassword
233                (
234                    wrap_under_get_rule!(kSecClass),
235                    wrap_under_get_rule!(kSecClassGenericPassword).as_CFType(),
236                ),
237                // with Service = "wire.com"
238                (
239                    wrap_under_get_rule!(kSecAttrService),
240                    CFString::from(WIRE_SERVICE_NAME).as_CFType(),
241                ),
242                // Holding account name = `key` (in the following form: `keystore_salt_[sha256(file_path)]`)
243                (wrap_under_get_rule!(kSecAttrAccount), CFString::from(key).as_CFType()),
244            ]);
245
246            // And now we ask to update the following properties:
247            let payload_params = CFDictionary::from_CFType_pairs(&[(
248                // Keychain Accessibility setting
249                // See: https://developer.apple.com/documentation/security/ksecattraccessible
250                wrap_under_get_rule!(kSecAttrAccessible),
251                // Set to AccessibleAfterFirstUnlock (i.e. is accessible after the first post-boot unlock)
252                wrap_under_get_rule!(kSecAttrAccessibleAfterFirstUnlock).as_CFType(),
253            )]);
254
255            // Update the item in the keychain
256            //
257            // SAFETY: As before, the main source of unsafety here appears to simply be that this is an FFI function
258            // and nobody has constructed a safe wrapper aroud it. And sure, we can't trust the safety properties that
259            // external code has. But on the other hand, we can't not call this function. So the unsafe block should be fine.
260            match unsafe {
261                security_framework_sys::keychain_item::SecItemUpdate(
262                    query_params.as_concrete_TypeRef(),
263                    payload_params.as_concrete_TypeRef(),
264                )
265            } {
266                errSecSuccess => Ok(()),
267                err => Err(Error::from_code(err)),
268            }
269        }
270
271        let mut path_hash = sha2::Sha256::default();
272        path_hash.update(path.as_bytes());
273        let keychain_key = format!("{LEGACY_ACCT_NAME}_{}", hex::encode(&path_hash.finalize()));
274
275        // Old version compat fix
276        if let Ok(salt) = ios_keychain::get_generic_password(WIRE_SERVICE_NAME, LEGACY_ACCT_NAME) {
277            ios_keychain::set_generic_password(WIRE_SERVICE_NAME, &keychain_key, salt.as_slice())?;
278            ios_keychain::delete_generic_password(WIRE_SERVICE_NAME, LEGACY_ACCT_NAME)?;
279        }
280
281        match ios_keychain::get_generic_password(WIRE_SERVICE_NAME, &keychain_key) {
282            Ok(salt) => {
283                conn.pragma_update(None, "cipher_salt", format!("x'{}'", hex::encode(salt)))?;
284            }
285            Err(e) if e.code() == ERR_SEC_ITEM_NOT_FOUND => {
286                let salt = conn.pragma_query_value(None, "cipher_salt", |r| r.get::<_, String>(0))?;
287                let mut bytes = [0u8; 16];
288                hex::decode_to_slice(salt, &mut bytes)
289                    .map_err(|e| crate::CryptoKeystoreError::HexSaltDecodeError(e))?;
290
291                ios_keychain::set_generic_password(WIRE_SERVICE_NAME, &keychain_key, &bytes)?;
292            }
293            Err(e) => return Err(e.into()),
294        }
295
296        // We're doing it here to make sure we retroactively mark database salts as accessible
297        mark_password_as_accessible(&keychain_key)?;
298
299        const CIPHER_PLAINTEXT_BYTES: u32 = 32;
300        conn.pragma_update(None, "cipher_plaintext_header_size", CIPHER_PLAINTEXT_BYTES)?;
301        conn.pragma_update(None, "user_version", 2u32)?;
302
303        Ok(())
304    }
305
306    fn run_migrations(&mut self) -> CryptoKeystoreResult<()> {
307        let mut conn = self.conn_blocking();
308        conn.create_scalar_function("sha256_blob", 1, FunctionFlags::SQLITE_DETERMINISTIC, |ctx| {
309            let input_blob = ctx.get::<Vec<u8>>(0)?;
310            Ok(crate::sha256(&input_blob))
311        })?;
312        let report = migrations::runner().run(&mut *conn).map_err(Box::new)?;
313        if let Some(version) = report.applied_migrations().iter().map(|m| m.version()).max() {
314            conn.pragma_update(None, "schema_version", version)?;
315        }
316
317        Ok(())
318    }
319}

323#[cfg_attr(target_family = "wasm", async_trait::async_trait(?Send))]
324#[cfg_attr(not(target_family = "wasm"), async_trait::async_trait)]
325impl<'a> DatabaseConnection<'a> for SqlCipherConnection {
326    type Connection = MutexGuard<'a, rusqlite::Connection>;
327
328    async fn open(name: &str, key: &str) -> CryptoKeystoreResult<Self> {
329        let name = name.to_string();
330        let key = key.to_string();
331        Ok(unblock(move || Self::init_with_key(&name, &key)).await?)
332    }
333
334    async fn open_in_memory(name: &str, key: &str) -> CryptoKeystoreResult<Self> {
335        let name = name.to_string();
336        let key = key.to_string();
337        Ok(unblock(move || Self::init_with_key_in_memory(&name, &key)).await?)
338    }
339
340    async fn close(self) -> CryptoKeystoreResult<()> {
341        unblock(|| self.close()).await
342    }
343
344    async fn wipe(self) -> CryptoKeystoreResult<()> {
345        self.wipe().await?;
346        Ok(())
347    }
