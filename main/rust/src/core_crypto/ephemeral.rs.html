1//! Utilities for ephemeral CoreCrypto instances.
2//!
3//! Ephemeral instances are intended to support history sharing. History sharing works like this:
4//! every history-enabled conversation has a passive "history client" as a member. This client
5//! is a member of the MLS group (and can therefore decrypt messages), but it is not actively running
6//! on any device or decrypting any messages.
7//!
8//! Approximately once daily, and whenever a member is removed from the group, a new history-sharing era
9//! begins. The client submitting the commit which instantiates the new history-sharing era is responsible
10//! for ensuring that the old history client is removed from the group, and new one is added. Additionally,
11//! one of the first application messages in the new history-sharing era contains the serialized history
12//! secret.
13//!
14//! When a new client joins the history-enabled conversation, they receive a list of history secrets
15//! and their associated history-sharing eras (identified by the epoch number at which they start).
16//! For each history-sharing era, they can instantiate an ephemeral client from the history secret,
17//! and use that client to decrypt all messages in this era.
18//!
19//! Though ephemeral clients are full instances of `CoreCrypto` and contain the same API, they should
20//! not be used to generate messages for sending. They should also not be instantiated to follow along with
21//! new messages as they are received, as that's pointless; the individual credentials suffice.
22

34/// We always instantiate history clients with this prefix in their client id, so
35/// we can use prefix testing to determine with some accuracy whether or not something is a history client.
36const HISTORY_CLIENT_ID_PREFIX: &str = "history-client";
37
38/// A `HistorySecret` encodes sufficient client state that it can be used to instantiate an
39/// ephemeral client.
40#[derive(serde::Serialize, serde::Deserialize)]
41pub struct HistorySecret {
42    /// Client id of the associated history client
43    pub client_id: ClientId,
44    pub(crate) credential_bundle: CredentialBundle,
45    pub(crate) key_package: KeyPackageSecretEncapsulation,
46}
47
48/// Create a new [`CoreCrypto`] with an **uninitialized** mls session.
49///
50/// You must initialize the session yourself before using this!
51async fn in_memory_cc_with_ciphersuite(ciphersuite: impl Into<MlsCiphersuite>) -> Result<CoreCrypto> {
52    let ciphersuites = vec![ciphersuite.into()];
53
54    let configuration = MlsClientConfiguration {
55        // we know what ciphersuite we want, at least
56        ciphersuites: ciphersuites.clone(),
57        // we have the client id from the history secret, but we don't want to use it here because
58        // that kicks off the `init`, and we want to inject our secret keys into the keystore before then
59        client_id: None,
60        // not used in in-memory client
61        store_path: String::new(),
62        // important so our keys aren't memory-snooped, but its actual value is irrelevant
63        database_key: DatabaseKey::generate(),
64        // irrelevant for this case
65        external_entropy: None,
66        // don't generate any keypackages; we do not want to ever add this client to a different group
67        nb_init_key_packages: Some(0),
68    };
69
70    // Construct the MLS session, but don't initialize it. The implementation when `client_id` is `None` just
71    // does construction, which is what we need.
72    let session = Session::try_new_in_memory(configuration)
73        .await
74        .map_err(RecursiveError::mls("creating ephemeral session"))?;
75
76    Ok(session.into())
77}

79/// Generate a new [`HistorySecret`].
80///
81/// This is useful when it's this client's turn to generate a new history client.
82///
83/// The generated secret is cryptographically unrelated to the current CoreCrypto client.
84///
85/// Note that this is a crate-private function; the public interface for this feature is
86/// [`Conversation::generate_history_secret`][core_crypto::mls::conversation::Conversation::generate_history_secret].
87/// This implementation lives here instead of there for organizational reasons.
88pub(crate) async fn generate_history_secret(ciphersuite: MlsCiphersuite) -> Result<HistorySecret> {
89    // generate a new completely arbitrary client id
90    let client_id = uuid::Uuid::new_v4();
91    let client_id = format!("{HISTORY_CLIENT_ID_PREFIX}-{client_id}");
92    let client_id = ClientId::from(client_id.into_bytes());
93    let identifier = ClientIdentifier::Basic(client_id);
94
95    let cc = in_memory_cc_with_ciphersuite(ciphersuite).await?;
96    let tx = cc
97        .new_transaction()
98        .await
99        .map_err(RecursiveError::transaction("creating new transaction"))?;
100    cc.init(identifier.clone(), &[ciphersuite], &cc.crypto_provider, 0)
101        .await
102        .map_err(RecursiveError::mls_client("initializing ephemeral cc"))?;
103
104    // we can get a credential bundle from a provider and ciphersuite
105    let mut signature_schemes = HashSet::with_capacity(1);
106    signature_schemes.insert(SignatureScheme::from(ciphersuite.0));
107    let bundles = identifier
108        .generate_credential_bundles(&cc.crypto_provider, signature_schemes)
109        .map_err(RecursiveError::mls_client("generating credential bundles"))?;
110    let [(_signature_scheme, client_id, credential_bundle)] = bundles
111        .try_into()
112        .expect("given exactly 1 signature scheme we must get exactly 1 credential bundle");
113
114    // we can generate a key package from the ephemeral cc and ciphersutite
115    let [key_package] = tx
116        .get_or_create_client_keypackages(ciphersuite, MlsCredentialType::Basic, 1)
117        .await
118        .map_err(RecursiveError::transaction("generating keypackages"))?
119        .try_into()
120        .expect("generating 1 keypackage returns 1 keypackage");
121    let key_package = KeyPackageSecretEncapsulation::load(&cc.crypto_provider, key_package)
122        .await
123        .map_err(MlsError::wrap("encapsulating key package"))?;
124
125    // we don't need to finish the transaction here--the point of the ephemeral CC was that no mutations would be saved there
126
127    Ok(HistorySecret {
128        client_id,
129        credential_bundle,
130        key_package,
131    })
132}

134impl CoreCrypto {
135    /// Instantiate a history client.
136    ///
137    /// This client exposes the full interface of `CoreCrypto`, but it should only be used to decrypt messages.
138    /// Other use is a logic error.
139    pub async fn history_client(history_secret: HistorySecret) -> Result<Self> {
140        if !history_secret
141            .client_id
142            .starts_with(HISTORY_CLIENT_ID_PREFIX.as_bytes())
143        {
144            return Err(Error::InvalidHistorySecret("client id has invalid format"));
145        }
146
147        let session = in_memory_cc_with_ciphersuite(history_secret.key_package.ciphersuite()).await?;
148        let tx = session
149            .new_transaction()
150            .await
151            .map_err(RecursiveError::transaction("creating new transaction"))?;
152
153        session
154            .restore_from_history_secret(history_secret)
155            .await
156            .map_err(RecursiveError::mls_client(
157                "restoring ephemeral session from history secret",
158            ))?;
159
160        tx.finish()
161            .await
162            .map_err(RecursiveError::transaction("finishing transaction"))?;
163
164        Ok(session)
165    }
166}

168#[cfg(test)]
169mod tests {
170    use rstest::rstest;
171    use rstest_reuse::apply;
172
173    use crate::test_utils::{SessionContext, TestContext, all_cred_cipher};
174
175    use super::*;
176
177    /// Create a history secret, and restore it into a CoreCrypto instance
178    #[apply(all_cred_cipher)]
179    async fn can_create_ephemeral_client(case: TestContext) {
180        if case.credential_type != MlsCredentialType::Basic {
181            // history client will only ever have basic credentials, so not much point in testing
182            // how it interacts with an x509-only conversation
183            return;
184        }
185
186        use crate::mls::conversation::Conversation as _;
187
188        let [alice] = case.sessions().await;
189        let conversation = case.create_conversation([&alice]).await;
190
191        let conversation_guard = conversation.guard().await;
192        let history_secret = conversation_guard.generate_history_secret().await.unwrap();
193
194        // the history secret has to survive encoding and decoding into some arbitrary serde format,
195        // so round-trip it
196        // note: we're not testing the serialization format
197        let encoded = rmp_serde::to_vec(&history_secret).unwrap();
198        let history_secret = rmp_serde::from_slice::<HistorySecret>(&encoded).unwrap();
199
200        let ephemeral_client = CoreCrypto::history_client(history_secret).await.unwrap();
201
202        // so how can we test that this has actually worked, given that we have not yet implemented the
203        // bit where we can actually enable history for a conversation, adding a history client? Well,
204        // with the caveat that
205        // WE SHOULD NOT DO THIS OUTSIDE A TESTING CONTEXT
206        // , we may as well try to
207        // roundtrip a conversation with Alice; that should at least prove that the ephemeral client
208        // has the basic minimal set of data in its keystore set properly.
209        let ephemeral_identifier = ClientIdentifier::Basic(ephemeral_client.mls.id().await.unwrap());
210        let ephemeral_session_context = SessionContext::new_with_identifier(&case, ephemeral_identifier, None)
211            .await
212            .unwrap();
213
214        let conversation = conversation.invite_notify([&ephemeral_session_context]).await;
215
216        assert!(
217            conversation
218                .is_functional_and_contains([&alice, &ephemeral_session_context])
219                .await
220        );
221    }

243        let ephemeral_session_context = SessionContext::new_with_identifier(&case, ephemeral_identifier, None)
244            .await
245            .unwrap();

248        let conversation = conversation.invite_notify([&ephemeral_session_context]).await;
249

251            conversation
