7#[derive(Debug, thiserror::Error, uniffi::Error)]
8#[uniffi(flat_error)]
9pub enum EpochChangedReportingError {
10    #[error("panic or otherwise unexpected error from foreign code")]
11    Ffi(#[from] uniffi::UnexpectedUniFFICallbackError),
12}
13
14/// An `EpochObserver` is notified whenever a conversation's epoch changes.
15#[uniffi::export(with_foreign)]
16#[cfg_attr(target_family = "wasm", async_trait(?Send))]
17#[cfg_attr(not(target_family = "wasm"), async_trait)]
18pub trait EpochObserver: Send + Sync {
19    /// This function will be called every time a conversation's epoch changes.
20    ///
21    /// The `epoch` parameter is the new epoch.
22    ///
23    /// <div class="warning">
24    /// This function must not block! Foreign implementors of this interface can
25    /// spawn a task indirecting the notification, or (unblocking) send the notification
26    /// on some kind of channel, or anything else, as long as the operation completes
27    /// quickly.
28    /// </div>
29    ///
30    /// Though the signature includes an error type, that error is only present because
31    /// it is required by `uniffi` in order to handle panics. This function should suppress
32    /// and ignore internal errors instead of propagating them, to the maximum extent possible.
33    async fn epoch_changed(
34        &self,
35        conversation_id: Arc<ConversationId>,
36        epoch: u64,
37    ) -> Result<(), EpochChangedReportingError>;
38}
39
40/// This shim bridges the public `EpochObserver` interface with the internal one defined by `core-crypto`.
41///
42/// This is slightly unfortunate, as it introduces an extra layer of indirection before a change notice can
43/// actually reach its foreign target. However, the orphan rule prevents us from just tying the two traits
44/// together directly, so this is the straightforward way to accomplish that.
45struct ObserverShim(Arc<dyn EpochObserver>);
46
47#[cfg_attr(target_family = "wasm", async_trait::async_trait(?Send))]
48#[cfg_attr(not(target_family = "wasm"), async_trait::async_trait)]
49impl core_crypto::mls::EpochObserver for ObserverShim {
50    async fn epoch_changed(&self, conversation_id: core_crypto::ConversationId, epoch: u64) {
51        if let Err(err) = self
52            .0
53            .epoch_changed(Arc::new(ConversationId(conversation_id.as_ref().to_owned())), epoch)
54            .await
55        {
56            // we don't _care_ if an error is thrown by the notification function, per se,
57            // but this would probably be useful information for downstream debugging efforts
58            log::warn!(
59                conversation_id = &conversation_id,
60                epoch,
61                err = log::kv::Value::from_dyn_error(&err);
62                "caught an error when attempting to notify the epoch observer of an epoch change"
63            );
64        }
65    }
66}
67
68#[uniffi::export]
69impl CoreCryptoFfi {
70    /// Add an epoch observer to this client.
71    ///
72    /// This function should be called 0 or 1 times in a session's lifetime. If called
73    /// when an epoch observer already exists, this will return an error.
74    pub async fn register_epoch_observer(&self, epoch_observer: Arc<dyn EpochObserver>) -> CoreCryptoResult<()> {
75        let shim = Arc::new(ObserverShim(epoch_observer));
76        self.inner
77            .mls_session()
78            .await?
79            .register_epoch_observer(shim)
80            .await
81            .map_err(CoreCryptoError::generic())
82    }
