10#[cfg(not(target_family = "wasm"))]
11#[macro_rules_attribute::apply(smol_macros::main)]
12async fn main() -> anyhow::Result<()> {
13    #[derive(Debug, clap::Parser)]
14    #[command(author, version, about, long_about = None)]
15    struct Args {
16        #[arg(short, long)]
17        /// The 256-bit database key, hex-encoded.
18        key: String,
19
20        path: String,
21    }
22
23    use chrono::TimeZone;
24    use clap::Parser as _;
25    use core_crypto_keystore::{
26        ConnectionType, Database as Keystore, DatabaseKey, connection::FetchFromDatabase, entities::*,
27    };
28    use openmls::prelude::TlsDeserializeTrait;
29    use serde::ser::{SerializeMap, Serializer};
30
31    let args = Args::parse();
32
33    if !std::path::Path::new(&args.path).exists() {
34        bail!("File not found: {}", args.path);
35    }
36
37    let key = DatabaseKey::try_from(hex::decode(&args.key)?.as_slice())?;
38    let keystore = Keystore::open(ConnectionType::Persistent(&args.path), &key)
39        .await
40        .map_err(|e| anyhow!("The passkey is probably wrong; [err: {e}]"))?;
41
42    let mut json_serializer = serde_json::Serializer::pretty(std::io::stdout());
43    let mut json_map = json_serializer.serialize_map(None)?;
44
45    let mut credentials: Vec<serde_json::Value> = vec![];
46    for cred in keystore
47        .find_all::<MlsCredential>(Default::default())
48        .await?
49        .into_iter()
50    {
51        let mls_credential = openmls::prelude::Credential::tls_deserialize(&mut cred.credential.as_slice())?;
52        let date = chrono::Utc
53            .timestamp_opt(cred.created_at as i64, 0)
54            .single()
55            .ok_or_else(|| anyhow!("Cannot parse credential creation date"))?;
56
57        credentials.push(serde_json::json!({
58            "id": cred.id,
59            "credential": mls_credential,
60            "created_at": date
61        }));
62    }
63    json_map.serialize_entry("mls_credentials", &credentials)?;
64
65    let mut signature_keypairs: Vec<serde_json::Value> = vec![];
66    for kp in keystore
67        .find_all::<MlsSignatureKeyPair>(Default::default())
68        .await?
69        .into_iter()
70    {
71        let mls_keypair = openmls_basic_credential::SignatureKeyPair::tls_deserialize(&mut kp.keypair.as_slice())?;
72        signature_keypairs.push(serde_json::json!({
73            "signature_scheme": kp.signature_scheme,
74            "mls_keypair": mls_keypair,
75            "credential_id": kp.credential_id,
76        }));
77    }
78    json_map.serialize_entry("mls_signature_keypairs", &signature_keypairs)?;
79
80    let hpke_sks: Vec<openmls_traits::types::HpkePrivateKey> = keystore
81        .find_all::<MlsHpkePrivateKey>(Default::default())
82        .await?
83        .into_iter()
84        .map(|hpke_sk| postcard::from_bytes::<openmls_traits::types::HpkePrivateKey>(&hpke_sk.sk))
85        .collect::<postcard::Result<_>>()?;
86    json_map.serialize_entry("mls_hpke_private_keys", &hpke_sks)?;
87
88    let hpke_keypairs: Vec<openmls_traits::types::HpkeKeyPair> = keystore
89        .find_all::<MlsEncryptionKeyPair>(Default::default())
90        .await?
91        .into_iter()
92        .map(|hpke_kp| postcard::from_bytes::<openmls_traits::types::HpkeKeyPair>(&hpke_kp.sk))
93        .collect::<postcard::Result<_>>()?;
94    json_map.serialize_entry("mls_hpke_keypairs", &hpke_keypairs)?;
95
96    let mut external_psks: std::collections::HashMap<String, openmls::schedule::psk::PskBundle> = Default::default();
97    for psk in keystore.find_all::<MlsPskBundle>(Default::default()).await?.into_iter() {
98        let mls_psk = postcard::from_bytes::<openmls::schedule::psk::PskBundle>(&psk.psk)?;
99        external_psks.insert(hex::encode(&psk.psk_id), mls_psk);
100    }
101
102    json_map.serialize_entry("external_psks", &external_psks)?;
103
104    let keypackages: Vec<openmls::prelude::KeyPackage> = keystore
105        .find_all::<MlsKeyPackage>(Default::default())
106        .await?
107        .into_iter()
108        .map(|kp| postcard::from_bytes::<openmls::prelude::KeyPackage>(&kp.keypackage))
109        .collect::<postcard::Result<_>>()?;
110    json_map.serialize_entry("mls_keypackages", &keypackages)?;
111
112    let e2ei_enrollments: Vec<core_crypto::prelude::E2eiEnrollment> = keystore
113        .find_all::<E2eiEnrollment>(Default::default())
114        .await?
115        .into_iter()
116        .map(|enrollment| serde_json::from_slice::<core_crypto::prelude::E2eiEnrollment>(&enrollment.content))
117        .collect::<serde_json::Result<_>>()?;
118    json_map.serialize_entry("e2ei_enrollments", &e2ei_enrollments)?;
119
120    let pgroups: Vec<openmls::prelude::MlsGroup> = keystore
121        .find_all::<PersistedMlsGroup>(Default::default())
122        .await?
123        .into_iter()
124        .map(|pgroup| core_crypto_keystore::deser::<openmls::prelude::MlsGroup>(&pgroup.state))
125        .collect::<core_crypto_keystore::CryptoKeystoreResult<_>>()?;
126    json_map.serialize_entry("mls_groups", &pgroups)?;
127
128    let pegroups: Vec<openmls::prelude::MlsGroup> = keystore
129        .find_all::<PersistedMlsPendingGroup>(Default::default())
130        .await?
131        .into_iter()
132        .map(|pgroup| core_crypto_keystore::deser::<openmls::prelude::MlsGroup>(&pgroup.state))
133        .collect::<core_crypto_keystore::CryptoKeystoreResult<_>>()?;
134    json_map.serialize_entry("mls_pending_groups", &pegroups)?;
135
136    if let Some(proteus_identity) = keystore.find::<ProteusIdentity>(ProteusIdentity::ID).await? {
137        let identity = {
138            let sk = proteus_identity.sk_raw();
139            let pk = proteus_identity.pk_raw();
140            proteus_wasm::keys::IdentityKeyPair::from_raw_key_pair(*sk, *pk)?
141        };
142        json_map.serialize_entry("proteus_identity", &identity)?;
143
144        let prekeys: Vec<proteus_wasm::keys::PreKey> = keystore
145            .find_all::<ProteusPrekey>(Default::default())
146            .await?
147            .into_iter()
148            .map(|pk| proteus_wasm::keys::PreKey::deserialise(&pk.prekey))
149            .collect::<Result<Vec<_>, proteus_wasm::DecodeError>>()?;
150        json_map.serialize_entry("proteus_prekeys", &prekeys)?;
151
152        let proteus_sessions: Vec<proteus_wasm::session::Session<proteus_wasm::keys::IdentityKeyPair>> = keystore
153            .find_all::<ProteusSession>(Default::default())
154            .await?
155            .into_iter()
156            .map(|session| proteus_wasm::session::Session::deserialise(identity.clone(), &session.session))
157            .collect::<Result<Vec<_>, proteus_wasm::DecodeError>>()?;
158
159        json_map.serialize_entry("proteus_sessions", &proteus_sessions)?;
160    }
161
162    json_map.end()?;
163
164    Ok(())
165}
