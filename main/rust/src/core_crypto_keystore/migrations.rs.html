53/// Try to extract the relevant data from the v5 credential and signature keypair to determine whether they correspond
54/// to each other. If succeeding, and they do match, return Ok(Some) with the new credential, return Ok(None) if they
55/// don't, and return an Error if the data extraction fails.
56pub(crate) fn migrate_to_new_credential(
57    v5_credential: &V5Credential,
58    stored_keypair: &StoredSignatureKeypair,
59) -> CryptoKeystoreResult<Option<V6Credential>> {
60    let mls_keypair = SignatureKeyPair::tls_deserialize_exact(&stored_keypair.keypair)
61        .map_err(|e| CryptoKeystoreError::MigrationFailed(format!("Deserializing keypair: {e}")))?;
62
63    if !v5_credential_matches_signature_key(v5_credential, stored_keypair, &mls_keypair)? {
64        return Ok(None);
65    }
66
67    let new_credential = V6Credential {
68        id: v5_credential.id.clone(),
69        credential: v5_credential.credential.clone(),
70        created_at: v5_credential.created_at,
71        signature_scheme: stored_keypair.signature_scheme,
72        secret_key: mls_keypair.private().to_owned(),
73        public_key: stored_keypair.pk.clone(),
74    };
75
76    Ok(Some(new_credential))
77}

79pub(crate) fn v5_credential_matches_signature_key(
80    v5_credential: &V5Credential,
81    stored_keypair: &StoredSignatureKeypair,
82    signature_key: &SignatureKeyPair,
83) -> CryptoKeystoreResult<bool> {
84    let mls_credential = MlsCredential::tls_deserialize_exact(&v5_credential.credential)
85        .map_err(|e| CryptoKeystoreError::MigrationFailed(format!("deserializing stored credential: {e}")))?;
86
87    match mls_credential.mls_credential() {
88        MlsCredentialType::Basic(_) => {
89            let keypair_identity = &stored_keypair.credential_id;
90            if keypair_identity != mls_credential.identity() {
91                return Ok(false);
92            }
93        }
94
95        MlsCredentialType::X509(cert) => {
96            let certificate_bytes = cert
97                .certificates
98                .first()
99                .ok_or(CryptoKeystoreError::MigrationFailed("No leaf certificate".into()))?;
100            let certificate_inner = x509_cert::Certificate::from_der(certificate_bytes.as_slice())
101                .map_err(|e| CryptoKeystoreError::MigrationFailed(format!("decoding x509 certificate: {e}")))?;
102            let credential_public_key = certificate_inner
103                .public_key()
104                .map_err(|e| CryptoKeystoreError::MigrationFailed(format!("extracting public key from cert: {e}")))?;
105
106            if signature_key.public() != credential_public_key {
107                return Ok(false);
108            }
109        }
110    }
111
112    Ok(true)
113}

115/// Make a function which inverts [`Credential::signature_algorithm`](https://github.com/wireapp/openmls/blob/c9cde17076508968c9cbead5728454f0a1f60c4f/traits/src/types.rs#L472-L474)
116///
117/// We need to map a signature scheme to a ciphersuite in our migrations. Most of these mappings are unambiguous.
118/// However, we still need to consider the possibility that users are using unexpected signature schemes.
119///
120/// The strategy here is to perform an analysis of the existing conversations, counting the ciphersuites actually in
121/// use. For each ambiguous signature scheme, there are these possibilities to consider:
122///
123/// | Ciphersuite A | Ciphersuite B | Outcome |
124/// |---------------|---------------|---------|
125/// |             0 |             0 | Ciphersuite unused, drop this credential |
126/// |      non-zero |             0 | As all conversations use ciphersuite A, it is safe to uniquely map the credential to this ciphersuite |
127/// |             0 |      non-zero | As all conversations use ciphersuite B, it is safe to uniquely map the credential to this ciphersuite |
128/// |    non-zero n |    non-zero n | We arbitrarily choose the ciphersuite with smaller canonical number, breaking half the user conversations |
129/// |    non-zero a |    non-zero b | We choose the ciphersuite with the greater population, breaking those conversations with the lesser |
130///
131/// However, we expect the cases where both A and B have non-zero counts to be very rare.
132pub(crate) fn make_ciphersuite_for_signature_scheme(
133    persisted_mls_groups: impl IntoIterator<Item = PersistedMlsGroup>,
134) -> CryptoKeystoreResult<impl Fn(u16) -> Option<u16>> {
135    let mut ed25519_aes = 0; // MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519
136    let mut ed25519_chacha = 0; // MLS_128_DHKEMX25519_CHACHA20POLY1305_SHA256_Ed25519
137    let mut ed448_aes = 0; // MLS_256_DHKEMX448_AES256GCM_SHA512_Ed448
138    let mut ed448_chacha = 0; // MLS_256_DHKEMX448_CHACHA20POLY1305_SHA512_Ed448
139
140    for mls_group in persisted_mls_groups {
141        let mls_group = deser::<MlsGroup>(&mls_group.state)?;
142        match mls_group.ciphersuite() {
143            Ciphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519 => ed25519_aes += 1,
144            Ciphersuite::MLS_128_DHKEMX25519_CHACHA20POLY1305_SHA256_Ed25519 => ed25519_chacha += 1,
145            Ciphersuite::MLS_256_DHKEMX448_AES256GCM_SHA512_Ed448 => ed448_aes += 1,
146            Ciphersuite::MLS_256_DHKEMX448_CHACHA20POLY1305_SHA512_Ed448 => ed448_chacha += 1,
147            _ => {}
148        }
149    }
150
151    let ed25519 = match (ed25519_aes, ed25519_chacha) {
152        (0, 0) => None,
153        // other 0 cases are degenerate and handled below
154        (a, b) if a >= b => Some(Ciphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519.into()),
155        _ => Some(Ciphersuite::MLS_128_DHKEMX25519_CHACHA20POLY1305_SHA256_Ed25519.into()),
156    };
157
158    let ed448 = match (ed448_aes, ed448_chacha) {
159        (0, 0) => None,
160        // other 0 cases are degenerate and handled below
161        (a, b) if a >= b => Some(Ciphersuite::MLS_256_DHKEMX448_AES256GCM_SHA512_Ed448.into()),
162        _ => Some(Ciphersuite::MLS_256_DHKEMX448_CHACHA20POLY1305_SHA512_Ed448.into()),
163    };
164
165    let ciphersuite_for_signature_scheme = move |signature_scheme: u16| -> Option<u16> {
166        let signature_scheme = SignatureScheme::try_from(signature_scheme).ok()?;
167        match signature_scheme {
168            SignatureScheme::ECDSA_SECP256R1_SHA256 => {
169                Some(Ciphersuite::MLS_128_DHKEMP256_AES128GCM_SHA256_P256.into())
170            }
171            SignatureScheme::ECDSA_SECP384R1_SHA384 => {
172                Some(Ciphersuite::MLS_256_DHKEMP384_AES256GCM_SHA384_P384.into())
173            }
174            SignatureScheme::ECDSA_SECP521R1_SHA512 => {
175                Some(Ciphersuite::MLS_256_DHKEMP521_AES256GCM_SHA512_P521.into())
176            }
177            SignatureScheme::ED25519 => ed25519,
178            SignatureScheme::ED448 => ed448,
179        }
180    };
181    Ok(ciphersuite_for_signature_scheme)
182}
