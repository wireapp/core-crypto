1use crate::{
2    CryptoKeystoreResult,

8pub(crate) const VERSION: i32 = 18;

10pub(crate) fn meta_migration(conn: &mut rusqlite::Connection) -> CryptoKeystoreResult<()> {
11    let tx = conn.transaction()?;
12
13    let mut group_stmt = tx.prepare(&format!(
14        "SELECT state FROM {mls_group_table}",
15        mls_group_table = PersistedMlsGroup::COLLECTION_NAME,
16    ))?;
17    let persisted_mls_groups = group_stmt
18        .query_map([], |row| {
19            Ok(PersistedMlsGroup {
20                state: row.get("state")?,
21                id: Vec::new(),  // not relevant for this application
22                parent_id: None, // not relevant for this application
23            })
24        })?
25        .filter_map(|row| row.ok()); // rows which can't load at the SQL level are skipped
26    let ciphersuite_for_signature_scheme = make_ciphersuite_for_signature_scheme(persisted_mls_groups)?;
27
28    let mut credential_stmt = tx.prepare(&format!(
29        "SELECT
30            session_id,
31            credential,
32            unixepoch(created_at) AS created_at,
33            signature_scheme,
34            public_key,
35            private_key
36         FROM {credential_table}",
37        credential_table = StoredCredential::COLLECTION_NAME,
38    ))?;
39
40    let mut rows = credential_stmt.query([])?;
41    while let Some(row) = rows.next()? {
42        let v6 = V6Credential {
43            session_id: row.get("session_id")?,
44            credential: row.get("credential")?,
45            created_at: row.get("created_at")?,
46            signature_scheme: row.get("signature_scheme")?,
47            public_key: row.get("public_key")?,
48            private_key: row.get("private_key")?,
49        };
50
51        // Insert the new credential into temporary mls_credentials_new table, that will be renamed in the next
52        // migration
53        if let Some(ciphersuite) = ciphersuite_for_signature_scheme(v6.signature_scheme) {
54            tx.execute(
55                "INSERT INTO mls_credentials_new (
56                        session_id,
57                        credential,
58                        created_at,
59                        ciphersuite,
60                        public_key,
61                        private_key
62                    )
63                    VALUES (?1, ?2, datetime(?3, 'unixepoch'), ?4, ?5, ?6)",
64                (
65                    v6.session_id.clone(),
66                    v6.credential.clone(),
67                    v6.created_at,
68                    ciphersuite,
69                    v6.public_key.clone(),
70                    v6.private_key.clone(),
71                ),
72            )?;
73        }
74    }
75
76    drop(rows);
77    drop(credential_stmt);
78    drop(ciphersuite_for_signature_scheme);
79    drop(group_stmt);
80
81    tx.commit()?;
82
83    Ok(())
84}
