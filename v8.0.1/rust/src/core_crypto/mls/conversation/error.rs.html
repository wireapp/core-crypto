1//! MLS conversation errors
2
3// We allow missing documentation in the error module because the types are generally self-descriptive.
4#![allow(missing_docs)]
5
6use super::config::MAX_PAST_EPOCHS;
7
8pub type Result<T, E = Error> = core::result::Result<T, E>;
9
10#[derive(Debug, thiserror::Error)]
11pub enum Error {
12    #[error("Message body type not suitable for restoration")]
13    InappropriateMessageBodyType,
14    #[error("The specified parent group has not been found in the keystore")]
15    ParentGroupNotFound,
16    #[error("The current client id isn't authorized to perform this action")]
17    Unauthorized,
18    /// Message rejected by the delivery service
19    #[error("Message rejected by the delivery service. Reason: {reason}")]
20    MessageRejected {
21        /// Why was the message rejected by the delivery service?
22        reason: String,
23    },
24    /// This is thrown when a client wants to retry sending an external join commit.
25    #[error(
26        "Cannot recreate a commit for retry without being in the conversation. Restart the operation from the beginning."
27    )]
28    CannotRetryWithoutConversation,
29    #[error("We already decrypted this message once")]
30    DuplicateMessage,
31    #[error("The epoch in which message was encrypted is older than {MAX_PAST_EPOCHS}")]
32    MessageEpochTooOld,
33    #[error("Incoming message is from a prior epoch")]
34    StaleMessage,
35    #[error(
36        "Incoming message is a commit for which we have not yet received all the proposals. Buffering until all proposals have arrived."
37    )]
38    BufferedCommit,
39    #[error("Incoming message is for a future epoch. We will buffer it until the commit for that epoch arrives")]
40    BufferedFutureMessage { message_epoch: u64 },
41    #[error(
42        "You tried to join with an external commit but did not merge it yet. We will reapply this message for you when you merge your external commit"
43    )]
44    BufferedForPendingConversation,
45    #[error("Incoming message is from an epoch too far in the future to buffer.")]
46    UnbufferedFarFutureMessage,
47    #[error("The received commit is deemed stale and is from an older epoch.")]
48    StaleCommit,
49    #[error("The received proposal is deemed stale and is from an older epoch.")]
50    StaleProposal,
51    #[error("An application message failed to decrypt")]
52    DecryptionError,
53    #[error("MLS Client was not initialized the right way")]
54    IdentityInitializationError,
55    #[error("MLS group is in an invalid state: {0}")]
56    MlsGroupInvalidState(&'static str),
57    #[error("MLS message is in an invalid state: {0}")]
58    MlsMessageInvalidState(&'static str),
59    #[error("The group lacks an ExternalSender extension whereas it should have at least one")]
60    MissingExternalSenderExtension,
61    #[error("Couldn't find pending proposal {0}")]
62    PendingProposalNotFound(crate::mls::proposal::MlsProposalRef),
63    #[error("Couldn't find pending commit")]
64    PendingCommitNotFound,
65    #[error(
66        "Happens when a client creates a commit, sends it to the DS which accepts it but then client \
67    clears this pending commit and creates another commit. This is triggered when the client tries to decrypt the original commit.\
68    This means something is very wrong in the client's code and has to be fixed immediately"
69    )]
70    ClearingPendingCommitError,
71    #[error("Tried to decrypt a commit created by self which is likely to have been replayed by the DS")]
72    SelfCommitIgnored,
73    #[error("This proposal variant cannot be renewed")]
74    ProposalVariantCannotBeRenewed,
75    #[error("caller error: {0}")]
76    CallerError(&'static str),
77    /// This happens when the DS cannot flag KeyPackages as claimed or not. It this scenario, a client
78    /// requests their old KeyPackages to be deleted but one has already been claimed by another client to create a Welcome.
79    /// In that case the only solution is that the client receiving such a Welcome tries to join the group
80    /// with an External Commit instead
81    #[error(
82        "Although this Welcome seems valid, the local KeyPackage it references has already been deleted locally. Join this group with an external commit"
83    )]
84    OrphanWelcome,
85    #[error("Serializing {item} for TLS")]
86    TlsSerialize {
87        item: &'static str,
88        #[source]
89        source: tls_codec::Error,
90    },
91    #[error("Deserializing {item} for TLS")]
92    TlsDeserialize {
93        item: &'static str,
94        #[source]
95        source: tls_codec::Error,
96    },
97    #[error(transparent)]
98    Mls(#[from] crate::MlsError),
99    #[error(transparent)]
100    Keystore(#[from] crate::KeystoreError),
101    #[error("{0}")]
102    Leaf(#[from] crate::LeafError),
103    #[error(transparent)]
104    Recursive(#[from] crate::RecursiveError),
105}
106
107impl Error {
108    pub fn tls_serialize(item: &'static str) -> impl FnOnce(tls_codec::Error) -> Self {
109        move |source| Self::TlsSerialize { item, source }
110    }
111
112    pub fn tls_deserialize(item: &'static str) -> impl FnOnce(tls_codec::Error) -> Self {
113        move |source| Self::TlsDeserialize { item, source }
114    }
115}
