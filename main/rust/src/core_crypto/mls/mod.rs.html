26    /// Configuration parameters for [Session].
27    #[derive(Debug, Clone)]
28    #[non_exhaustive]
29    pub struct MlsClientConfiguration {
30        /// Location where the SQLite/IndexedDB database will be stored
31        pub store_path: String,
32        /// Database key to be used to instantiate the [MlsCryptoProvider]
33        pub database_key: DatabaseKey,
34        /// Identifier for the client to be used by [Session]
35        pub client_id: Option<ClientId>,
36        /// Entropy pool seed for the internal PRNG
37        pub external_entropy: Option<EntropySeed>,
38        /// All supported ciphersuites
39        pub ciphersuites: Vec<ciphersuite::MlsCiphersuite>,
40        /// Number of [openmls::prelude::KeyPackage] to create when creating a MLS client.
41        /// Defaults to [crate::prelude::INITIAL_KEYING_MATERIAL_COUNT].
42        pub nb_init_key_packages: Option<usize>,
43    }
44
45    impl MlsClientConfiguration {
46        /// Creates a new instance of the configuration.
47        ///
48        /// # Arguments
49        /// * `store_path` - location where the SQLite/IndexedDB database will be stored
50        /// * `database_key` - key to be used to instantiate the [MlsCryptoProvider]
51        /// * `client_id` - identifier for the client to be used by [Session]
52        /// * `ciphersuites` - Ciphersuites supported by this device
53        /// * `entropy` - External source of entropy for platforms where default source insufficient
54        ///
55        /// # Errors
56        /// Any empty string parameter will result in a [Error::MalformedIdentifier] error.
57        ///
58        /// # Examples
59        ///
60        /// ```
61        /// use core_crypto::prelude::{MlsClientConfiguration, MlsCiphersuite};
62        /// use core_crypto::DatabaseKey;
63        ///
64        /// let result = MlsClientConfiguration::try_new(
65        ///     "/tmp/crypto".to_string(),
66        ///     DatabaseKey::generate(),
67        ///     Some(b"MY_CLIENT_ID".to_vec().into()),
68        ///     vec![MlsCiphersuite::default()],
69        ///     None,
70        ///     Some(100),
71        /// );
72        /// assert!(result.is_ok());
73        /// ```
74        pub fn try_new(
75            store_path: String,
76            database_key: DatabaseKey,
77            client_id: Option<ClientId>,
78            ciphersuites: Vec<MlsCiphersuite>,
79            entropy: Option<Vec<u8>>,
80            nb_init_key_packages: Option<usize>,
81        ) -> Result<Self> {
82            // TODO: probably more complex rules to enforce. Tracking issue: WPB-9598
83            if store_path.trim().is_empty() {
84                return Err(Error::MalformedIdentifier("store_path"));
85            }
86            // TODO: probably more complex rules to enforce. Tracking issue: WPB-9598
87            if let Some(client_id) = client_id.as_ref() {
88                if client_id.is_empty() {
89                    return Err(Error::MalformedIdentifier("client_id"));
90                }
91            }
92            let external_entropy = entropy
93                .as_deref()
94                .map(|seed| &seed[..EntropySeed::EXPECTED_LEN])
95                .map(EntropySeed::try_from_slice)
96                .transpose()
97                .map_err(MlsError::wrap("gathering external entropy"))?;
98            Ok(Self {
99                store_path,
100                database_key,
101                client_id,
102                ciphersuites,
103                external_entropy,
104                nb_init_key_packages,
105            })
106        }
107
108        /// Sets the entropy seed
109        pub fn set_entropy(&mut self, entropy: EntropySeed) {
110            self.external_entropy = Some(entropy);
111        }

115        /// Creates temporary file to prevent test collisions which would happen with hardcoded file path
116        /// Intended to be used only in tests.
117        pub(crate) fn tmp_store_path(tmp_dir: &tempfile::TempDir) -> String {
118            let path = tmp_dir.path().join("store.edb");
119            std::fs::File::create(&path).unwrap();
120            path.to_str().unwrap().to_string()
121        }

288    #[apply(all_cred_cipher)]
289    async fn can_2_phase_init_central(mut case: TestContext) {
290        let tmp_dir = case.tmp_dir().await;
291        Box::pin(async move {
292            let x509_test_chain = X509TestChain::init_empty(case.signature_scheme());
293            let configuration = MlsClientConfiguration::try_new(
294                tmp_dir,
295                DatabaseKey::generate(),
296                None,
297                vec![case.ciphersuite()],
298                None,
299                Some(INITIAL_KEYING_MATERIAL_COUNT),
300            )
301            .unwrap();
302            // phase 1: init without initialized mls_client
303            let client = Session::try_new(configuration).await.unwrap();
304            let cc = CoreCrypto::from(client);
305            let context = cc.new_transaction().await.unwrap();
306            x509_test_chain.register_with_central(&context).await;
307
308            assert!(!context.session().await.unwrap().is_ready().await);
309            // phase 2: init mls_client
310            let client_id = "alice";
311            let identifier = match case.credential_type {
312                MlsCredentialType::Basic => ClientIdentifier::Basic(client_id.into()),
313                MlsCredentialType::X509 => {
314                    CertificateBundle::rand_identifier(client_id, &[x509_test_chain.find_local_intermediate_ca()])
315                }
316            };
317            context
318                .mls_init(
319                    identifier,
320                    vec![case.ciphersuite()],
321                    Some(INITIAL_KEYING_MATERIAL_COUNT),
322                )
323                .await
324                .unwrap();
325            assert!(context.session().await.unwrap().is_ready().await);
326            // expect mls_client to work
327            assert_eq!(
328                context
329                    .get_or_create_client_keypackages(case.ciphersuite(), case.credential_type, 2)
330                    .await
331                    .unwrap()
332                    .len(),
333                2
334            );
335        })
336        .await
337    }
