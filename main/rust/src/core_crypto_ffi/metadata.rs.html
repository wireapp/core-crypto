9// Unfortunately, we need to define the `BuildMetadata` struct twice:

13
14#[derive(uniffi::Record)]
15/// Metadata describing the conditions of the build of this software.
16pub struct BuildMetadata {
17    /// Build Timestamp
18    pub timestamp: String,
19    /// Whether this build was in Debug mode (true) or Release mode (false)
20    pub cargo_debug: String,
21    /// Features enabled for this build
22    pub cargo_features: String,
23    /// Optimization level
24    pub opt_level: String,
25    /// Build target triple
26    pub target_triple: String,
27    /// Git branch
28    pub git_branch: String,
29    /// Output of `git describe`
30    pub git_describe: String,
31    /// Hash of current git commit
32    pub git_sha: String,
33    /// `true` when the source code differed from the commit at the most recent git hash
34    pub git_dirty: String,
35}
36
37/// Returns build data for CoreCrypto
38#[uniffi::export]
39pub fn build_metadata() -> BuildMetadata {
40    BuildMetadata {
41        timestamp: core_crypto::BUILD_METADATA.timestamp.into(),
42        cargo_debug: core_crypto::BUILD_METADATA.cargo_debug.into(),
43        cargo_features: core_crypto::BUILD_METADATA.cargo_features.into(),
44        opt_level: core_crypto::BUILD_METADATA.opt_level.into(),
45        target_triple: core_crypto::BUILD_METADATA.target_triple.into(),
