1//! Persistence for [`CredentialRef`], i.e. loading actual credentials from the keystore given a ref.
2//!
3//! It is not logically required that these methods are crate-private, but they aren't likely to be
4//! useful to end users. Clients building on the CC API can't do anything useful with a full [`Credential`],
5//! and it's wasteful to transfer one across the FFI boundary.
6
7use core_crypto_keystore::{
8    connection::FetchFromDatabase as _,
9    entities::{EntityFindParams, StoredCredential, StoredSignatureKeypair},
10};
11use mls_crypto_provider::Database;
12use openmls::prelude::Credential as MlsCredential;
13use tls_codec::Deserialize as _;
14
15use super::{super::keypairs, Error, Result};
16use crate::{Credential, CredentialRef, KeystoreError, RecursiveError};
17
18/// Helper struct caching relevant data from the keystore.
19///
20/// This is generated by [`CredentialRef::load_cache`]. It's only relevant when you want to
21/// load multiple credentials at once. For single loads, prefer [`CredentialRef::load`].
22//
23// We'd very much like it if in the future we could do filtering at the database level,
24// obviating the requirement for this cache structure. See WPB-20839 and WPB-20844.
25pub(crate) struct Cache {
26    keypairs: Vec<StoredSignatureKeypair>,
27    credentials: Vec<StoredCredential>,
28}
29
30impl CredentialRef {
31    /// Helper to prefetch relevant keypairs when loading multiple credentials at a time.
32    ///
33    /// Only useful when preparing to call [`Self::load_with_cache`] multiple times.
34    /// For loading a single credential, prefer [`Self::load`].
35    pub(crate) async fn load_cache(database: &Database) -> Result<Cache> {
36        let keypairs = keypairs::load_all(database)
37            .await
38            .map_err(RecursiveError::mls_credential("loading all keypairs for cache"))?;
39        let credentials = database
40            .find_all::<StoredCredential>(EntityFindParams::default())
41            .await
42            .map_err(KeystoreError::wrap("finding all mls credentials"))?;
43        Ok(Cache { keypairs, credentials })
44    }
45
46    /// Load all credentials which match this ref from the database.
47    ///
48    /// Note that this does not attach the credential to any Session; it just does the data manipulation.
49    ///
50    /// The database schema currently permits multiple credentials to exist simultaneously which match a given credential ref.
51    /// Therefore, this function returns all of them, ordered by `earliest_validity`.
52    ///
53    /// Due to database limitations we currently cannot efficiently retrieve only those keypairs of interest;
54    /// if you are going to be loading several references in a row, it is more efficient to first fetch all
55    /// stored keypairs with [`Self::load_cache`] and then call [`Self::load_with_cache`].
56    pub(crate) async fn load(&self, database: &Database) -> Result<Vec<Credential>> {
57        let cache = Self::load_cache(database).await?;
58        let credentials = self.load_with_cache(&cache).await?.collect::<Result<Vec<_>>>()?;
59        Ok(credentials)
60    }
61
62    /// Load this credential from the database.
63    ///
64    /// Note that this does not attach the credential to any Session; it just does the data manipulation.
65    ///
66    /// The database schema currently permits multiple credentials to exist simultaneously which match a given credential ref.
67    /// Therefore, this function returns a possibly-empty iterator over all of them, ordered by `earliest_validity`.
68    ///
69    /// If you are only loading a single credential ref, it may be simpler to call [`Self::load`].
70    pub(crate) async fn load_with_cache<'cache, 'cref>(
71        &'cref self,
72        Cache { keypairs, credentials }: &'cache Cache,
73    ) -> Result<impl 'cache + Iterator<Item = Result<Credential>>>
74    where
75        'cref: 'cache,
76    {
77        let signature_key_pair =
78            keypairs::find_matching(keypairs, self.client_id(), self.signature_scheme(), self.public_key())
79                .await
80                .map_err(RecursiveError::mls_credential(
81                    "finding matching key pairs while loading credential",
82                ))?
83                .ok_or(Error::KeypairNotFound)?;
84
85        let iter = credentials
86            .iter()
87            // this is the only check we can currently do at the DB level: match the client id and creation timestamp
88            .filter(|stored_credential| stored_credential.id == self.client_id().as_slice() && stored_credential.created_at == self.earliest_validity)
89            // from here we can at least deserialize the credential
90            .map(move |stored_credential| {
91                let mls_credential = MlsCredential::tls_deserialize(&mut stored_credential.credential.as_slice())
92                    .map_err(Error::tls_deserialize("mls credential"))?;
93                let credential_type = mls_credential.credential_type().try_into().map_err(RecursiveError::mls_credential("loading credential from db"))?;
94                let earliest_validity = stored_credential.created_at;
95                Ok(Credential {
96                    signature_key_pair: signature_key_pair.clone(),
97                    credential_type,
98                    mls_credential,
99                    earliest_validity,
100                })
101            })
102            // after deserialization, we can filter out any results which do not match the conditions in the credential ref
103            // but pass through any errors
104            .filter(|credential_result| {
105                credential_result.as_ref().ok().is_none_or(|credential| {
106                    self.signature_scheme == credential.signature_key_pair.signature_scheme()
107                            && self.r#type == credential.mls_credential.credential_type()
108                })
109            })
110            // we also need to ensure that the credential validates
111            .map(|credential_result| {
112                credential_result.and_then(|credential| {
113                    Credential::validate_mls_credential(
114                        &credential.mls_credential,
115                        self.client_id(),
116                        &credential.signature_key_pair,
117                    )?;
118                    Ok(credential)
119                })
120            });
121
122        Ok(iter)
123    }
124}
