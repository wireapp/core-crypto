1use core_crypto::CredentialRef as CryptoCredentialRef;
2
3use crate::{Ciphersuite, ClientId, CredentialType, SignatureScheme};

5/// A reference to a credential which has been persisted in CC.
6///
7/// This is because credentials can be quite large; we'd really like to avoid passing them
8/// back and forth across the FFI boundary more than is strictly required.
9/// Therefore, we use this type which is substantially more compact.
10///
11/// Created with [`CoreCryptoContext::add_credential`][crate::CoreCryptoContext::add_credential].
12///
13/// This reference is _not_ a literal reference in memory.
14/// It is instead the key from which a credential can be retrieved.
15/// This means that it is stable over time and across the FFI boundary.
16#[derive(Debug, Clone, derive_more::From, derive_more::Into, uniffi::Object)]
17#[uniffi::export(Debug)]
18pub struct CredentialRef(pub(crate) CryptoCredentialRef);
19
20#[uniffi::export]
21impl CredentialRef {
22    /// Get the client id associated with this credential ref
23    pub fn client_id(&self) -> ClientId {
24        self.0.client_id().to_owned().into()
25    }
26
27    /// Get the public key associated with this credential ref
28    pub fn public_key(&self) -> Vec<u8> {
29        self.0.public_key().to_owned()
30    }
31
32    /// Get the type of this credential ref.
33    pub fn r#type(&self) -> CredentialType {
34        self.0.r#type().into()
35    }
36
37    /// Get the signature scheme of this credential ref.
38    pub fn signature_scheme(&self) -> SignatureScheme {
39        self.0.signature_scheme().into()
40    }
41
42    /// Get the ciphersuite of this credential ref.
43    pub fn ciphersuite(&self) -> Ciphersuite {
44        self.0.ciphersuite().into()
45    }
46
47    /// Get the earliest possible validity of this credential, expressed as seconds after the unix epoch.
48    ///
49    /// Basic credentials have no defined earliest validity and will always return 0.
50    pub fn earliest_validity(&self) -> u64 {
51        self.0.earliest_validity()
52    }
53}
