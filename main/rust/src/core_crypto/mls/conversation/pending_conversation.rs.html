33/// A pending conversation is a conversation that has been created via an external join commit
34/// locally, while this commit has not yet been approved by the DS.
35#[derive(Debug)]
36pub struct PendingConversation {
37    inner: PersistedMlsPendingGroup,
38    context: TransactionContext,
39}
40
41impl PendingConversation {
42    pub(crate) fn new(inner: PersistedMlsPendingGroup, context: TransactionContext) -> Self {
43        Self { inner, context }
44    }
45
46    pub(crate) fn from_mls_group(
47        group: MlsGroup,
48        custom_cfg: MlsCustomConfiguration,
49        context: TransactionContext,
50    ) -> Result<Self> {
51        let serialized_cfg = serde_json::to_vec(&custom_cfg).map_err(MlsError::wrap("serializing custom config"))?;
52        let serialized_group =
53            core_crypto_keystore::ser(&group).map_err(KeystoreError::wrap("serializing mls group"))?;
54        let group_id = group.group_id().to_vec();
55
56        let inner = PersistedMlsPendingGroup {
57            id: group_id,
58            state: serialized_group,
59            custom_configuration: serialized_cfg,
60            parent_id: None,
61        };
62        Ok(Self::new(inner, context))
63    }
64
65    async fn mls_provider(&self) -> Result<MlsCryptoProvider> {
66        self.context
67            .mls_provider()
68            .await
69            .map_err(RecursiveError::transaction("getting mls provider"))
70            .map_err(Into::into)
71    }
72
73    async fn keystore(&self) -> Result<Database> {
74        let backend = self.mls_provider().await?;
75        Ok(backend.keystore())
76    }
77
78    fn id(&self) -> &ConversationIdRef {
79        ConversationIdRef::new(&self.inner.id)
80    }
81
82    pub(crate) async fn save(&self) -> Result<()> {
83        let keystore = self.keystore().await?;
84        keystore
85            .mls_pending_groups_save(self.id(), &self.inner.state, &self.inner.custom_configuration, None)
86            .await
87            .map_err(KeystoreError::wrap("saving mls pending groups"))
88            .map_err(Into::into)
89    }

91    /// Send the commit via [crate::MlsTransport] and handle the response.
92    pub(crate) async fn send_commit(&self, commit: MlsCommitBundle) -> Result<()> {
93        let transport = self
94            .context
95            .mls_transport()
96            .await
97            .map_err(RecursiveError::transaction("getting mls transport"))?;
98        let transport = transport.as_ref().ok_or::<Error>(
99            RecursiveError::root("getting mls transport")(crate::Error::MlsTransportNotProvided).into(),
100        )?;
101
102        match transport
103            .send_commit_bundle(commit.clone())
104            .await
105            .map_err(RecursiveError::root("sending commit bundle"))?
106        {
107            MlsTransportResponse::Success => Ok(()),
108            MlsTransportResponse::Abort { reason } => Err(Error::MessageRejected { reason }),
109            MlsTransportResponse::Retry => Err(Error::CannotRetryWithoutConversation),
110        }
111    }

113    /// If the given message is the commit generated by [TransactionContext::join_by_external_commit],
114    /// merge the pending group and restore any buffered messages.
115    ///
116    /// Otherwise, the given message will be buffered.
117    pub async fn try_process_own_join_commit(
118        &mut self,
119        message: impl AsRef<[u8]>,
120    ) -> Result<MlsConversationDecryptMessage> {
121        // If the confirmation tag of the pending group and this incoming message are identical, we can merge the pending group.
122        if self.incoming_message_is_own_join_commit(message.as_ref()).await? {
123            return self.merge_and_restore_messages().await;
124        }
125
126        let keystore = self.keystore().await?;
127
128        let pending_msg = MlsPendingMessage {
129            foreign_id: self.id().as_ref().to_owned(),
130            message: message.as_ref().to_vec(),
131        };
132        keystore
133            .save::<MlsPendingMessage>(pending_msg)
134            .await
135            .map_err(KeystoreError::wrap("saving mls pending message"))?;
136        Err(Error::BufferedForPendingConversation)
137    }
138
139    /// If the message confirmation tag and the group confirmation tag are the same, it means that
140    /// the external join commit has been accepted by the DS and the pending group can be merged.
141    async fn incoming_message_is_own_join_commit(&self, message: impl AsRef<[u8]>) -> Result<bool> {
142        let backend = self.mls_provider().await?;
143        let keystore = backend.keystore();
144        // Instantiate the pending group
145        let (group, _cfg) = keystore
146            .mls_pending_groups_load(self.id())
147            .await
148            .map_err(KeystoreError::wrap("loading mls pending groups"))?;
149        let mut mls_group = core_crypto_keystore::deser::<MlsGroup>(&group)
150            .map_err(KeystoreError::wrap("deserializing mls pending groups"))?;
151
152        // The commit is only merged on this temporary instance of the pending group, to enable
153        // calculation of the confirmation tag.
154        mls_group
155            .merge_pending_commit(&backend)
156            .await
157            .map_err(MlsError::wrap("merging pending commit"))?;
158        let message_in = MlsMessageIn::tls_deserialize(&mut message.as_ref())
159            .map_err(MlsError::wrap("deserializing mls message"))?;
160        let MlsMessageInBody::PublicMessage(public_message) = message_in.extract() else {
161            return Ok(false);
162        };
163        let Some(msg_ct) = public_message.confirmation_tag() else {
164            return Ok(false);
165        };
166        let group_ct = mls_group
167            .compute_confirmation_tag(&backend)
168            .map_err(MlsError::wrap("computing confirmation tag"))?;
169        Ok(*msg_ct == group_ct)
170    }

172    /// Merges the [Self] instance and restores any buffered messages.
173    async fn merge_and_restore_messages(&mut self) -> Result<MlsConversationDecryptMessage> {
174        let buffered_messages = self.merge().await?;
175        let context = &self.context;
176        let backend = self.mls_provider().await?;
177        let id = self.id();
178
179        // This is the now merged conversation
180        let conversation = context
181            .conversation(id)
182            .await
183            .map_err(RecursiveError::transaction("getting conversation by id"))?;
184        let conversation = conversation.conversation().await;
185        let own_leaf = conversation.group.own_leaf().ok_or(LeafError::InternalMlsError)?;
186
187        // We return self identity here, probably not necessary to check revocation
188        let own_leaf_credential_with_key = CredentialWithKey {
189            credential: own_leaf.credential().clone(),
190            signature_key: own_leaf.signature_key().clone(),
191        };
192        let identity = own_leaf_credential_with_key
193            .extract_identity(conversation.ciphersuite(), None)
194            .map_err(RecursiveError::mls_credential("extracting identity"))?;
195
196        let crl_new_distribution_points = get_new_crl_distribution_points(
197            &backend,
198            extract_crl_uris_from_group(&conversation.group)
199                .map_err(RecursiveError::mls_credential("extracting crl uris from group"))?,
200        )
201        .await
202        .map_err(RecursiveError::mls_credential("getting new crl distribution points"))?;
203
204        // Note that though we return `has_epoch_changed: true` here, we don't notify the observer.
205        // This function can only be reached via a code path going through `MlsConversation::decrypt_message`
206        // which already notifies the observer; it would be redundant to notify here also.
207
208        // we still support the `has_epoch_changed` field, though we'll remove it later
209        #[expect(deprecated)]
210        Ok(MlsConversationDecryptMessage {
211            app_msg: None,
212            proposals: vec![],
213            is_active: conversation.group.is_active(),
214            delay: conversation.compute_next_commit_delay(),
215            sender_client_id: None,
216            has_epoch_changed: true,
217            identity,
218            buffered_messages,
219            crl_new_distribution_points,
220        })
221    }

223    /// This merges the commit generated by [TransactionContext::join_by_external_commit],
224    /// persists the group permanently and deletes the temporary one. After merging, the group
225    /// is fully functional.
226    ///
227    /// # Errors
228    /// Errors resulting from OpenMls, the KeyStore calls and deserialization
229    pub(crate) async fn merge(&mut self) -> Result<Option<Vec<MlsBufferedConversationDecryptMessage>>> {
230        let mls_provider = self.mls_provider().await?;
231        let id = self.id();
232        let group = self.inner.state.clone();
233        let cfg = self.inner.custom_configuration.clone();
234
235        let mut mls_group =
236            core_crypto_keystore::deser::<MlsGroup>(&group).map_err(KeystoreError::wrap("deserializing mls group"))?;
237
238        // Merge it aka bring the MLS group to life and make it usable
239        mls_group
240            .merge_pending_commit(&mls_provider)
241            .await
242            .map_err(MlsError::wrap("merging pending commit"))?;
243
244        // Restore the custom configuration and build a conversation from it
245        let custom_cfg =
246            serde_json::from_slice(&cfg).map_err(MlsError::wrap("deserializing mls custom configuration"))?;
247        let configuration = MlsConversationConfiguration {
248            ciphersuite: mls_group.ciphersuite().into(),
249            custom: custom_cfg,
250            ..Default::default()
251        };
252
253        // We have to determine the restore policy before we persist the group, because it depends
254        // on whether the group already exists.
255        let restore_policy = if mls_provider.key_store().mls_group_exists(id).await {
256            // If the group already exists, it means the external commit is about rejoining the group.
257            // This is most of the time a last resort measure (for example when a commit is dropped,
258            // and you go out of sync), so there's no point in decrypting buffered messages
259            trace!("External commit trying to rejoin group");
260            MessageRestorePolicy::ClearOnly
261        } else {
262            MessageRestorePolicy::DecryptAndClear
263        };
264
265        // Persist the now usable MLS group in the keystore
266        let conversation = MlsConversation::from_mls_group(mls_group, configuration, &mls_provider)
267            .await
268            .map_err(RecursiveError::mls_conversation(
269                "constructing conversation from mls group",
270            ))?;
271
272        let context = &self.context;
273
274        context
275            .mls_groups()
276            .await
277            .map_err(RecursiveError::transaction("getting mls groups"))?
278            .insert(id, conversation);
279
280        // This is the now merged conversation
281        let mut conversation = context
282            .conversation(id)
283            .await
284            .map_err(RecursiveError::transaction("getting conversation by id"))?;
285        let pending_messages = conversation
286            .restore_pending_messages(restore_policy)
287            .await
288            .map_err(RecursiveError::mls_conversation("restoring pending messages"))?;
289
290        if pending_messages.is_some() {
291            mls_provider
292                .keystore()
293                .remove_pending_messages_by_conversation_id(id)
294                .await
295                .map_err(KeystoreError::wrap("deleting mls pending messages by conversation id"))?;
296        }
297
298        // cleanup the pending group we no longer need
299        self.clear().await?;
300
301        Ok(pending_messages)
302    }
303
304    /// In case the external commit generated by [TransactionContext::join_by_external_commit] is
305    /// rejected by the Delivery Service, and we want to abort this external commit,
306    /// we can wipe out the pending group from the keystore.
307    ///
308    /// # Errors
309    /// Errors resulting from the KeyStore calls
310    pub(crate) async fn clear(&mut self) -> Result<()> {
311        self.keystore()
312            .await?
313            .mls_pending_groups_delete(self.id())
314            .await
315            .map_err(KeystoreError::wrap("deleting pending groups by id"))?;
316        Ok(())
317    }

325    #[apply(all_cred_cipher)]
326    async fn should_buffer_and_reapply_messages_after_external_commit_merged(case: TestContext) {
327        let [alice, bob, charlie, debbie] = case.sessions().await;
328        Box::pin(async move {
329            let conversation = case.create_conversation([&alice]).await;
330            // Bob tries to join Alice's group with an external commit
331            let (commit_guard, _pending_conversation) = conversation.external_join_unmerged(&bob).await;
332            let external_commit = commit_guard.message();
333
334            // Alice decrypts the external commit...
335            let conversation = commit_guard
336                .notify_member(&alice)
337                .await
338                .process_member_changes()
339                .await
340                .finish();
341
342            // Meanwhile Debbie joins the party by creating an external proposal
343            let proposal_guard = conversation.external_join_proposal(&debbie).await;
344            let external_proposal = proposal_guard.message();
345
346            // ...then Alice generates new messages for this epoch
347            let app_msg = proposal_guard
348                .conversation()
349                .guard()
350                .await
351                .encrypt_message(b"Hello Bob !")
352                .await
353                .unwrap();
354
355            let conversation = proposal_guard.notify_member(&alice).await.finish();
356            let proposal_guard = conversation.update_proposal().await;
357            let proposal = proposal_guard.message();
358            let conversation = proposal_guard.finish();
359
360            let commit_guard = conversation.invite([&charlie]).await;
361            let commit = commit_guard.message();
362            let conversation = commit_guard.process_member_changes().await.finish();
363
364            // And now Bob will have to decrypt those messages while he hasn't yet merged its external commit
365            // To add more fun, he will buffer the messages in exactly the wrong order (to make
366            // sure he reapplies them in the right order afterwards)
367            let messages = vec![commit, external_proposal, proposal]
368                .into_iter()
369                .map(|m| m.to_bytes().unwrap());
370            let Err(crate::transaction_context::Error::PendingConversation(mut pending_conversation)) =
371                bob.transaction.conversation(conversation.id()).await
372            else {
373                panic!("Bob should not have the conversation yet")
374            };
375            for m in messages {
376                let decrypt = pending_conversation.try_process_own_join_commit(m).await;
377                assert!(matches!(decrypt.unwrap_err(), Error::BufferedForPendingConversation));
378            }
379            let decrypt = pending_conversation.try_process_own_join_commit(app_msg).await;
380            assert!(matches!(decrypt.unwrap_err(), Error::BufferedForPendingConversation));
381
382            // Bob should have buffered the messages
383            assert_eq!(bob.transaction.count_entities().await.pending_messages, 4);
384
385            let observer = TestEpochObserver::new();
386            bob.session()
387                .await
388                .register_epoch_observer(observer.clone())
389                .await
390                .unwrap();
391
392            // Finally, Bob receives the green light from the DS and he can merge the external commit
393            let MlsConversationDecryptMessage {
394                buffered_messages: Some(restored_messages),
395                ..
396            } = pending_conversation
397                .try_process_own_join_commit(external_commit.to_bytes().unwrap())
398                .await
399                .unwrap()
400            else {
401                panic!("Alice's messages should have been restored at this point");
402            };
403
404            let observed_epochs = observer.observed_epochs().await;
405            assert_eq!(
406                observed_epochs.len(),
407                1,
408                "we should see exactly 1 epoch change in these 4 messages"
409            );
410            assert_eq!(observed_epochs[0].0, *conversation.id(), "conversation id must match");
411
412            for (idx, msg) in restored_messages.iter().enumerate() {
413                if idx == 0 {
414                    // the only application message
415                    assert_eq!(msg.app_msg.as_deref(), Some(b"Hello Bob !" as _));
416                } else {
417                    assert!(msg.app_msg.is_none());
418                }
419            }
420
421            assert_eq!(conversation.member_count().await, 4);
422            assert!(
423                conversation
424                    .is_functional_and_contains([&alice, &bob, &charlie, &debbie])
425                    .await
426            );
427
428            // After merging we should erase all those pending messages
429            assert_eq!(bob.transaction.count_entities().await.pending_messages, 0);
430        })
431        .await
432    }

434    #[apply(all_cred_cipher)]
435    async fn should_not_reapply_buffered_messages_when_rejoining(case: TestContext) {
436        use crate::mls;
437
438        let [alice, bob] = case.sessions().await;
439        Box::pin(async move {
440            let conversation = case.create_conversation([&alice, &bob]).await;
441
442            // Alice will never see this commit
443            let conversation = conversation.acting_as(&bob).await;
444            let commit_guard = conversation.update().await;
445            let conversation = commit_guard.conversation();
446
447            let msg1 = conversation.guard().await.encrypt_message("A").await.unwrap();
448            let msg2 = conversation.guard().await.encrypt_message("B").await.unwrap();
449
450            let conversation = commit_guard.finish();
451            // Since Alice missed Bob's commit she should buffer this message
452            let decrypt = conversation.guard().await.decrypt_message(msg1).await;
453            assert!(matches!(
454                decrypt.unwrap_err(),
455                mls::conversation::Error::BufferedFutureMessage { .. }
456            ));
457            let decrypt = conversation.guard().await.decrypt_message(msg2).await;
458            assert!(matches!(
459                decrypt.unwrap_err(),
460                mls::conversation::Error::BufferedFutureMessage { .. }
461            ));
462            assert_eq!(alice.transaction.count_entities().await.pending_messages, 2);
463
464            let conversation = conversation.acting_as(&bob).await.external_join_notify(&alice).await;
465            // Alice should have deleted all her buffered messages
466            assert_eq!(alice.transaction.count_entities().await.pending_messages, 0);
467            assert!(conversation.is_functional_and_contains([&alice, &bob]).await);
468        })
469        .await
470    }
