14#[derive(Debug, serde::Serialize, serde::Deserialize)]
15pub struct CredentialBundle {
16    pub(crate) credential: Credential,
17    pub(crate) signature_key: SignatureKeyPair,
18    pub(crate) created_at: u64,
19}
20
21impl CredentialBundle {
22    pub fn credential(&self) -> &Credential {
23        &self.credential
24    }
25
26    pub(crate) fn signature_key(&self) -> &SignatureKeyPair {
27        &self.signature_key
28    }
29
30    pub fn to_mls_credential_with_key(&self) -> CredentialWithKey {
31        CredentialWithKey {
32            credential: self.credential.clone(),
33            signature_key: self.signature_key.to_public_vec().into(),
34        }
35    }
36}
37
38impl From<CredentialBundle> for CredentialWithKey {
39    fn from(cb: CredentialBundle) -> Self {
40        Self {
41            credential: cb.credential,
42            signature_key: cb.signature_key.public().into(),
43        }
44    }
45}
46
47impl Clone for CredentialBundle {
48    fn clone(&self) -> Self {
49        Self {
50            credential: self.credential.clone(),
51            signature_key: SignatureKeyPair::from_raw(
52                self.signature_key.signature_scheme(),
53                self.signature_key.private().to_vec(),
54                self.signature_key.to_public_vec(),
55            ),
56            created_at: self.created_at,
57        }
58    }
59}
60
61impl Eq for CredentialBundle {}
62impl PartialEq for CredentialBundle {
63    fn eq(&self, other: &Self) -> bool {
64        self.credential.eq(&other.credential)
65            && self.created_at.eq(&other.created_at)
66            && self
67                .signature_key
68                .signature_scheme()
69                .eq(&other.signature_key.signature_scheme())
70            && self.signature_key.public().eq(other.signature_key.public())
71    }
72}
73
74impl Hash for CredentialBundle {
75    fn hash<H: Hasher>(&self, state: &mut H) {
76        self.created_at.hash(state);
77        self.signature_key.signature_scheme().hash(state);
78        self.signature_key.public().hash(state);
79        self.credential().identity().hash(state);
80        match self.credential().mls_credential() {
81            openmls::prelude::MlsCredentialType::X509(cert) => {
82                cert.certificates.hash(state);
83            }
84            openmls::prelude::MlsCredentialType::Basic(_) => {}
85        };
86    }
87}
88
89impl Ord for CredentialBundle {
90    fn cmp(&self, other: &Self) -> Ordering {
91        self.created_at.cmp(&other.created_at)
92    }
93}
94
95impl PartialOrd for CredentialBundle {
96    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
97        Some(self.cmp(other))
98    }
99}

238    async fn should_not_fail_but_degrade_when_certificate_expired(case: TestContext) {
239        if !case.is_x509() {
240            return;
241        }
242        Box::pin(async move {
243            let mut x509_test_chain = X509TestChain::init_empty(case.signature_scheme());
244
245            let expiration_time = core::time::Duration::from_secs(14);
246            let start = web_time::Instant::now();
247
248            let (alice_identifier, _) = x509_test_chain.issue_simple_certificate_bundle("alice", None);
249            let (bob_identifier, _) = x509_test_chain.issue_simple_certificate_bundle("bob", Some(expiration_time));
250            let alice = SessionContext::new_with_identifier(&case, alice_identifier, Some(&x509_test_chain))
251                .await
252                .unwrap();
253            let bob = SessionContext::new_with_identifier(&case, bob_identifier, Some(&x509_test_chain))
254                .await
255                .unwrap();
256
257            let conversation = case.create_conversation([&alice, &bob]).await;
258            // this should work since the certificate is not yet expired
259            assert!(conversation.is_functional_and_contains([&alice, &bob]).await);
260
261            assert_eq!(
262                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
263                E2eiConversationState::Verified
264            );
265
266            let elapsed = start.elapsed();
267            // Give time to the certificate to expire
268            if expiration_time > elapsed {
269                async_std::task::sleep(expiration_time - elapsed + core::time::Duration::from_secs(2)).await;
270            }
271
272            assert!(conversation.is_functional_and_contains([&alice, &bob]).await);
273            assert_eq!(
274                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
275                E2eiConversationState::NotVerified
276            );
277        })
278        .await;
279    }

282    async fn should_not_fail_but_degrade_when_basic_joins(case: TestContext) {
283        if !case.is_x509() {
284            return;
285        }
286        Box::pin(async {
287            let ([alice, bob], [charlie]) = case.sessions_mixed_credential_types().await;
288            let conversation = case.create_conversation([&alice, &bob]).await;
289
290            // this should work since the certificate is not yet expired
291            assert!(conversation.is_functional_and_contains([&alice, &bob]).await);
292            assert_eq!(
293                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
294                E2eiConversationState::Verified
295            );
296            assert_eq!(
297                conversation
298                    .guard_of(&bob)
299                    .await
300                    .e2ei_conversation_state()
301                    .await
302                    .unwrap(),
303                E2eiConversationState::Verified
304            );
305
306            // Charlie is a basic client that tries to join (i.e. emulates guest links in Wire)
307            let conversation = conversation
308                .invite_with_credential_type_notify(MlsCredentialType::Basic, [&charlie])
309                .await;
310
311            assert_eq!(
312                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
313                E2eiConversationState::NotVerified
314            );
315            assert!(conversation.is_functional_and_contains([&alice, &bob, &charlie]).await);
316            assert_eq!(
317                conversation.guard().await.e2ei_conversation_state().await.unwrap(),
318                E2eiConversationState::NotVerified
319            );
320        })
321        .await;
322    }

325    async fn should_fail_when_certificate_not_valid_yet(case: TestContext) {
326        use crate::MlsErrorKind;
327
328        if !case.is_x509() {
329            return;
330        }
331        let x509_test_chain = X509TestChain::init_empty(case.signature_scheme());
332
333        let tomorrow = now_std() + core::time::Duration::from_secs(3600 * 24);
334        let local_ca = x509_test_chain.find_local_intermediate_ca();
335        let alice_cert = {
336            let name = "alice";
337            let common_name = format!("{name} Smith");
338            let handle = format!("{}_wire", name.to_lowercase());
339            let client_id: String = crate::e2e_identity::id::QualifiedE2eiClientId::generate_with_domain("wire.com")
340                .try_into()
341                .unwrap();
342            local_ca.create_and_sign_end_identity(CertificateParams {
343                common_name: Some(common_name.clone()),
344                handle: Some(handle.clone()),
345                client_id: Some(client_id.clone()),
346                validity_start: Some(tomorrow),
347                ..Default::default()
348            })
349        };
350        let cb = CertificateBundle::from_certificate_and_issuer(&alice_cert, local_ca);
351        let alice_identifier = ClientIdentifier::X509(HashMap::from([(case.signature_scheme(), cb)]));
352        let err = SessionContext::new_with_identifier(&case, alice_identifier, Some(&x509_test_chain))
353            .await
354            .unwrap_err();
355
356        assert!(innermost_source_matches!(
357            err,
358            MlsErrorKind::MlsCryptoError(openmls::prelude::CryptoError::ExpiredCertificate),
359        ))
360    }
