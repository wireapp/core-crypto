8/// Entities implementing `SearchableEntity` have a distinct search key which
9/// can produce multiple items.
10///
11/// Effectively, this is a way at the type-system level to implement `WHERE`-clause
12/// searching.
13///
14/// This trait can potentially be implemented multiple times per entity, in case there are
15/// a variety of interesting searches.
16///
17/// While the trait design does not require it, implementations should take advantage of
18/// database features such as indices to ensure that searching by a search key is efficient.
19#[cfg_attr(target_family = "wasm", async_trait(?Send))]
20#[cfg_attr(not(target_family = "wasm"), async_trait)]
21pub trait SearchableEntity<SearchKey: KeyType>: Entity {
22    /// Find all entities matching the search key.
23    ///
24    /// The specific meaning of "matching" the search key will depend on the entity in question,
25    /// but generally the search key will have one or more fields which effectively act
26    /// as a `WHERE`-clause for the search.
27    async fn find_all_matching(
28        conn: &mut Self::ConnectionType,
29        search_key: &SearchKey,
30    ) -> CryptoKeystoreResult<Vec<Self>>;
31}
