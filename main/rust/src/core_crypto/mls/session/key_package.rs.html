25fn from_stored(stored_keypackage: &StoredKeypackage) -> Result<Keypackage> {
26    core_crypto_keystore::deser::<Keypackage>(&stored_keypackage.keypackage)
27        .map_err(KeystoreError::wrap("deserializing keypackage"))
28        .map_err(Into::into)
29}

33    async fn credential_from_ref(&self, credential_ref: &CredentialRef) -> Result<Arc<Credential>> {
34        let signature_public_key = credential_ref.public_key().into();
35        self.find_credential_by_public_key(&signature_public_key).await
36    }

38    /// Generate a [Keypackage] from the referenced credential.
39    ///
40    /// Makes no attempt to look up or prune existing keypackges.
41    ///
42    /// If `lifetime` is set, the keypackages will expire that span into the future.
43    /// If it is unset, [`KEYPACKAGE_DEFAULT_LIFETIME`] is used.
44    ///
45    /// As a side effect, stores the keypackages and some related data in the keystore.
46    ///
47    /// Must not be fully public, only crate-public, because as it mutates the keystore it must only ever happen within
48    /// a transaction.
49    pub(crate) async fn generate_keypackage(
50        &self,
51        credential_ref: &CredentialRef,
52        lifetime: Option<Duration>,
53    ) -> Result<Keypackage> {
54        let lifetime = Lifetime::new(lifetime.unwrap_or(KEYPACKAGE_DEFAULT_LIFETIME).as_secs());
55        let credential = self.credential_from_ref(credential_ref).await?;
56
57        let config = CryptoConfig {
58            ciphersuite: credential.ciphersuite.into(),
59            version: openmls::versions::ProtocolVersion::default(),
60        };
61
62        Keypackage::builder()
63            .leaf_node_capabilities(MlsConversationConfiguration::default_leaf_capabilities())
64            .key_package_lifetime(lifetime)
65            .build(
66                config,
67                &self.crypto_provider,
68                &credential.signature_key_pair,
69                credential.to_mls_credential_with_key(),
70            )
71            .await
72            .map_err(Error::keypackage_new())
73    }

75    /// Get all [`Keypackage`]s in the database.
76    pub(crate) async fn get_keypackages(&self) -> Result<Vec<Keypackage>> {
77        let stored_keypackages: Vec<StoredKeypackage> = self
78            .crypto_provider
79            .keystore()
80            .load_all()
81            .await
82            .map_err(KeystoreError::wrap("finding all keypackages"))?;
83
84        let keypackages = stored_keypackages
85            .iter()
86            .map(from_stored)
87            // if any ref from loading all fails to load now, skip it
88            // strictly we could panic, but this is safer--maybe someone removed it concurrently
89            .filter_map(|kp| kp.ok())
90            .collect();
91
92        Ok(keypackages)
93    }
94
95    /// Get all [`KeypackageRef`]s in the database.
96    pub async fn get_keypackage_refs(&self) -> Result<Vec<KeypackageRef>> {
97        self.get_keypackages()
98            .await?
99            .iter()
100            .map(|keypackage| keypackage.make_ref().map_err(Into::into))
101            .collect()
102    }
103
104    /// Load one [`Keypackage`] from its [`KeypackageRef`]
105    pub(crate) async fn load_keypackage(&self, kp_ref: &KeypackageRef) -> Result<Option<Keypackage>> {
106        self.crypto_provider
107            .keystore()
108            .get_borrowed::<StoredKeypackage>(kp_ref.hash_ref())
109            .await
110            .map_err(KeystoreError::wrap("loading keypackage from database"))?
111            .map(|stored_keypackage| from_stored(&stored_keypackage))
112            .transpose()
113    }
114
115    /// Remove one [`Keypackage`] from the database.
116    ///
117    /// Succeeds silently if the keypackage does not exist in the database.
118    ///
119    /// Implementation note: this must first load and deserialize the keypackage,
120    /// then remove items from three distinct tables.
121    pub(crate) async fn remove_keypackage(&self, kp_ref: &KeypackageRef) -> Result<()> {
122        let Some(kp) = self.load_keypackage(kp_ref).await? else {
123            return Ok(());
124        };
125
126        let db = self.crypto_provider.keystore();
127        db.remove_borrowed::<StoredKeypackage>(kp_ref.hash_ref())
128            .await
129            .map_err(KeystoreError::wrap("removing key package from keystore"))?;
130        db.remove_borrowed::<StoredHpkePrivateKey>(kp.hpke_init_key().as_slice())
131            .await
132            .map_err(KeystoreError::wrap("removing private key from keystore"))?;
133        db.remove_borrowed::<StoredEncryptionKeyPair>(kp.leaf_node().encryption_key().as_slice())
134            .await
135            .map_err(KeystoreError::wrap("removing encryption keypair from keystore"))?;
136
137        Ok(())
138    }

140    /// Remove all keypackages associated with this credential.
141    ///
142    /// This is fairly expensive as it must first load all keypackages, then delete those matching the credential.
143    ///
144    /// Implementation note: once it makes it as far as having a list of keypackages, does _not_ short-circuit
145    /// if removing one returns an error. In that case, only the first produced error is returned.
146    /// This helps ensure that as many keypackages for the given credential ref are removed as possible.
147    pub(crate) async fn remove_keypackages_for(&self, credential_ref: &CredentialRef) -> Result<()> {
148        let credential = self.credential_from_ref(credential_ref).await?;
149        let signature_public_key = credential.signature_key_pair.public();
150
151        let mut first_err = None;
152        macro_rules! try_retain_err {
153            ($e:expr) => {
154                match $e {
155                    Err(err) => {
156                        if first_err.is_none() {
157                            first_err = Some(Error::from(err));
158                        }
159                        continue;
160                    }
161                    Ok(val) => val,
162                }
163            };
164        }
165
166        for keypackage in self
167            .get_keypackages()
168            .await?
169            .into_iter()
170            .filter(|keypackage| keypackage.leaf_node().signature_key().as_slice() == signature_public_key)
171        {
172            let kp_ref = try_retain_err!(keypackage.make_ref());
173            try_retain_err!(self.remove_keypackage(&kp_ref).await);
174        }
175
176        match first_err {
177            None => Ok(()),
178            Some(err) => Err(err),
179        }
180    }

201        // 90-day standard expiration
202        let kp_std_exp = session.new_keypackage(&case).await;
203        assert!(kp_std_exp.is_valid());

216    async fn requesting_x509_key_packages_after_basic(case: TestContext) {
217        // Basic test case
218        if !case.is_basic() {
219            return;
220        }
221
222        let [session_context] = case.sessions_basic_with_pki_env().await;
223        Box::pin(async move {
224            // Generate 5 Basic key packages first
225            let mut initial_kp_refs = Vec::new();
226            for _ in 0..5 {
227                let kp = session_context.new_keypackage(&case).await;
228                initial_kp_refs.push(kp.make_ref().unwrap());
229            }
230            initial_kp_refs.sort_by_key(|kp_ref| kp_ref.hash_ref().to_owned());
231
232            // Set up E2E identity
233            let test_chain = session_context.x509_chain_unchecked();
234
235            let (mut enrollment, cert_chain) = e2ei_enrollment(
236                &session_context.transaction,
237                &case,
238                test_chain,
239                &session_context.get_e2ei_client_id().await.to_uri(),
240                false,
241                init_activation_or_rotation,
242                noop_restore,
243            )
244            .await
245            .unwrap();
246
247            let _rotate_bundle = session_context
248                .transaction
249                .save_x509_credential(&mut enrollment, cert_chain)
250                .await
251                .unwrap();
252
253            // E2E identity has been set up correctly
254            assert!(
255                session_context
256                    .transaction
257                    .e2ei_is_enabled(case.ciphersuite())
258                    .await
259                    .unwrap()
260            );
261
262            // Request X509 key packages
263            let key_packages = session_context.transaction.get_keypackage_refs().await.unwrap();
264            let (mut from_initial_set, x509_key_packages) = key_packages
265                .into_iter()
266                .partition::<Vec<_>, _>(|kp_ref| initial_kp_refs.contains(kp_ref));
267
268            from_initial_set.sort_by_key(|kp_ref| kp_ref.hash_ref().to_owned());
269            assert_eq!(initial_kp_refs, from_initial_set);
270
271            // Verify that the key packages are X509
272            assert!(
273                x509_key_packages
274                    .iter()
275                    .all(|kp| CredentialType::X509 == kp.credential_type())
276            );
277        })
278        .await
279    }

281    #[apply(all_cred_cipher)]
282    async fn new_keypackage_has_correct_extensions(case: TestContext) {
283        let [cc] = case.sessions().await;
284        Box::pin(async move {
285            let kp = cc.new_keypackage(&case).await;
286
287            // make sure it's valid
288            let _ = KeyPackageIn::from(kp.clone())
289                .standalone_validate(
290                    &cc.transaction.mls_provider().await.unwrap(),
291                    ProtocolVersion::Mls10,
292                    true,
293                )
294                .await
295                .unwrap();
296
297            // see https://www.rfc-editor.org/rfc/rfc9420.html#section-10-10
298            assert!(kp.extensions().is_empty());
299
300            assert_eq!(kp.leaf_node().capabilities().versions(), &[ProtocolVersion::Mls10]);
301            assert_eq!(
302                kp.leaf_node().capabilities().ciphersuites().to_vec(),
303                MlsConversationConfiguration::DEFAULT_SUPPORTED_CIPHERSUITES
304                    .iter()
305                    .map(|c| VerifiableCiphersuite::from(*c))
306                    .collect::<Vec<_>>()
307            );
308            assert!(kp.leaf_node().capabilities().proposals().is_empty());
309            assert!(kp.leaf_node().capabilities().extensions().is_empty());
310            assert_eq!(
311                kp.leaf_node().capabilities().credentials(),
312                MlsConversationConfiguration::DEFAULT_SUPPORTED_CREDENTIALS
313            );
314        })
315        .await
316    }
