1//! Implement the MLS Transport interface.
2//!
3//! Unfortunately, there is no good way to reuse code between uniffi and wasm for an interface
4//! like this; the fundamental techniques in use are very different. So we just have to feature-gate
5//! everything.
6use std::{fmt, sync::Arc};
7
8use core_crypto::{HistorySecret, MlsCommitBundle};
9
10use crate::{ClientId, CommitBundle, HistorySecret as HistorySecretFfi};

12/// MLS transport may or may not succeeed; this response indicates to CC the outcome of the transport attempt.
13#[derive(Debug, Clone, PartialEq, Eq, uniffi::Enum)]
14pub enum MlsTransportResponse {
15    /// The message was accepted by the distribution service
16    Success,
17    /// A client should have consumed all incoming messages before re-trying.
18    Retry,
19    /// The message was rejected by the delivery service and there's no recovery.
20    Abort {
21        /// Why was this message rejected
22        reason: String,
23    },
24}
25
26/// An entity / data which has been packaged by the application to be encrypted
27/// and transmitted in an application message.
28//
29// TODO: We derive Constructor here only because we need to construct an instance in interop.
30// Remove it once we drop the FFI client from interop.
31#[derive(Debug, derive_more::From, derive_more::Into, derive_more::Deref, derive_more::Constructor)]
32pub struct MlsTransportData(core_crypto::MlsTransportData);
33
34uniffi::custom_type!(MlsTransportData, Vec<u8>, {
35    lower: |key| key.0.to_vec(),
36    try_lift: |vec| {
37        Ok(MlsTransportData(core_crypto::MlsTransportData::from(vec)))
38    }
39});
40
41impl From<MlsTransportResponse> for core_crypto::MlsTransportResponse {
42    fn from(value: MlsTransportResponse) -> Self {
43        match value {
44            MlsTransportResponse::Success => Self::Success,
45            MlsTransportResponse::Retry => Self::Retry,
46            MlsTransportResponse::Abort { reason } => Self::Abort { reason },
47        }
48    }
49}

51/// Used by core crypto to send commits or application messages to the delivery service.
52/// This trait must be implemented before calling any functions that produce commits.
53#[uniffi::export(with_foreign)]
54#[cfg_attr(target_family = "wasm", async_trait::async_trait(?Send))]
55#[cfg_attr(not(target_family = "wasm"), async_trait::async_trait)]
56pub trait MlsTransport: Send + Sync {
57    /// Send a commit bundle to the corresponding endpoint.
58    async fn send_commit_bundle(&self, commit_bundle: CommitBundle) -> MlsTransportResponse;
59    /// Send a message to the corresponding endpoint.
60    async fn send_message(&self, mls_message: Vec<u8>) -> MlsTransportResponse;
61    /// Prepare a history secret before being sent
62    async fn prepare_for_transport(&self, history_secret: HistorySecretFfi) -> MlsTransportData;
63}
64
65#[derive(derive_more::Constructor)]
66struct MlsTransportShim(Arc<dyn MlsTransport>);
67
68impl std::fmt::Debug for MlsTransportShim {
69    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> std::fmt::Result {
70        f.debug_tuple("MlsTransportShim")
71            .field(&"Arc<dyn MlsTransport>")
72            .finish()
73    }
74}
75
76#[cfg_attr(target_family = "wasm", async_trait::async_trait(?Send))]
77#[cfg_attr(not(target_family = "wasm"), async_trait::async_trait)]
78impl core_crypto::MlsTransport for MlsTransportShim {
79    async fn send_commit_bundle(
80        &self,
81        commit_bundle: MlsCommitBundle,
82    ) -> core_crypto::Result<core_crypto::MlsTransportResponse> {
83        let commit_bundle = CommitBundle::try_from(commit_bundle)
84            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
85        Ok(self.0.send_commit_bundle(commit_bundle).await.into())
86    }
87
88    async fn send_message(&self, mls_message: Vec<u8>) -> core_crypto::Result<core_crypto::MlsTransportResponse> {
89        Ok(self.0.send_message(mls_message).await.into())
90    }
91
92    async fn prepare_for_transport(
93        &self,
94        secret: &HistorySecret,
95    ) -> core_crypto::Result<core_crypto::MlsTransportData> {
96        let client_id = ClientId::from(secret.client_id.clone()).into();
97        let history_secret = rmp_serde::to_vec(&secret)
98            .map(|secret| HistorySecretFfi {
99                client_id,
100                data: secret,
101            })
102            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
103        Ok(self.0.prepare_for_transport(history_secret).await.into())
104    }
105}

107/// In uniffi, `MlsTransport` is a trait which we need to wrap
108pub(crate) fn callback_shim(callbacks: Arc<dyn MlsTransport>) -> Arc<dyn core_crypto::MlsTransport> {
109    Arc::new(MlsTransportShim::new(callbacks))
