17/// Implements the `Entity` trait and related traits for the given struct.
18///
19/// Intended to be used only within `core-crypto-keystore`.
20///
21/// All fields in an entity using this derive macro must have one of three types:
22///
23/// - `Vec<u8>`
24/// - `String`
25/// - `Option<Vec<u8>>`
26///
27/// There is one exception: the identity field can only be a `Vec<u8>` or `String`; the optional
28/// variant is forbidden in that context.
29///
30/// This is intentionally restrictive. If your entity has other field types, implement these traits manually.
31///
32/// ## Attributes
33///
34/// Attributes modify the code generated by this proc macro by adding annotations to the whole struct
35/// or to distinct fields. All attributes accepted by this proc macro are keyed with `#[entity]`.
36/// The attribute itself appears in the following parenthesis.
37///
38/// It is acceptable to combine multiple entity attributes together, simply by combining them
39/// within the parenthesis, separated by commas: `#[entity(id, hex)]`, for example.
40///
41/// ### `#[entity(collection_name = "my_collection")]`
42///
43/// This is an **outer attribute** which modifies the entire item. Its parameters:
44///
45/// - `collection_name`: Optional. When set, this specifies the collection name which will be used. Otherwise, one will
46///   be generated by transforming the item name to snake case and adding an `s`.
47///
48/// ### `#[entity(id)]`
49///
50/// This is a **field attribute** which can be applied to at most 1 field in the item.
51///
52/// The field to which this attribute is applied must have type `Vec<u8>` or `String`.
53/// Otherwise, you must implement all relevant entity traits manually instead of relying on this macro.
54///
55/// If this attribute does not appear, the macro will search for a field whose name is `id` and assume
56/// that is an implicit id.
57///
58/// This attribute does not accept a value.
59///
60/// ### `#[entity(hex)]`
61///
62/// This is a **field attribute** which can be applied to any number of fields in the struct.
63///
64/// When set, this indicates that the actual data in the database is hex-encoded.
65/// The encoding and decoding will be applied transparently when storing or loading this field.
66///
67/// This may only be applied to fields of type `Vec<u8>`.
68///
69/// This attribute does not accept a value.
70///
71/// **NOTE**: this is mainly intended for backwards compatibility. It is not recommended that new fields be
72/// hex-encoded; doing so provides no safety but adds slight serialization costs on store/load.
73///
74/// ### `#[entity(column = "my_column")]`
75///
76/// This is a **field attribute** which can be applied to any number of fields in the struct.
77///
78/// The `column` value will be used as the SQL column name for this field. If this is unset, the field's name
79/// will be used as the sql column name.
80///
81/// ## Trait Implementations
82///
83/// ### Implements
84///
85/// - `EntityBase`
86/// - `Entity`
87/// - `BorrowPrimaryKey`
88/// - `EntityDatabaseMutation`
89/// - `EntityDeleteBorrowed`
90/// - `Encrypting` (+ relevant associated struct)
91/// - `Decrypting` (+ relevant associated struct)
92/// - `Decryptable`
93///
94/// ### `EntityBase`
95///
96/// This trait effectively stores only the collection name, and generates the transaction entity based on the item name.
97///
98/// No fields are autogenerated. If any fields should be autogenerated, you must implement all relevant traits manually.
99///
100/// ### `Entity`
101///
102/// The primary key for all derived entities is `Vec<u8>`. For a more complicated primary key, you need to implement
103/// all these traits manually.
104///
105/// Implementations of non-wasm methods defer to the helper functions defined in `entities::platform::generic`.
106///
107/// ### `BorrowPrimaryKey`
108///
109/// The borrowed form of the primary key for all derived entities is `&[u8]`. For a more complicated borrowed primary
110/// key, you need to implement all these traits manually.
111///
112/// Implementation of `get_borrowed` defers to the helper function defined in `entities::platform::generic`.
113///
114/// ### `EntityDatabaseMutation`
115///
116/// The derived `save` method (non-wasm) generates sql like:
117///
118/// ```sql
119/// INSERT OR REPLACE INTO entities (id, foo, bar) VALUES (?, ?, ?)
120/// ```
121///
122/// The wasm implementation delegates to the transaction's `save` method, which serializes via serde.
123///
124/// ### Item-level encryption (i.e. `Encrypting`, `Decrypting`)
125///
126/// All fields except the id field are encrypted. If you need more unencrypted fields, implement this trait family
127/// yourself.
128///
129/// This macro generates an encryption helper associated struct. If this item is `Foo`, the associated struct is
130/// `FooEncrypt`. `FooEncrypt` implements `serde::Serialize`. Expected usage:
131///
132/// ```rust,ignore
133/// let encrypted = serde_json::to_string(&foo.encrypt(cipher)?)?;
134/// ```
135///
136/// This macro generates a decryption helper associated struct. If this item is `Foo`, the associated struct is
137/// `FooDecrypt`. `FooDecrypt` implements `serde::Deserialize` and `Decrypting`, with `FooDecrypt::DecryptedForm:
138/// Foo`. `Foo` implements `Decryptable`, with `Decryptable::DecryptableFrom: FooDecrypt`. Expected usage:
139///
140/// ```rust,ignore
141/// let decrypted = serde_json::from_str::<Foo::DecryptableFrom>(&encrypted)?.decrypt(cipher)?;
142/// ```
143#[proc_macro_derive(Entity, attributes(entity))]
144pub fn derive_entity(input: TokenStream) -> TokenStream {
145    let input = parse_macro_input!(input as DeriveInput);
146    match entity_derive::Entity::from_derive_input(&input) {
147        Ok(entity) => quote! { #entity },
148        Err(err) => err.write_errors(),
149    }
150    .into()
151}

153/// Will drop current MLS group in memory and replace it with the one in the keystore.
154/// This simulates an application crash. Once restarted, everything has to be loaded from the
155/// keystore, memory is lost.
156///
157/// Requires the `MlsConversation` method to have a parameter exactly like `backend: &MlsCryptoProvider`
158///
159/// This helps spotting:
160/// * when one has forgotten to call `persist_group_when_changed`
161/// * if persisted fields are sufficient to pursue normally after a crash
162///
163/// **IF** you mark a method `#[durable]`, remove its call to
164/// `persist_group_when_changed` and tests still pass, you either:
165/// * have unit tests not covering the method enough
166/// * do not require this method to be durable
167#[proc_macro_attribute]
168pub fn durable(_args: TokenStream, item: TokenStream) -> TokenStream {
169    durable::durable(item)
170}

172/// !!! Not literally idempotent !!!
173///
174/// Marker for methods on 'core_crypto::Client' which leave the number of entities in the keystore even.
175/// They can create/destroy some but always compensate.
176/// So they are not idempotent, they cannot be safely replayed and they might leave the keystore in
177/// a different state.
178#[proc_macro_attribute]
179pub fn idempotent(_args: TokenStream, item: TokenStream) -> TokenStream {
180    idempotent::idempotent(item)
181}
182
183/// Neologism to mean the opposite of idempotent. Methods of 'core_crypto::Client' marked with this have to
184/// insert/delete an entity in the keystore and change the number of entities persisted.
185#[proc_macro_attribute]
186pub fn dispotent(_args: TokenStream, item: TokenStream) -> TokenStream {
187    idempotent::dispotent(item)
188}

190pub(crate) fn doc_attributes(ast: &ItemFn) -> Vec<Attribute> {
191    ast.attrs
192        .iter()
193        .filter(|attr| attr.path().is_ident("doc"))
194        .cloned()
195        .collect::<Vec<syn::Attribute>>()
196}

198pub(crate) fn compile_error(mut item: TokenStream, err: syn::Error) -> TokenStream {
199    let compile_err = TokenStream::from(err.to_compile_error());
200    item.extend(compile_err);
201    item
202}

204#[allow(clippy::type_complexity)]
205pub(crate) fn items(
206    ast: &ItemFn,
207) -> (
208    &ReturnType,
209    &Ident,
210    &Punctuated<FnArg, Comma>,
211    &Box<Block>,
212    &Vec<Attribute>,
213    &Visibility,
214) {
215    let ret = &ast.sig.output;
216    let name = &ast.sig.ident;
217    let inputs = &ast.sig.inputs;
218    let body = &ast.block;
219    let attrs = &ast.attrs;
220    let vis = &ast.vis;
221    (ret, name, inputs, body, attrs, vis)
222}
