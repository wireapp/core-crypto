1//! A MLS group can be merged (aka committed) when it has a pending commit. The latter is a commit
2//! we created which is still waiting to be "committed". By doing so, we will apply all the
3//! modifications present in the commit to the ratchet tree and also persist the new group in the
4//! keystore. Like this, even if the application crashes we will be able to restore.
5//!
6//! This table summarizes when a MLS group can be merged:
7//!
8//! | can be merged ?   | 0 pend. Commit | 1 pend. Commit |
9//! |-------------------|----------------|----------------|
10//! | 0 pend. Proposal  | ❌              | ✅              |
11//! | 1+ pend. Proposal | ❌              | ✅              |
12
13use core_crypto_keystore::{Database, entities::StoredEncryptionKeyPair};
14use openmls_traits::OpenMlsCryptoProvider;
15
16use super::Result;
17use crate::{MlsError, Session, mls::MlsConversation, mls_provider::MlsCryptoProvider};
18
19/// Abstraction over a MLS group capable of merging a commit
20impl MlsConversation {
21    pub(crate) async fn commit_accepted(
22        &mut self,
23        session: &Session<Database>,
24        backend: &MlsCryptoProvider,
25    ) -> Result<()> {
26        // openmls stores here all the encryption keypairs used for update proposals..
27        let previous_own_leaf_nodes = self.group.own_leaf_nodes.clone();
28
29        self.group
30            .merge_pending_commit(backend)
31            .await
32            .map_err(MlsError::wrap("merging pending commit"))?;
33
34        // ..so if there's any, we clear them after the commit is merged
35        for oln in &previous_own_leaf_nodes {
36            let ek = oln.encryption_key().as_slice();
37            let _ = backend.key_store().remove_borrowed::<StoredEncryptionKeyPair>(ek).await;
38        }
39
40        session
41            .notify_epoch_changed(self.id.clone(), self.group.epoch().as_u64())
42            .await;
43
44        Ok(())
45    }
46}

48#[cfg(test)]
49mod tests {
50    use crate::test_utils::*;
51
52    mod commit_accepted {
53        use super::*;
54
55        #[apply(all_cred_cipher)]
56        async fn should_apply_pending_commit(case: TestContext) {
57            let [alice, bob] = case.sessions().await;
58            Box::pin(async move {
59                let conversation = case.create_conversation([&alice, &bob]).await;
60                let commit_guard = conversation.update_unmerged().await.notify_member(&bob).await;
61                let conversation = commit_guard.conversation();
62
63                assert!(conversation.has_pending_commit().await);
64
65                let session = alice.session().await;
66
67                conversation
68                    .guard()
69                    .await
70                    .conversation_mut(async |conversation| {
71                        conversation.commit_accepted(&session, &session.crypto_provider).await
72                    })
73                    .await
74                    .unwrap();
75
76                assert_eq!(conversation.member_count().await, 2);
77                assert!(conversation.is_functional_and_contains([&alice, &bob]).await);
78            })
79            .await
80        }
81
82        #[apply(all_cred_cipher)]
83        async fn should_clear_pending_commit_and_proposals(case: TestContext) {
84            let [alice, bob] = case.sessions().await;
85            Box::pin(async move {
86                let commit = case
87                    .create_conversation([&alice, &bob])
88                    .await
89                    .remove_proposal_notify(&bob)
90                    .await
91                    .update_unmerged()
92                    .await;
93
94                let conversation = commit.conversation();
95
96                assert!(conversation.has_pending_proposals().await);
97                assert!(conversation.has_pending_commit().await);
98
99                let session = alice.session().await;
100
101                conversation
102                    .guard()
103                    .await
104                    .conversation_mut(async |conversation| {
105                        conversation.commit_accepted(&session, &session.crypto_provider).await
106                    })
107                    .await
108                    .unwrap();
109                assert!(!conversation.has_pending_proposals().await);
110                assert!(!conversation.has_pending_commit().await);
111            })
112            .await
113        }
114    }
