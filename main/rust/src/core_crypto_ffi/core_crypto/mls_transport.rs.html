25/// MLS transport may or may not succeeed; this response indicates to CC the outcome of the transport attempt.
26#[cfg(not(target_family = "wasm"))]
27#[derive(Debug, Clone, PartialEq, Eq, uniffi::Enum)]
28pub enum MlsTransportResponse {
29    /// The message was accepted by the distribution service
30    Success,
31    /// A client should have consumed all incoming messages before re-trying.
32    Retry,
33    /// The message was rejected by the delivery service and there's no recovery.
34    Abort {
35        /// Why was this message rejected
36        reason: String,
37    },
38}

79/// Used by core crypto to send commits or application messages to the delivery service.
80/// This trait must be implemented before calling any functions that produce commits.
81#[cfg(not(target_family = "wasm"))]
82#[uniffi::export(with_foreign)]
83#[async_trait::async_trait]
84pub trait MlsTransport: Send + Sync {
85    /// Send a commit bundle to the corresponding endpoint.
86    async fn send_commit_bundle(&self, commit_bundle: CommitBundle) -> MlsTransportResponse;
87    /// Send a message to the corresponding endpoint.
88    async fn send_message(&self, mls_message: Vec<u8>) -> MlsTransportResponse;
89    /// Prepare a history secret before being sent
90    async fn prepare_for_transport(&self, history_secret: HistorySecretFfi) -> MlsTransportData;
91}

106#[cfg(not(target_family = "wasm"))]
107#[async_trait::async_trait]
108impl core_crypto::prelude::MlsTransport for MlsTransportShim {
109    async fn send_commit_bundle(
110        &self,
111        commit_bundle: MlsCommitBundle,
112    ) -> core_crypto::Result<core_crypto::MlsTransportResponse> {
113        let commit_bundle = CommitBundle::try_from(commit_bundle)
114            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
115        Ok(self.0.send_commit_bundle(commit_bundle).await.into())
116    }
117
118    async fn send_message(&self, mls_message: Vec<u8>) -> core_crypto::Result<core_crypto::MlsTransportResponse> {
119        Ok(self.0.send_message(mls_message).await.into())
120    }
121
122    async fn prepare_for_transport(
123        &self,
124        secret: &HistorySecret,
125    ) -> core_crypto::Result<core_crypto::MlsTransportData> {
126        let client_id = ClientId::from_cc(secret.client_id.clone());
127        let history_secret = rmp_serde::to_vec(&secret)
128            .map(|secret| HistorySecretFfi {
129                client_id,
130                data: secret,
131            })
132            .map_err(|e| core_crypto::Error::ErrorDuringMlsTransport(e.to_string()))?;
133        Ok(self.0.prepare_for_transport(history_secret).await.into())
134    }
135}

166        })

206/// Used by core crypto to send commits or application messages to the delivery service.
207/// This must be instantiated and set before calling any functions which produce commits.
208#[cfg(target_family = "wasm")]
209#[derive(Debug)]
210#[wasm_bindgen]
211pub struct MlsTransport {
212    this_context: JsValue,
213    send_commit_bundle: js_sys::Function,
214    send_message: js_sys::Function,
215    prepare_for_transport: js_sys::Function,
216}

219// SAFETY: we promise that we're only ever using this in a single-threaded context
220unsafe impl Send for MlsTransport {}
221#[cfg(target_family = "wasm")]
222// SAFETY: we promise that we're only ever using this in a single-threaded context
223unsafe impl Sync for MlsTransport {}

225#[cfg(target_family = "wasm")]
226#[wasm_bindgen]
227impl MlsTransport {
228    /// Create a new MLS Transport instance.
229    ///
230    /// This function should be hidden on the JS side of things! The JS bindings should have an `interface MlsTransport`
231    /// which has the two methods defined, and the bindings themselves should destructure an instance implementing that
232    /// interface appropriately to construct this.
233    ///
234    /// - `this_context` is the instance itself, which will be bound to `this` within the function bodies
235    /// - `send_commit_bundle`: A function of the form `(CommitBundle) -> Promise<MlsTransportResponse>`.
236    ///   Sends a commit bundle to the corresponding endpoint.
237    /// - `send_message`: A function of the form `(Uint8Array) -> Promise<MlsTransportResponse>`
238    ///   Sends a message to the corresponding endpoint.
239    /// - `prepare_for_transport`: A function of the form `(HistorySecret) -> Promise<MlsTransportData>`
240    ///   Prepare a history secret to be sent over the transport.
241    #[wasm_bindgen(constructor)]
242    pub fn new(
243        this_context: JsValue,
244        send_commit_bundle: js_sys::Function,
245        send_message: js_sys::Function,
246        prepare_for_transport: js_sys::Function,
247    ) -> CoreCryptoResult<Self> {
248        // we can't do much type-checking here unfortunately, but we can at least validate that the incoming functions have the right length
249        if send_commit_bundle.length() != 1 {
250            return Err(CoreCryptoError::ad_hoc(format!(
251                "`send_commit_bundle` must accept 1 argument but accepts {}",
252                send_commit_bundle.length()
253            )));
254        }
255        if send_message.length() != 1 {
256            return Err(CoreCryptoError::ad_hoc(format!(
257                "`send_message` must accept 1 argument but accepts {}",
258                send_message.length()
259            )));
260        }
261        if prepare_for_transport.length() != 1 {
262            return Err(CoreCryptoError::ad_hoc(format!(
263                "`prepare_for_transport` must accept 1 argument but accepts {}",
264                prepare_for_transport.length()
265            )));
266        }
267        Ok(Self {
268            this_context,
269            send_commit_bundle,
270            send_message,
271            prepare_for_transport,
272        })
273    }
274}

276#[cfg(target_family = "wasm")]
277#[async_trait::async_trait(?Send)]
278impl core_crypto::MlsTransport for MlsTransport {
279    /// Send a commit bundle to the corresponding endpoint.
280    async fn send_commit_bundle(
281        &self,
282        commit_bundle: MlsCommitBundle,
283    ) -> core_crypto::Result<core_crypto::MlsTransportResponse> {
284        let commit_bundle = JsValue::from(CommitBundle::try_from(commit_bundle).map_err(|err| {
285            core_crypto::Error::ErrorDuringMlsTransport(format!("converting commit bundle to wasm-compatible: {err}"))
286        })?);
287
288        let promise = self
289            .send_commit_bundle
290            .call1(&self.this_context, &commit_bundle)
291            .map_err(|js_err| {
292                core_crypto::Error::ErrorDuringMlsTransport(format!(
293                    "send_commit_bundle received error from js when constructing promise: {js_err:?}"
294                ))
295            })?
296            .dyn_into::<Promise>()
297            .map_err(|not_promise| {
298                core_crypto::Error::ErrorDuringMlsTransport(format!(
299                    "send_commit_bundle received a value that was not a promise: {not_promise:?}"
300                ))
301            })?;
302
303        let response = JsFuture::from(promise).await.map_err(|err| {
304            core_crypto::Error::ErrorDuringMlsTransport(format!(
305                "send_commit_bundle received error from js when executing promise: {err:?}"
306            ))
307        })?;
308        let response = serde_wasm_bindgen::from_value::<MlsTransportResponse>(response)
309            .map_err(|not_transport_response| {
310                core_crypto::Error::ErrorDuringMlsTransport(format!(
311                    "send_commit_bundle received a value which was not an MlsTransportResponse after awaiting js promise: {not_transport_response:?}"
312                ))
313            })?;
314
315        Ok(response.into())
316    }
317
318    /// Send a message to the corresponding endpoint.
319    async fn send_message(&self, mls_message: Vec<u8>) -> core_crypto::Result<core_crypto::MlsTransportResponse> {
320        let mls_message = Uint8Array::from(mls_message.as_slice());
321
322        let promise = self
323            .send_message
324            .call1(&self.this_context, &mls_message)
325            .map_err(|err| {
326                core_crypto::Error::ErrorDuringMlsTransport(format!(
327                    "send_message received error from js when constructing promise: {err:?}"
328                ))
329            })?
330            .dyn_into::<Promise>()
331            .map_err(|not_promise| {
332                core_crypto::Error::ErrorDuringMlsTransport(format!(
333                    "send_message received a value that was not a promise: {not_promise:?}"
334                ))
335            })?;
336
337        let response = JsFuture::from(promise).await.map_err(|err| {
338            core_crypto::Error::ErrorDuringMlsTransport(format!(
339                "send_message received error from js when executing promise: {err:?}"
340            ))
341        })?;
342        let response = serde_wasm_bindgen::from_value::<MlsTransportResponse>(response)
343            .map_err(|not_transport_response| {
344                core_crypto::Error::ErrorDuringMlsTransport(format!(
345                    "send_message received a value which was not an MlsTransportResponse after awaiting js promise: {not_transport_response:?}"
346                ))
347            })?;
348
349        Ok(response.into())
350    }
351
352    /// prepare a history secret to be sent via the mls transport
353    async fn prepare_for_transport(
354        &self,
355        secret: &HistorySecret,
356    ) -> core_crypto::Result<core_crypto::MlsTransportData> {
357        let history_secret = JsValue::from(HistorySecretFfi::try_from(secret).map_err(|err| {
358            core_crypto::Error::ErrorDuringMlsTransport(format!("converting history secret to wasm-compatible: {err}"))
359        })?);
360
361        let promise = self
362            .prepare_for_transport
363            .call1(&self.this_context, &history_secret)
364            .map_err(|err| {
365                core_crypto::Error::ErrorDuringMlsTransport(format!(
366                    "prepare_for_transport received error from js when constructing promise: {err:?}"
367                ))
368            })?
369            .dyn_into::<Promise>()
370            .map_err(|not_promise| {
371                core_crypto::Error::ErrorDuringMlsTransport(format!(
372                    "prepare_for_transport received a value that was not a promise: {not_promise:?}"
373                ))
374            })?;
375
376        let response = JsFuture::from(promise).await.map_err(|err| {
377            core_crypto::Error::ErrorDuringMlsTransport(format!(
378                "prepare_for_transport received error from js when executing promise: {err:?}"
379            ))
380        })?;
381        let response = serde_wasm_bindgen::from_value::<MlsTransportData>(response)
382            .map_err(|not_transport_data| {
383                core_crypto::Error::ErrorDuringMlsTransport(format!(
384                    "prepare_for_transport received a value which was not an MlsTransportData after awaiting js promise: {not_transport_data:?}"
385                ))
386            })?;
387
388        Ok(core_crypto::MlsTransportData::from(response.data))
389    }
390}

392/// In uniffi, `MlsTransport` is a trait which we need to wrap
393#[cfg(not(target_family = "wasm"))]

396#[cfg(not(target_family = "wasm"))]
397fn callback_shim(callbacks: Callbacks) -> Arc<dyn core_crypto::MlsTransport> {
398    Arc::new(MlsTransportShim::new(callbacks))

401/// In wasm, `MlsTransport` is an object with some callable members defined.

403type Callbacks = MlsTransport;

405#[cfg(target_family = "wasm")]
406fn callback_shim(callbacks: Callbacks) -> Arc<dyn core_crypto::MlsTransport> {
