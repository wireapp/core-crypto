8#[cfg(not(target_family = "wasm"))]
9#[tokio::main]
10async fn main() -> Result<()> {
11    #[derive(Debug, clap::Parser)]
12    #[command(author, version, about, long_about = None)]
13    struct Args {
14        #[arg(short, long)]
15        /// The 256-bit database key, hex-encoded.
16        key: String,
17
18        path: String,
19    }
20
21    use chrono::TimeZone;
22    use clap::Parser as _;
23    use color_eyre::eyre::eyre;
24    use core_crypto_keystore::{
25        Connection as Keystore, ConnectionType, DatabaseKey, connection::FetchFromDatabase, entities::*,
26    };
27    use openmls::prelude::TlsDeserializeTrait;
28    use serde::ser::{SerializeMap, Serializer};
29
30    color_eyre::install()?;
31
32    let args = Args::parse();
33
34    if !tokio::fs::try_exists(&args.path).await.unwrap_or_default() {
35        return Err(eyre!("File not found: {}", args.path));
36    }
37
38    let key = DatabaseKey::try_from(hex::decode(&args.key)?.as_slice())?;
39    let keystore = Keystore::open(ConnectionType::Persistent(&args.path), &key)
40        .await
41        .map_err(|e| eyre!("The passkey is probably wrong; [err: {e}]"))?;
42
43    let mut json_serializer = serde_json::Serializer::pretty(std::io::stdout());
44    let mut json_map = json_serializer.serialize_map(None)?;
45
46    let mut credentials: Vec<serde_json::Value> = vec![];
47    for cred in keystore
48        .find_all::<MlsCredential>(Default::default())
49        .await?
50        .into_iter()
51    {
52        let mls_credential = openmls::prelude::Credential::tls_deserialize(&mut cred.credential.as_slice())?;
53        let date = chrono::Utc
54            .timestamp_opt(cred.created_at as i64, 0)
55            .single()
56            .ok_or_else(|| eyre!("Cannot parse credential creation date"))?;
57
58        credentials.push(serde_json::json!({
59            "id": cred.id,
60            "credential": mls_credential,
61            "created_at": date
62        }));
63    }
64    json_map.serialize_entry("mls_credentials", &credentials)?;
65
66    let mut signature_keypairs: Vec<serde_json::Value> = vec![];
67    for kp in keystore
68        .find_all::<MlsSignatureKeyPair>(Default::default())
69        .await?
70        .into_iter()
71    {
72        let mls_keypair = openmls_basic_credential::SignatureKeyPair::tls_deserialize(&mut kp.keypair.as_slice())?;
73        signature_keypairs.push(serde_json::json!({
74            "signature_scheme": kp.signature_scheme,
75            "mls_keypair": mls_keypair,
76            "credential_id": kp.credential_id,
77        }));
78    }
79    json_map.serialize_entry("mls_signature_keypairs", &signature_keypairs)?;
80
81    let hpke_sks: Vec<openmls_traits::types::HpkePrivateKey> = keystore
82        .find_all::<MlsHpkePrivateKey>(Default::default())
83        .await?
84        .into_iter()
85        .map(|hpke_sk| postcard::from_bytes::<openmls_traits::types::HpkePrivateKey>(&hpke_sk.sk))
86        .collect::<postcard::Result<_>>()?;
87    json_map.serialize_entry("mls_hpke_private_keys", &hpke_sks)?;
88
89    let hpke_keypairs: Vec<openmls_traits::types::HpkeKeyPair> = keystore
90        .find_all::<MlsEncryptionKeyPair>(Default::default())
91        .await?
92        .into_iter()
93        .map(|hpke_kp| postcard::from_bytes::<openmls_traits::types::HpkeKeyPair>(&hpke_kp.sk))
94        .collect::<postcard::Result<_>>()?;
95    json_map.serialize_entry("mls_hpke_keypairs", &hpke_keypairs)?;
96
97    let mut external_psks: std::collections::HashMap<String, openmls::schedule::psk::PskBundle> = Default::default();
98    for psk in keystore.find_all::<MlsPskBundle>(Default::default()).await?.into_iter() {
99        let mls_psk = postcard::from_bytes::<openmls::schedule::psk::PskBundle>(&psk.psk)?;
100        external_psks.insert(hex::encode(&psk.psk_id), mls_psk);
101    }
102
103    json_map.serialize_entry("external_psks", &external_psks)?;
104
105    let keypackages: Vec<openmls::prelude::KeyPackage> = keystore
106        .find_all::<MlsKeyPackage>(Default::default())
107        .await?
108        .into_iter()
109        .map(|kp| postcard::from_bytes::<openmls::prelude::KeyPackage>(&kp.keypackage))
110        .collect::<postcard::Result<_>>()?;
111    json_map.serialize_entry("mls_keypackages", &keypackages)?;
112
113    let e2ei_enrollments: Vec<core_crypto::prelude::E2eiEnrollment> = keystore
114        .find_all::<E2eiEnrollment>(Default::default())
115        .await?
116        .into_iter()
117        .map(|enrollment| serde_json::from_slice::<core_crypto::prelude::E2eiEnrollment>(&enrollment.content))
118        .collect::<serde_json::Result<_>>()?;
119    json_map.serialize_entry("e2ei_enrollments", &e2ei_enrollments)?;
120
121    let pgroups: Vec<openmls::prelude::MlsGroup> = keystore
122        .find_all::<PersistedMlsGroup>(Default::default())
123        .await?
124        .into_iter()
125        .map(|pgroup| core_crypto_keystore::deser::<openmls::prelude::MlsGroup>(&pgroup.state))
126        .collect::<core_crypto_keystore::CryptoKeystoreResult<_>>()?;
127    json_map.serialize_entry("mls_groups", &pgroups)?;
128
129    let pegroups: Vec<openmls::prelude::MlsGroup> = keystore
130        .find_all::<PersistedMlsPendingGroup>(Default::default())
131        .await?
132        .into_iter()
133        .map(|pgroup| core_crypto_keystore::deser::<openmls::prelude::MlsGroup>(&pgroup.state))
134        .collect::<core_crypto_keystore::CryptoKeystoreResult<_>>()?;
135    json_map.serialize_entry("mls_pending_groups", &pegroups)?;
136
137    if let Some(proteus_identity) = keystore.find::<ProteusIdentity>(&[]).await? {
138        let identity = {
139            let sk = proteus_identity.sk_raw();
140            let pk = proteus_identity.pk_raw();
141            proteus_wasm::keys::IdentityKeyPair::from_raw_key_pair(*sk, *pk)?
142        };
143        json_map.serialize_entry("proteus_identity", &identity)?;
144
145        let prekeys: Vec<proteus_wasm::keys::PreKey> = keystore
146            .find_all::<ProteusPrekey>(Default::default())
147            .await?
148            .into_iter()
149            .map(|pk| proteus_wasm::keys::PreKey::deserialise(&pk.prekey))
150            .collect::<Result<Vec<_>, proteus_wasm::DecodeError>>()?;
151        json_map.serialize_entry("proteus_prekeys", &prekeys)?;
152
153        let proteus_sessions: Vec<proteus_wasm::session::Session<proteus_wasm::keys::IdentityKeyPair>> = keystore
154            .find_all::<ProteusSession>(Default::default())
155            .await?
156            .into_iter()
157            .map(|session| proteus_wasm::session::Session::deserialise(identity.clone(), &session.session))
158            .collect::<Result<Vec<_>, proteus_wasm::DecodeError>>()?;
159
160        json_map.serialize_entry("proteus_sessions", &proteus_sessions)?;
161    }
162
163    json_map.end()?;
164
165    Ok(())
166}
