43impl From<RecursiveError> for CoreCryptoError {
44    fn from(error: RecursiveError) -> Self {
45        log_error(&error);
46
47        // check if the innermost error is any kind of e2e error
48        let innermost = {
49            let mut err: &dyn std::error::Error = &error;
50            while let Some(inner) = err.source() {
51                err = inner;
52            }
53            err
54        };
55
56        if let Some(err) = innermost.downcast_ref::<core_crypto::e2e_identity::Error>() {
57            return CoreCryptoError::E2ei(err.to_string());
58        }
59
60        // What now? We only really care about the innermost variants, not the error stack, but that produces
61        // an arbitrary set of types. We can't match against that!
62        //
63        // Or at least, not without the power of macros. We can use them to match against heterogenous types.
64
65        /// Like [`matches!`], but with an out expression which can reference items captured by the pattern.
66        ///
67        /// Hopefully only ever use this in conjunction with `interior_matches!`, because for most sane
68        /// circumstances, `if let` is the better design pattern.
69        macro_rules! matches_option {
70            ($val:expr, $pattern:pat $(if $guard:expr)? => $out:expr) => {
71                match ($val) {
72                    $pattern $(if $guard)? => Some($out),
73                    _ => None,
74                }
75            };
76        }
77
78        /// This is moderately horrific and we hopefully will not require it anywhere else, but
79        /// it solves a real problem here: how do we match against the innermost error variants,
80        /// when we have a heterogenous set of types to match against?
81        macro_rules! match_heterogenous {
82            ($err:expr => {
83                $( $pattern:pat $(if $guard:expr)? => $var:expr, )*
84                ||=> $default:expr,
85            }) => {{
86                if false {unreachable!()}
87                $(
88                    else if let Some(v) = matches_option!($err.downcast_ref(), Some($pattern) $(if $guard)? => $var) {
89                        v
90                    }
91                )*
92                else {
93                    $default
94                }
95            }};
96        }
97
98        match_heterogenous!(innermost => {
99            core_crypto::LeafError::ConversationAlreadyExists(id) => MlsError::ConversationAlreadyExists(id.clone()).into(),
100            core_crypto::mls::conversation::Error::BufferedFutureMessage{..} => MlsError::BufferedFutureMessage.into(),
101            core_crypto::mls::conversation::Error::DuplicateMessage => MlsError::DuplicateMessage.into(),
102            core_crypto::mls::conversation::Error::MessageEpochTooOld => MlsError::MessageEpochTooOld.into(),
103            core_crypto::mls::conversation::Error::SelfCommitIgnored => MlsError::SelfCommitIgnored.into(),
104            core_crypto::mls::conversation::Error::StaleCommit => MlsError::StaleCommit.into(),
105            core_crypto::mls::conversation::Error::StaleProposal => MlsError::StaleProposal.into(),
106            core_crypto::mls::conversation::Error::UnbufferedFarFutureMessage => MlsError::WrongEpoch.into(),
107            core_crypto::mls::conversation::Error::BufferedCommit => MlsError::BufferedCommit.into(),
108            core_crypto::mls::conversation::Error::MessageRejected { reason } => MlsError::MessageRejected { reason: reason.clone() }.into(),
109            core_crypto::mls::conversation::Error::OrphanWelcome => MlsError::OrphanWelcome.into(),
110            // The internal name is what we want, but renaming the external variant is a breaking change.
111            // Since we're re-designing the `BufferedMessage` errors soon, it's not worth producing
112            // an additional breaking change until then, so the names are inconsistent.
113            core_crypto::mls::conversation::Error::BufferedForPendingConversation => MlsError::UnmergedPendingGroup.into(),
114            ||=> MlsError::Other(error.innermost_error_message()).into(),
115        })
116    }
117}

119// This implementation is intended to be temporary; we're going to be completely restructuring the way we handle
120// errors in `core-crypto` soon. We can replace this with better error patterns when we do.
121//
122// Certain error mappings could apply to both MLS and Proteus. In all such cases, we map them to the MLS variant.
123// When we redesign the errors in `core-crypto`, these ambiguities should disappear anyway.
124impl From<core_crypto::Error> for CoreCryptoError {
125    fn from(error: core_crypto::Error) -> Self {
126        log_error(&error);
127
128        match error {
129            #[cfg(feature = "proteus")]
130            core_crypto::Error::ProteusNotInitialized => Self::Other(error.to_string()),
131            #[cfg(not(feature = "proteus"))]
132            core_crypto::Error::ProteusNotInitialized => Self::Other("proteus not initialized".into()),
133            #[cfg(feature = "proteus")]
134            core_crypto::Error::Proteus(proteus) => {
135                let error_code = proteus.source.error_code();
136                if let Some(proteus_error) = ProteusError::from_error_code(error_code) {
137                    Self::Proteus(proteus_error)
138                } else {
139                    Self::Other(format!("unknown proteus error code: {error_code:?}"))
140                }
141            }
142            #[cfg(not(feature = "proteus"))]
143            core_crypto::Error::Proteus(_proteus) => {
144                unreachable!("we don't raise proteus errors when building without proteus")
145            }
146            core_crypto::Error::Mls(mls) => Self::Mls(MlsError::from(mls)),
147            core_crypto::Error::InvalidContext => Self::Other(error.to_string()),
148            core_crypto::Error::MlsTransportNotProvided => Self::Other(error.to_string()),
149            core_crypto::Error::ErrorDuringMlsTransport(error_message) => Self::Other(error_message),
150            core_crypto::Error::Keystore(keystore_error) => Self::Other(keystore_error.innermost_error_message()),
151            core_crypto::Error::CryptoboxMigration(cryptobox) => Self::Other(cryptobox.innermost_error_message()),
152            core_crypto::Error::Recursive(recursive_error) => recursive_error.into(),
153            core_crypto::Error::FeatureDisabled(_) => Self::Other(error.to_string()),
154        }
155    }
156}

158/// We can't do a generic `impl<E: ToRecursiveError> From<E> for CoreCryptoError`
159/// because that has the potential to cause breaking conflicts later on: what if
160/// core-crypto later did `impl ToRecursiveError for core_crypto::Error`? That would
161/// cause a duplicate `From` impl.
162///
163/// Instead, we explicitly specify every variant which can be converted to a
164/// `CoreCryptoError`, and implement its `From` block directly.
165macro_rules! impl_from_via_recursive_error {
166    ($($t:ty),+ $(,)?) => {
167        $(
168            impl From<$t> for CoreCryptoError {
169                fn from(error: $t) -> Self {
170                    use core_crypto::ToRecursiveError;
171                    error
172                        .construct_recursive("this context string does not matter and gets immediately stripped")
173                        .into()
174                }
175            }
176        )*
177    };
178}

198    #[cfg(target_family = "wasm")]
199    pub(crate) fn variant_name(&self) -> String {
200        let mut out = self.as_ref().to_string() + "Error";
201        match self {
202            Self::Mls(mls) => out += mls.as_ref(),
203            Self::Proteus(proteus) => out += proteus.as_ref(),
204            _ => {}
205        }
206        out
207    }
208
209    #[cfg(target_family = "wasm")]
210    pub(crate) fn stack(&self) -> Vec<String> {
211        let mut stack = Vec::new();
212        let mut err: &dyn std::error::Error = self;
213        stack.push(err.to_string());
214
215        while let Some(source) = err.source() {
216            stack.push(source.to_string());
217            err = source;
218        }
219
220        stack
221    }
