121impl PkiKeypair {
122    pub fn signing_key_bytes(&self) -> Vec<u8> {
123        match self {
124            Self::P256(sk) => sk.to_bytes().to_vec(),
125            Self::P384(sk) => sk.to_bytes().to_vec(),
126            Self::P521(sk) => sk.0.to_bytes().to_vec(),
127            Self::Ed25519(sk) => sk.0.to_bytes().to_vec(),

131    pub fn public_key_bytes(&self) -> Vec<u8> {
132        match self {
133            Self::P256(sk) => sk.verifying_key().to_sec1_bytes().to_vec(),
134            Self::P384(sk) => sk.verifying_key().to_sec1_bytes().to_vec(),
135            Self::P521(sk) => sk.0.verifying_key().to_sec1_bytes().to_vec(),
136            Self::Ed25519(sk) => sk.0.verifying_key().to_bytes().to_vec(),
137        }
138    }
139
140    pub fn public_key_identifier(&self) -> Vec<u8> {
141        use sha1::Digest as _;
142        sha1::Sha1::digest(self.public_key_bytes()).to_vec()
143    }

148pub struct CertificateGenerationArgs<'a> {
149    pub signature_scheme: SignatureScheme,
150    pub profile: CertProfile,
151    pub serial: u64,
152    /// Duration since UNIX EPOCH
153    pub validity_start: Option<std::time::Duration>,
154    /// Duration relative to `validity_start` if present. Otherwise relative to now
155    pub validity_from_start: std::time::Duration,
156    pub org: &'a str,
157    pub common_name: Option<&'a str>,
158    pub alternative_names: Option<&'a [&'a str]>,
159    pub domain: Option<&'a str>,
160    pub crl_dps: Option<&'a [&'a str]>,
161    pub signer: Option<&'a PkiKeypair>,
162    pub is_ca: bool,
163    pub is_root: bool,

166fn get_extended_keyusage(is_ca: bool) -> x509_cert::ext::pkix::ExtendedKeyUsage {
167    let mut ext_keyusages = vec![];
168    if !is_ca {
169        ext_keyusages.push(x509_cert::der::oid::db::rfc5280::ID_KP_CLIENT_AUTH);
170    }
171
172    x509_cert::ext::pkix::ExtendedKeyUsage(ext_keyusages)
173}

175macro_rules! impl_certgen {
176    (
177        $signer:expr, $signer_keypair:expr, $sig_type:path,
178        $profile:expr, $own_spki:expr, $serial:expr,
179        $subject:expr, $org:expr, $domain:expr, $validity:expr, $alt_names:expr,
180        $crl_dps:expr, $is_ca:expr, $is_root:expr
181    ) => {{
182        let add_akid = $is_ca && $profile == x509_cert::builder::Profile::Root;
183
184        let mut builder = x509_cert::builder::CertificateBuilder::new(
185            $profile,
186            $serial,
187            $validity,
188            $subject,
189            $own_spki,
190            $signer_keypair,
191        )
192        .map_err(|_| MlsProviderError::CertificateGenerationError)?;
193
194        if add_akid {
195            builder
196                .add_extension(&$signer.akid()?)
197                .map_err(|_| MlsProviderError::CertificateGenerationError)?;
198        }
199
200        builder
201            .add_extension(&get_extended_keyusage($is_ca))
202            .map_err(|_| MlsProviderError::CertificateGenerationError)?;
203
204        if !$is_ca {
205            if let Some(alt_names) = $alt_names {
206                let mut alt_names_list = vec![];
207                for alt_name in alt_names {
208                    alt_names_list.push(x509_cert::ext::pkix::name::GeneralName::UniformResourceIdentifier(
209                        alt_name
210                            .to_string()
211                            .try_into()
212                            .map_err(|_| MlsProviderError::CertificateGenerationError)?,
213                    ));
214                }
215
216                builder
217                    .add_extension(&x509_cert::ext::pkix::SubjectAltName(alt_names_list))
218                    .map_err(|_| MlsProviderError::CertificateGenerationError)?;
219            }
220        } else {
221            let mut permitted_subtrees = vec![
222                x509_cert::ext::pkix::name::GeneralName::UniformResourceIdentifier(
223                    format!(".{}", $org)
224                        .try_into()
225                        .map_err(|_| MlsProviderError::CertificateGenerationError)?,
226                ),
227                x509_cert::ext::pkix::name::GeneralName::UniformResourceIdentifier(
228                    format!("{}", $org)
229                        .try_into()
230                        .map_err(|_| MlsProviderError::CertificateGenerationError)?,
231                ),
232            ];
233
234            if let Some(domain) = $domain {
235                // Add Domain DNS SAN
236                builder
237                    .add_extension(&x509_cert::ext::pkix::SubjectAltName(vec![
238                        x509_cert::ext::pkix::name::GeneralName::DnsName(
239                            domain
240                                .to_string()
241                                .try_into()
242                                .map_err(|_| MlsProviderError::CertificateGenerationError)?,
243                        ),
244                    ]))
245                    .map_err(|_| MlsProviderError::CertificateGenerationError)?;
246
247                permitted_subtrees.push(x509_cert::ext::pkix::name::GeneralName::DnsName(
248                    domain
249                        .to_string()
250                        .try_into()
251                        .map_err(|_| MlsProviderError::CertificateGenerationError)?,
252                ));
253            }
254
255            if !$is_root {
256                builder
257                    .add_extension(&x509_cert::ext::pkix::NameConstraints {
258                        permitted_subtrees: Some(
259                            permitted_subtrees
260                                .into_iter()
261                                .map(|base| x509_cert::ext::pkix::constraints::name::GeneralSubtree {
262                                    base,
263                                    minimum: 0,
264                                    maximum: None,
265                                })
266                                .collect(),
267                        ),
268
269                        excluded_subtrees: None,
270                    })
271                    .map_err(|_| MlsProviderError::CertificateGenerationError)?;
272            }
273        }
274
275        if let Some(crl_dps) = $crl_dps {
276            let mut crl_distribution_points = vec![];
277            for dp in crl_dps {
278                crl_distribution_points.push(x509_cert::ext::pkix::crl::dp::DistributionPoint {
279                    distribution_point: Some(x509_cert::ext::pkix::name::DistributionPointName::FullName(vec![
280                        x509_cert::ext::pkix::name::GeneralName::UniformResourceIdentifier(
281                            dp.to_string()
282                                .try_into()
283                                .map_err(|_| MlsProviderError::CertificateGenerationError)?,
284                        ),
285                    ])),
286                    crl_issuer: None,
287                    reasons: None,
288                });
289            }
290            builder
291                .add_extension(&x509_cert::ext::pkix::CrlDistributionPoints(crl_distribution_points))
292                .map_err(|_| MlsProviderError::CertificateGenerationError)?;
293        }
294
295        builder
296            .build::<$sig_type>()
297            .map_err(|_| MlsProviderError::CertificateGenerationError)?
298    }};
299}

301impl PkiKeypair {
302    pub fn new(signature_scheme: SignatureScheme, sk: Vec<u8>) -> MlsProviderResult<Self> {
303        match signature_scheme {
304            SignatureScheme::ECDSA_SECP256R1_SHA256 => Ok(PkiKeypair::P256(
305                p256::ecdsa::SigningKey::from_slice(sk.as_slice())
306                    .map_err(|_| MlsProviderError::CertificateGenerationError)?,
307            )),
308            SignatureScheme::ECDSA_SECP384R1_SHA384 => Ok(PkiKeypair::P384(
309                p384::ecdsa::SigningKey::from_slice(sk.as_slice())
310                    .map_err(|_| MlsProviderError::CertificateGenerationError)?,
311            )),
312            SignatureScheme::ECDSA_SECP521R1_SHA512 => Ok(PkiKeypair::P521(P521PkiKeypair(
313                ecdsa::SigningKey::<p521::NistP521>::from_slice(sk.as_slice())
314                    .map_err(|_| MlsProviderError::CertificateGenerationError)?,
315            ))),
316            SignatureScheme::ED25519 => Ok(PkiKeypair::Ed25519(Ed25519PkiKeypair(
317                super::RustCrypto::normalize_ed25519_key(sk.as_slice())
318                    .map_err(|_| MlsProviderError::CertificateGenerationError)?,
319            ))),
320            _ => Err(MlsProviderError::UnsupportedSignatureScheme),
321        }
322    }
323
324    pub fn signature_algorithm(&self) -> spki::AlgorithmIdentifierRef<'_> {
325        match self {
326            Self::P256(_) => p256::ecdsa::SigningKey::SIGNATURE_ALGORITHM_IDENTIFIER,
327            Self::P384(_) => p384::ecdsa::SigningKey::SIGNATURE_ALGORITHM_IDENTIFIER,
328            Self::P521(_) => spki::AlgorithmIdentifierRef {
329                oid: ecdsa::ECDSA_SHA512_OID,
330                parameters: None,
331            },
332            Self::Ed25519(_) => ed25519_dalek::pkcs8::ALGORITHM_ID,
333        }
334    }
335
336    pub fn spki(&self) -> MlsProviderResult<spki::SubjectPublicKeyInfoOwned> {
337        match self {
338            Self::P256(sk) => Ok(spki::SubjectPublicKeyInfoOwned::from_key(*sk.verifying_key())
339                .map_err(|_| MlsProviderError::CertificateGenerationError)?),
340            Self::P384(sk) => Ok(spki::SubjectPublicKeyInfoOwned::from_key(*sk.verifying_key())
341                .map_err(|_| MlsProviderError::CertificateGenerationError)?),
342            Self::P521(sk) => Ok(spki::SubjectPublicKeyInfoOwned::from_key(*sk.0.verifying_key())
343                .map_err(|_| MlsProviderError::CertificateGenerationError)?),
344            Self::Ed25519(sk) => Ok(spki::SubjectPublicKeyInfoOwned::from_key(sk.0.verifying_key())
345                .map_err(|_| MlsProviderError::CertificateGenerationError)?),
346        }
347    }

349    pub fn akid(&self) -> MlsProviderResult<x509_cert::ext::pkix::AuthorityKeyIdentifier> {
350        Ok(x509_cert::ext::pkix::AuthorityKeyIdentifier {
351            key_identifier: Some(
352                spki::der::asn1::OctetString::new(self.public_key_identifier())

354            ),
355            authority_cert_issuer: None,
356            authority_cert_serial_number: None,
357        })
358    }

360    pub fn revoke_certs(
361        &self,
362        issuer_cert: &x509_cert::Certificate,
363        revoked_cert_serial_numbers: Vec<Vec<u8>>,
364    ) -> MlsProviderResult<x509_cert::crl::CertificateList> {
365        let signature_algorithm = self.signature_algorithm();
366        let now = web_time::SystemTime::now()
367            .duration_since(web_time::UNIX_EPOCH)
368            .map_err(|_| MlsProviderError::CertificateGenerationError)?;
369        let now = x509_cert::der::asn1::GeneralizedTime::from_unix_duration(now)
370            .map_err(|_| MlsProviderError::CertificateGenerationError)?;
371        let now = x509_cert::time::Time::GeneralTime(now);
372
373        let revoked_certificates = revoked_cert_serial_numbers
374            .into_iter()
375            .map(|serial_number| x509_cert::crl::RevokedCert {
376                serial_number: x509_cert::serial_number::SerialNumber::new(&serial_number)
377                    .expect("Non-positive serial number"),
378                revocation_date: now,
379                crl_entry_extensions: None,
380            })
381            .collect();
382
383        let tbs_cert_list = x509_cert::crl::TbsCertList {
384            version: x509_cert::Version::V3,
385            signature: signature_algorithm.ref_to_owned(),
386            issuer: issuer_cert.tbs_certificate.subject.clone(),
387            this_update: now,
388            next_update: None,
389            revoked_certificates: Some(revoked_certificates),
390            crl_extensions: None,
391        };
392
393        use spki::der::Encode as _;
394
395        let tbs = tbs_cert_list
396            .to_der()
397            .map_err(|_| MlsProviderError::CertificateGenerationError)?;
398
399        use signature::Signer as _;
400
401        let signature: Vec<u8> = match self {
402            PkiKeypair::P256(sk) => signature::Signer::<p256::ecdsa::DerSignature>::try_sign(sk, &tbs)?
403                .to_der()
404                .map_err(|_| MlsProviderError::CertificateGenerationError),
405            PkiKeypair::P384(sk) => signature::Signer::<p384::ecdsa::DerSignature>::try_sign(sk, &tbs)?
406                .to_der()
407                .map_err(|_| MlsProviderError::CertificateGenerationError),
408            PkiKeypair::P521(sk) => {
409                let sk = p521::ecdsa::SigningKey::from(sk.0.clone());
410                let signature: p521::ecdsa::DerSignature = sk.try_sign(&tbs)?.to_der();
411
412                signature
413                    .to_der()
414                    .map_err(|_| MlsProviderError::CertificateGenerationError)
415            }
416            PkiKeypair::Ed25519(sk) => Ok(sk.try_sign(&tbs)?.0.to_vec()),
417        }?;
418
419        let signature =
420            spki::der::asn1::BitString::new(0, signature).map_err(|_| MlsProviderError::CertificateGenerationError)?;
421
422        Ok(x509_cert::crl::CertificateList {
423            tbs_cert_list,
424            signature_algorithm: signature_algorithm.ref_to_owned(),
425            signature,
426        })
427    }

429    pub fn generate_cert(&self, args: CertificateGenerationArgs) -> MlsProviderResult<x509_cert::Certificate> {
430        use std::str::FromStr as _;
431
432        use x509_cert::builder::Builder as _;
433        let mut subject_fmt = format!("O={}", args.org);
434        if let Some(cn) = args.common_name {
435            subject_fmt.push_str(&format!(",CN={cn}"));
436        }
437
438        let subject =
439            x509_cert::name::Name::from_str(&subject_fmt).map_err(|_| MlsProviderError::CertificateGenerationError)?;
440
441        let validity_start = if let Some(validity_start) = args.validity_start {
442            validity_start
443        } else {
444            web_time::SystemTime::now()
445                .duration_since(web_time::UNIX_EPOCH)
446                .map_err(|_| MlsProviderError::CertificateGenerationError)?
447        } - std::time::Duration::from_secs(1); // to prevent time clipping
448
449        let validity = {
450            let not_before = x509_cert::der::asn1::GeneralizedTime::from_unix_duration(validity_start)
451                .map_err(|_| MlsProviderError::CertificateGenerationError)?
452                .into();
453            let not_after =
454                x509_cert::der::asn1::GeneralizedTime::from_unix_duration(validity_start + args.validity_from_start)
455                    .map_err(|_| MlsProviderError::CertificateGenerationError)?
456                    .into();
457            x509_cert::time::Validity { not_before, not_after }
458        };
459
460        let serial_number = x509_cert::serial_number::SerialNumber::from(args.serial);
461        let spki = self.spki()?;
462
463        let signer = args.signer.unwrap_or(self);
464
465        let cert = match signer {
466            PkiKeypair::P256(kp) => {
467                impl_certgen!(
468                    signer,
469                    kp,
470                    p256::ecdsa::DerSignature,
471                    args.profile,
472                    spki,
473                    serial_number,
474                    subject,
475                    args.org,
476                    args.domain,
477                    validity,
478                    args.alternative_names,
479                    args.crl_dps,
480                    args.is_ca,
481                    args.is_root
482                )
483            }
484            PkiKeypair::P384(kp) => {
485                impl_certgen!(
486                    signer,
487                    kp,
488                    p384::ecdsa::DerSignature,
489                    args.profile,
490                    spki,
491                    serial_number,
492                    subject,
493                    args.org,
494                    args.domain,
495                    validity,
496                    args.alternative_names,
497                    args.crl_dps,
498                    args.is_ca,
499                    args.is_root
500                )
501            }
502            PkiKeypair::P521(kp) => {
503                impl_certgen!(
504                    signer,
505                    kp,
506                    p521::ecdsa::DerSignature,
507                    args.profile,
508                    spki,
509                    serial_number,
510                    subject,
511                    args.org,
512                    args.domain,
513                    validity,
514                    args.alternative_names,
515                    args.crl_dps,
516                    args.is_ca,
517                    args.is_root
518                )
519            }
520            PkiKeypair::Ed25519(kp) => {
521                impl_certgen!(
522                    signer,
523                    kp,
524                    Ed25519PkiSignature,
525                    args.profile,
526                    spki,
527                    serial_number,
528                    subject,
529                    args.org,
530                    args.domain,
531                    validity,
532                    args.alternative_names,
533                    args.crl_dps,
534                    args.is_ca,
535                    args.is_root
536                )
537            }
538        };
539
540        Ok(cert)
541    }
