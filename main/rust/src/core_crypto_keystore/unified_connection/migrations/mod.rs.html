13    Latest,

17pub(super) fn run_migrations(conn: &mut rusqlite::Connection, target: MigrationTarget) -> CryptoKeystoreResult<()> {
18    conn.create_scalar_function("sha256_blob", 1, FunctionFlags::SQLITE_DETERMINISTIC, |ctx| {
19        let input_blob = ctx.get::<Vec<u8>>(0)?;
20        Ok(crate::sha256(&input_blob))
21    })?;
22
23    let mut runner = migrations::runner();
24    let Some(latest_migration_version) = runner
25        .get_migrations()
26        .iter()
27        .map(|migration| migration.version())
28        .max()
29    else {
30        // No migrations means nothing to do.
31        return Ok(());
32    };
33
34    let target_version = match target {
35        MigrationTarget::Latest => latest_migration_version,
36        MigrationTarget::Version(target_argument) => (latest_migration_version).min(target_argument as i32),
37    };
38
39    for version in 1..=target_version {
40        runner = runner.set_target(Target::Version(version));
41        let report = runner.run(conn).map_err(Box::new)?;
42
43        let Some(updated_version) = report.applied_migrations().iter().map(|m| m.version()).max() else {
44            continue;
45        };
46
47        // If the version has been updated by the runner, first run the meta migration, then update the schema
48        // version.
49        run_meta_migration(updated_version, conn)?;
50        conn.pragma_update(None, "schema_version", updated_version)?;
51    }
52
53    Ok(())
54}

56/// Add a new match arm here if you want to run a meta migration (i.e., addtional work implemented in rust)
57/// after a regular SQL migration.
58fn run_meta_migration(sql_migration_version: i32, conn: &mut rusqlite::Connection) -> CryptoKeystoreResult<()> {
59    match sql_migration_version {
60        meta_migrations::v16::VERSION => meta_migrations::v16::meta_migration(conn),
61        meta_migrations::v18::VERSION => meta_migrations::v18::meta_migration(conn),
62        meta_migrations::v19::VERSION => meta_migrations::v19::meta_migration(conn),
63        _ => Ok(()),
64    }
65}

67pub async fn migrate_db_key_type_to_bytes(
68    path: &str,
69    old_key: &str,
70    new_key: &DatabaseKey,
71) -> CryptoKeystoreResult<()> {
72    let mut conn = rusqlite::Connection::open(path)?;
73
74    conn.pragma_update(None, "key", old_key)?;
75
76    // ? iOS WAL journaling fix; see details here: https://github.com/sqlcipher/sqlcipher/issues/255
77    #[cfg(target_os = "ios")]
78    super::ios_wal_compat::handle_ios_wal_compat(&conn, path)?;
79
80    /// This is the latest schema version our test db dump is compatible with.
81    const MAX_SUPPORTED_SCHEMA_VERSION: u8 = 15;
82
83    let version = conn.query_row("PRAGMA schema_version;", [], |row| row.get::<_, i32>(0))?;
84    if version >= MAX_SUPPORTED_SCHEMA_VERSION as i32 {
85        return Err(CryptoKeystoreError::MigrationFailed(
86            "key type migration from string to bytes can and should only be done once and on database versions
87                    corresponding to a core crypto version <= 9."
88                .to_string(),
89        ));
90    }
91
92    // Enable WAL journaling mode
93    conn.pragma_update(None, "journal_mode", "wal")?;
94
95    // Disable FOREIGN KEYs - The 2 step blob writing process invalidates foreign key checks unfortunately
96    conn.pragma_update(None, "foreign_keys", "OFF")?;
97
98    // Now update the database to the latest compatible schema version. The other, following migrations
99    // will be run when the database is opened regularly.
100    run_migrations(&mut conn, MigrationTarget::Version(MAX_SUPPORTED_SCHEMA_VERSION as u16))?;
101
102    // Rekey the database.
103    super::encryption::rekey(&mut conn, new_key)
104}

141    #[test]
142    fn deduplicating_credentials() {
143        let mut db_file = NamedTempFile::new().unwrap();
144        db_file.write_all(DB).unwrap();
145        let path = db_file
146            .path()
147            .to_str()
148            .expect("tmpfile path is representable in unicode");
149
150        let new_key = DatabaseKey::generate();
151        smol::block_on(migrate_db_key_type_to_bytes(path, OLD_KEY, &new_key)).unwrap();
152
153        smol::block_on(async {
154            let db = Database::open_at_schema_version(path, &new_key, MigrationTarget::Version(18))
155                .await
156                .unwrap();
157
158            let mut stmt = db
159                .conn
160                .prepare(&format!(
161                    "SELECT
162                        session_id,
163                        credential,
164                        unixepoch(created_at) AS created_at,
165                        ciphersuite,
166                        public_key,
167                        private_key
168                     FROM {credential_table}",
169                    credential_table = "mls_credentials_new",
170                ))
171                .expect("preparing statement");
172
173            let credential = stmt
174                .query_one([], |row| {
175                    Ok(StoredCredential {
176                        session_id: row.get("session_id")?,
177                        credential: row.get("credential")?,
178                        created_at: row.get("created_at")?,
179                        ciphersuite: row.get("ciphersuite")?,
180                        public_key: row.get("public_key")?,
181                        private_key: row.get("private_key")?,
182                    })
183                })
184                .expect("credential from row");
185
186            // Ciphersuites need to be ambiguous w.r.t their signature scheme to be a relevant duplicate
187            db.conn
188                .execute(
189                    "UPDATE mls_credentials_new SET ciphersuite = ?1",
190                    [Ciphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519 as u16],
191                )
192                .expect("updating ciphersuite");
193
194            // Create a duplicate from this credential
195            db.conn
196                .execute(
197                    "INSERT INTO mls_credentials_new (
198                        session_id,
199                        credential,
200                        created_at,
201                        ciphersuite,
202                        public_key,
203                        private_key
204                    )
205                    VALUES (?1, ?2, datetime(?3, 'unixepoch'), ?4, ?5, ?6)",
206                    (
207                        credential.session_id.clone(),
208                        credential.credential.clone(),
209                        credential.created_at,
210                        Ciphersuite::MLS_128_DHKEMX25519_CHACHA20POLY1305_SHA256_Ed25519 as u16,
211                        credential.public_key.clone(),
212                        credential.private_key.clone(),
213                    ),
214                )
215                .expect("inserting duplicate");
216
217            let count = db
218                .conn
219                .query_row("SELECT COUNT(*) FROM mls_credentials_new", [], |row| {
220                    row.get::<_, i32>(0)
221                })
222                .unwrap();
223
224            assert_eq!(count, 2);
225
226            drop(stmt);
227            drop(db);
228
229            let db = Database::open(path, &new_key).await.unwrap();
230            let deduplicated_credentials = StoredCredential::load_all(&db.conn).expect("deduplicated credentials");
231
232            let deduplicated_count = deduplicated_credentials.len();
233
234            let deduplicated_credential = deduplicated_credentials.first().expect("first credential");
235
236            assert_eq!(deduplicated_count, 1);
237
238            // In case of equal occurence, the credential with the numerically lower ciphersuite is kept.
239            assert_eq!(
240                deduplicated_credential.ciphersuite,
241                Ciphersuite::MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519 as u16
242            );
243        });
244    }
