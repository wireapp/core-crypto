3/// How a `GroupInfo` is encrypted in a commit bundle.
4#[derive(Debug, Clone, Copy, uniffi::Enum)]
5#[repr(u8)]
6pub enum MlsGroupInfoEncryptionType {
7    /// Unencrypted `GroupInfo`
8    Plaintext = 1,
9    /// `GroupInfo` encrypted in a JWE
10    JweEncrypted = 2,
11}
12
13impl From<core_crypto::GroupInfoEncryptionType> for MlsGroupInfoEncryptionType {
14    fn from(value: core_crypto::GroupInfoEncryptionType) -> Self {
15        match value {
16            core_crypto::GroupInfoEncryptionType::Plaintext => Self::Plaintext,
17            core_crypto::GroupInfoEncryptionType::JweEncrypted => Self::JweEncrypted,
18        }
19    }
20}
21
22impl From<MlsGroupInfoEncryptionType> for core_crypto::GroupInfoEncryptionType {
23    fn from(value: MlsGroupInfoEncryptionType) -> Self {
24        match value {
25            MlsGroupInfoEncryptionType::Plaintext => Self::Plaintext,
26            MlsGroupInfoEncryptionType::JweEncrypted => Self::JweEncrypted,
27        }
28    }
29}

31/// How the ratchet tree is represented in a `GroupInfo`.
32#[derive(Debug, Clone, Copy, uniffi::Enum)]
33#[repr(u8)]
34pub enum MlsRatchetTreeType {
35    /// The full ratchet tree is included.
36    Full = 1,
37    /// Only changes since the previous epoch are included.
38    ///
39    /// Not yet implemented. See the draft proposal:
40    /// <https://github.com/rohan-wire/ietf-drafts/blob/main/mahy-mls-ratchet-tree-delta/draft-mahy-mls-ratchet-tree-delta.md>
41    Delta = 2,
42    /// The ratchet tree is identified by an external reference rather than included inline.
43    ByRef = 3,
44}
45
46impl From<core_crypto::RatchetTreeType> for MlsRatchetTreeType {
47    fn from(value: core_crypto::RatchetTreeType) -> Self {
48        match value {
49            core_crypto::RatchetTreeType::Full => Self::Full,
50            core_crypto::RatchetTreeType::Delta => Self::Delta,
51            core_crypto::RatchetTreeType::ByRef => Self::ByRef,
52        }
53    }
54}
55
56impl From<MlsRatchetTreeType> for core_crypto::RatchetTreeType {
57    fn from(value: MlsRatchetTreeType) -> Self {
58        match value {
59            MlsRatchetTreeType::Full => Self::Full,
60            MlsRatchetTreeType::Delta => Self::Delta,
61            MlsRatchetTreeType::ByRef => Self::ByRef,
62        }
63    }
64}

66/// A `GroupInfo` with associated metadata.
67#[derive(Debug, Clone, uniffi::Record)]
68pub struct GroupInfoBundle {
69    /// How the group info is encrypted.
70    pub encryption_type: MlsGroupInfoEncryptionType,
71    /// What kind of ratchet tree is used.
72    pub ratchet_tree_type: MlsRatchetTreeType,
73    /// The group info payload.
74    pub payload: Vec<u8>,
75}
76
77impl From<CcGroupInfoBundle> for GroupInfoBundle {
78    fn from(group_info_bundle: CcGroupInfoBundle) -> Self {
79        // single variant => no match stmt necessary
80        let core_crypto::GroupInfoPayload::Plaintext(payload) = group_info_bundle.payload;
81        Self {
82            encryption_type: group_info_bundle.encryption_type.into(),
83            ratchet_tree_type: group_info_bundle.ratchet_tree_type.into(),
84            payload,
85        }
86    }
