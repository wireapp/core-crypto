7/// The `HistoryObserver` will be called when updating the history client in a conversation
8#[cfg_attr(target_family = "wasm", async_trait(?Send))]
9#[cfg_attr(not(target_family = "wasm"), async_trait)]
10pub trait HistoryObserver: Send + Sync {
11    /// This function will be called when a new history client has been created and accepted
12    /// by the delivery service.
13    ///
14    /// The `secret` parameter contains the history client's secrets.
15    async fn history_client_created(&self, conversation_id: ConversationId, secret: &HistorySecret);
16}
17
18impl Session {
19    /// Add an history observer to this session.
20    /// (see [HistoryObserver]).
21    ///
22    /// This function should be called 0 or 1 times in a session's lifetime. If called
23    /// when an epoch observer already exists, this will return an error.
24    pub async fn register_history_observer(
25        &self,
26        history_observer: Arc<dyn HistoryObserver>,
27    ) -> crate::prelude::Result<()> {
28        let mut history_guard = self.history_observer.write().await;
29        if history_guard.is_some() {
30            return Err(Error::HistoryObserverAlreadyExists);
31        }
32        history_guard.replace(history_observer);
33        Ok(())
34    }
35
36    /// Notify the history handler that the history client has been replaced, if one is present.
37    #[allow(dead_code)]
38    pub(crate) async fn notify_new_history_client(
39        &self,
40        conversation_id: ConversationId,
41        history_secret: &HistorySecret,
42    ) {
43        if let Some(handler) = self.history_observer.read().await.as_ref() {
44            handler.history_client_created(conversation_id, history_secret).await;
45        }
46    }
47}

52    use crate::test_utils::{TestContext, TestHistoryObserver, all_cred_cipher};
53    use rstest::rstest;
54    use rstest_reuse::apply;
55
56    #[apply(all_cred_cipher)]
57    pub async fn observe_new_history_client(case: TestContext) {
58        let [session_context] = case.sessions().await;
59        Box::pin(async move {
60            let test_conv = case.create_conversation([&session_context]).await;
61
62            let observer = TestHistoryObserver::new();
63            session_context
64                .session()
65                .await
66                .register_history_observer(observer.clone())
67                .await
68                .unwrap();
69
70            // generate a new history secret
71            let history_secret = test_conv.guard().await.generate_history_secret().await.unwrap();
72            let client_id = history_secret.client_id.clone();
73            let id = test_conv.advance_epoch().await.id;
74
75            session_context
76                .session()
77                .await
78                .notify_new_history_client(id.clone(), &history_secret)
79                .await;
80
81            // ensure we have observed the epoch change
82            let observed_history_clients = observer.observed_history_clients().await;
83            assert_eq!(
84                observed_history_clients.len(),
85                1,
86                "we triggered exactly one epoch change and so should observe one epoch change"
87            );
88            assert_eq!(
89                observed_history_clients[0].0, id,
90                "conversation id of observed epoch change must match"
91            );
92            assert_eq!(
93                observed_history_clients[0].1, client_id,
94                "history client id of observed history client change must match"
95            );
96        })
97        .await
98    }
