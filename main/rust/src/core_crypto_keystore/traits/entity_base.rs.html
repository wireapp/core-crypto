3/// A supertrait that all entities must implement. This handles multiplexing over the two different database backends.
4///
5/// This trait should be removed once the persistence layers are unified. See WPB-16241.
6pub trait EntityBase: 'static + Sized {
7    type ConnectionType: for<'a> DatabaseConnection<'a>;
8    type AutoGeneratedFields: Default;
9
10    /// Beware: if you change the value of this constant on any WASM entity, you'll need to do a data migration
11    ///     not only because it is used as reference to the object store names but also for the value of the aad.
12    const COLLECTION_NAME: &'static str;
13
14    fn downcast<T: EntityBase>(&self) -> Option<&T> {
15        let as_dyn_any: &dyn std::any::Any = self;
16        as_dyn_any.downcast_ref()
17    }
18
19    fn to_transaction_entity(self) -> crate::transaction::dynamic_dispatch::Entity;
20}
