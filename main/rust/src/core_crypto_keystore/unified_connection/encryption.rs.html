5/// Format a key pragma according to the raw form supported by sqlcipher and sqlite3mc
6fn set_key_pragma(conn: &mut rusqlite::Connection, key: &DatabaseKey, pragma_name: &str) -> CryptoKeystoreResult<()> {
7    // Make sqlite use raw key data, without key derivation. Also make sure to zeroize
8    // the string containing the key after the call.
9    let mut x_hex_key = format!("x'{}'", hex::encode(key));
10    let result = conn.pragma_update(None, pragma_name, &x_hex_key);
11    x_hex_key.zeroize();
12    result.map_err(Into::into)
13}
14
15/// Decrypt the database by setting the key pragma appropriately
16pub(super) fn decrypt(conn: &mut rusqlite::Connection, key: &DatabaseKey) -> CryptoKeystoreResult<()> {
17    set_key_pragma(conn, key, "key")
18}
19
20/// Reencrypt the database with a new key.
21pub(super) fn rekey(conn: &mut rusqlite::Connection, new_key: &DatabaseKey) -> CryptoKeystoreResult<()> {
22    set_key_pragma(conn, new_key, "rekey")
23}
