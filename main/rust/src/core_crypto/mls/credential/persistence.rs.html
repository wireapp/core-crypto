1use core_crypto_keystore::{Sha256Hash, entities::StoredCredential, traits::FetchFromDatabase};
2use openmls::prelude::SignaturePublicKey;
3use tls_codec::Serialize as _;
4
5use super::{Error, Result};
6use crate::{Credential, CredentialRef, KeystoreError, mls_provider::Database};
7
8impl Credential {
9    /// Loads a credential with the given public key from the database.
10    ///
11    /// Should not be passed over the ffi boundary.
12    pub(crate) async fn find_by_public_key(
13        database: &impl FetchFromDatabase,
14        public_key: &SignaturePublicKey,
15    ) -> Result<Self> {
16        let stored_credential = &database
17            .get::<StoredCredential>(&Sha256Hash::hash_from(public_key.as_slice()))
18            .await
19            .map_err(KeystoreError::wrap("getting credential by public key"))?
20            .ok_or_else(|| Error::CredentialNotFound(public_key.clone()))?;
21        stored_credential.try_into()
22    }
23
24    /// Persist this credential into the database.
25    ///
26    /// Returns a reference which is stable over time and across the FFI boundary.
27    ///
28    /// Normally this is called internally by [`Session::add_credential`][crate::Session::add_credential];
29    /// use caution if calling it from elsewhere.
30    pub(crate) async fn save(&mut self, database: &Database) -> Result<CredentialRef> {
31        let credential_data = self
32            .mls_credential
33            .tls_serialize_detached()
34            .map_err(Error::tls_serialize("credential"))?;
35
36        self.earliest_validity = database
37            .save(StoredCredential {
38                session_id: self.client_id().to_owned().into_inner(),
39                credential: credential_data,
40                created_at: Default::default(), // updated by the `.save` impl
41                ciphersuite: u16::from(self.ciphersuite),
42                private_key: self.signature_key_pair.private().to_owned(),
43                public_key: self.signature_key().public().to_owned(),
44            })
45            .await
46            .map_err(KeystoreError::wrap("saving credential"))?;
47
48        Ok(CredentialRef::from_credential(self))
49    }
50
51    /// Delete this credential from the database
52    pub(crate) async fn delete(self, database: &Database) -> Result<()> {
53        database
54            .remove::<StoredCredential>(&Sha256Hash::hash_from(self.signature_key_pair.public()))
55            .await
56            .map_err(KeystoreError::wrap("deleting credential"))?;
57
58        Ok(())
59    }
60}
