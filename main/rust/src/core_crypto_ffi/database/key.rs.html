5/// The key used to encrypt the database.
6#[derive(Debug, derive_more::From, derive_more::Into, PartialEq, Eq, Clone, derive_more::Deref, uniffi::Object)]
7#[uniffi::export(Eq)]
8pub struct DatabaseKey(core_crypto_keystore::DatabaseKey);
9
10#[uniffi::export]
11impl DatabaseKey {
12    /// Construct a new instance from a byte vector.
13    #[uniffi::constructor]
14    pub fn new(key: Vec<u8>) -> CoreCryptoResult<Self> {
15        core_crypto_keystore::DatabaseKey::try_from(key.as_slice())
16            .map(Self)
17            .map_err(CoreCryptoError::generic())
18    }
19}

21/// Updates the key of the CoreCrypto database.
22/// To be used only once, when moving from CoreCrypto <= 5.x to CoreCrypto 6.x.
23#[uniffi::export]
24pub async fn migrate_database_key_type_to_bytes(
25    path: &str,
26    old_key: &str,
27    new_key: &DatabaseKey,
28) -> CoreCryptoResult<()> {
29    Database::migrate_db_key_type_to_bytes(path, old_key, &new_key.0)
30        .await
31        .map_err(CoreCryptoError::generic())
32}
33
34/// Updates the key of the CoreCrypto database.
35#[uniffi::export]
36pub async fn update_database_key(name: &str, old_key: &DatabaseKey, new_key: &DatabaseKey) -> CoreCryptoResult<()> {
37    let mut db = Database::open(ConnectionType::Persistent(name), &old_key.0)
38        .await
39        .map_err(CoreCryptoError::generic())?;
40    db.update_key(&new_key.0).await.map_err(CoreCryptoError::generic())?;
41    db.close().await.map_err(CoreCryptoError::generic())?;
42    Ok(())
43}
