10/// Information returned when a commit is created.
11#[derive(Debug)]
12#[cfg_attr(
13    target_family = "wasm",
14    wasm_bindgen(getter_with_clone),
15    derive(serde::Serialize, serde::Deserialize)
16)]
17#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Record))]
18pub struct CommitBundle {
19    /// A welcome message if there are pending Add proposals
20    pub welcome: Option<WelcomeMaybeArc>,
21    /// The commit message
22    pub commit: Vec<u8>,
23    /// `GroupInfo` if the commit is merged
24    pub group_info: GroupInfoBundle,
25    /// An encrypted message to fan out to all other conversation members in the new epoch
26    #[cfg_attr(target_family = "wasm", wasm_bindgen(js_name = "encryptedMessage", readonly))]
27    pub encrypted_message: Option<Vec<u8>>,
28}
29
30impl TryFrom<MlsCommitBundle> for CommitBundle {
31    type Error = CoreCryptoError;
32
33    fn try_from(msg: MlsCommitBundle) -> Result<Self, Self::Error> {
34        let encrypted_message = msg.encrypted_message.clone();
35        let (welcome, commit, group_info) = msg.to_bytes_triple()?;
36        let welcome = welcome.map(welcome_coerce_maybe_arc);
37        let group_info = group_info.into();
38        Ok(Self {
39            welcome,
40            commit,
41            group_info,
42            encrypted_message,
43        })
44    }
45}
