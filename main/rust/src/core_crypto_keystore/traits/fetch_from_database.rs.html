9/// Interface to fetch from the database either from the connection directly or through a
10/// transaction.
11///
12/// Fundamentally these are convenience methods, allowing you to do `let n_foos = database.count::<Foo>()`
13/// instead of `Foo::count(&mut database.conn)`.
14#[cfg_attr(target_family = "wasm", async_trait(?Send))]
15#[cfg_attr(not(target_family = "wasm"), async_trait)]
16pub trait FetchFromDatabase: Send + Sync {
17    /// Get an instance of `E` from the database by its primary key.
18    async fn get<E>(&self, id: &<E as Entity>::PrimaryKey) -> CryptoKeystoreResult<Option<E>>
19    where
20        E: Entity<ConnectionType = KeystoreDatabaseConnection>;
21
22    /// Count the number of `E`s in the database.
23    async fn count<E>(&self) -> CryptoKeystoreResult<u32>
24    where
25        E: Entity<ConnectionType = KeystoreDatabaseConnection>;
26
27    /// Load all `E`s from the database.
28    async fn load_all<E>(&self) -> CryptoKeystoreResult<Vec<E>>
29    where
30        E: Entity<ConnectionType = KeystoreDatabaseConnection>;
31
32    /// Get the requested unique entity from the database.
33    async fn get_unique<U>(&self) -> CryptoKeystoreResult<Option<U>>
34    where
35        U: UniqueEntity<ConnectionType = KeystoreDatabaseConnection>;
36}
