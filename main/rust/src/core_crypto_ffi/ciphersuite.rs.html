1//! Ciphersuites in bindings
2//!
3//! Both wasm-bindgen and uniffi support emitting enums, as long as they directly implement the enum;
4//! it doesn't work on newtypes around external enums. We therefore redefine the ciphersuites enum
5//! here with appropriate annotations such that it gets exported to all relevant bindings.

15/// MLS ciphersuites.
16#[allow(non_camel_case_types)]
17#[allow(clippy::upper_case_acronyms)]
18#[derive(Default, Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, derive_more::TryFrom)]
19#[try_from(repr)]
20#[repr(u16)]
21#[cfg_attr(target_family = "wasm", wasm_bindgen, derive(serde::Serialize, serde::Deserialize))]
22#[cfg_attr(not(target_family = "wasm"), derive(uniffi::Enum))]
23pub enum Ciphersuite {
24    /// DH KEM x25519 | AES-GCM 128 | SHA2-256 | Ed25519
25    #[default]
26    MLS_128_DHKEMX25519_AES128GCM_SHA256_Ed25519 = 0x0001,
27
28    /// DH KEM P256 | AES-GCM 128 | SHA2-256 | EcDSA P256
29    MLS_128_DHKEMP256_AES128GCM_SHA256_P256 = 0x0002,
30
31    /// DH KEM x25519 | Chacha20Poly1305 | SHA2-256 | Ed25519
32    MLS_128_DHKEMX25519_CHACHA20POLY1305_SHA256_Ed25519 = 0x0003,
33
34    /// DH KEM x448 | AES-GCM 256 | SHA2-512 | Ed448
35    MLS_256_DHKEMX448_AES256GCM_SHA512_Ed448 = 0x0004,
36
37    /// DH KEM P521 | AES-GCM 256 | SHA2-512 | EcDSA P521
38    MLS_256_DHKEMP521_AES256GCM_SHA512_P521 = 0x0005,
39
40    /// DH KEM x448 | Chacha20Poly1305 | SHA2-512 | Ed448
41    MLS_256_DHKEMX448_CHACHA20POLY1305_SHA512_Ed448 = 0x0006,
42
43    /// DH KEM P384 | AES-GCM 256 | SHA2-384 | EcDSA P384
44    MLS_256_DHKEMP384_AES256GCM_SHA384_P384 = 0x0007,
45}

118#[cfg_attr(target_family = "wasm", wasm_bindgen)]
119impl Ciphersuites {
120    pub fn from_u16s(ids: Vec<u16>) -> CoreCryptoResult<Self> {
121        let ciphersuites = ids
122            .into_iter()
123            .map(Ciphersuite::try_from)
124            .collect::<Result<_, _>>()
125            .map_err(CoreCryptoError::generic())?;
126        Ok(Self(ciphersuites))
127    }
128
129    /// The default `Ciphersuites` set contains only the single default ciphersuite.
130    //
131    // We implement this here instead of in the `impl Default` section to expose it to wasm_bindgen
132    // But we use `allow` here instead of `expect` because for some reason when building for WASM
133    // the expectation fails whether the allow line is present or not.
134    #[allow(clippy::should_implement_trait)]
135    pub fn default() -> Self {
136        let mut cs = HashSet::new();
137        cs.insert(Ciphersuite::default());
138        Self(cs)
139    }
140}

142impl Default for Ciphersuites {
143    fn default() -> Self {

145        <Self>::default()
146    }
147}
