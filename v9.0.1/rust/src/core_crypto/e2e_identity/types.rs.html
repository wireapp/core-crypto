5/// See [RFC 8555 Section 7.1.1](https://www.rfc-editor.org/rfc/rfc8555.html#section-7.1.1)
6#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
7#[serde(rename_all = "camelCase")]
8pub struct E2eiAcmeDirectory {
9    /// For fetching a new nonce used in [crate::prelude::E2eiEnrollment::new_account_request]
10    pub new_nonce: String,
11    /// URL to call with [crate::prelude::E2eiEnrollment::new_account_request]
12    pub new_account: String,
13    /// URL to call with [crate::prelude::E2eiEnrollment::new_order_request]
14    pub new_order: String,
15    /// Not yet used
16    pub revoke_cert: String,
17}
18
19impl From<wire_e2e_identity::prelude::AcmeDirectory> for E2eiAcmeDirectory {
20    fn from(directory: wire_e2e_identity::prelude::AcmeDirectory) -> Self {
21        Self {
22            new_nonce: directory.new_nonce.to_string(),
23            new_account: directory.new_account.to_string(),
24            new_order: directory.new_order.to_string(),
25            revoke_cert: directory.revoke_cert.to_string(),
26        }
27    }
28}

43/// Result of an order creation
44/// see [RFC 8555 Section 7.4](https://www.rfc-editor.org/rfc/rfc8555.html#section-7.4)
45#[derive(Debug, serde::Serialize, serde::Deserialize)]
46#[serde(rename_all = "camelCase")]
47pub struct E2eiNewAcmeOrder {
48    /// Opaque raw json value
49    pub delegate: super::Json,
50    /// Authorizations to create with [crate::prelude::E2eiEnrollment::new_authz_request]
51    pub authorizations: Vec<String>,
52}
53
54impl TryFrom<wire_e2e_identity::prelude::E2eiNewAcmeOrder> for E2eiNewAcmeOrder {
55    type Error = Error;
56
57    fn try_from(new_order: wire_e2e_identity::prelude::E2eiNewAcmeOrder) -> Result<Self> {
58        Ok(Self {
59            authorizations: new_order.authorizations.iter().map(url::Url::to_string).collect(),
60            delegate: serde_json::to_vec(&new_order.delegate)?,
61        })
62    }
63}
64
65impl TryFrom<E2eiNewAcmeOrder> for wire_e2e_identity::prelude::E2eiNewAcmeOrder {
66    type Error = Error;
67
68    fn try_from(new_order: E2eiNewAcmeOrder) -> Result<Self> {
69        let authorizations = new_order
70            .authorizations
71            .iter()
72            .map(|a| a.parse())
73            .collect::<Result<Vec<url::Url>, url::ParseError>>()?
74            .try_into()
75            .map_err(|_| Error::ImplementationError)?;
76
77        Ok(Self {
78            authorizations,
79            delegate: serde_json::to_value(new_order.delegate)?,
80        })
81    }
82}

84/// Result of an authorization creation
85/// see [RFC 8555 Section 7.5](https://www.rfc-editor.org/rfc/rfc8555.html#section-7.5)
86#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
87#[serde(rename_all = "camelCase")]
88pub struct E2eiNewAcmeAuthz {
89    /// DNS entry associated with those challenge
90    pub identifier: String,
91    /// ACME challenge + ACME key thumbprint
92    pub keyauth: Option<String>,
93    /// ACME Challenge
94    pub challenge: E2eiAcmeChallenge,
95}
96
97impl TryFrom<wire_e2e_identity::prelude::E2eiAcmeAuthorization> for E2eiNewAcmeAuthz {
98    type Error = Error;
99
100    fn try_from(authz: wire_e2e_identity::prelude::E2eiAcmeAuthorization) -> Result<Self> {
101        Ok(match authz {
102            wire_e2e_identity::prelude::E2eiAcmeAuthorization::User {
103                identifier,
104                keyauth,
105                challenge,
106            } => Self {
107                identifier,
108                keyauth: Some(keyauth),
109                challenge: challenge.try_into()?,
110            },
111            wire_e2e_identity::prelude::E2eiAcmeAuthorization::Device { identifier, challenge } => Self {
112                identifier,
113                keyauth: None,
114                challenge: challenge.try_into()?,
115            },
116        })
117    }
118}
119
120impl TryFrom<&E2eiNewAcmeAuthz> for wire_e2e_identity::prelude::E2eiAcmeAuthorization {
121    type Error = Error;
122
123    fn try_from(authz: &E2eiNewAcmeAuthz) -> Result<Self> {
124        Ok(match &authz.keyauth {
125            None => Self::Device {
126                identifier: authz.identifier.clone(),
127                challenge: (&authz.challenge).try_into()?,
128            },
129            Some(keyauth) => Self::User {
130                identifier: authz.identifier.clone(),
131                keyauth: keyauth.clone(),
132                challenge: (&authz.challenge).try_into()?,
133            },
134        })
135    }
136}

138/// For creating a challenge
139/// see [RFC 8555 Section 7.5.1](https://www.rfc-editor.org/rfc/rfc8555.html#section-7.5.1)
140#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
141#[serde(rename_all = "camelCase")]
142pub struct E2eiAcmeChallenge {
143    /// Opaque raw json value
144    pub delegate: super::Json,
145    /// URL to call for the acme server to complete the challenge
146    pub url: String,
147    /// Non-standard, Wire specific claim. Indicates the consumer from where it should get the challenge
148    /// proof. Either from wire-server "/access-token" endpoint in case of a DPoP challenge, or from
149    /// an OAuth token endpoint for an OIDC challenge
150    pub target: String,
151}
152
153impl TryFrom<wire_e2e_identity::prelude::E2eiAcmeChallenge> for E2eiAcmeChallenge {
154    type Error = Error;
155
156    fn try_from(chall: wire_e2e_identity::prelude::E2eiAcmeChallenge) -> Result<Self> {
157        Ok(Self {
158            delegate: serde_json::to_vec(&chall.delegate)?,
159            url: chall.url.to_string(),
160            target: chall.target.to_string(),
161        })
162    }
163}
164
165impl TryFrom<&E2eiAcmeChallenge> for wire_e2e_identity::prelude::E2eiAcmeChallenge {
166    type Error = Error;
167
168    fn try_from(chall: &E2eiAcmeChallenge) -> Result<Self> {
169        Ok(Self {
170            delegate: serde_json::from_slice(&chall.delegate[..])?,
171            url: chall.url.parse()?,
172            target: chall.target.parse()?,
173        })
174    }
